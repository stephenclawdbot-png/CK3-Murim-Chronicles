murim_calamity_ai_expansion2.0001 = {
	type = character_event
	title = murim_calamity_ai_exp2.0001.t
	desc = murim_calamity_ai_exp2.0001.desc
	theme = intrigue
	left_portrait = { character = root animation = scheme }

	trigger = {
		has_global_variable = murim_calamity_active
		murim_is_martial_artist_trigger = yes
		is_ruler = yes
		any_ally = { murim_is_martial_artist_trigger = yes is_ruler = yes }
		NOT = { has_character_flag = calamity_betrayal_considered }
	}

	weight_multiplier = {
		base = 30
		modifier = { add = 30 has_trait = deceitful }
		modifier = { add = 20 intrigue >= 15 }
		modifier = { add = -20 has_trait = honest }
	}

	immediate = {
		set_character_flag = calamity_betrayal_considered
		random_ally = {
			limit = { murim_is_martial_artist_trigger = yes is_ruler = yes }
			save_scope_as = betrayal_target
		}
	}

	option = {
		name = murim_calamity_ai_exp2.0001.a
		trigger = { exists = scope:betrayal_target }
		add_character_modifier = { modifier = murim_calamity_betrayer years = 20 }
		scope:betrayal_target = {
			add_opinion = { modifier = murim_betrayed_during_calamity target = root opinion = -100 }
			add_character_modifier = { modifier = murim_betrayal_victim years = 10 }
			random = { chance = 40 add_trait = wounded_2 }
		}
		add_gold = 500
		add_prestige = -500
		trigger_event = { id = murim_calamity_ai_expansion2.0002 days = { 30 60 } }
	}
	option = {
		name = murim_calamity_ai_exp2.0001.b
		add_character_modifier = { modifier = murim_calamity_loyal_ally years = 20 }
		if = { limit = { exists = scope:betrayal_target }
			scope:betrayal_target = { add_opinion = { modifier = murim_stood_by_in_crisis target = root opinion = 30 } }
		}
		add_piety = 300
	}
	option = {
		name = murim_calamity_ai_exp2.0001.c
		add_character_modifier = { modifier = murim_calamity_lone_wolf years = 15 }
		add_prowess = 3
		add_martial = 2
	}
}
