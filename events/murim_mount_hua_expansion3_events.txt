murim_mount_hua_expansion3.0001 = {
	type = character_event
	title = murim_mount_hua_expansion3.0001.t
	desc = murim_mount_hua_expansion3.0001.desc
	theme = martial
	
	# PLUM BLOSSOM FORMATION MASTERY — Group sword formation training
	# Mount Hua's signature formation requires 5 swordsmen in sync
	
	trigger = {
		has_trait = mount_hua_sect
		has_trait = expert_rank
		NOT = { has_character_flag = plum_formation_trained }
		any_courtier = {
			count >= 4
			has_trait = mount_hua_sect
			prowess >= 12
		}
	}
	
	immediate = {
		set_character_flag = {
			flag = plum_formation_trained
			days = 3650
		}
		every_courtier = {
			limit = {
				has_trait = mount_hua_sect
				prowess >= 12
			}
			add_to_list = formation_members
		}
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.a
		# Train the full 5-man Plum Blossom Formation
		custom_tooltip = murim_plum_formation_full_tt
		every_in_list = {
			list = formation_members
			limit = { count <= 4 }
			add_character_modifier = {
				modifier = murim_plum_formation_member
				years = 10
			}
			change_prowess = 3
		}
		add_character_modifier = {
			modifier = murim_plum_formation_leader
			years = 10
		}
		change_prowess = 5
		add_prestige = 200
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.b
		# Develop a new variation — 3-man swift formation
		trigger = {
			martial >= 16
		}
		custom_tooltip = murim_swift_formation_tt
		add_trait = formation_innovator
		add_character_modifier = {
			modifier = murim_swift_plum_formation
			years = 15
		}
		change_prowess = 4
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.c
		# Solo sword dance — transcend the formation entirely
		trigger = {
			has_trait = grandmaster_rank
		}
		custom_tooltip = murim_solo_sword_dance_tt
		add_character_modifier = {
			modifier = murim_transcendent_swordsman
			years = 20
		}
		change_prowess = 8
		add_prestige = 500
	}
}

murim_mount_hua_expansion3.0002 = {
	type = character_event
	title = murim_mount_hua_expansion3.0002.t
	desc = murim_mount_hua_expansion3.0002.desc
	theme = intrigue
	
	# SWORD VS QI FACTION — Internal Mount Hua power struggle deepens
	# The Sword faction (pure sword technique) vs Qi faction (internal energy focus)
	
	trigger = {
		has_trait = mount_hua_sect
		has_trait = clan_leader
		NOT = { has_character_flag = mount_hua_faction_war_3 }
	}
	
	immediate = {
		set_character_flag = {
			flag = mount_hua_faction_war_3
			days = 7300
		}
		# Find sword faction leader
		random_courtier = {
			limit = {
				has_trait = mount_hua_sect
				prowess >= 18
			}
			weight = {
				base = 10
				modifier = { add = 20 prowess >= 22 }
			}
			save_scope_as = sword_faction_leader
		}
		# Find qi faction leader
		random_courtier = {
			limit = {
				has_trait = mount_hua_sect
				learning >= 16
				NOT = { this = scope:sword_faction_leader }
			}
			weight = {
				base = 10
				modifier = { add = 20 learning >= 20 }
			}
			save_scope_as = qi_faction_leader
		}
	}
	
	option = {
		name = murim_mount_hua_expansion3.0002.a
		# Support the Sword Faction — pure technique is Mount Hua's identity
		custom_tooltip = murim_support_sword_faction_tt
		if = {
			limit = { exists = scope:sword_faction_leader }
			scope:sword_faction_leader = {
				add_opinion = {
					modifier = murim_faction_support_opinion
					target = root
					opinion = 40
				}
				add_prestige = 200
			}
		}
		if = {
			limit = { exists = scope:qi_faction_leader }
			scope:qi_faction_leader = {
				add_opinion = {
					modifier = murim_faction_opposed_opinion
					target = root
					opinion = -30
				}
			}
		}
		add_character_modifier = {
			modifier = murim_sword_faction_supporter
			years = 10
		}
		# All sword-focused disciples gain power
		every_courtier = {
			limit = {
				has_trait = mount_hua_sect
				prowess >= 15
			}
			change_prowess = 2
		}
	}
	
	option = {
		name = murim_mount_hua_expansion3.0002.b
		# Support the Qi Faction — internal energy is the true foundation
		custom_tooltip = murim_support_qi_faction_tt
		if = {
			limit = { exists = scope:qi_faction_leader }
			scope:qi_faction_leader = {
				add_opinion = {
					modifier = murim_faction_support_opinion
					target = root
					opinion = 40
				}
				add_prestige = 200
			}
		}
		if = {
			limit = { exists = scope:sword_faction_leader }
			scope:sword_faction_leader = {
				add_opinion = {
					modifier = murim_faction_opposed_opinion
					target = root
					opinion = -30
				}
				# Sword faction leader may rebel
				random = {
					chance = 30
					trigger_event = {
						id = murim_mount_hua_expansion3.0003
						days = { 30 90 }
					}
				}
			}
		}
		add_character_modifier = {
			modifier = murim_qi_faction_supporter
			years = 10
		}
	}
	
	option = {
		name = murim_mount_hua_expansion3.0002.c
		# Unify both factions — sword and qi are one
		trigger = {
			diplomacy >= 18
			martial >= 14
		}
		custom_tooltip = murim_unify_factions_tt
		add_prestige = 400
		add_character_modifier = {
			modifier = murim_unified_mount_hua
			years = 20
		}
		if = {
			limit = { exists = scope:sword_faction_leader }
			scope:sword_faction_leader = {
				add_opinion = {
					modifier = murim_faction_unified_opinion
					target = root
					opinion = 20
				}
			}
		}
		if = {
			limit = { exists = scope:qi_faction_leader }
			scope:qi_faction_leader = {
				add_opinion = {
					modifier = murim_faction_unified_opinion
					target = root
					opinion = 20
				}
			}
		}
	}
}
