###############################
# CULTIVATION REALM CHECKS
###############################

murim_is_mortal_realm_trigger = {
	OR = {
		has_variable = murim_cultivation_stage
		NOT = { has_variable = murim_cultivation_stage }
	}
	trigger_if = {
		limit = { has_variable = murim_cultivation_stage }
		var:murim_cultivation_stage < 4
	}
	trigger_else = {
		always = yes
	}
}

murim_is_earth_realm_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 4
	var:murim_cultivation_stage < 7
}

murim_is_heaven_realm_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 7
	var:murim_cultivation_stage < 10
}

murim_is_transcendent_realm_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 10
}

murim_is_at_least_earth_realm_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 4
}

murim_is_at_least_heaven_realm_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 7
}

###############################
# CULTIVATION STAGE CHECKS
###############################

murim_has_cultivation_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 1
}

murim_at_stage_cap_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 12
}

murim_can_advance_stage_trigger = {
	murim_has_cultivation_trigger = yes
	NOT = { murim_at_stage_cap_trigger = yes }
	custom_description = {
		text = murim_not_in_qi_deviation_tt
		NOT = { has_character_modifier = murim_qi_deviation_penalty }
	}
	custom_description = {
		text = murim_not_cultivation_cooldown_tt
		NOT = { has_character_modifier = murim_advancement_cooldown }
	}
}

murim_meets_qi_threshold_trigger = {
	has_variable = murim_qi_amount
	trigger_if = {
		limit = { murim_is_mortal_realm_trigger = yes }
		var:murim_qi_amount >= 100
	}
	trigger_else_if = {
		limit = { murim_is_earth_realm_trigger = yes }
		var:murim_qi_amount >= 300
	}
	trigger_else_if = {
		limit = { murim_is_heaven_realm_trigger = yes }
		var:murim_qi_amount >= 600
	}
	trigger_else = {
		var:murim_qi_amount >= 1000
	}
}

murim_is_early_stage_cultivator_trigger = {
	murim_has_cultivation_trigger = yes
	trigger_if = {
		limit = { has_variable = murim_cultivation_stage }
		var:murim_cultivation_stage <= 3
	}
}

murim_is_mid_stage_cultivator_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 4
	var:murim_cultivation_stage <= 7
}

murim_is_late_stage_cultivator_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 8
	var:murim_cultivation_stage <= 10
}

murim_is_peak_cultivator_trigger = {
	has_variable = murim_cultivation_stage
	var:murim_cultivation_stage >= 11
}

###############################
# BREAKTHROUGH READINESS
###############################

murim_ready_for_breakthrough_trigger = {
	murim_can_advance_stage_trigger = yes
	murim_meets_qi_threshold_trigger = yes
	age >= 16
	is_alive = yes
	is_imprisoned = no
	custom_description = {
		text = murim_not_incapacitated_tt
		NOR = {
			has_trait = incapable
			has_trait = wounded_3
		}
	}
}

murim_breakthrough_high_chance_trigger = {
	murim_ready_for_breakthrough_trigger = yes
	OR = {
		has_trait = genius
		has_trait = quick
		has_trait = shrewd
		learning >= 16
	}
	stress < 50
}

murim_breakthrough_dangerous_trigger = {
	murim_ready_for_breakthrough_trigger = yes
	OR = {
		has_character_modifier = murim_unstable_foundation
		stress >= 80
		AND = {
			murim_is_at_least_heaven_realm_trigger = yes
			has_trait = impatient
		}
	}
}

murim_breakthrough_guaranteed_failure_trigger = {
	OR = {
		has_trait = incapable
		has_character_modifier = murim_qi_deviation_penalty
		AND = {
			stress >= 95
			has_trait = lunatic
		}
		AND = {
			has_character_modifier = murim_meridian_shattered
			NOT = { has_character_modifier = murim_meridian_repaired }
		}
	}
}

murim_breakthrough_has_catalyst_trigger = {
	OR = {
		has_character_modifier = murim_breakthrough_pill_active
		has_character_modifier = murim_heavenly_lightning_survivor
		has_character_modifier = murim_enlightenment_state
		any_held_artifact = {
			has_variable = murim_breakthrough_catalyst
		}
		trigger_if = {
			limit = { has_variable = murim_insight_points }
			var:murim_insight_points >= 10
		}
	}
}

murim_can_duel_trigger = {
	is_alive = yes

	NOT = { has_trait = incapable }
	NOT = { has_character_modifier = murim_duel_cooldown }

murim_can_accept_formal_duel_trigger = {
	murim_can_duel_trigger = yes
	custom_description = {
		text = murim_not_recovering_tt
		NOR = {
			has_trait = wounded_3
			has_character_modifier = murim_critical_injury
			has_character_modifier = murim_near_death_recovery
		}
	}
}

murim_is_duel_favorite_trigger = {
	prowess >= 20
	OR = {
		murim_is_at_least_earth_realm_trigger = yes
		has_trait = blademaster_3
		has_trait = herculean
	}
}

murim_should_duel_be_lethal_trigger = {
	OR = {
		# Blood feud
		trigger_if = {
			limit = { exists = scope:duel_opponent }
			has_relation_rival = scope:duel_opponent
		}
		# Death match declared
		has_character_flag = murim_death_match_declared
		# Demonic cultivators fight to the death
		AND = {
			OR = {
				has_trait = murim_blood_demon_art
				has_trait = murim_soul_devouring_art
			}
			trigger_if = {
				limit = { exists = scope:duel_opponent }
				scope:duel_opponent = {
					OR = {
						has_trait = murim_righteous_heart
						has_trait = just
					}
				}
			}
		}
	}
}

murim_duel_should_cause_trauma_trigger = {
	OR = {
		# Close loss
		trigger_if = {
			limit = { has_variable = murim_last_duel_margin }
			var:murim_last_duel_margin <= 5
		}
		# Already traumatized
		has_trait = duel_ptsd
		# Young and inexperienced
		AND = {
			age < 20
			trigger_if = {
				limit = { has_variable = murim_duel_count }
				var:murim_duel_count < 3
			}
		}
	}
}

murim_is_undefeated_duelist_trigger = {
	trigger_if = {
		limit = { has_variable = murim_duel_losses }
		var:murim_duel_losses = 0
	}
	trigger_if = {
		limit = { has_variable = murim_duel_wins }
		var:murim_duel_wins >= 5
	}
}
