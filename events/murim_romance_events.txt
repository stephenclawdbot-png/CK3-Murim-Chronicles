######################################
# murim_romance.0001 - Dual Cultivation Proposal
# A fellow cultivator of opposite/complementary qi proposes dual cultivation
######################################
murim_romance.0001 = {
	type = character_event
	title = murim_romance.0001.t
	desc = murim_romance.0001.desc
	theme = love
	left_portrait = {
		character = root
		animation = personality_compassionate
	}
	right_portrait = {
		character = scope:romance_partner
		animation = personality_compassionate
	}

	trigger = {
		is_available_adult = yes
		NOT = { has_character_flag = dual_cultivation_cooldown }
		any_relation = {
			type = friend
			is_available_adult = yes
			has_trait = murim_cultivator
			NOT = { has_character_flag = dual_cultivation_cooldown }
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = qi_harmony_adept
		}
		modifier = {
			add = 1
			has_trait = murim_inner_disciple
		}
	}

	immediate = {
		random_relation = {
			type = friend
			limit = {
				is_available_adult = yes
				has_trait = murim_cultivator
				NOT = { has_character_flag = dual_cultivation_cooldown }
			}
			save_scope_as = romance_partner
		}
		add_character_flag = {
			flag = dual_cultivation_cooldown
			years = 3
		}
	}

	# Accept dual cultivation
	option = {
		name = murim_romance.0001.a
		trigger = {
			is_attracted_to_gender_of = scope:romance_partner
		}
		add_opinion = {
			target = scope:romance_partner
			modifier = murim_dual_cultivation_bond
			opinion = 30
		}
		scope:romance_partner = {
			add_opinion = {
				target = root
				modifier = murim_dual_cultivation_bond
				opinion = 30
			}
		}
		create_character_memory = {
			type = murim_dual_cultivation_memory
			participants = {
				partner = scope:romance_partner
			}
		}
		# Cultivation boost from harmonized qi
		add_prowess = 1
		scope:romance_partner = {
			add_prowess = 1
		}
		# Chance of deeper connection
		random_list = {
			50 = {
				set_relation_lover = scope:romance_partner
				trigger_event = {
					id = murim_romance.0003
					days = { 60 120 }
				}
			}
			30 = {
				# Successful but platonic
				trigger_event = {
					id = murim_cultivation_events.0001
					days = { 30 60 }
				}
			}
			20 = {
				# Qi deviation from incompatible energies
				add_trait = qi_deviation_scarred
				create_character_memory = {
					type = murim_qi_deviation_memory
					participants = {
						partner = scope:romance_partner
					}
				}
				trigger_event = {
					id = murim_cultivation_events.0005
					days = { 5 15 }
				}
			}
		}
		ai_chance = {
			base = 70
			modifier = {
				add = 20
				has_trait = lustful
			}
		}
	}

	# Decline respectfully
	option = {
		name = murim_romance.0001.b
		add_opinion = {
			target = scope:romance_partner
			modifier = murim_declined_dual_cultivation
			opinion = -10
		}
		create_character_memory = {
			type = murim_declined_proposal_memory
			participants = {
				partner = scope:romance_partner
			}
		}
		stress_impact = {
			shy = minor_stress_impact_loss
			chaste = minor_stress_impact_loss
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = chaste
			}
		}
	}

	# Attempt to steal their cultivation technique instead
	option = {
		name = murim_romance.0001.c
		trait = deceitful
		add_dread = 10
		duel = {
			skill = intrigue
			target = scope:romance_partner
			50 = {
				desc = murim_romance.0001.c.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 0.5
				}
				# Successfully steal technique insight
				add_prowess = 2
				create_character_memory = {
					type = murim_stolen_technique_memory
					participants = {
						victim = scope:romance_partner
					}
				}
				scope:romance_partner = {
					add_opinion = {
						target = root
						modifier = murim_stolen_technique_hatred
						opinion = -50
					}
				}
				trigger_event = {
					id = murim_duel_events.0001
					days = { 30 90 }
				}
			}
			50 = {
				desc = murim_romance.0001.c.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -0.5
				}
				# Caught! They attack you
				add_prestige = -200
				scope:romance_partner = {
					set_relation_rival = root
				}
				random_list = {
					40 = {
						add_trait = wounded_1
					}
					30 = {
						add_trait = wounded_2
					}
					30 = {
						# Nothing physical but massive shame
						add_trait = craven
					}
				}
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				add = 15
				has_trait = deceitful
			}
		}
	}
}

######################################
# murim_romance.0002 - Forbidden Love: Rival Sect Romance
# Fall in love with someone from a rival/enemy sect
######################################

	type = character_event
	title = murim_romance.0002.t

		character = root
		animation = worry
	}

		character = scope:forbidden_lover
		animation = personality_compassionate
	}

	trigger = {
		is_available_adult = yes
		has_trait = murim_cultivator
		NOT = { has_character_flag = forbidden_love_cooldown }

			is_available_adult = yes
			has_trait = murim_cultivator
			has_relation_rival = root
			NOR = {
				has_relation_lover = root
				has_relation_soulmate = root
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = lustful
		}
		modifier = {
			add = 2
			has_trait = compassionate
		}
	}

	immediate = {
		random_living_character = {
			limit = {
				is_available_adult = yes
				has_trait = murim_cultivator
				has_relation_rival = root
				NOR = {
					has_relation_lover = root
					has_relation_soulmate = root
				}
			}
			save_scope_as = forbidden_lover
		}
		add_character_flag = {
			flag = forbidden_love_cooldown
			years = 5
		}
	}

	# Pursue the forbidden romance despite all risks
	option = {
		name = murim_romance.0002.a
		set_relation_lover = scope:forbidden_lover
		add_opinion = {
			target = scope:forbidden_lover
			modifier = murim_forbidden_love_passion
			opinion = 40
		}
		scope:forbidden_lover = {
			add_opinion = {
				target = root
				modifier = murim_forbidden_love_passion
				opinion = 40
			}
		}
		create_character_memory = {
			type = murim_forbidden_love_memory
			participants = {
				beloved = scope:forbidden_lover
			}
		}
		add_prestige = -100
		# Risk of discovery by your sect
		random_list = {
			40 = {
				# Secret is safe for now
				add_character_flag = {
					flag = secret_forbidden_love
					years = 2
				}
				trigger_event = {
					id = murim_romance.0003
					days = { 90 180 }
				}
			}
			35 = {
				# Discovered! Sect demands you choose
				trigger_event = {
					id = murim_romance.0006
					days = { 30 60 }
				}
			}
			25 = {
				# Ambushed by both sects
				add_trait = wounded_2
				scope:forbidden_lover = {
					add_trait = wounded_1
				}
				trigger_event = {
					id = murim_sect_events.0001
					days = { 10 30 }
				}
			}
		}
		stress_impact = {
			brave = minor_stress_impact_loss
			craven = medium_stress_impact_gain
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = lustful
			}
		}
	}

	# Suppress your feelings for the sake of your sect
	option = {
		name = murim_romance.0002.b
		add_stress = 30
		add_opinion = {
			target = scope:forbidden_lover
			modifier = murim_suppressed_feelings
			opinion = -5
		}
		create_character_memory = {
			type = murim_suppressed_love_memory
			participants = {
				beloved = scope:forbidden_lover
			}
		}
		# Channel pain into training
		add_prowess = 1
		trigger_event = {
			id = murim_training_events.0001
			days = { 15 30 }
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = temperate
			}
		}
	}

	# Use the romance as a ploy to infiltrate their sect
	option = {
		name = murim_romance.0002.c
		trait = deceitful

		add_dread = 15
		create_character_memory = {
			type = murim_romance_betrayal_scheme_memory
			participants = {
				target = scope:forbidden_lover
			}
		}
		trigger_event = {
			id = murim_jianghu_events.0001
			days = { 60 120 }
		}
		stress_impact = {
			compassionate = major_stress_impact_gain

		}
		ai_chance = {
			base = 20
			modifier = {
				add = 25
				has_trait = deceitful
			}
		}
	}
}

######################################
# murim_romance.0004 - Marriage Duel Challenge
# Someone challenges you for your beloved's hand
######################################
murim_romance.0004 = {
	type = character_event
	title = murim_romance.0004.t
	desc = murim_romance.0004.desc
	theme = martial
	left_portrait = {
		character = root
		animation = anger
	}
	right_portrait = {
		character = scope:marriage_rival
		animation = personality_bold
	}

	trigger = {
		is_available_adult = yes
		has_trait = murim_cultivator
		any_relation = {
			type = lover
			is_available_adult = yes
		}
		NOT = { has_character_flag = marriage_duel_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			any_relation = {
				type = lover
				has_trait = beauty_good
			}
		}
	}

	immediate = {
		random_relation = {
			type = lover
			limit = {
				is_available_adult = yes
			}
			save_scope_as = beloved
		}
		# Create the rival
		create_character = {
			culture = root.culture
			location = root.location
			age = { 22 40 }

			random_traits = yes

			dynasty = none

			faith = root.faith

			gender_female_chance = 50

			trait = murim_martial_artist
			save_scope_as = marriage_rival
		}
		scope:marriage_rival = {
			add_trait = murim_cultivator
			add_trait = ambitious
			set_relation_rival = root
		}
		add_character_flag = {
			flag = marriage_duel_cooldown
			years = 5
		}
	}

	# Accept the marriage duel with honor
	option = {
		name = murim_romance.0004.a
		duel = {
			skill = prowess
			target = scope:marriage_rival
			50 = {
				desc = murim_romance.0004.a.victory
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				add_prestige = 300
				add_opinion = {
					target = scope:beloved
					modifier = murim_defended_love_honor
					opinion = 30
				}
				create_character_memory = {
					type = murim_marriage_duel_victory_memory
					participants = {
						rival = scope:marriage_rival
						beloved = scope:beloved
					}
				}
				# Defeated rival
				scope:marriage_rival = {
					random_list = {
						40 = {
							add_trait = wounded_2
						}
						30 = {
							add_trait = incapable
						}
						30 = {
							death = {
								death_reason = death_duel
								killer = root
							}
							root = {
								create_character_memory = {
									type = murim_killed_marriage_rival_memory
									participants = {
										victim = scope:marriage_rival
									}
								}
								trigger_event = {
									id = murim_ranker_death_events.0001
									days = { 5 15 }
								}
							}
						}
					}
				}
				trigger_event = {
					id = murim_ranking_events.0001
					days = { 30 60 }
				}
			}
			50 = {
				desc = murim_romance.0004.a.defeat
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				add_prestige = -200
				random_list = {
					50 = {
						add_trait = wounded_2
					}
					30 = {
						add_trait = wounded_3
					}
					20 = {
						add_trait = incapable
					}
				}
				create_character_memory = {
					type = murim_marriage_duel_defeat_memory
					participants = {
						rival = scope:marriage_rival
						beloved = scope:beloved
					}
				}
				# Beloved may be taken or choose to stay
				random_list = {
					60 = {
						scope:beloved = {
							remove_relation_lover = root
						}
					}
					40 = {
						# Beloved stays loyal despite defeat
						scope:beloved = {
							add_opinion = {
								target = root
								modifier = murim_loyal_despite_defeat
								opinion = 20
							}
						}
					}
				}
				trigger_event = {
					id = murim_training_events.0001
					days = { 30 60 }
				}
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}

	# Refuse -- resolve this through diplomacy
	option = {
		name = murim_romance.0004.b
		add_prestige = -100
		add_opinion = {
			target = scope:beloved
			modifier = murim_refused_to_fight_for_love
			opinion = -20
		}
		create_character_memory = {
			type = murim_refused_marriage_duel_memory
			participants = {
				rival = scope:marriage_rival
				beloved = scope:beloved
			}
		}
		scope:marriage_rival = {
			add_prestige = 100
		}
		trigger_event = {
			id = murim_politics_events.0001
			days = { 30 60 }
		}
		stress_impact = {
			brave = major_stress_impact_gain
			craven = minor_stress_impact_loss
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 20
				has_trait = craven
			}
		}
	}

	# Elope -- flee together under cover of night
	option = {
		name = murim_romance.0004.c
		add_prestige = -300
		add_opinion = {
			target = scope:beloved
			modifier = murim_eloped_together
			opinion = 40
		}
		scope:beloved = {
			add_opinion = {
				target = root
				modifier = murim_eloped_together
				opinion = 40
			}
		}
		create_character_memory = {
			type = murim_elopement_memory

			}
		}
		# Both become wanderers
		add_trait = wandering_cultivator
		trigger_event = {
			id = murim_daily_life_events.0001
			days = { 30 90 }
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 20
				has_trait = lustful
			}
		}
	}
}
