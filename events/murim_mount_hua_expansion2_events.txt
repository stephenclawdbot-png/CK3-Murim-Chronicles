murim_mount_hua_expansion2.0001 = {
	type = character_event
	title = murim_mount_hua_expansion2.0001.t
	desc = murim_mount_hua_expansion2.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:mount_hua_elder
		animation = personality_rational
	}
	
	trigger = {
		has_trait = mount_hua_disciple
		NOT = { has_character_flag = plum_blossom_trial_complete }
		age >= 16
		prowess >= 12
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = sword_prodigy
		}
		modifier = {
			add = 1
			prowess >= 18
		}
	}
	
	immediate = {
		random_relation = {
			type = mentor
			save_scope_as = mount_hua_elder
		}
		if = {
			limit = { NOT = { exists = scope:mount_hua_elder } }
			random_courtier = {
				limit = {
					has_trait = mount_hua_elder
					age >= 50
				}
				save_scope_as = mount_hua_elder
			}
		}
		set_variable = {
			name = plum_blossom_petals_cut
			value = 0
		}
	}
	
	# Option A: Face the trial with pure sword technique
	option = {
		name = murim_mount_hua_expansion2.0001.a
		
		duel = {
			skill = prowess
			target = scope:mount_hua_elder
			50 = {
				desc = murim_mount_hua_expansion2.0001.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_mount_hua_expansion2.0001.a.success.toast
					left_icon = root
					add_character_modifier = {
						modifier = plum_blossom_sword_initiate
						years = 10
					}
					add_prowess = 2
					add_prestige = 200
				}
				set_character_flag = plum_blossom_trial_complete
			}
			50 = {
				desc = murim_mount_hua_expansion2.0001.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_mount_hua_expansion2.0001.a.failure.toast
					left_icon = root
					add_character_modifier = {
						modifier = plum_blossom_trial_failed
						years = 3
					}
					add_stress = 20
				}
			}
		}
		
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				prowess >= 20
			}
		}
	}
	
	# Option B: Use inner qi to enhance the sword
	option = {
		name = murim_mount_hua_expansion2.0001.b
		trigger = {
			has_trait = qi_cultivator
		}
		
		add_character_modifier = {
			modifier = qi_enhanced_plum_blossom
			years = 5
		}
		add_prowess = 1
		add_prestige = 150
		set_character_flag = plum_blossom_trial_complete
		
		stress_impact = {
			ambitious = medium_stress_loss
			lazy = medium_stress_gain
		}
		
		ai_chance = {
			base = 40
		}
	}
	
	# Option C: Refuse the trial - not ready yet
	option = {
		name = murim_mount_hua_expansion2.0001.c
		
		add_stress = 10
		add_character_flag = {
			flag = plum_blossom_trial_refused
			years = 2
		}
		
		ai_chance = {
			base = 10
		}
	}
}
