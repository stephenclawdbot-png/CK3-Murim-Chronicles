# 1. Inner Demon Tribulation
# During meditation, cultivator's inner demons manifest as a spiritual trial.
# Must confront past sins, fears, and desires to advance.
##############################
murim_cult_exp.0001 = {
	type = character_event
	title = murim_cult_exp.0001.t
	desc = murim_cult_exp.0001.desc
	theme = mystic
	left_portrait = {
		character = root
		animation = personality_zealous
	}

	trigger = {
		is_alive = yes
		has_variable = murim_cultivation_rank
		var:murim_cultivation_rank >= 3
		NOT = { has_character_modifier = inner_demon_tribulation_cooldown }
		NOT = { has_trait = incapable }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_trait = wrathful
		}
		modifier = {
			add = 0.5
			has_trait = lustful
		}
		modifier = {
			add = 0.3
			has_trait = murim_vengeful
		}
		modifier = {
			add = 0.5
			stress >= 50
		}
	}

	immediate = {
		add_character_modifier = {
			modifier = inner_demon_tribulation_cooldown
			years = 5
		}
	}

	# Face the inner demon head-on with iron will
	option = {
		name = murim_cult_exp.0001.a
		duel = {
			skill = learning
			target = root
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				min = 5
				desc = murim_cult_exp.0001.a.success
				# Demons conquered - massive cultivation breakthrough
				add_piety = 500
				add_learning = 2
				change_variable = {
					name = murim_cultivation_xp
					add = 300
				}
				add_character_modifier = {
					modifier = inner_demon_conquered
					years = 10
				}
				if = {
					limit = { has_trait = wrathful }
					remove_trait = wrathful
					add_trait = calm
				}
				create_character_memory = {
					type = murim_inner_demon_conquered_memory
				}
				# Chain: breakthrough attempt after conquering demons
				trigger_event = { id = murim_cult.0003 days = { 30 60 } }
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min = 5
				desc = murim_cult_exp.0001.a.failure
				# Inner demon overwhelms - qi deviation
				add_stress = 80
				random_list = {
					40 = {
						add_trait = lunatic
						create_character_memory = {
							type = murim_inner_demon_possession_memory
						}
					}
					35 = {
						add_trait = wounded_2
						add_character_modifier = {
							modifier = qi_deviation_recovery
							years = 3
						}
					}
					15 = {
						add_trait = incapable
						create_character_memory = {
							type = murim_inner_demon_shattered_mind_memory
						}
					}
					10 = {
						# Demon takes partial control - gain dark power at cost
						add_trait = sadistic
						add_prowess = 5
						change_variable = {
							name = murim_cultivation_xp
							add = -200
						}
						create_character_memory = {
							type = murim_demon_pact_memory
						}
						# Chain: may trigger jianghu dark path
						trigger_event = { id = murim_jianghu.0001 days = { 60 120 } }
					}
				}
			}
		}
		ai_chance = {
			base = 60
			modifier = { add = 20 has_trait = brave }
			modifier = { add = -20 has_trait = craven }
		}
	}

	# Seek a master's guidance to suppress the demons
	option = {
		name = murim_cult_exp.0001.b
		trigger = {
			any_relation = {
				type = mentor
				is_alive = yes
			}
		}
		random_relation = {
			type = mentor
			save_scope_as = guiding_master
		}
		add_piety = 100
		add_stress = -30
		scope:guiding_master = {
			add_opinion = {
				target = root
				modifier = murim_grateful_disciple_opinion
				opinion = 30
			}
		}
		random_list = {
			70 = {
				# Master helps suppress demons
				add_character_modifier = {
					modifier = demons_suppressed_not_conquered
					years = 5
				}
				create_character_memory = {
					type = murim_master_demon_suppression_memory
					participants = { master = scope:guiding_master }
				}
			}
			30 = {
				# Master's method backfires - both injured
				add_trait = wounded_1
				scope:guiding_master = {
					add_trait = wounded_1
				}
				create_character_memory = {
					type = murim_demon_suppression_backfire_memory
					participants = { master = scope:guiding_master }
				}
				# Chain: mentor relationship strained
				trigger_event = { id = murim_mentor.0005 days = { 14 30 } }
			}
		}
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = humble }
		}
	}

	# Use forbidden demonic cultivation to absorb the demon
	option = {
		name = murim_cult_exp.0001.c
		add_piety = -300
		add_prowess = 3
		add_intrigue = 2
		change_variable = {
			name = murim_cultivation_xp
			add = 150
		}
		add_trait = deviant
		add_character_modifier = {
			modifier = demonic_cultivation_taint
			years = 15
		}
		create_character_memory = {
			type = murim_demonic_absorption_memory
		}
		# Reputation destroyed if discovered
		hidden_effect = {
			random_list = {
				50 = {
					every_courtier = {
						limit = { is_alive = yes }
						add_opinion = {
							target = root
							modifier = murim_demonic_cultivator_discovered_opinion
							opinion = -40
						}
					}
				}
				50 = { }
			}
		}
		# Chain: jianghu underworld notices
		trigger_event = { id = murim_jianghu.0005 days = { 30 90 } }
		ai_chance = {
			base = 10
			modifier = { add = 20 has_trait = ambitious }
			modifier = { add = 15 has_trait = sadistic }
		}
	}

	# Abandon the tribulation and retreat
	option = {
		name = murim_cult_exp.0001.d
		add_stress = 40
		change_variable = {
			name = murim_cultivation_xp
			add = -100
		}
		add_character_modifier = {
			modifier = qi_stagnation
			years = 3
		}
		create_character_memory = {
			type = murim_inner_demon_fled_memory
		}
		stress_impact = {
			brave = major_stress_impact_gain
		}
		ai_chance = {
			base = 10
			modifier = { add = 30 has_trait = craven }
		}
	}
}
