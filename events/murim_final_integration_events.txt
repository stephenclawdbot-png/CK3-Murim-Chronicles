namespace = murim_final_integration

# Event 0001: The Murim World Reacts to Ascension
# Triggered after immortal ascension trial declared
# Links: ascension + tournament + tribunal + all clans
murim_final_integration.0001 = {
    type = character_event
    title = murim_final_integration.0001.t
    desc = murim_final_integration.0001.desc
    theme = murim_martial_theme

    trigger = {
        has_variable = murim_martial_artist
        NOT = { has_character_flag = murim_reacted_to_ascension }
        any_ruler = {
            has_character_flag = murim_ascension_trial_declared
            NOT = { this = root }
        }
    }

    immediate = {
        set_character_flag = murim_reacted_to_ascension
        random_ruler = {
            limit = { has_character_flag = murim_ascension_trial_declared }
            save_scope_as = ascension_candidate
        }
    }

    # Support the ascension
    option = {
        name = murim_final_integration.0001.a
        add_opinion = { modifier = supports_ascension target = scope:ascension_candidate opinion = 25 }
        add_prestige = 100
        hidden_effect = {
            scope:ascension_candidate = {
                add_opinion = { modifier = grateful_opinion target = root opinion = 15 }
            }
        }
        ai_chance = {
            base = 40
            modifier = {
                add = 20
                opinion = { target = scope:ascension_candidate value >= 30 }
            }
        }
    }

    # Challenge the candidate — only the worthy should ascend
    option = {
        name = murim_final_integration.0001.b
        trigger = { prowess >= 20 }
        add_prestige = 200
        hidden_effect = {
            trigger_event = { id = murim_final_integration.0005 days = 15 }
        }
        ai_chance = {
            base = 25
            modifier = {
                add = 30
                prowess >= 25
            }
            modifier = {
                add = 20
                has_rivalry = { target = scope:ascension_candidate }
            }
        }
    }

    # Observe from the shadows
    option = {
        name = murim_final_integration.0001.c
        add_intrigue = 1
        stress_impact = { paranoid = medium_stress_impact_loss }
        ai_chance = {
            base = 35
            modifier = {
                add = 20
                has_trait = murim_shadow_sect_leader
            }
        }
    }
}

# Event 0002: Post-Calamity Murim Reconstruction Summit
# Triggered after calamity ends
# Links: calamity + tribunal + blood oaths + trade
murim_final_integration.0002 = {
    type = character_event
    title = murim_final_integration.0002.t
    desc = murim_final_integration.0002.desc
    theme = murim_martial_theme

    trigger = {
        has_variable = murim_martial_artist
        has_global_variable = murim_calamity_ended
        NOT = { has_character_flag = murim_attended_reconstruction }
        OR = {
            has_trait = murim_grandmaster
            has_trait = murim_sect_leader
            has_trait = murim_elder
        }
    }

    immediate = {
        set_character_flag = murim_attended_reconstruction
    }

    # Lead the reconstruction — unite all factions
    option = {
        name = murim_final_integration.0002.a
        add_prestige = 500
        add_character_modifier = {
            modifier = murim_post_calamity_rebuilding_modifier
            years = 10
        }
        hidden_effect = {
            every_vassal = {
                limit = { has_variable = murim_martial_artist }
                add_opinion = { modifier = grateful_opinion target = root opinion = 20 }
            }
            # Forge new blood oath alliances from shared survival
            trigger_event = { id = murim_final_integration.0006 days = 30 }
        }
        ai_chance = {
            base = 50
            modifier = {
                add = 30
                diplomacy >= 18
            }
        }
    }

    # Seize power in the chaos
    option = {
        name = murim_final_integration.0002.b
        add_prestige = 300
        add_dread = 30
        add_character_modifier = {
            modifier = murim_murim_world_in_chaos_modifier
            years = 5
        }
        hidden_effect = {
            # Rivals will resist
            every_vassal = {
                limit = {
                    has_variable = murim_martial_artist
                    opinion = { target = root value < 0 }
                }
                trigger_event = { id = murim_final_integration.0007 days = 20 }
            }
        }
        ai_chance = {
            base = 25
            modifier = {
                add = 30
                has_trait = murim_tyrant
            }
            modifier = {
                add = 20
                has_trait = ambitious
            }
        }
    }

    # Focus on rebuilding our own sect first
    option = {
        name = murim_final_integration.0002.c
        add_stewardship = 2
        every_held_title = {
            limit = { tier = tier_county }
            add_county_modifier = {
                modifier = murim_trade_route_secured_modifier
                years = 5
            }
        }
        ai_chance = {
            base = 25
            modifier = {
                add = 20
                has_trait = content
            }
        }
    }
}

        name = murim_final_integration.0003.a

            # Gather all martial artists

                limit = {
                    has_variable = murim_martial_artist

# Event 0005: Ascension Challenge Duel
murim_final_integration.0005 = {
    type = character_event
    title = murim_final_integration.0005.t
    desc = murim_final_integration.0005.desc
    theme = murim_martial_theme

    immediate = {
        random_ruler = {
            limit = { has_character_flag = murim_ascension_trial_declared }
            save_scope_as = ascension_candidate
        }
    }

    option = {
        name = murim_final_integration.0005.a
        duel = {
            target = scope:ascension_candidate
            skill = prowess
            50 = {
                desc = murim_final_integration.0005.a.success
                compare_modifier = { value = scope:duel_value }
                add_prestige = 1000
                add_prowess = 3
                scope:ascension_candidate = {
                    add_character_modifier = { modifier = murim_ascension_failed_modifier years = 5 }
                    remove_character_flag = murim_ascension_trial_declared
                }
            }
            50 = {
                desc = murim_final_integration.0005.a.failure
                compare_modifier = { value = scope:duel_value factor = -1 }
                add_prestige = -200
                add_prowess = 1
            }
        }
        ai_chance = { base = 100 }
    }
}
