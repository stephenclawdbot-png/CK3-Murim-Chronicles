# SHATTERED MERIDIANS
# Cannot cultivate above current rank ever again
# Caused by: devastating qi explosion, critical breakthrough failure during combat
######################################

shattered_meridians = {
	index = 1
	category = personality

	# Permanent cultivation ceiling
	modifier = {
		monthly_piety = -0.5
		stress_gain_mult = 0.15
		health = -0.25
		learning = -3
	}

	# Cannot be removed by normal means
	can_have = {
		is_alive = yes
	}

	# Incompatible with qi_crippled (meridians must exist to be shattered vs fully gone)
	opposites = {
		qi_crippled
		meridian_blockage
	}

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_trait = shattered_meridians
				}
				desc = trait_shattered_meridians_desc
			}
		}
	}

	# AI impact: will never attempt breakthrough, avoids cultivation decisions
	ai_energy = -25
	ai_boldness = -20
	ai_honor = 0
	ai_greed = 0
	ai_rationality = 15

	flag = murim_combat_trauma
	flag = murim_permanent_injury
	flag = cannot_advance_cultivation
}
