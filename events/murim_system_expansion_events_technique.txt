# Forbidden technique discovered
murim_technique.1400 = {
	type = character_event
	title = murim_technique.1400.t
	desc = murim_technique.1400.desc
	theme = skull
	left_portrait = { character = root animation = personality_callous }

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 4
	}

	# Study the forbidden technique despite the risks
	option = {
		name = murim_technique.1400.a
		add_prowess = 4
		change_variable = { name = murim_cultivation_xp add = 100 }
		add_stress = 40
		add_dread = 25
		ai_chance = {
			base = 40
			modifier = { add = 30 has_trait = ambitious }
		}
	}

	# Destroy it — too dangerous for anyone
	option = {
		name = murim_technique.1400.b
		add_piety = 150
		add_prestige = 100
		add_stress = -20
		ai_chance = {
			base = 35
			modifier = { add = 25 has_trait = just }
		}
	}

	# Seal it away for safekeeping
	option = {
		name = murim_technique.1400.c
		add_intrigue = 2
		add_prestige = 50
		ai_chance = { base = 25 }
	}
}
