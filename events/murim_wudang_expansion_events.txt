# EVENT 1: TAI CHI GRANDMASTER ENLIGHTENMENT
# The ultimate Wudang achievement: mastering the unity of yin and yang in combat.
# Decades of meditation and practice condensed into a single moment of clarity.
# The practitioner's movements become water - soft yet unstoppable.
##############################
murim_wudang_exp.0001 = {
	type = character_event
	title = murim_wudang_exp.0001.t
	desc = murim_wudang_exp.0001.desc
	theme = martial_arts
	
	trigger = {
		is_wudang_member_trigger = yes
		prowess >= 22
		learning >= 14
		age >= 35
		NOT = { has_character_flag = wudang_tai_chi_enlightenment }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			prowess >= 26
		}
		modifier = {
			add = 2
			has_trait = patient
		}
		modifier = {
			add = 2
			has_trait = calm
		}
	}
	
	immediate = {
		set_character_flag = wudang_tai_chi_enlightenment
		create_character_memory = {
			type = wudang_tai_chi_enlightenment_memory
		}
	}
	
	# Option A: Embrace the moment of perfect stillness
	option = {
		name = murim_wudang_exp.0001.a
		random_list = {
			40 = {
				modifier = {
					add = 15
					prowess >= 28
				}
				modifier = {
					add = 10
					learning >= 18
				}
				# True Tai Chi Grandmaster!
				add_prowess_xp = 1200
				add_learning_xp = 600
				add_trait = tai_chi_grandmaster
				add_character_modifier = {
					modifier = wudang_tai_chi_transcendence_modifier
					years = -1
				}
				add_prestige = 700
				create_character_memory = {
					type = wudang_tai_chi_mastery_memory
				}
				# All martial artists know
				every_living_character = {
					limit = {
						murim_is_martial_artist_trigger = yes
					}
					add_opinion = {
						modifier = wudang_tai_chi_master_awe_opinion
						target = root
						opinion = 15
					}
				}
				# Cross-ref: ranking challenge
				trigger_event = {
					id = murim_rank.0001
					days = { 30 90 }
				}
			}
			35 = {
				# Partial enlightenment - close but not complete
				add_prowess_xp = 600
				add_learning_xp = 300
				add_character_modifier = {
					modifier = wudang_near_enlightenment_modifier
					years = 10
				}
			}
			25 = {
				# Qi deviation during meditation - internal injury
				add_stress = 40
				add_character_modifier = {
					modifier = wudang_qi_deviation_modifier
					years = 3
				}
			}
		}
		ai_chance = {
			base = 55
		}
	}
	
	# Option B: Seek a sparring partner to test the insight in combat
	option = {
		name = murim_wudang_exp.0001.b
		# Find a worthy opponent to test Tai Chi in combat
		random_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
				prowess >= 18
				NOT = { is_wudang_member_trigger = yes }
				NOT = { this = root }
			}
			weight = {
				base = 1
				modifier = { add = 2 is_shaolin_member_trigger = yes }
				modifier = { add = 2 is_iron_fist_member_trigger = yes }
			}
			save_scope_as = test_opponent
		}
		duel = {
			skill = prowess
			target = scope:test_opponent
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				min_compared_to = root.prowess
				# Tai Chi proves supreme
				add_prowess_xp = 1000
				add_trait = tai_chi_grandmaster
				add_prestige = 500
				create_character_memory = {
					type = wudang_tai_chi_combat_proof_memory
					participants = { defeated = scope:test_opponent }
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min_compared_to = scope:test_opponent.prowess
				# Enlightenment fades under pressure
				add_prowess_xp = 400
				add_stress = 30
			}
		}
		ai_chance = {
			base = 30
		}
	}
	
	# Option C: Return to meditation - not yet ready
	option = {
		name = murim_wudang_exp.0001.c
		add_learning_xp = 200
		add_stress = -15
		remove_character_flag = wudang_tai_chi_enlightenment
		ai_chance = {
			base = 15
		}
	}
}

# EVENT 3: WUDANG-SHAOLIN SUMMIT
# The two greatest orthodox sects meet to coordinate against unorthodox threats.
# Political maneuvering between the most prestigious orthodox clans.
# Who leads the Orthodox Alliance - Wudang's wisdom or Shaolin's tradition?
##############################
murim_wudang_exp.0003 = {
	type = character_event
	title = murim_wudang_exp.0003.t
	desc = murim_wudang_exp.0003.desc
	theme = diplomacy
	
	trigger = {
		is_wudang_member_trigger = yes
		murim_is_clan_leader_trigger = yes
		any_living_character = {
			is_shaolin_member_trigger = yes
			murim_is_clan_leader_trigger = yes
		}
		NOT = { has_character_flag = wudang_shaolin_summit_recent }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			any_living_character = {
				murim_is_unorthodox_trigger = yes

		set_character_flag = {
			flag = wudang_shaolin_summit_recent

##############################
# EVENT 6: WUDANG SUCCESSOR SELECTION
# The Sect Master of Wudang chooses their successor.
# Unlike violent sects, Wudang favors wisdom and balance.
# But not everyone agrees with the selection.
##############################
murim_wudang_exp.0006 = {
	type = character_event
	title = murim_wudang_exp.0006.t
	desc = murim_wudang_exp.0006.desc
	theme = diplomacy
	
	trigger = {
		is_wudang_member_trigger = yes
		murim_is_clan_leader_trigger = yes
		age >= 50
		NOT = { has_character_flag = wudang_successor_selected }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			age >= 60
		}
	}
	
	immediate = {
		set_character_flag = wudang_successor_selected
		# Find the best candidate
		random_courtier = {
			limit = {
				is_wudang_member_trigger = yes
				age >= 20
				age <= 40
				NOT = { this = root }
			}
			weight = {
				base = 1
				modifier = { add = 3 learning >= 14 }
				modifier = { add = 2 prowess >= 16 }
				modifier = { add = 2 has_trait = patient }
				modifier = { add = 2 has_trait = just }
			}
			save_scope_as = chosen_successor
		}
		# Find a rival candidate
		random_courtier = {
			limit = {
				is_wudang_member_trigger = yes
				has_trait = ambitious
				NOT = { this = root }
				NOT = { this = scope:chosen_successor }
			}
			save_scope_as = rival_candidate
		}
		create_character_memory = {
			type = wudang_successor_selection_memory
			participants = {
				chosen = scope:chosen_successor
				rival = scope:rival_candidate
			}
		}
	}
	
	# Option A: Announce successor publicly (transparent, may cause jealousy)
	option = {
		name = murim_wudang_exp.0006.a
		scope:chosen_successor = {
			add_character_modifier = {
				modifier = wudang_designated_successor_modifier
				years = -1
			}
			add_prestige = 300
		}
		scope:rival_candidate = {
			add_opinion = {
				modifier = wudang_passed_over_opinion
				target = root
				opinion = -25
			}
			add_stress = 30
		}
		add_diplomacy_xp = 300
		change_variable = {
			name = murim_clan_harmony
			add = -5
		}
		ai_chance = { base = 40 }
	}
	
	# Option B: Train both candidates and let merit decide
	option = {
		name = murim_wudang_exp.0006.b
		scope:chosen_successor = {
			add_prowess_xp = 400
			add_learning_xp = 400
		}
		scope:rival_candidate = {
			add_prowess_xp = 400
			add_learning_xp = 400
			add_opinion = {
				modifier = wudang_fair_chance_opinion
				target = root
				opinion = 10
			}
		}
		add_character_modifier = {
			modifier = wudang_wise_selection_modifier
			years = 5
		}
		change_variable = {
			name = murim_clan_harmony
			add = 10
		}
		# Trigger training event
		trigger_event = {
			id = murim_training.0001
			days = { 30 60 }
		}
		ai_chance = { base = 40 }
	}
	
	# Option C: Delay the decision (avoid conflict for now)
	option = {
		name = murim_wudang_exp.0006.c
		add_stress = 20
		remove_character_flag = wudang_successor_selected
		ai_chance = { base = 20 }
	}
}
