# Murim Chronicles - Top 50 Martial Artist Ranking Events
# Entry/exit notifications, death promotions, #1 announcements,
# ranked challenges, yearly summary, hidden talent emergence
# Load-safe: all events check exists/has_variable before access

namespace = murim_rank

######################################
# 0001 - ENTERED THE TOP 50
# Player notification when crossing threshold
######################################

murim_rank.0001 = {
	type = character_event
	title = murim_rank.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = this
					has_variable = murim_martial_rank_position
					var:murim_martial_rank_position <= 10
				}
				desc = murim_rank.0001.desc_top_10
			}
			triggered_desc = {
				trigger = {
					exists = this
					has_variable = murim_martial_rank_position
					var:murim_martial_rank_position <= 20
				}
				desc = murim_rank.0001.desc_top_20
			}
			desc = murim_rank.0001.desc_top_50
		}
	}
	theme = martial
	left_portrait = {
		character = root
		animation = personality_bold
	}
	override_background = {
		reference = throne_room
	}

	trigger = {
		exists = this
		is_alive = yes
		is_ai = no
		murim_is_martial_artist_trigger = yes
	}

	immediate = {
		# Safety: initialize variables if missing
		if = {
			limit = { NOT = { has_variable = murim_martial_rank_score } }
			set_variable = {
				name = murim_martial_rank_score
				value = murim_martial_rank_score
			}
		}
	}

	# Option A: Embrace the fame
	option = {
		name = murim_rank.0001.a
		tooltip = murim_rank_embrace_fame_tt
		add_prestige = 300
		add_character_modifier = {
			modifier = murim_ranked_fame_modifier
			years = 2
		}
		# Ambitious characters gain extra
		if = {
			limit = { has_trait = ambitious }
			add_prestige = 100
			add_stress = -10
		}
		# Cross-system: ranking fame ripples outward
		hidden_effect = {
			trigger_event = { id = murim_hero.0001 days = { 14 30 } }
			trigger_event = { id = murim_renown.0001 days = { 7 21 } }
		}
		stress_impact = {
			shy = minor_stress_impact_gain
			humble = minor_stress_impact_gain
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 30
				has_trait = ambitious
			}
			modifier = {
				add = 20
				has_trait = arrogant
			}
		}
	}

	# Option B: Stay humble, train harder
	option = {
		name = murim_rank.0001.b
		tooltip = murim_rank_stay_humble_tt
		add_prestige = 100
		add_character_modifier = {
			modifier = murim_ranked_focused_training_modifier
			years = 2
		}
		# Humble/patient characters benefit more
		if = {
			limit = {
				OR = {
					has_trait = humble
					has_trait = patient
					has_trait = temperate
				}
			}
			add_prowess = 1
		}
		stress_impact = {
			arrogant = minor_stress_impact_gain
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 30
				has_trait = humble
			}
			modifier = {
				add = 20
				has_trait = patient
			}
		}
	}

	# Option C: Use the rank for political leverage
	option = {
		name = murim_rank.0001.c
		tooltip = murim_rank_political_leverage_tt
		add_prestige = 150
		add_character_modifier = {
			modifier = murim_ranked_political_influence_modifier
			years = 3
		}
		# Diplomacy/intrigue characters gain extra
		if = {
			limit = { diplomacy >= 12 }
			add_diplomacy = 1
		}
		stress_impact = {
			honest = minor_stress_impact_gain
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = gregarious
			}
			modifier = {
				add = 20
				diplomacy >= 15
			}
		}
	}
}

######################################
# 0005 - AI RANKED CHALLENGE
# A ranked fighter challenges another ranked fighter for position
######################################

murim_rank.0005 = {
	type = character_event
	title = murim_rank.0005.t
	desc = murim_rank.0005.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:rank_challenger
		animation = personality_bold
	}

	trigger = {
		exists = this
		is_alive = yes
		is_ai = no
		murim_is_top_50_martial_artist_trigger = yes
	}

	immediate = {
		# Find a valid challenger
		random_living_character = {
			limit = {
				murim_ai_should_challenge_for_rank_trigger = yes
				NOT = { this = root }
				is_ai = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					has_trait = ambitious
				}
				modifier = {
					add = 3
					prowess >= 20
				}
			}
			save_scope_as = rank_challenger
		}
	}

	# Option A: Accept the challenge
	option = {
		name = murim_rank.0005.a
		tooltip = murim_rank_accept_challenge_tt
		trigger = {
			NOR = {
				has_trait = incapable
				has_trait = broken_spirit
			}
		}
		# Duel resolution
		duel = {
			skill = prowess
			target = scope:rank_challenger
			50 = {
				desc = murim_rank_challenge_win_desc
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2
				}
				# Player wins
				add_prestige = 200
				scope:rank_challenger = {
					add_prestige = -100
					murim_roll_combat_injuries_effect = yes
					murim_roll_combat_trauma_effect = yes
				}
				murim_update_rankings_after_duel_effect = yes
			}
			50 = {
				desc = murim_rank_challenge_lose_desc
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2
				}
				# Player loses
				add_prestige = -200
				murim_roll_combat_injuries_effect = yes
				murim_roll_combat_trauma_effect = yes
				scope:rank_challenger = {
					add_prestige = 200
				}
				# Swap rank positions
				if = {
					limit = {
						has_variable = murim_martial_rank_position
						scope:rank_challenger = { has_variable = murim_martial_rank_position }
					}
					save_temporary_scope_value_as = {
						name = temp_rank
						value = var:murim_martial_rank_position
					}
					set_variable = {
						name = murim_martial_rank_position
						value = scope:rank_challenger.var:murim_martial_rank_position
					}
					scope:rank_challenger = {
						set_variable = {
							name = murim_martial_rank_position
							value = scope:temp_rank
						}
					}
				}
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				has_trait = brave
			}
			modifier = {
				add = -40
				has_trait = craven
			}
		}
	}

	# Option B: Decline the challenge
	option = {
		name = murim_rank.0005.b
		tooltip = murim_rank_decline_challenge_tt
		add_prestige = -300
		add_character_modifier = {
			modifier = murim_declined_rank_challenge_modifier
			years = 2
		}
		# Lose face with martial world
		every_ruler = {
			limit = {
				murim_is_top_50_martial_artist_trigger = yes
				NOT = { this = root }
			}
			add_opinion = {
				target = root
				modifier = murim_coward_declined_challenge_opinion
				years = 5
			}
		}
		stress_impact = {
			brave = major_stress_impact_gain
			ambitious = major_stress_impact_gain
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 40
				has_trait = craven
			}
			modifier = {
				add = 20
				has_trait = wounded
			}
		}
	}
}
