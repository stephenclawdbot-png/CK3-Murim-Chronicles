# murim_sect_alliance.0001 - Grand Alliance Formation Ceremony
# Multiple sects gather to form a martial alliance against a common threat
######################################
murim_sect_alliance.0001 = {
	type = character_event
	title = murim_sect_alliance.0001.t
	desc = murim_sect_alliance.0001.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		is_landed = yes
		has_variable = murim_cultivation_level
		var:murim_cultivation_level >= 4
		NOT = { has_character_flag = attended_alliance_ceremony }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			has_trait = diplomat
		}
		modifier = {
			factor = 1.5
			is_ruler = yes
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = attended_alliance_ceremony
			days = 1825
		}
		create_character_memory = {
			type = murim_attended_alliance_ceremony
		}
	}
	
	# Option A: Take the blood oath and become Alliance Leader
	option = {
		name = murim_sect_alliance.0001.a
		trigger = {
			var:murim_cultivation_level >= 6
		}
		
		random_list = {
			40 = {
				desc = murim_sect_alliance.0001.a.leader
				add_prestige = 1500
				add_character_modifier = {
					modifier = martial_alliance_leader

				}
				create_character_memory = {
					type = murim_became_alliance_leader
				}
				trigger_event = {
					id = murim_politics.0001
					days = { 30 90 }
				}
			}
			30 = {
				desc = murim_sect_alliance.0001.a.challenged
				add_prestige = 500
				create_character_memory = {
					type = murim_leadership_challenged_at_ceremony
				}
				trigger_event = {
					id = murim_duel.0005
					days = { 1 5 }
				}
			}
			30 = {
				desc = murim_sect_alliance.0001.a.assassination_attempt
				add_trait = wounded
				add_prestige = 300
				create_character_memory = {
					type = murim_assassination_attempt_at_ceremony
