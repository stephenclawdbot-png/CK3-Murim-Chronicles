# Murim Chronicles - China-Centric Geography Triggers
# HARD RULE: China (Song Dynasty) is the CENTER of the martial world.
# Everything radiates outward. Goryeo is secondary hub. Everything else is periphery.
# INTEGRATION: Referenced by ALL event files, ALL recruitment, ALL tournament logic,
#              ALL ranking systems, ALL scout routes, ALL travel events.
# VANILLA SEPARATION: Geography tiers affect MARTIAL systems only, not political.

# ===========================
# TIER 1 — CENTER: China / Song Dynasty Region
# 60-70% of martial characters. All 6 clan HQs. Murim Alliance seat.
# Grand Tournament, Champions Battle ALWAYS held here.
# ===========================

murim_is_in_china_trigger = {
    # Character is in the Song Dynasty / Chinese cultural region
    OR = {
        # Primary check: Chinese culture group
        culture = { has_cultural_pillar = heritage_chinese }
        # Geographic check: key Chinese provinces
        location = {
            OR = {
                geographical_region = world_china    # If CK3 defines this
                # Fallback: specific title regions representing Song Dynasty
                any_county_province = {
                    barony = {
                        de_jure_liege = {
                            OR = {
                                this = title:k_china    # Kingdom-level
                                this = title:e_china    # Empire-level
                                # Individual duchies in Chinese heartland
                                this = title:d_kaifeng
                                this = title:d_luoyang
                                this = title:d_chengdu
                                this = title:d_hangzhou
                                this = title:d_nanjing
                                this = title:d_xian
                                this = title:d_guangzhou
                            }
                        }
                    }
                }
            }
        }
    }
}

murim_is_china_province_trigger = {
    # Province-level check for a title/location
    OR = {
        geographical_region = world_china
        de_jure_liege = title:e_china
    }
}

murim_martial_density_tier_1_trigger = {
    # Used for seeding: 60-70% of martial characters here
    murim_is_in_china_trigger = yes
}

# ===========================
# RANKING ELIGIBILITY
# The Top 50 is a CHINESE institution.
# Foreign fighters are UNRANKED until they prove themselves.
# ===========================

murim_is_eligible_for_ranking_trigger = {
    # To be ranked, you must EITHER:
    OR = {
        # 1. Be in China (local fighters automatically eligible)
        murim_is_in_china_trigger = yes
        # 2. Have defeated a ranked fighter anywhere in the world
        has_character_modifier = murim_defeated_ranked_fighter
        # 3. Have a clan vouch for you
        OR = {
            has_trait = outer_disciple
            has_trait = inner_disciple
            has_trait = sect_elder
            has_trait = vice_sect_leader
            has_trait = sect_leader
        }
        # 4. Have won a major tournament (Grand Tournament or Champions Battle)
        has_character_modifier = murim_tournament_champion
        # 5. Have traveled to China and fought there (journey complete)
        has_variable = murim_fought_in_china
    }
}

murim_is_unranked_foreign_fighter_trigger = {
    # Strong fighter who hasn't been to China or proven themselves
    has_trait = martial_artist
    prowess >= 15
    NOR = {
        murim_is_in_china_trigger = yes
        has_variable = murim_ranking_position
        has_character_modifier = murim_defeated_ranked_fighter
        OR = {
            has_trait = outer_disciple
            has_trait = inner_disciple
            has_trait = sect_elder
        }
    }
}

# ===========================
# TECHNIQUE CEILING BY REGION
# Chinese arts: Tier 1-8. Foreign arts: cap at Tier 5-6 unless hybrid.
# ===========================

murim_technique_ceiling_trigger = {
    # Returns whether a character can access high-tier techniques
    # Tier 7-8 techniques require Chinese training or hybrid fusion
    OR = {
        # Trained in China
        murim_is_in_china_trigger = yes
        has_variable = murim_trained_in_china
        # Hybrid style (foreign + Chinese fusion)
        has_character_modifier = murim_hybrid_technique_master
        # Learned from a Chinese master abroad
        has_variable = murim_learned_from_chinese_master
    }
}

murim_foreign_technique_cap_trigger = {
    # Character is limited to Tier 5-6 techniques (foreign cap)
    NOR = {

        has_variable = murim_trained_in_china

        has_variable = murim_learned_from_chinese_master

# ===========================
# TRAVEL DISTANCE VALUES
# Used for martial journey duration calculations
# ===========================

murim_travel_distance_to_china_value = {
    # Returns approximate months of martial travel to China
    if = {
        limit = { murim_is_in_china_trigger = yes }
        value = 1    # Already there
    }
    else_if = {
        limit = { murim_is_in_goryeo_trigger = yes }
        value = 3    # Short trip
    }
    else_if = {
        limit = { murim_is_in_japan_trigger = yes }
        value = 6    # Sea crossing
    }
    else_if = {
        limit = { murim_is_in_southeast_asia_trigger = yes }
        value = 6    # Southern route
    }
    else_if = {
        limit = { murim_is_in_india_trigger = yes }
        value = 9    # Long overland
    }
    else_if = {
        limit = { murim_is_in_persia_trigger = yes }
        value = 12    # Silk Road
    }
    else_if = {
        limit = { murim_is_in_steppe_trigger = yes }
        value = 9    # Northern route
    }
    else_if = {
        limit = { murim_is_in_europe_trigger = yes }
        value = 18    # Epic journey
    }
    else_if = {
        limit = { murim_is_in_africa_trigger = yes }
        value = 18    # Epic journey
    }
    else = {
        value = 24    # Unknown far land
    }
}

        value = 4

        limit = { murim_is_in_persia_trigger = yes }
        value = 3
