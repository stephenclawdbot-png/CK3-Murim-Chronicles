# Murim Chronicles - Clan Scout Character Effects
# Spawns and manages NAMED traveling recruiter characters for each clan.
# Scouts are REAL characters on the map with traits, not abstract events.
# INTEGRATION: Called by murim_game_start_effects.txt (initial spawn)
# INTEGRATION: Called by murim_on_actions.txt (replacement spawn on death)
# INTEGRATION: Feeds murim_clan_scout_events.txt (scout travel/encounter events)
# INTEGRATION: Uses murim_clan_triggers.txt (clan membership checks)
# INTEGRATION: Uses murim_clan_membership_effects.txt (recruit join effects)
# DESIGN: VANILLA SEPARATION — scouts do NOT affect titles, wars, succession.
#         They are martial-layer characters only. Their travel is martial travel,
#         not political movement. Courts they visit get martial events, not diplomatic ones.
# GENDER: All scouts can be any gender. No sex restrictions.

# ============================================================
# SPAWN SCOUT EFFECTS — Called on game start and on scout death
# ============================================================

# ============================================================
# SCOUT MANAGEMENT EFFECTS
# ============================================================

# Yearly tick: advance scout timer, check for return
murim_scout_yearly_tick_effect = {
	# Increment timer
	change_variable = {
		name = murim_scout_return_timer
		add = 1
	}
	
	# Check if scout should return (timer >= 5 OR max recruits reached)
	if = {
		limit = {
			OR = {
				var:murim_scout_return_timer >= 5
				var:murim_scout_recruits_found >= var:murim_scout_max_recruits
			}
		}
		# Trigger return event
		trigger_event = {
			id = murim_scout.0010  # Scout returns to clan HQ
		}
		# Reset timer
		set_variable = {
			name = murim_scout_return_timer
			value = 0
		}
	}
}

# Scout death: spawn replacement after delay
murim_scout_death_replacement_effect = {
	# Save which clan lost a scout
	save_scope_as = dead_scout
	# Set clan variable: scout_replacement_timer = 2 (years)
	# On yearly tick, decrement. When 0, spawn new scout.
	# The clan that lost the scout gets a "mourning" event
	trigger_event = {
		id = murim_scout.0020  # Clan mourns lost scout, begins training replacement
		days = 30
	}
}

# Scout captured: ransom/rescue chain
murim_scout_captured_effect = {
	# Set captured flag
	set_variable = {
		name = murim_scout_captured
		value = yes
	}
	# Trigger rescue event chain for the clan
	# Clan leader decides: pay ransom, send rescue team, abandon scout
	trigger_event = {
		id = murim_scout.0030  # Clan learns scout was captured
		days = 30
	}
}

# Scout expelled from realm: blacklist that realm for 10 years
murim_scout_expelled_effect = {
	# Save the realm that expelled the scout
	# Set realm variable: murim_scout_blacklisted_by_[clan] = 10
	# Decrement yearly. Scout won't visit that realm until 0.
	scope:expelling_ruler = {
		set_variable = {
			name = murim_scout_blacklist_timer
			value = 10
		}
	}
	# Scout moves to next realm on their circuit
	trigger_event = {
		id = murim_scout.0040  # Scout expelled, moves on
		days = 7
	}
}

# Scout finds a recruit: increment counter, flag the recruit
murim_scout_found_recruit_effect = {
	# Increment scout's recruit counter
	change_variable = {
		name = murim_scout_recruits_found
		add = 1
	}
	# Flag the recruit to travel with scout
	scope:recruit = {
		set_variable = {
			name = murim_traveling_with_scout
			value = scope:scout
		}
		set_variable = {
			name = murim_destination_clan
			value = scope:scout.var:murim_clan_scout
		}
	}
}

# Scout returns with recruits: process all recruits into clan
murim_scout_process_recruits_effect = {
	# For each character flagged as traveling with this scout:
	# 1. Move them to clan HQ region
	# 2. Apply outer_disciple trait via murim_join_sect_as_outer_disciple_effect
	# 3. Remove traveling_with_scout variable
	# 4. Trigger culture shock event if foreign
	# 5. Reset scout's recruit counter
	set_variable = {
		name = murim_scout_recruits_found
		value = 0
	}
}
