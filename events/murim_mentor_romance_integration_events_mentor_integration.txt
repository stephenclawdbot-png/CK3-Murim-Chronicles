# Mentor's teachings reach a critical moment
murim_mentor_integration.0001 = {
	type = character_event
	title = murim_mentor_integration.0001.t
	desc = murim_mentor_integration.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_compassionate }

	trigger = {
		has_variable = murim_martial_artist
		has_character_flag = murim_is_mentor
	}

	# Share your ultimate technique with your disciple
	option = {
		name = murim_mentor_integration.0001.a
		add_prestige = 200
		add_stress = 15
		change_variable = { name = murim_cultivation_xp add = 30 }
		ai_chance = {
			base = 45
			modifier = { add = 25 has_trait = compassionate }
		}
	}

	# Hold back — they must earn it through hardship
	option = {
		name = murim_mentor_integration.0001.b
		add_martial = 1
		add_stress = -10
		ai_chance = {
			base = 35
			modifier = { add = 20 has_trait = patient }
		}
	}

	# Test them with a challenge first
	option = {
		name = murim_mentor_integration.0001.c
		add_prowess = 1
		add_prestige = 100
		ai_chance = { base = 20 }
	}
}
