namespace = murim_calamity_ai

# Great Calamity AI Autonomous Events
# These events fire autonomously for AI characters during the Great Calamity
# All clans wage war, betray each other, and die without player input
# 70-90% kill rate. 1-3 survivors per clan. Player dynasty always survives.

###################################
# AI Clan War Declaration
# Fired monthly during calamity for AI clan leaders
###################################
murim_calamity_ai.0001 = {
	type = character_event
	title = murim_calamity_ai.0001.t
	desc = murim_calamity_ai.0001.desc
	theme = war
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_variable = murim_clan_leader
		NOT = { has_trait = murim_calamity_survivor }
		NOT = { has_character_flag = calamity_war_declared_this_year }
		any_living_character = {
			has_variable = murim_clan_leader
			NOT = { this = root }
			NOT = { has_trait = murim_calamity_survivor }
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = calamity_war_declared_this_year
			days = 365
		}
		random_living_character = {
			limit = {
				has_variable = murim_clan_leader
				NOT = { this = root }
				NOT = { has_trait = murim_calamity_survivor }
			}
			save_scope_as = calamity_war_target
		}
	}
	
	option = {
		name = murim_calamity_ai.0001.a
		# Declare total war on rival clan
		start_war = {
			casus_belli = murim_clan_annihilation_cb
			target = scope:calamity_war_target
		}
		add_prestige = -500
		stress_impact = {
			ambitious = medium_stress_impact_loss
			craven = massive_stress_impact_gain
		}
		ai_chance = {
			base = 80
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = -30
				has_trait = craven
			}
		}
	}
	
	option = {
		name = murim_calamity_ai.0001.b
		# Attempt to fortify and defend
		add_character_modifier = {
			modifier = murim_calamity_defensive_stance
			years = 2
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = craven
			}
		}
	}
}

###################################
# AI Clan Betrayal During Calamity
# Allied clans backstab each other
###################################
murim_calamity_ai.0002 = {
	type = character_event
	title = murim_calamity_ai.0002.t
	desc = murim_calamity_ai.0002.desc
	theme = intrigue
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_variable = murim_clan_leader
		any_ally = {
			has_variable = murim_clan_leader
			NOT = { has_trait = murim_calamity_survivor }
		}
		NOT = { has_character_flag = calamity_betrayal_cooldown }
	}
	
	immediate = {
		add_character_flag = {
			flag = calamity_betrayal_cooldown
			days = 730
		}
		random_ally = {
			limit = {
				has_variable = murim_clan_leader
				NOT = { has_trait = murim_calamity_survivor }
			}
			save_scope_as = betrayal_target
		}
	}
	
	option = {
		name = murim_calamity_ai.0002.a
		# Betray the alliance — poison their supplies
		scope:betrayal_target = {
			add_character_modifier = {
				modifier = murim_calamity_poisoned_supplies
				years = 1
			}
			add_opinion = {
				target = root
				modifier = murim_calamity_betrayed_opinion
				opinion = -100
			}
		}
		remove_relation_ally = scope:betrayal_target
		add_trait = murim_betrayer
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				has_trait = deceitful
			}
			modifier = {
				add = -40
				has_trait = honest
			}
		}
	}
	
	option = {
		name = murim_calamity_ai.0002.b
		# Hold the alliance — for now
		add_prestige = 100
		scope:betrayal_target = {
			add_opinion = {
				target = root
				modifier = murim_calamity_honored_alliance
				opinion = 20
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 40
				has_trait = honest
			}
		}
	}
}

# AI Mass Casualty Event
# Random martial artists die during calamity chaos
###################################
murim_calamity_ai.0003 = {
	type = character_event
	title = murim_calamity_ai.0003.t
	desc = murim_calamity_ai.0003.desc
	theme = death
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		OR = {
			has_trait = murim_martial_artist
			has_variable = murim_clan_member
		}
		NOT = { has_trait = murim_calamity_survivor }
		NOT = { is_ai = no } # Player protected
		NOT = { has_character_flag = calamity_death_check_done }
	}
	
	immediate = {
		add_character_flag = {
			flag = calamity_death_check_done
			days = 180
		}
	}
	
	# 70-90% chance of death based on cultivation level
	option = {
		name = murim_calamity_ai.0003.a
		# Die in the chaos
		random_list = {
			70 = {
				modifier = {
					add = 10
					has_trait = murim_rank_unranked
				}
				modifier = {
					add = -20
					has_trait = murim_rank_peak_master
				}
				modifier = {
					add = -30
					has_trait = murim_rank_transcendent
				}
				death = {
					death_reason = death_murim_calamity
				}
			}
			20 = {
				# Severely wounded but alive
				add_trait = wounded_3
				add_trait = murim_calamity_survivor
				add_character_modifier = {
					modifier = murim_calamity_broken_meridians
					years = 10
				}
			}
			10 = {
				# Emerged stronger
				add_trait = murim_calamity_survivor
				add_trait = murim_calamity_witness
				add_prowess = 4
				add_prestige = 500
			}
		}
		ai_chance = { base = 100 }
	}
}

###################################
# Survivor Selection — Clan Remnants
# 1-3 survivors per clan after calamity winds down
###################################
murim_calamity_ai.0006 = {
	type = character_event
	title = murim_calamity_ai.0006.t
	desc = murim_calamity_ai.0006.desc
	theme = skull
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_global_variable = murim_calamity_winding_down
		has_variable = murim_clan_member
		NOT = { has_trait = murim_calamity_survivor }
		is_alive = yes
	}
	
	immediate = {
		# Count living clan members
		every_living_character = {
			limit = {
				var:murim_clan_id = root.var:murim_clan_id
				has_trait = murim_calamity_survivor
			}
			root = {
				if = {
					limit = { NOT = { has_variable = calamity_survivors_count } }
					set_variable = { name = calamity_survivors_count value = 0 }
				}
				change_variable = { name = calamity_survivors_count add = 1 }
			}
		}
	}
	
	option = {
		name = murim_calamity_ai.0006.a
		# If fewer than 3 survivors, this character lives
		trigger = {
			OR = {
				NOT = { has_variable = calamity_survivors_count }
				var:calamity_survivors_count < 3
			}
		}
		add_trait = murim_calamity_survivor
		add_character_modifier = {
			modifier = murim_calamity_last_of_clan
			years = 50
		}
		add_prestige = 1000
		ai_chance = { base = 100 }
	}
	
	option = {
		name = murim_calamity_ai.0006.b
		# Too many survivors already — this one dies
		trigger = {
			has_variable = calamity_survivors_count
			var:calamity_survivors_count >= 3
		}
		death = {
			death_reason = death_murim_calamity_final_battle
		}
		ai_chance = { base = 100 }
	}
}

###################################
# Calamity Cycle Reset
# The great cycle begins anew — new era dawns
###################################
murim_calamity_ai.0010 = {
	type = character_event
	title = murim_calamity_ai.0010.t
	desc = murim_calamity_ai.0010.desc
	theme = faith
	
	trigger = {
		has_global_variable = murim_calamity_ended
		has_trait = murim_calamity_survivor
		NOT = { has_character_flag = calamity_new_era_witnessed }
		current_date >= global_var:murim_calamity_end_date
	}
	
	immediate = {
		add_character_flag = calamity_new_era_witnessed
		# Reset global calamity variables for next cycle
		if = {
			limit = { has_variable = murim_clan_leader }
			set_global_variable = {
				name = murim_next_calamity_year
				value = 150
			}
		}
	}
	
	option = {
		name = murim_calamity_ai.0010.a
		# Embrace the new era — swear the cycle will not repeat
		add_trait = murim_calamity_era_founder
		add_prestige = 3000
		add_piety = 1000
		add_character_modifier = {
			modifier = murim_new_era_founder
			years = 50
		}
		# Remove calamity active flag
		remove_global_variable = murim_great_calamity_active
		remove_global_variable = murim_calamity_winding_down
		remove_global_variable = murim_calamity_ended
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				has_trait = ambitious
			}
		}
	}
	
	option = {
		name = murim_calamity_ai.0010.b
		# The cycle is eternal — accept fate
		add_trait = murim_fatalist
		add_piety = 500
		add_character_modifier = {
			modifier = murim_calamity_fatalist_wisdom
			years = 30
		}
		remove_global_variable = murim_great_calamity_active
		remove_global_variable = murim_calamity_winding_down
		remove_global_variable = murim_calamity_ended
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = content
			}
		}
	}
}
