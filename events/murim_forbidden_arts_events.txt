###############################################
# murim_forbidden_arts.0001 - Demonic Qi Awakening
# Character discovers a dark power source within themselves
###############################################
murim_forbidden_arts.0001 = {
	type = character_event
	title = murim_forbidden_arts.0001.t
	desc = murim_forbidden_arts.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = personality_vengeful
	}
	
	trigger = {
		is_alive = yes
		NOT = { has_trait = demonic_cultivator }
		NOT = { has_character_flag = demonic_awakening_cooldown }
		OR = {
			has_trait = wounded
			has_trait = incapable
			has_trait = lunatic
			stress >= 50
		}
		prowess >= 10
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = ambitious
		}
		modifier = {
			add = 2
			has_trait = murim_vengeful
		}
		modifier = {
			add = 2
			stress >= 80
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = demonic_awakening_cooldown
			years = 10
		}
	}
	
	# Option A: Embrace the demonic qi fully
	option = {
		name = murim_forbidden_arts.0001.a
		add_trait = demonic_cultivator
		add_prowess = 5
		add_character_modifier = {
			modifier = demonic_qi_surge
			years = 3
		}
		create_character_memory = {
			type = embraced_demonic_qi_memory
		}
		# Healing through dark power
		if = {
			limit = { has_trait = wounded }
			remove_trait = wounded
		}
		if = {
			limit = { has_trait = incapable }
			remove_trait = incapable
		}
		# But at a cost - sanity
		random_list = {
			40 = { add_trait = lunatic }
			30 = {
				add_trait = sadistic
			}
			30 = { }
		}
		# Orthodox sects react
		every_relation = {
			type = friend
			limit = {
				NOT = { has_trait = demonic_cultivator }
			}
			add_opinion = {
				target = root
				modifier = embraced_dark_path
				opinion = -40
			}
		}
		# Chain: jianghu underworld welcomes you
		trigger_event = {
			id = murim_jianghu.0001
			days = { 14 30 }
		}
		# Chain: fate turns dark
		trigger_event = {
			id = murim_fate.0001
			days = { 30 60 }
		}
		stress_impact = {
			zealous = major_stress_impact_gain
			compassionate = major_stress_impact_gain
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = ambitious
			}
			modifier = {
				add = 20
				has_trait = murim_vengeful
			}
		}
	}
	
	# Option B: Partially absorb the demonic qi - controlled corruption
	option = {
		name = murim_forbidden_arts.0001.b
		add_prowess = 2
		add_character_modifier = {
			modifier = partially_corrupted_qi
			years = 5
		}
		create_character_memory = {
			type = partial_demonic_absorption_memory
		}
		# Unstable - periodic qi deviations
		random_list = {
			30 = {
				trigger_event = {
					id = murim_cultivation.0003
					days = { 60 180 }
				}
			}
			40 = {
				add_trait = murim_scarred
			}
			30 = { }
		}
		# Chain: training to control it
		trigger_event = {
			id = murim_training.0001
			days = { 14 30 }
		}
		stress_impact = {
			brave = medium_stress_loss
		}
		ai_chance = {
			base = 40
		}
	}
	
	# Option C: Seal the demonic qi and purge it
	option = {
		name = murim_forbidden_arts.0001.c
		add_piety = 100
		create_character_memory = {
			type = purged_demonic_qi_memory
		}
		# Purging causes backlash
		random_list = {
			40 = { add_trait = wounded }
			20 = {
				add_trait = incapable
				trigger_event = {
					id = murim_mentor.0001
					days = { 7 14 }
				}
			}
			40 = { }
		}
		# Sect respects your choice
		add_prestige = 100
		stress_impact = {
			zealous = medium_stress_loss
			compassionate = medium_stress_loss
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = just
			}
		}
	}
}

# murim_forbidden_arts.0003 - Soul Devouring Technique
# A forbidden technique that absorbs the martial power of defeated opponents
###############################################
murim_forbidden_arts.0003 = {
	type = character_event
	title = murim_forbidden_arts.0003.t
	desc = murim_forbidden_arts.0003.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = personality_vengeful
	}
	right_portrait = {
		character = scope:soul_devour_victim
		animation = fear
	}
	
	trigger = {
		is_alive = yes
		has_trait = demonic_cultivator
		NOT = { has_character_flag = soul_devour_cooldown }
		any_prisoner = {
			prowess >= 12
			is_alive = yes
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 5
			has_trait = sadistic
		}
	}
	
	immediate = {
		random_prisoner = {
			limit = {
				prowess >= 12
				is_alive = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					prowess >= 20
				}
			}
			save_scope_as = soul_devour_victim
		}
		add_character_flag = {
			flag = soul_devour_cooldown
			years = 2
		}
	}
	
	# Option A: Devour their martial soul completely
	option = {
		name = murim_forbidden_arts.0003.a
		add_prowess = 4
		add_dread = 50
		add_tyranny = 30
		create_character_memory = {
			type = devoured_martial_soul_memory
			participants = {
				soul_victim = scope:soul_devour_victim
			}
		}
		scope:soul_devour_victim = {
			death = {
				death_reason = death_murder
				killer = root
			}
		}
		# Power surge but instability
		random_list = {
			30 = {
				add_character_modifier = {
					modifier = unstable_absorbed_power
					years = 3
				}
				add_trait = lunatic
			}
			30 = {
				add_character_modifier = {
					modifier = soul_devour_mastery
					years = 5
				}
			}
			20 = {
				# Victim's memories haunt you
				add_trait = lunatic
				trigger_event = {
					id = murim_fate.0001
					days = { 30 90 }
				}
			}
			20 = {
				# Perfect absorption
				add_trait_xp = {
					trait = martial_arts_master
					track = martial_arts_track
					value = 50
				}
			}
		}
		# Chains: whole martial world reacts
		trigger_event = {
			id = murim_jianghu.0001
			days = { 7 14 }
		}
		trigger_event = {
			id = murim_ranking.0001
			days = { 3 7 }
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = sadistic
			}
		}
	}
	
	# Option B: Partially absorb - leave them alive but crippled
	option = {
		name = murim_forbidden_arts.0003.b
		add_prowess = 2
		add_dread = 25
		create_character_memory = {
			type = partial_soul_devour_memory
			participants = {
				weakened_victim = scope:soul_devour_victim
			}
		}
		scope:soul_devour_victim = {
			add_trait = incapable
			add_prowess = -5
			create_character_memory = {
				type = had_soul_partially_devoured_memory
				participants = {
					soul_devourer = root
				}
			}
			# Chain: victim seeks revenge years later
			trigger_event = {
				id = murim_legendary_duel.0004
				days = { 365 730 }
			}
		}
		random_list = {
			40 = { add_trait = lunatic }
			60 = { }
		}
		ai_chance = {
			base = 35
		}
	}
	
	# Option C: Release the prisoner instead
	option = {
		name = murim_forbidden_arts.0003.c
		scope:soul_devour_victim = {
			release_from_prison = yes
			add_opinion = {
				target = root
				modifier = showed_mercy_released_me
				opinion = 30
			}
		}
		add_piety = 50
		create_character_memory = {
			type = showed_mercy_to_prisoner_memory
			participants = {
				freed_prisoner = scope:soul_devour_victim
			}
		}
		stress_impact = {
			sadistic = major_stress_impact_gain
			compassionate = medium_stress_loss
		}
		ai_chance = {
			base = 25
			modifier = {
				add = 25
				has_trait = compassionate
			}
		}
	}
}
