# Decision: Activate Murim-Wide Intelligence Network
# Links: espionage + black market + shadow networks
activate_murim_intelligence_network = {
    picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
    desc = activate_murim_intelligence_network_desc
    selection_weight = { base = 15 }
    ai_check_interval = 60

    is_shown = {
        has_variable = murim_martial_artist
        OR = {
            has_trait = murim_spymaster
            has_trait = murim_shadow_sect_leader
            has_trait = murim_information_broker
        }
        NOT = { has_character_flag = murim_intel_network_active }
    }

    is_valid = {
        intrigue >= 15
        gold >= 200
        any_vassal = {
            count >= 3
            has_variable = murim_martial_artist
        }
    }

    cost = { gold = 200 }

    effect = {
        add_character_flag = murim_intel_network_active
        add_character_modifier = {
            modifier = murim_intelligence_network_modifier
            years = 5
        }
        hidden_effect = {
            every_vassal = {
                limit = {
                    has_variable = murim_martial_artist
                    OR = {
                        has_trait = murim_spy
                        has_trait = murim_informant
                    }
                }
                add_opinion = { modifier = grateful_opinion target = root opinion = 15 }
            }
            # Cross-system: alert black market contacts
            if = {
                limit = { has_character_flag = murim_black_market_contact }
                trigger_event = { id = murim_cross_system.0010 days = 15 }
            }
            # Cross-system: shadow network synergy
            if = {
                limit = { has_character_flag = murim_shadow_network_member }
                trigger_event = { id = murim_cross_system.0011 days = 30 }
            }
        }
    }

    ai_potential = {
        intrigue >= 18
        gold >= 300
    }
    ai_will_do = {
        base = 30
        modifier = {
            add = 20
            has_trait = murim_shadow_sect_leader
        }
        modifier = {
            add = 15
            any_relation = { type = rival count >= 2 }
        }
    }
}
