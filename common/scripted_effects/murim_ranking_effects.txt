# Murim Chronicles - Top 50 Martial Ranking Effects
# Recalculation, entry/exit, death removal, duel updates
# Load-safe: all effects check exists before variable access
# POLISH FIX: All script value and modifier references verified against actual files
# Cross-references: murim_ranking_values.txt, murim_modifiers.txt, murim_ranking_events.txt

######################################
# RECALCULATE MARTIAL RANK
# Yearly pulse - recalculates score for a single character
######################################

murim_recalculate_martial_rank_effect = {
	# Safety check
	if = {
		limit = {
			exists = this
			is_alive = yes
			murim_is_martial_artist_trigger = yes
		}
		# Calculate new score using the composite formula from murim_ranking_values.txt
		set_variable = {
			name = murim_martial_rank_score
			value = murim_total_ranking_score_value
		}
		# Check if score crosses top 50 threshold
		if = {
			limit = {
				var:murim_martial_rank_score >= murim_is_top_50_threshold_value
				NOT = { has_character_modifier = murim_ranked_top_50_modifier }
			}
			# New entry into Top 50
			murim_enter_top_50_effect = yes
		}
		# Check if dropped below threshold
		if = {
			limit = {
				var:murim_martial_rank_score < murim_is_top_50_threshold_value
				has_character_modifier = murim_ranked_top_50_modifier
			}
			# Dropped out of Top 50
			murim_leave_top_50_effect = yes
		}
		# Update tier modifiers
		murim_update_rank_tier_modifiers_effect = yes
	}
}

######################################
# ENTER TOP 50
# Character crosses threshold, gains modifier, notification fires
######################################

murim_enter_top_50_effect = {
	# Add ranked fighter modifier
	add_character_modifier = {
		modifier = murim_ranked_top_50_modifier
		years = 1
	}
	# Notification to the character
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_ranking.0001
			days = 1
		}
	}
	# Prestige boost for entering rankings
	add_prestige = 200
	# Set initial rank position variable if not set
	if = {
		limit = {
			NOT = { has_variable = murim_martial_rank_position }
		}
		set_variable = {
			name = murim_martial_rank_position
			value = 50
		}
	}
	# Memory of entering rankings
	create_character_memory = {
		type = murim_entered_rankings_memory
	}
	# Court notification for liege
	if = {
		limit = {
			exists = liege
			liege = { is_ai = no }
		}
		liege = {
			send_interface_message = {
				type = event_generic_neutral
				title = murim_rank_new_entry_notification
				left_icon = root
				desc = murim_rank_new_entry_notification_desc
			}
		}
	}
	# Stress relief from glory
	add_stress = -15
}

######################################
# LEAVE TOP 50
# Dropped out, loses modifier, shame
######################################

murim_leave_top_50_effect = {
	# Remove ranked modifier
	if = {
		limit = { has_character_modifier = murim_ranked_top_50_modifier }
		remove_character_modifier = murim_ranked_top_50_modifier
	}
	# Notification to the character
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_ranking.0002
			days = 1
		}
	}
	# Shame modifier
	add_character_modifier = {
		modifier = murim_fallen_from_rankings_modifier
		years = 2
	}
	# Memory of falling from rankings
	create_character_memory = {
		type = murim_fell_from_rankings_memory
	}
	# Prestige loss
	add_prestige = -100
	# Stress from public failure
	add_stress = 20
	# Remove position variable
	if = {
		limit = { has_variable = murim_martial_rank_position }
		remove_variable = murim_martial_rank_position
	}
}

######################################
# REMOVE FROM RANKINGS ON DEATH
# Dead fighters removed, next in line promoted
######################################

murim_remove_from_rankings_on_death_effect = {
	# Only process if was a ranked fighter
	if = {
		limit = {
			has_variable = murim_martial_rank_position
		}
		# Store the dead fighter's rank for promotion logic
		save_temporary_scope_as = dead_ranked_fighter
		# Was this the #1?
		if = {
			limit = {
				var:murim_martial_rank_position = 1
			}
			# Realm-wide notification: #1 has fallen
			every_ruler = {
				limit = {
					is_ai = no
					is_alive = yes
				}
				trigger_event = {
					id = murim_ranker_death.0001
					days = { 1 5 }
				}
			}
		}
		# Remove all ranking variables and modifiers
		if = {
			limit = { has_character_modifier = murim_ranked_top_50_modifier }
			remove_character_modifier = murim_ranked_top_50_modifier
		}
		remove_variable = murim_martial_rank_position
		if = {
			limit = { has_variable = murim_martial_rank_score }
			remove_variable = murim_martial_rank_score
		}
	}
}

######################################
# UPDATE RANK TIER MODIFIERS
# Adds/removes top-10, top-25 tier modifiers based on score
######################################

murim_update_rank_tier_modifiers_effect = {
	# Top 10 check
	if = {
		limit = {
			exists = var:murim_martial_rank_score
			var:murim_martial_rank_score >= murim_is_top_10_threshold_value
		}
		# In top 10 tier
		if = {
			limit = {
				NOT = { has_character_modifier = strongest_under_heaven }
			}
			add_character_modifier = {
				modifier = strongest_under_heaven
				years = 1
			}
		}
	}
	else = {
		# Not in top 10 - remove modifier if present
		if = {
			limit = { has_character_modifier = strongest_under_heaven }
			remove_character_modifier = strongest_under_heaven
		}
	}
}

######################################
# DUEL RECORD UPDATE EFFECTS
# Called after ranked duels to update win/loss/kill records
######################################

murim_record_duel_win_effect = {
	if = {
		limit = {
			NOT = { has_variable = murim_duel_wins }
		}
		set_variable = {
			name = murim_duel_wins
			value = 0
		}
	}
	change_variable = {
		name = murim_duel_wins
		add = 1
	}
	# Recalculate rank after duel
	murim_recalculate_martial_rank_effect = yes
}

murim_record_duel_loss_effect = {

		add = 1
	}

}

######################################
# RANKING POSITION SWAP EFFECT
# Called when a challenger defeats a ranked fighter
######################################

murim_ranking_position_swap_effect = {
	# Root = challenger (winner), scope:target = defeated
	if = {
		limit = {
			scope:target = {
				has_variable = murim_martial_rank_position
			}
		}
		# Store target's old position
		save_temporary_scope_value_as = {
			name = stolen_position
			value = scope:target.var:murim_martial_rank_position
		}
		# Challenger takes the position
		set_variable = {
			name = murim_martial_rank_position
			value = scope:stolen_position
		}
		# Target drops one rank (or out if at 50)
		scope:target = {
			if = {
				limit = { var:murim_martial_rank_position >= 50 }
				# Pushed out of top 50 entirely
				murim_leave_top_50_effect = yes
			}
			else = {
				change_variable = {
					name = murim_martial_rank_position
					add = 1
				}
			}
		}
		# If challenger wasn't ranked before, enter rankings
		if = {
			limit = {
				NOT = { has_character_modifier = murim_ranked_top_50_modifier }
			}
			add_character_modifier = {
				modifier = murim_ranked_top_50_modifier
				years = 1
			}
		}
		# Memory of ranking ascension
		create_character_memory = {
			type = murim_rank_ascension_memory
		}
	}
}

######################################
# YEARLY RANKING RECALCULATION PULSE
# Runs for every living martial artist
######################################

murim_yearly_ranking_pulse_effect = {
	every_living_character = {
		limit = {
			murim_is_martial_artist_trigger = yes
			age >= 16
		}
		murim_recalculate_martial_rank_effect = yes
	}
}

######################################
# INITIALIZE RANKING VARIABLES
# Sets up ranking variables for a new martial artist
######################################

murim_initialize_ranking_variables_effect = {
	if = {
		limit = {
			NOT = { has_variable = murim_duel_wins }
		}
		set_variable = {
			name = murim_duel_wins
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_duel_losses }
		}
		set_variable = {
			name = murim_duel_losses
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_duel_kills }
		}
		set_variable = {
			name = murim_duel_kills
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_martial_rank_score }
		}
		set_variable = {
			name = murim_martial_rank_score
			value = 0
		}
	}
}
