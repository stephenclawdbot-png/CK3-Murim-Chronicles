# Entry Path Expansion 3 - Deep Chain for All 4 Paths into Martial World
# Path 1: Born into Clan, Path 2: Street Orphan, Path 3: Scholar Discovers Arts, Path 4: Imperial Agent Sent to Infiltrate
# 8 events covering advancement in each path

### EVENT 1: Born Into Clan - Legacy Pressure ###
murim_entry_path_exp3.0001 = {
	type = character_event
	title = murim_entry_path_exp3.0001.t
	desc = murim_entry_path_exp3.0001.desc
	theme = martial_arts
	
	trigger = {
		has_character_flag = entry_path_born_clan
		NOT = { has_character_flag = legacy_pressure_faced }
		age >= 14
		age <= 20
	}
	
	weight_multiplier = {
		base = 3
		modifier = { add = 2 any_parent = { martial >= 20 } }
	}
	
	# Option A: Embrace the family legacy
	option = {
		name = murim_entry_path_exp3.0001.a
		add_character_flag = { flag = legacy_pressure_faced days = -1 }
		add_martial_skill = 2
		add_prowess = 3
		add_prestige = 300
		add_stress = 15
		stress_impact = { lazy = major_stress_impact_gain }
		ai_chance = { base = 50 }
	}
	
	# Option B: Reject the family path - forge your own
	option = {
		name = murim_entry_path_exp3.0001.b
		add_character_flag = { flag = legacy_pressure_faced days = -1 }
		add_character_flag = { flag = rejected_family_legacy days = -1 }
		add_prestige = -200
		add_stress = -10
		add_learning_skill = 2
		reverse_add_opinion = {
			target = scope:clan_elder
			modifier = rejected_legacy_opinion
			opinion = -40
		}
		ai_chance = { base = 25 }
	}
	
	# Option C: Secretly study forbidden techniques
	option = {
		name = murim_entry_path_exp3.0001.c
		trigger = { intrigue >= 10 }
		add_character_flag = { flag = legacy_pressure_faced days = -1 }
		add_character_flag = { flag = studying_forbidden days = -1 }
		add_intrigue_skill = 2
		add_martial_skill = 1
		add_prowess = 2
		ai_chance = { base = 20 }
	}
	
	# Option D: Ask to train under a different clan's master
	option = {
		name = murim_entry_path_exp3.0001.d
		trigger = { diplomacy >= 12 }
		add_character_flag = { flag = legacy_pressure_faced days = -1 }
		add_character_flag = { flag = cross_clan_training days = -1 }
		add_diplomacy_skill = 1
		add_prestige = 100
		ai_chance = { base = 15 }
	}
}
