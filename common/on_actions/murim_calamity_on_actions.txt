# Murim Chronicles - Great Calamity On-Actions
# Hooks for calamity detection, phase transitions, mass death processing, survivor selection.
# INTEGRATION: murim_calamity_triggers.txt, murim_great_calamity_events.txt, murim_calamity_traits.txt
# DESIGN: Yearly check for calamity conditions. Phase transitions on timer.
# MASS DEATH: Iterates all martial artists during Phase 3, kills non-survivors.

# ========================================
# YEARLY CALAMITY CHECK
# Runs every year to detect calamity conditions
# ========================================

on_yearly_pulse = {
	effect = {
		# === OMEN PHASE: Start warnings 10-20 years before ===
		if = {
			limit = {
				murim_calamity_omens_should_start_trigger = yes
			}
			set_global_variable = {
				name = murim_calamity_omens_active
				value = yes
			}
			# Fire omen events to random martial artists
			every_living_character = {
				limit = {
					murim_is_martial_artist_trigger = yes
					is_ruler = yes
				}
				random = {
					chance = 15
					trigger_event = { id = murim_calamity.0001 days = { 1 180 } }
				}
				random = {
					chance = 10
					trigger_event = { id = murim_calamity.0010 days = { 30 270 } }
				}
				random = {
					chance = 8
					trigger_event = { id = murim_calamity.0020 days = { 60 300 } }
				}
			}
		}
		
		# === CALAMITY TRIGGER: Conditions met, start Phase 1 ===
		if = {
			limit = {
				murim_calamity_conditions_met_trigger = yes
				NOT = { has_global_variable = murim_calamity_delayed }
			}
			# Fire Phase 1: Gathering Storm to all rulers
			every_living_character = {
				limit = { is_ruler = yes }
				trigger_event = { id = murim_calamity.1000 days = { 1 30 } }
			}
			# Schedule Phase 2 transition (5 years later)
			random_living_character = {
				limit = { is_ruler = yes }
				trigger_event = { id = murim_calamity_on_action.0001 days = 1825 }
			}
		}
		
		# === DELAYED CALAMITY: If delayed, check if delay expired ===
		if = {
			limit = {
				has_global_variable = murim_calamity_delayed
				# Delay lasts 10 years, then triggers anyway
				murim_calamity_conditions_met_trigger = yes
			}
			# After delay, remove the delay flag (calamity fires next year)
			random = {
				chance = 10  # 10% per year after delay = ~10 year average
				remove_global_variable = murim_calamity_delayed
			}
		}
	}
}

namespace = murim_calamity_on_action

# Transition to Phase 2: The Great War
murim_calamity_on_action.0001 = {
	type = character_event
	hidden = yes
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_global_variable = murim_calamity_phase_storm
		NOT = { has_global_variable = murim_calamity_phase_war }
	}
	
	immediate = {
		# Fire Phase 2 kickoff to all rulers
		every_living_character = {
			limit = { is_ruler = yes }
			trigger_event = { id = murim_calamity.2000 days = { 1 30 } }
		}
		# Fire death events to AI martial artists during war
		every_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
				is_ai = yes
				NOT = { has_trait = murim_fate_blessed }
				NOT = { has_character_flag = murim_calamity_hiding }
			}
			random = {
				chance = 30  # 30% chance per martial artist
				trigger_event = { id = murim_calamity.2010 days = { 30 730 } }
			}
		}
		# Heavenly Demon event for all rulers
		every_living_character = {
			limit = { is_ruler = yes }
			trigger_event = { id = murim_calamity.2020 days = { 180 540 } }
		}
		# Schedule Phase 3 transition (2-3 years later)
		trigger_event = { id = murim_calamity_on_action.0002 days = 912 }
	}
	
	option = { name = ok }
}

# Transition to Phase 3: The Silence
murim_calamity_on_action.0002 = {
	type = character_event
	hidden = yes
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_global_variable = murim_calamity_phase_war
		NOT = { has_global_variable = murim_calamity_phase_silence }
	}
	
	immediate = {
		# === SURVIVOR SELECTION ===
		# Step 1: Mark player dynasty as survivors
		every_living_character = {
			limit = { is_ai = no }
			add_character_flag = murim_calamity_survivor
		}
		every_living_character = {
			limit = {
				dynasty = {
					any_dynasty_member = {
						is_ai = no
					}
				}
			}
			# At least 1 dynasty member survives
			random = {
				chance = 50
				add_character_flag = murim_calamity_survivor
			}
		}
		
		# Step 2: Mark fate-blessed as survivors
		every_living_character = {
			limit = {
				has_trait = murim_fate_blessed
			}
			add_character_flag = murim_calamity_survivor
		}
		
		# Step 3: Mark 1-3 per clan as survivors (random)
		# Shaolin
		random_living_character = {
			limit = {
				has_trait = shaolin_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				has_trait = shaolin_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 50
				add_character_flag = murim_calamity_survivor
			}
		}
		# Wudang
		random_living_character = {
			limit = {
				has_trait = wudang_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				has_trait = wudang_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 50
				add_character_flag = murim_calamity_survivor
			}
		}
		# Mount Hua
		random_living_character = {
			limit = {
				has_trait = mount_hua_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				has_trait = mount_hua_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 40
				add_character_flag = murim_calamity_survivor
			}
		}
		# Beggar Sect
		random_living_character = {
			limit = {
				has_trait = beggar_sect_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				has_trait = beggar_sect_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 60  # Beggar Sect has better survival (spy network)
				add_character_flag = murim_calamity_survivor
			}
		}
		# Iron Fist Gate
		random_living_character = {
			limit = {
				has_trait = iron_fist_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		# Tang Clan
		random_living_character = {
			limit = {
				OR = {
					has_trait = tang_clan_member
					has_trait = tang_clan_disciple
				}
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				OR = {
					has_trait = tang_clan_member
					has_trait = tang_clan_disciple
				}
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 40
				add_character_flag = murim_calamity_survivor
			}
		}
		# Heavenly Demon Sect — lower survival, they were at the center
		random_living_character = {
			limit = {
				has_trait = heavenly_demon_disciple
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		# Poison Flower Palace
		random_living_character = {
			limit = {
				has_trait = poison_flower_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		# Phantom Blade Sect — assassins survive well
		random_living_character = {
			limit = {
				has_trait = phantom_blade_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			add_character_flag = murim_calamity_survivor
		}
		random_living_character = {
			limit = {
				has_trait = phantom_blade_member
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 60  # Phantom Blade good at hiding
				add_character_flag = murim_calamity_survivor
			}
		}
		
		# Step 4: Characters in seclusion survive
		every_living_character = {
			limit = {
				has_character_flag = murim_calamity_hiding
			}
			add_character_flag = murim_calamity_survivor
		}
		
		# Step 5: Young geniuses survive (hope for the future)
		every_living_character = {
			limit = {
				age <= 20
				OR = {
					has_trait = genius
					has_trait = quick
					has_trait = brave
				}
			}
			random = {
				chance = 60
				add_character_flag = murim_calamity_survivor
			}
		}
		
		# === MASS DEATH PROCESSING ===
		# Kill all non-survivor martial artists
		every_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
				is_ai = yes
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			death = {
				death_reason = death_battle
			}
		}
		
		# Kill 50-70% of non-martial-artist AI characters (civilians)
		every_living_character = {
			limit = {
				is_ai = yes
				NOT = { murim_is_martial_artist_trigger = yes }
				NOT = { has_character_flag = murim_calamity_survivor }
			}
			random = {
				chance = 60
				death = {
					death_reason = death_battle
				}
			}
		}
		
		# Fire Phase 3 event to survivors
		every_living_character = {
			limit = { is_ruler = yes }
			trigger_event = { id = murim_calamity.3000 days = { 1 30 } }
		}
		
		# Fire ruin wandering events
		every_living_character = {
			limit = {
				has_character_flag = murim_calamity_survivor
			}
			trigger_event = { id = murim_calamity.3020 days = { 60 270 } }
		}
		
		# Schedule Phase 4 transition (1 year later)
		trigger_event = { id = murim_calamity_on_action.0003 days = 365 }
	}
	
	option = { name = ok }
}

# Transition to Phase 4: The Renewal
murim_calamity_on_action.0003 = {
	type = character_event
	hidden = yes
	
	trigger = {
		has_global_variable = murim_great_calamity_active
		has_global_variable = murim_calamity_phase_silence
	}
	
	immediate = {
		# Fire renewal events
		every_living_character = {
			limit = { is_ruler = yes }
			trigger_event = { id = murim_calamity.4000 days = { 1 30 } }
		}
		
		# Fire legacy events
		every_living_character = {
			limit = {
				has_trait = murim_calamity_survivor
			}
			trigger_event = { id = murim_calamity.4030 days = { 60 365 } }
		}
		
		# Fire training events for survivors with children
		every_living_character = {
			limit = {
				has_trait = murim_calamity_survivor
				any_child = {
					age >= 10
					age <= 16
				}
			}
			trigger_event = { id = murim_calamity.4020 days = { 90 365 } }
		}
		
		# Assign hidden fate traits to newborns for next cycle
		every_living_character = {
			limit = {
				age <= 5
			}
			random_list = {
				10 = { add_trait = murim_fate_blessed }
				10 = { add_trait = murim_fate_doomed }
				80 = { }  # Most have no special fate
			}
		}
	}
	
	option = { name = ok }
}

# ========================================
# BIRTH HOOK: Assign fate traits to new characters
# ========================================

on_birth_child = {
	effect = {
		# During non-calamity times, rarely assign fate traits
		random_list = {
			5 = { add_trait = murim_fate_blessed }
			5 = { add_trait = murim_fate_doomed }
			90 = { }
		}
	}
}

# DEATH HOOK: Calamity death processing
# ========================================

on_death = {
	effect = {
		# During calamity, track clan extinction
		if = {
			limit = {
				has_global_variable = murim_great_calamity_active
				murim_is_martial_artist_trigger = yes
			}
			# If this was the last member of a clan, set a global flag
			if = {
				limit = { has_trait = shaolin_disciple }
				if = {
					limit = {
						NOT = {
							any_living_character = {
								has_trait = shaolin_disciple
								NOT = { this = root }
							}
						}
					}
					set_global_variable = { name = murim_shaolin_extinct value = yes }
