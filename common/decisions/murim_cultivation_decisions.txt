#   var:murim_qi_reserves, var:murim_technique_mastery_minor,

# ==============================================================
# 1. MEDITATE IN SECLUSION
# Enter 30-day meditation retreat for cultivation XP.
# Safe, reliable, bread-and-butter progression.
# Cooldown: 6 months.
# ==============================================================

meditate_in_seclusion = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = meditate_in_seclusion_desc
	selection_tooltip = meditate_in_seclusion_tooltip
	major = no

	ai_check_interval = 60

	is_shown = {
		has_character_flag = murim_initiated
		is_available_adult = yes
	}

	is_valid = {
		is_available_adult = yes
		is_imprisoned = no
		is_at_war = no
		NOR = {
			has_character_flag = murim_in_seclusion
			has_character_flag = murim_closed_door_cultivation
			has_character_modifier = murim_meditating
		}
		custom_description = {
			text = "murim_meditation_cooldown_not_expired"
			NOT = { has_character_flag = murim_meditation_cooldown }
		}
	}

	cost = {
		prestige = {
			value = 50
			# Higher ranks demand more prestige (reputation cost of vanishing)
			add = {
				value = murim_cultivation_rank_value
				multiply = 15
			}
		}
	}

	effect = {
		# Lock for 30 days
		add_character_flag = {
			flag = murim_in_seclusion
		}
		add_character_modifier = {
			modifier = murim_meditating
			days = 30
		}

		# Calculate XP gain based on learning
		custom_tooltip = murim_meditation_xp_gain_tooltip
		hidden_effect = {
			# Base XP: learning * 0.8, clamped
			change_variable = {
				name = murim_cultivation_xp
				add = {
					value = learning
					multiply = 0.8
					# Talent modifier
					multiply = murim_cultivation_talent_modifier
					# Meditation efficiency (traits, stress, etc.)
					multiply = murim_meditation_efficiency
					min = 2
					max = 30
					round = yes
				}
			}

			# Bonus for patient/temperate characters
			if = {
				limit = { has_trait = patient }
				change_variable = {
					name = murim_cultivation_xp
					add = 3
				}
			}
			if = {
				limit = { has_trait = temperate }
				change_variable = {
					name = murim_cultivation_xp
					add = 2
				}
			}

			# Small stress loss (meditation is calming)
			add_stress = -10

			# Set cooldown flag (6 months)
			add_character_flag = {
				flag = murim_meditation_cooldown
				months = 6
			}

			# Remove seclusion lock after 30 days via event
			trigger_event = {
				id = murim_cultivation.001
				days = 30
			}
		}

		# Flavor text
		custom_tooltip = murim_meditation_flavor_tooltip
	}

	ai_potential = {
		has_character_flag = murim_initiated
		is_at_war = no
		stress >= 20
	}

	ai_will_do = {
		base = 30

		# AI more likely if high stress or low XP
		modifier = {
			add = 20
			stress >= 50
		}
		modifier = {
			add = 15
			has_trait = patient
		}
		modifier = {
			add = 10
			has_trait = temperate
		}
		modifier = {
			add = -20
			has_trait = impatient
		}
		modifier = {
			add = -15
			has_trait = lazy
		}
		# Don't meditate if close to breakthrough (save for that)
		modifier = {
			add = -30
			exists = var:murim_cultivation_xp
			var:murim_cultivation_xp >= murim_breakthrough_threshold
		}
	}
}

# ==============================================================
# 2. ATTEMPT BREAKTHROUGH
# THE key progression decision. Advance cultivation rank.
# Available when XP >= threshold. Complex outcomes.
# ==============================================================

attempt_breakthrough = {
	picture = "gfx/interface/illustrations/decisions/decision_destiny_quest.dds"
	desc = attempt_breakthrough_desc
	selection_tooltip = attempt_breakthrough_tooltip
	major = yes

	ai_check_interval = 30

	is_shown = {
		has_character_flag = murim_initiated
		# Not already at max rank
		NOT = { has_trait = murim_spirit_severing }
	}

	is_valid = {
		is_available_adult = yes
		is_imprisoned = no
		NOT = { has_character_flag = murim_breakthrough_in_progress }
		# Must have enough XP
		custom_description = {
			text = "murim_xp_meets_breakthrough_threshold"
			exists = var:murim_cultivation_xp
			var:murim_cultivation_xp >= murim_breakthrough_threshold
		}
		# Health gate: can't attempt while severely injured
		custom_description = {
			text = "murim_health_sufficient_for_breakthrough"
			health >= 2
		}
	}

	cost = {
		gold = {
			value = 75
			add = {
				value = murim_cultivation_rank_value
				multiply = 50
			}
		}
		prestige = {
			value = 100
			add = {
				value = murim_cultivation_rank_value
				multiply = 75
			}
		}
	}

	effect = {
		add_character_flag = {
			flag = murim_breakthrough_in_progress
		}

		custom_tooltip = murim_breakthrough_attempt_tooltip

		hidden_effect = {
			# Main breakthrough roll
			random_list = {
				# --- CRITICAL SUCCESS ---
				# Breakthrough + bonus XP carry-over + trait
				5 = {
					modifier = {
						add = {
							value = learning
							multiply = 0.5
						}
					}
					modifier = {
						add = 5
						has_trait = murim_heavenly_body
					}
					modifier = {
						add = 3
						has_character_flag = murim_manual_studied_for_breakthrough
					}
					trigger_event = {
						id = murim_cultivation.101
						days = 3
					}
				}

				# --- STANDARD SUCCESS ---
				# Breakthrough, XP resets, rank advances
				murim_breakthrough_success_chance = {
					trigger_event = {
						id = murim_cultivation.102
						days = 3
					}
				}

				# --- FAILURE (No Deviation) ---
				# Lose 30% XP, no rank change, try again later
				20 = {
					modifier = {
						add = 10
						stress >= 60
					}
					modifier = {
						add = 10
						age >= 55
					}
					modifier = {
						add = -10
						has_trait = patient
					}
					trigger_event = {
						id = murim_cultivation.103
						days = 3
					}
				}

				# --- FAILURE WITH QI DEVIATION ---
				# Lose XP + health damage + possible trait damage
				murim_qi_deviation_risk = {
					trigger_event = {
						id = murim_cultivation.104
						days = 3
					}
				}

				# --- CATASTROPHIC FAILURE ---
				# Permanent injury, massive XP loss, rank may drop
				3 = {
					modifier = {
						add = 5
						has_character_flag = murim_demonic_cultivator
					}
					modifier = {
						add = 3
						has_character_flag = murim_used_forbidden_technique
					}
					modifier = {
						add = -3
						has_character_modifier = murim_sect_master_guidance
					}
					trigger_event = {
						id = murim_cultivation.105
						days = 3
					}
				}
			}
		}
	}

	ai_potential = {
		has_character_flag = murim_initiated
		NOT = { has_trait = murim_spirit_severing }
		exists = var:murim_cultivation_xp
		var:murim_cultivation_xp >= murim_breakthrough_threshold
	}

	ai_will_do = {
		base = 80

		# AI almost always attempts when ready
		modifier = {
			add = 15
			has_trait = ambitious
		}
		modifier = {
			add = -30
			health <= 3
		}
		modifier = {
			add = -20
			stress >= 70
		}
		# Demonic cultivators are more reckless
		modifier = {
			add = 10
			has_character_flag = murim_demonic_cultivator
		}
		modifier = {
			add = 10
			has_character_modifier = murim_sect_master_guidance
		}
	}
}

# 3. STUDY MARTIAL MANUAL
# Spend time with a martial arts manual.
# Gain technique mastery XP. Cooldown: 3 months.
# ==============================================================

study_martial_manual = {
	picture = "gfx/interface/illustrations/decisions/decision_misc.dds"
	desc = study_martial_manual_desc
	selection_tooltip = study_martial_manual_tooltip
	major = no

	ai_check_interval = 90

	is_shown = {
		has_character_flag = murim_initiated
		is_available_adult = yes
	}

	is_valid = {
		is_available_adult = yes
		is_imprisoned = no
		NOR = {
			has_character_modifier = murim_studying_manual
			has_character_flag = murim_in_seclusion
			has_character_flag = murim_closed_door_cultivation
		}
		custom_description = {
			text = "murim_manual_study_cooldown_not_expired"
			NOT = { has_character_flag = murim_manual_study_cooldown }
		}
	}

	cost = {
		gold = {
			value = 30
			# Higher quality manuals cost more
			add = {
				value = murim_cultivation_rank_value
				multiply = 20
			}
		}
	}

	effect = {
		# Add studying modifier (lasts 90 days, gives ongoing bonuses)
		add_character_modifier = {
			modifier = murim_studying_manual
			days = 90
		}

		custom_tooltip = murim_study_manual_effects_tooltip

		hidden_effect = {
			# Determine manual quality based on rank + luck
			random_list = {
				# Common manual (quality 1-2)
				50 = {
					set_variable = {
						name = murim_manual_quality
						value = 1
					}
					set_variable = {
						name = murim_manual_min_rank
						value = 0
					}
				}
				# Rare manual (quality 3-4)
				30 = {
					modifier = {
						add = 10
						murim_cultivation_rank_value >= 3
					}
					set_variable = {
						name = murim_manual_quality
						value = 3
					}
					set_variable = {
						name = murim_manual_min_rank
						value = 2
					}
				}
				# Legendary manual (quality 5)
				10 = {
					modifier = {
						add = 5
						murim_cultivation_rank_value >= 4
					}
					modifier = {
						add = 5
						has_character_flag = murim_sect_master
					}
					set_variable = {
						name = murim_manual_quality
						value = 5
					}
					set_variable = {
						name = murim_manual_min_rank
						value = 3
					}
				}
				# Forbidden manual (quality 5 + demonic)
				10 = {
					modifier = {
						add = 10
						has_character_flag = murim_demonic_cultivator
					}
					modifier = {
						add = -5
						has_character_flag = murim_orthodox_cultivator
					}
					set_variable = {
						name = murim_manual_quality
						value = 5
					}
					set_variable = {
						name = murim_manual_min_rank
						value = 2
					}
					# Forbidden manuals shift alignment
					change_variable = {
						name = murim_alignment
						add = -8
					}
					add_character_flag = {
						flag = murim_used_forbidden_technique
					}
				}
			}

			# Reset study XP for new manual
			set_variable = {
				name = murim_manual_study_xp
				value = 0
			}

			# Immediate technique mastery XP boost
			change_variable = {
				name = murim_technique_mastery_minor_xp
				add = {
					value = learning
					multiply = 0.3
					min = 1
					max = 5
					round = yes
				}
			}

			# Set cooldown (3 months)
			add_character_flag = {
				flag = murim_manual_study_cooldown
				months = 3
			}
		}
	}

	ai_potential = {
		has_character_flag = murim_initiated
		gold >= 30
	}

	ai_will_do = {
		base = 25

		modifier = {
			add = 15
			has_trait = diligent
		}
		modifier = {
			add = 10
			learning >= 15
		}
		modifier = {
			add = -15
			has_trait = lazy
		}
		# Prioritize if approaching breakthrough
		modifier = {
			add = 20
			exists = var:murim_cultivation_xp
			var:murim_cultivation_xp >= {
				value = murim_breakthrough_threshold
				multiply = 0.7
			}
		}
	}
}

# 4. TEMPER BODY
# Physical cultivation. Costs gold (medicine) + health risk.
# Increases body_tempering variable. Cooldown: 6 months.
# ==============================================================

temper_body = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = temper_body_desc
	selection_tooltip = temper_body_tooltip
	major = no

	ai_check_interval = 90

	is_shown = {
		has_character_flag = murim_initiated
		is_available_adult = yes
		# Max 9 body tempering levels
		exists = var:murim_body_tempering_level
		var:murim_body_tempering_level < 9
	}

	is_valid = {
		is_available_adult = yes
		is_imprisoned = no
		NOT = { has_character_flag = murim_in_seclusion }
		NOT = { has_character_flag = murim_closed_door_cultivation }
		custom_description = {
			text = "murim_body_tempering_cooldown_not_expired"
			NOT = { has_character_flag = murim_body_tempering_cooldown }
		}
		# Need minimum health (this is physically brutal)
		custom_description = {
			text = "murim_health_sufficient_for_tempering"
			health >= 3
		}
	}

	cost = {
		gold = {
			# Medicinal baths get progressively more expensive
			value = 40
			add = {
				value = var:murim_body_tempering_level
				multiply = 25
			}
		}
	}

	effect = {
		custom_tooltip = murim_body_tempering_effects_tooltip

		hidden_effect = {
			# Body tempering XP gain (martial + prowess based)
			change_variable = {
				name = murim_body_tempering_xp
				add = {
					value = martial
					multiply = 0.5
					add = {
						value = prowess
						multiply = 0.4
					}
					# Strong/Herculean traits help
					if = {
						limit = { has_trait = strong }
						add = 3
					}
					if = {
						limit = { has_trait = herculean }
						add = 6
					}
					min = 3
					max = 25
					round = yes
				}
			}

			# Health risk -- higher levels are more dangerous
			random_list = {
				# Safe -- no complications
				60 = {
					modifier = {
						add = 10
						has_trait = robust
					}
					modifier = {
						add = 10
						has_trait = strong
					}
					modifier = {
						add = -5
						var:murim_body_tempering_level >= 5
					}
					modifier = {
						add = -10
						var:murim_body_tempering_level >= 7
					}
					# Everything went fine
					custom_tooltip = murim_tempering_safe_tooltip
				}
				# Minor injury -- small health loss
				25 = {
					modifier = {
						add = 5
						var:murim_body_tempering_level >= 5
					}
					add_character_modifier = {
						modifier = murim_tempering_strain
						days = 30
					}
				}
				# Severe reaction -- significant health damage
				10 = {
					modifier = {
						add = 5
						var:murim_body_tempering_level >= 7
					}
					modifier = {
						add = 5
						has_trait = weak
					}
					modifier = {
						add = -5
						has_trait = herculean
					}
					add_character_modifier = {
						modifier = murim_tempering_severe_strain
						days = 60
					}
					# Small XP penalty (body rejected the tempering)
					change_variable = {
						name = murim_body_tempering_xp
						add = -3
					}
				}
				# Critical failure -- meridian damage
				5 = {
					modifier = {
						add = 3
						var:murim_body_tempering_level >= 8
					}
					modifier = {
						add = -3
						has_character_modifier = murim_sect_master_guidance
					}
					add_character_modifier = {
						modifier = murim_meridian_damage_minor
						years = 1
					}
					trigger_event = {
						id = murim_cultivation.401
						days = 1
					}
				}
			}

			# Prowess gain from the training
			change_skill = {
				prowess = 1
			}

			# Set cooldown (6 months)
			add_character_flag = {
				flag = murim_body_tempering_cooldown
				months = 6
			}
		}
	}

	ai_potential = {
		has_character_flag = murim_initiated
		exists = var:murim_body_tempering_level
		var:murim_body_tempering_level < 9
		health >= 4
	}

	ai_will_do = {
		base = 20

		modifier = {
			add = 15
			has_trait = brave
		}
		modifier = {
			add = 10
			has_trait = strong
		}
		modifier = {
			add = 10
			prowess >= 15
		}
		modifier = {
			add = -20
			has_trait = craven
		}
		modifier = {
			add = -10
			health <= 4
		}
	}
}

# ==============================================================
# 5. OPEN MERIDIAN
# Attempt to open next meridian (12 total).
# Each grants specific bonus. Cooldown: 1 year.
# ==============================================================

open_meridian = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = open_meridian_desc
	selection_tooltip = open_meridian_tooltip
	major = no

	ai_check_interval = 120

	is_shown = {
		has_character_flag = murim_initiated
		is_available_adult = yes
	}

	is_valid = {
		is_available_adult = yes
		is_imprisoned = no
		NOT = { has_character_flag = murim_in_seclusion }
		NOT = { has_character_flag = murim_closed_door_cultivation }
		# Must have meridians left to open (max 12)
		custom_description = {
			text = "murim_meridians_remaining_to_open"
			OR = {
				NOT = { exists = var:murim_meridians_opened }
				var:murim_meridians_opened < 12
			}
		}
		# Cooldown check
		custom_description = {
			text = "murim_meridian_cooldown_not_expired"
			NOT = { has_character_flag = murim_meridian_cooldown }
		}
		# Minimum cultivation rank scales with meridians opened
		custom_description = {
			text = "murim_rank_sufficient_for_meridian"
			trigger_if = {
				limit = {
					exists = var:murim_meridians_opened
					var:murim_meridians_opened >= 6
				}
				murim_cultivation_rank_value >= 3
			}
			trigger_if = {
				limit = {
					exists = var:murim_meridians_opened
					var:murim_meridians_opened >= 9
				}
				murim_cultivation_rank_value >= 4
			}
		}
	}

	cost = {
		prestige = {
			value = 75
			if = {
				limit = { exists = var:murim_meridians_opened }
				add = {
					value = var:murim_meridians_opened
					multiply = 25
				}
			}
		}
	}

	effect = {
		custom_tooltip = murim_open_meridian_effects_tooltip

		hidden_effect = {
			# Initialize if first meridian
			if = {
				limit = {
					NOT = { exists = var:murim_meridians_opened }
				}
				set_variable = {
					name = murim_meridians_opened
					value = 0
				}
			}

			# Qi cost -- uses reserves
			if = {
				limit = { exists = var:murim_qi_reserves }
				change_variable = {
					name = murim_qi_reserves
					add = {
						value = -10
						subtract = {
							value = var:murim_meridians_opened
							multiply = 3
						}
					}
				}
				# Clamp to 0
				if = {
					limit = { var:murim_qi_reserves < 0 }
					set_variable = {
						name = murim_qi_reserves
						value = 0
					}
				}
			}

			# Success roll
			random_list = {
				# Success -- meridian opens
				60 = {
					modifier = {
						add = {
							value = learning
							multiply = 1.0
						}
					}
					modifier = {
						add = 10
						has_trait = murim_heavenly_body
					}
					modifier = {
						add = 5
						has_trait = murim_pure_yin_body
					}
					modifier = {
						add = 5
						has_trait = murim_pure_yang_body
					}
					modifier = {
						add = -5
						exists = var:murim_meridians_opened
						var:murim_meridians_opened >= 8
					}

					# Open the meridian
					change_variable = {
						name = murim_meridians_opened
						add = 1
					}

					# Grant bonus based on which meridian (1-12)
					# Meridians 1-4: stat bonuses
					# Meridians 5-8: cultivation bonuses
					# Meridians 9-12: transcendent bonuses
					trigger_event = {
						id = murim_cultivation.501
