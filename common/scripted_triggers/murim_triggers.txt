######################################

murim_is_martial_artist_trigger = {
	OR = {
		# Base martial artist trait (from murim_clan_traits.txt)
		# Characters entering via entry paths get this first
		has_trait = martial_artist
		# Variable-based check (from murim_variable_init.txt)
		has_variable = murim_cultivator
		# Cultivation realm traits (murim_cultivation_traits.txt)
		has_trait = mortal_realm
		has_trait = qi_condensation_realm
		has_trait = foundation_realm
		has_trait = core_formation_realm
		has_trait = nascent_soul_realm
		has_trait = ascension_realm
		# Alternate cultivation traits (01_martial_traits.txt)
		has_trait = murim_outer_qi
		has_trait = murim_qi_gathering
		has_trait = murim_foundation
		has_trait = murim_golden_core
		has_trait = murim_nascent_soul
		has_trait = murim_spirit_severing
		has_trait = murim_heavenly_tribulation
		# Ranking traits (murim_ranking_traits.txt)
		has_trait = murim_rank_top_50
		has_trait = murim_rank_top_20
		has_trait = murim_rank_top_10
		has_trait = murim_rank_top_5
		has_trait = murim_rank_first
		# Clan membership traits (any clan member is a martial artist)
		has_trait = mount_hua_disciple
		has_trait = wudang_disciple
		has_trait = shaolin_monk
		has_trait = heavenly_demon_disciple
		has_trait = beggar_sect_member
		has_trait = iron_fist_disciple
		has_trait = poison_flower_disciple
		has_trait = phantom_blade_disciple
		has_trait = tang_clan_disciple
		has_trait = tang_clan_member
	}
}

# COMBAT & CHALLENGE ELIGIBILITY
######################################

murim_can_challenge_for_rank_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	NOT = { has_trait = craven }
	murim_is_martial_artist_trigger = yes
	NOT = {
		has_character_modifier = murim_recent_duel_cooldown
	}
	prowess >= 10
}

murim_can_be_challenged_for_rank_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	murim_is_top_50_martial_artist_trigger = yes
	NOT = {
		has_character_modifier = murim_recent_duel_cooldown
	}
}

murim_ai_should_challenge_for_rank_trigger = {
	is_ai = yes
	is_alive = yes
	murim_can_challenge_for_rank_trigger = yes
	NOT = { has_character_modifier = murim_recent_challenge_cooldown }
	OR = {
		has_trait = ambitious
		has_trait = brave
		prowess >= 18
	}
}

######################################
# GEOGRAPHY TRIGGERS
######################################

murim_is_china_region_trigger = {
	OR = {
		has_variable = murim_china_region
		# Capital is in East Asia cultural region
		capital_province = {
			OR = {
				geographical_region = world_china
				geographical_region = world_tibet
			}
		}
	}
}

murim_is_goryeo_region_trigger = {
	OR = {
		has_variable = murim_goryeo_region
		capital_province = {
			geographical_region = world_korea
		}
	}
}

murim_is_martial_heartland_trigger = {
	OR = {
		murim_is_china_region_trigger = yes
		murim_is_goryeo_region_trigger = yes
	}
}

murim_is_periphery_trigger = {
	NOT = { murim_is_martial_heartland_trigger = yes }
}

######################################

	NOT = { murim_is_orthodox_aligned_trigger = yes }
