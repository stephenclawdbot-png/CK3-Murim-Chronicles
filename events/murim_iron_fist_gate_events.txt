# IRON FIST GATE — Full Clan Event Chain
# The Iron Fist Gate is a martial clan renowned for body cultivation and fist arts.
# Orthodox faction. Known for unyielding discipline, iron body techniques, and honor-bound combat.
# Located in the mountainous northern reaches of the Murim world.

# ============================================
# EVENT 0001 — Iron Body Initiation
# New disciples undergo the Iron Body Tempering, submerging in medicinal baths and enduring strikes
# ============================================
murim_iron_fist_gate.0001 = {
	type = character_event
	title = murim_iron_fist_gate.0001.t
	desc = murim_iron_fist_gate.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = pain
	}
	right_portrait = {
		character = scope:iron_fist_elder
		animation = personality_bold
	}
	
	trigger = {
		is_alive = yes
		murim_is_martial_artist_trigger = yes
		NOT = { has_trait = iron_body_initiate }
		NOT = { has_character_flag = iron_fist_initiation_done }
		any_relation = {
			type = mentor
			has_trait = iron_fist_gate_member
		}
	}
	
	immediate = {
		random_relation = {
			type = mentor
			limit = { has_trait = iron_fist_gate_member }
			save_scope_as = iron_fist_elder
		}
		set_character_flag = {
			flag = iron_fist_initiation_done
			days = 3650
		}
	}
	
	# Option A: Endure the full tempering
	option = {
		name = murim_iron_fist_gate.0001.a
		add_trait = iron_body_initiate
		add_character_modifier = {
			modifier = iron_body_tempering_modifier
			days = 365
		}
		stress_impact = {
			brave = minor_stress_impact_loss
			craven = major_stress_impact_gain
		}
		add_internal_qi = 5
		trigger_event = {
			id = murim_iron_fist_gate.0002
			days = { 60 120 }
		}
	}
	
	# Option B: Refuse — too painful
	option = {
		name = murim_iron_fist_gate.0001.b
		add_prestige = -200
		add_opinion = {
			target = scope:iron_fist_elder
			modifier = disappointed_opinion
			opinion = -30
		}
		add_character_flag = {
			flag = iron_fist_rejected_tempering
			days = 7300
		}
	}
	
	# Option C: Attempt a shortcut with herbs
	option = {
		name = murim_iron_fist_gate.0001.c
		trigger = {
			OR = {
				has_trait = deceitful
				learning >= 12
			}
		}
		random_list = {
			60 = {
				add_trait = iron_body_initiate
				add_character_modifier = {
					modifier = herbal_iron_body_modifier
					days = 180
				}
			}
			40 = {
				add_trait = qi_deviation_minor
				add_character_modifier = {
					modifier = failed_shortcut_modifier
					days = 365
				}
				add_prestige = -100
			}
		}
	}
}
