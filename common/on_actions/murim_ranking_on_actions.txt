# Murim Chronicles - Top 50 Martial Ranking On Actions
# Yearly recalculation pulse, death removal, duel completion hooks
# AI ranked challenges, tournament scheduling
# Load-safe: all on_actions check exists before variable access

######################################
# YEARLY RANKING RECALCULATION
# Every January, recalculate all martial artist scores
######################################

on_yearly_pulse = {
	effect = {
		# Recalculate rankings for all living martial artists
		every_living_character = {
			limit = {
				exists = this
				is_alive = yes
				murim_is_martial_artist_trigger = yes
			}
			# Initialize variables if first time
			murim_initialize_ranking_variables_effect = yes
			# Recalculate score
			murim_recalculate_martial_rank_effect = yes
		}

		# Player yearly ranking summary
		every_player = {
			limit = {
				exists = this
				is_alive = yes
				murim_is_martial_artist_trigger = yes
			}
			trigger_event = {
				id = murim_rank.0006
				days = { 5 15 }
			}
		}

		# AI ranked fighters challenge each other (3-5 per year)
		random_living_character = {
			limit = {
				is_ai = yes
				is_alive = yes
				murim_is_top_50_martial_artist_trigger = yes
				murim_ai_should_challenge_for_rank_trigger = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					has_trait = ambitious
				}
				modifier = {
					add = 3
					has_trait = brave
				}
				modifier = {
					add = -10
					has_trait = craven
				}
			}
			# Find a target to challenge
			random_living_character = {
				limit = {
					is_alive = yes
					murim_is_top_50_martial_artist_trigger = yes
					murim_can_be_challenged_for_rank_trigger = yes
					NOT = { this = prev }
					# Challenge someone ranked higher
					has_variable = murim_martial_rank_position
					prev = {
						has_variable = murim_martial_rank_position
						var:murim_martial_rank_position > prev.var:murim_martial_rank_position
					}
				}
				save_scope_as = rank_challenge_target
			}
			# Execute the background duel
			if = {
				limit = {
					exists = scope:rank_challenge_target
				}
				save_scope_as = rank_challenger
				# Prowess duel resolution
				duel = {
					skill = prowess
					target = scope:rank_challenge_target
					50 = {
						# Challenger wins - swap positions
						compare_modifier = {
							value = scope:duel_value
							multiplier = 2
						}
						add_prestige = 150
						scope:rank_challenge_target = {
							add_prestige = -100
							save_scope_as = loser
							murim_roll_combat_injuries_effect = yes
							murim_roll_combat_trauma_effect = yes
						}
						murim_update_rankings_after_duel_effect = yes
						# Notify player if in same realm
						every_player = {
							limit = {
								is_alive = yes
								OR = {
									any_vassal = { this = scope:rank_challenger }
									any_vassal = { this = scope:rank_challenge_target }
									any_courtier = { this = scope:rank_challenger }
									any_courtier = { this = scope:rank_challenge_target }
								}
							}
							send_interface_message = {
								type = event_generic_neutral
								title = murim_rank_ai_duel_winner_notification
								left_icon = scope:rank_challenger
								right_icon = scope:rank_challenge_target
								desc = murim_rank_ai_duel_winner_desc
							}
						}
					}
					50 = {
						# Challenger loses
						compare_modifier = {
							value = scope:duel_value
							multiplier = -2
						}
						add_prestige = -100
						save_scope_as = loser
						murim_roll_combat_injuries_effect = yes
						murim_roll_combat_trauma_effect = yes
						scope:rank_challenge_target = {
							add_prestige = 100
						}
						# Notify player
						every_player = {
							limit = {
								is_alive = yes
								OR = {
									any_vassal = { this = scope:rank_challenger }
									any_vassal = { this = scope:rank_challenge_target }
									any_courtier = { this = scope:rank_challenger }
									any_courtier = { this = scope:rank_challenge_target }
								}
							}
							send_interface_message = {
								type = event_generic_neutral
								title = murim_rank_ai_duel_defender_notification
								left_icon = scope:rank_challenge_target
								right_icon = scope:rank_challenger
								desc = murim_rank_ai_duel_defender_desc
							}
						}
					}
				}
			}
		}

		# Second AI ranked challenge (so 2+ happen per year)
		random_living_character = {
			limit = {
				is_ai = yes
				is_alive = yes
				murim_is_top_50_martial_artist_trigger = yes
				murim_ai_should_challenge_for_rank_trigger = yes
				NOT = { has_character_modifier = murim_rank_challenge_cooldown }
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					has_trait = ambitious
				}
				modifier = {
					add = 5
					prowess >= 25
				}
			}
			random_living_character = {
				limit = {
					is_alive = yes
					murim_is_top_50_martial_artist_trigger = yes
					murim_can_be_challenged_for_rank_trigger = yes
					NOT = { this = prev }
				}
				save_scope_as = rank_challenge_target_2
			}
			if = {
				limit = { exists = scope:rank_challenge_target_2 }
				save_scope_as = rank_challenger_2
				duel = {
					skill = prowess
					target = scope:rank_challenge_target_2
					50 = {
						compare_modifier = {
							value = scope:duel_value
							multiplier = 2
						}
						add_prestige = 150
						scope:rank_challenge_target_2 = {
							save_scope_as = loser
							murim_roll_combat_injuries_effect = yes
						}
						murim_update_rankings_after_duel_effect = yes
					}
					50 = {
						compare_modifier = {
							value = scope:duel_value
							multiplier = -2
						}
						save_scope_as = loser
						murim_roll_combat_injuries_effect = yes
						scope:rank_challenge_target_2 = {
							add_prestige = 100
						}
					}
				}
			}
		}

		# AI challenges PLAYER if player is ranked and AI is close in score
		every_player = {
			limit = {
				is_alive = yes
				murim_is_top_50_martial_artist_trigger = yes
			}
			random_living_character = {
				limit = {
					is_ai = yes
					is_alive = yes
					murim_ai_should_challenge_for_rank_trigger = yes
					prowess >= 12
					NOT = { has_character_modifier = murim_rank_challenge_cooldown }
				}
				weight = {
					base = 1
					modifier = {
						add = 5
						has_trait = ambitious
					}
					modifier = {
						add = 3
						has_relation_rival = prev
					}
					modifier = {
						add = 10
						has_trait = battle_madness
					}
				}
				save_scope_as = rank_challenger
			}
			if = {
				limit = {
					exists = scope:rank_challenger
					# ~30% chance per year
					random_chance = { chance = 30 }
				}
				trigger_event = {
					id = murim_rank.0005
					days = { 30 90 }
				}
			}
		}

		# Hidden talent emergence (once per year check)
		random_living_character = {
			limit = {
				is_alive = yes
				murim_is_martial_artist_trigger = yes
				prowess >= 18
				NOT = { has_character_modifier = murim_ranked_veteran_modifier }
				NOT = { murim_is_top_50_martial_artist_trigger = yes }
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					age <= 25
				}
				modifier = {
					add = 3
					prowess >= 25
				}
			}
			# Check if they qualify for rankings now
			if = {
				limit = {
					has_variable = murim_martial_rank_score
					var:murim_martial_rank_score >= murim_rank_threshold_top_50
				}
				murim_enter_top_50_effect = yes
				# Notify player about hidden talent
				every_player = {
					limit = { is_alive = yes }
					trigger_event = {
						id = murim_rank.0007
						days = { 5 20 }
					}
				}
			}
		}
	}
}

######################################
# ON CHARACTER DEATH
# Remove dead fighters from rankings, promote next
######################################

on_death = {
	effect = {
		# Check if the dead character was ranked
		if = {
			limit = {
				exists = this
				murim_is_martial_artist_trigger = yes
				has_variable = murim_martial_rank_position
			}
			murim_remove_from_rankings_on_death_effect = yes
		}
	}
}

######################################
# QUARTERLY TOURNAMENT
# Every 3 months, chance for a ranked tournament event
######################################

on_quarterly_pulse = {
	effect = {
		# 15% chance per quarter for a tournament
		random = {
			chance = 15
			# Find a suitable host (sect leader or ruler with martial background)
			random_living_character = {
				limit = {
					is_alive = yes
					is_ruler = yes
					OR = {
						murim_is_top_50_martial_artist_trigger = yes
						any_vassal = { murim_is_martial_artist_trigger = yes }
					}
				}
				weight = {
					base = 1
					modifier = {
						add = 5
						prestige >= 2000
					}
					modifier = {
						add = 3
						has_trait = gregarious
					}
				}
				save_scope_as = tournament_host
			}
			# If we found a host, announce tournament
			if = {
				limit = { exists = scope:tournament_host }
				every_player = {
					limit = {
						is_alive = yes
						murim_is_martial_artist_trigger = yes
					}
					send_interface_message = {
						type = event_generic_good
						title = murim_rank_tournament_announced
						left_icon = scope:tournament_host
						desc = murim_rank_tournament_announced_desc
					}
				}
				# AI tournament in background (3 rounds of duels)
				random_living_character = {
					limit = {
						is_ai = yes
						murim_is_top_50_martial_artist_trigger = yes
						murim_can_challenge_for_rank_trigger = yes
					}
					weight = {
						base = 1
						modifier = { add = 3 prowess >= 20 }
					}
					random_living_character = {
						limit = {
							is_ai = yes
							murim_is_top_50_martial_artist_trigger = yes
							murim_can_challenge_for_rank_trigger = yes
							NOT = { this = prev }
						}
						save_scope_as = tournament_opponent
					}
					if = {
						limit = { exists = scope:tournament_opponent }
						duel = {
							skill = prowess
							target = scope:tournament_opponent
							50 = {
								compare_modifier = {
									value = scope:duel_value
									multiplier = 2
								}
								add_prestige = 200
								if = {
									limit = { NOT = { has_trait = tournament_champion } }
									add_trait = tournament_champion
								}
								scope:tournament_opponent = {
									save_scope_as = loser
									murim_roll_combat_injuries_effect = yes
								}
								murim_update_rankings_after_duel_effect = yes
							}
							50 = {
								compare_modifier = {
									value = scope:duel_value
									multiplier = -2
								}
								save_scope_as = loser
								murim_roll_combat_injuries_effect = yes
								scope:tournament_opponent = {
									add_prestige = 200
									if = {
										limit = { NOT = { has_trait = tournament_champion } }
										add_trait = tournament_champion
									}
								}
							}
						}
					}
				}
			}
		}
	}
}

######################################
# MONTHLY - AI RANDOM DUELS (NON-RANKED)
# Keeps the martial world active with constant combat
######################################

on_monthly_pulse = {
	effect = {
		# 20% chance per month for a random duel between martial artists
		random = {
			chance = 20
			random_living_character = {
				limit = {
					is_ai = yes
					is_alive = yes
					murim_is_martial_artist_trigger = yes
					murim_can_challenge_for_rank_trigger = yes
					OR = {
						has_trait = brave
						has_trait = ambitious
						has_trait = wrathful
						has_trait = battle_madness
						prowess >= 15
					}
				}
				weight = {
					base = 1
					modifier = { add = 5 has_trait = wrathful }
					modifier = { add = 10 has_trait = battle_madness }
					modifier = { add = 3 has_trait = brave }
				}
				random_living_character = {
					limit = {
						is_alive = yes
						murim_is_martial_artist_trigger = yes
						NOT = { this = prev }
						NOT = { has_trait = incapable }
						NOT = { has_trait = broken_spirit }
					}
					save_scope_as = monthly_duel_target
				}
				if = {
					limit = { exists = scope:monthly_duel_target }
					save_scope_as = monthly_duel_initiator
					duel = {
						skill = prowess
						target = scope:monthly_duel_target
						50 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = 2
							}
							add_prestige = 50
							scope:monthly_duel_target = {
								save_scope_as = loser
								murim_roll_combat_injuries_effect = yes
							}
							# Both get recent combat modifier
							add_character_modifier = {
								modifier = murim_recent_combat_modifier
								days = 30
							}
							scope:monthly_duel_target = {
								add_character_modifier = {
									modifier = murim_recent_combat_modifier
									days = 30
								}
							}
						}
						50 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = -2
							}
							save_scope_as = loser
							murim_roll_combat_injuries_effect = yes
							scope:monthly_duel_target = {
								add_prestige = 50
							}
							add_character_modifier = {
								modifier = murim_recent_combat_modifier
								days = 30
							}
							scope:monthly_duel_target = {
								add_character_modifier = {
									modifier = murim_recent_combat_modifier
									days = 30
								}
							}
						}
					}
				}
			}
		}
	}
}
