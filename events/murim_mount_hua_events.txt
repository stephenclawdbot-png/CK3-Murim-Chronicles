# Murim Chronicles - Mount Hua Sect Events
# The Plum Blossom Sword Sect: righteous path, scholarly martial arts, elegant swordsmanship
# Fighting style: elegant, principled, strong 1v1 swordsmanship. High prowess, honor-bound.
# REFUSES ambush/poison. Fights fair or not at all.
# Counter system: Mount Hua precision COUNTERS Heavenly Demon raw power / Berserker rage
# COUNTERED BY: Tang Clan poison (can't fight what you can't sword), Beggar Sect intelligence (outmaneuvered)
# Cross-refs: murim_clan_war (righteous alliance leader), murim_duel (honor duels), murim_ranking (sword prestige),
#             murim_wudang_events (friendly rivalry), murim_heavenly_demon_events (mortal enemies),
#             murim_tang_clan_events (uneasy distrust), murim_training (sect training), murim_artifact (legendary swords),
#             murim_cultivation (sword intent = cultivation path), murim_global_martial (foreign sword exchange)
# All combat uses murim_injury_chance values. Every event chains to 3+ other files.

namespace = murim_mount_hua

# ============================================================
# EVENT 0001: Plum Blossom Tournament — Annual Sect Competition
# Entry point into Mount Hua storyline. Chains to: ranking, duel, training
# ============================================================
murim_mount_hua.0001 = {
	type = character_event
	title = murim_mount_hua.0001.t
	desc = murim_mount_hua.0001.desc
	theme = martial
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	
	trigger = {
		is_alive = yes
		is_landed = yes
		OR = {
			has_trait = mount_hua_disciple
			has_trait = plum_blossom_swordsman
			has_trait = mount_hua_elder
		}
		NOT = { has_character_flag = plum_blossom_tournament_cooldown }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_trait = plum_blossom_swordsman
		}
		modifier = {
			add = 1.0
			has_trait = mount_hua_elder
		}
	}
	
	immediate = {
		set_character_flag = {
			flag = plum_blossom_tournament_cooldown
			years = 3
		}
	}
	
	# Option A: Enter the tournament as a competitor
	option = {
		name = murim_mount_hua.0001.a
		
		# Prowess duel check — uses injury chance calculator
		duel = {
			skill = prowess
			value = average
			
			# WIN: glory, prestige, possible rank advancement
			50 = {
				desc = murim_mount_hua.0001.a.success
				send_interface_toast = {
					title = murim_mount_hua.0001.a.success.toast
				}
				add_prestige = 300
				add_character_flag = {
					flag = plum_blossom_champion
					years = 3
				}
				# Chain to: RANKING system — tournament victory may change rank
				if = {
					limit = {
						exists = var:murim_rank
						var:murim_rank <= 50
					}
					change_variable = {
						name = murim_rank_prestige
						add = 10
					}
				}
				# Chain to: TRAINING — winning reveals technique insight
				random = {
					chance = 30
					trigger_event = {
						id = murim_training.0006  # Master's special teaching
						days = { 15 30 }
					}
				}
				# Injury roll for winner — talented swordsman walks away clean
				random = {
					chance = murim_injury_chance_honor_duel_winner
					random_list = {
						70 = { add_trait = wounded }
						20 = { add_character_modifier = { modifier = minor_sword_cut days = 60 } }
						10 = { add_trait = one_eyed }
					}
				}
				# Fight memory — ALWAYS created
				create_character_memory = {
					type = plum_blossom_victory_memory
				}
			}
			
			# LOSE: humiliation, but learn from defeat
			50 = {
				desc = murim_mount_hua.0001.a.failure
				add_prestige = -100
				stress_impact = {
					base = 15
					ambitious = 10
					arrogant = 15
				}
				# Chain to: DUEL — defeated fighter may seek revenge match
				add_character_flag = {
					flag = tournament_defeat_memory
					years = 5
				}
				# Injury roll for loser — higher chance
				random = {
					chance = murim_injury_chance_honor_duel_loser
					random_list = {
						50 = { add_trait = wounded }
						25 = { add_character_modifier = { modifier = sword_slash_scar days = 365 } }
						15 = { add_trait = one_eyed }
						10 = { add_trait = disfigured }
					}
				}
				# Chain to: TRAINING motivation — defeat drives improvement
				random = {
					chance = 40
					trigger_event = {
						id = murim_training.0001  # Training montage triggered by loss
						days = { 30 60 }
					}
				}
				create_character_memory = {
					type = plum_blossom_defeat_memory
				}
			}
		}
		
		ai_chance = {
			base = 60
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = -20
				has_trait = craven
			}
		}
	}
	
	# Option B: Judge the tournament (elder path)
	option = {
		name = murim_mount_hua.0001.b
		
		trigger = {
			OR = {
				has_trait = mount_hua_elder
				age >= 50
			}
		}
		
		add_prestige = 150
		add_piety = 50
		# Chain to: SECT events — judging builds sect influence
		change_variable = {
			name = sect_influence
			add = 5
		}
		# Chain to: MENTOR — may spot a talented disciple
		random = {
			chance = 25
			trigger_event = {
				id = murim_mentor.0001  # Find talented student
				days = { 7 14 }
			}
		}
		
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				age >= 50
			}
		}
	}
	
	# Option C: Invite rival sects to watch — diplomacy play
	option = {
		name = murim_mount_hua.0001.c
		
		diplomacy >= 10
		
		add_prestige = 200
		# Chain to: CLAN WAR — showing strength deters enemies
		add_character_modifier = {
			modifier = sect_display_of_strength
			years = 2
		}
		# Chain to: GLOBAL — foreign masters may attend
		random = {
			chance = 10
			trigger_event = {
				id = murim_global.0001  # Foreign master arrives
				days = { 30 90 }
			}
		}
		# Chain to: POLITICS — rival sect may take offense
		random = {
			chance = 20
			trigger_event = {
				id = murim_politics.0002  # Sect tension rises
				days = { 60 120 }
			}
		}
		
		ai_chance = {
			base = 20
			modifier = {
				add = 20
				diplomacy >= 15
			}
		}
	}
	
	# Option D: Skip — focus on cultivation instead
	option = {
		name = murim_mount_hua.0001.d
		
		stress_impact = {
			base = -10
		}
		# Chain to: CULTIVATION — meditation instead of fighting
		trigger_event = {
			id = murim_cultivation.0001  # Quiet cultivation session
			days = { 3 7 }
		}
		
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = content
			}
		}
	}
}
