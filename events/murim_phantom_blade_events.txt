# BLADE SPIRIT COMMUNION
# Phantom Blade disciple communes with their blade's spirit
########################################
murim_phantom.0001 = {
	type = character_event
	title = murim_phantom.0001.t
	desc = murim_phantom.0001.desc
	theme = mystic
	left_portrait = {
		character = root
		animation = personality_rational
	}

	trigger = {
		has_trait = phantom_blade_disciple
		murim_is_martial_artist_trigger = yes
		NOT = { has_character_flag = blade_spirit_communion_done }
		age >= 16
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = cultivation_stage_3
		}
		modifier = {
			add = 3
			prowess >= 18
		}
	}

	immediate = {
		add_character_flag = {
			flag = blade_spirit_communion_done
		}
	}

	# Fully bond with the blade spirit
	option = {
		name = murim_phantom.0001.a
		add_prowess_skill = 3
		add_character_modifier = {
			modifier = murim_blade_spirit_bonded
			years = 10
		}
		add_prestige = 200
		stress_impact = {
			craven = medium_stress_impact_gain
		}
		hidden_effect = {
			add_to_variable_list = {
				name = murim_combat_memories
				target = flag:blade_spirit_full_bond
			}
		}
		ai_chance = {
			base = 60
		}
	}

	# Partial bond - maintain independence
	option = {
		name = murim_phantom.0001.b
		add_prowess_skill = 1
		add_martial_skill = 2
		add_character_modifier = {
			modifier = murim_blade_spirit_partial
			years = 5
		}
		hidden_effect = {
			add_to_variable_list = {
				name = murim_combat_memories
				target = flag:blade_spirit_partial_bond

		}

			base = 30

	# Reject the spirit - forge your own path
	option = {
		name = murim_phantom.0001.c

murim_phantom.0002 = {
	type = character_event
	title = murim_phantom.0002.t
	desc = murim_phantom.0002.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = personality_cynical
	}

	trigger = {
		has_trait = phantom_blade_disciple
		has_variable = murim_clan_rank
		var:murim_clan_rank >= 2
		intrigue >= 12
		NOT = { has_character_flag = assassination_contract_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			intrigue >= 18
		}
		modifier = {
			add = 2
			prowess >= 16
		}
	}

	immediate = {
		add_character_flag = {
			flag = assassination_contract_cooldown
			days = 1095
		}
	}

	# Accept the contract - target is an Orthodox elder
	option = {
		name = murim_phantom.0002.a
		random_list = {
			40 = {
				desc = murim_phantom.0002.a.success
				add_gold = 200
				add_prestige = 300
				add_dread = 25
				add_character_modifier = {
					modifier = murim_phantom_assassin_fame
					years = 5
				}
			}
			35 = {
				desc = murim_phantom.0002.a.partial
				add_gold = 100
				add_trait = wounded_1
				add_prestige = 100
			}
			25 = {
				desc = murim_phantom.0002.a.fail
				add_trait = wounded_2
				add_prestige = -200
				add_character_modifier = {
					modifier = murim_failed_assassin_hunted
					years = 3
				}
				# Cross-ref: Orthodox clans now hunting you
				trigger_event = {
					id = murim_faction_war.0004
					days = { 10 30 }
				}
			}
		}
		hidden_effect = {
			add_to_variable_list = {
				name = murim_combat_memories
				target = flag:assassination_contract_accepted
			}
		}
		stress_impact = {
			compassionate = major_stress_impact_gain
			just = major_stress_impact_gain
		}
		ai_chance = {
			base = 50
		}
	}

	# Refuse the contract - maintain honor
	option = {
		name = murim_phantom.0002.b
		add_piety = 50
		add_prestige = -50
		stress_impact = {
			greedy = medium_stress_impact_gain
		}
		hidden_effect = {
			add_to_variable_list = {
				name = murim_combat_memories
				target = flag:assassination_contract_refused
			}
		}
		ai_chance = {
			base = 30
		}
	}

	# Warn the target instead - betray the contract
	option = {
		name = murim_phantom.0002.c
		trigger = {
			OR = {
				has_trait = compassionate
				has_trait = just
				has_trait = honest
			}
		}
		add_piety = 100
		add_prestige = -100
		add_character_modifier = {
			modifier = murim_contract_betrayer

		}
