murim_duel_power_comparison_value = {
	# Compares ROOT's combat power against scope:target
	value = murim_effective_combat_power_value
	subtract = scope:target.murim_effective_combat_power_value
}

murim_duel_win_chance_value = {
	# Percentage chance ROOT wins against scope:target in a duel
	value = 50
	# Power differential shifts odds
	if = {

		limit = { murim_duel_power_comparison_value >= 50 }
