murim_beggar_sect_expansion2.0001 = {
	type = character_event
	title = murim_beggar_sect_expansion2.0001.t
	desc = murim_beggar_sect_expansion2.0001.desc
	theme = martial_arts
	left_portrait = { character = root animation = personality_bold }
	right_portrait = { character = scope:beggar_chief animation = personality_rational }
	
	trigger = {
		has_trait = beggar_sect_member
		NOT = { has_character_flag = dog_beating_staff_initiated }
		age >= 16
		prowess >= 10
	}
	weight_multiplier = { base = 1 modifier = { add = 2 has_trait = gregarious } }
	
	immediate = {
		set_character_flag = dog_beating_staff_initiated
		random_courtier = {
			limit = { has_trait = beggar_sect_elder age >= 40 }
			save_scope_as = beggar_chief
		}
	}
	
	# Option A: Learn the 36 Dog Beating Staff moves
	option = {
		name = murim_beggar_sect_expansion2.0001.a
		add_character_modifier = { modifier = dog_beating_staff_student years = 15 }
		change_prowess = 2
		add_prestige = 150
		set_character_flag = staff_combat_path
		ai_chance = { base = 45 }
	}
	
	# Option B: Focus on Eighteen Dragon Subduing Palms
	option = {
		name = murim_beggar_sect_expansion2.0001.b
		trigger = { prowess >= 14 }
		add_character_modifier = { modifier = dragon_palm_student years = 15 }
		change_prowess = 3
		add_prestige = 200
		set_character_flag = dragon_palm_path
		ai_chance = { base = 35 modifier = { add = 15 prowess >= 18 } }
	}
	
	# Option C: Specialize in intelligence gathering
	option = {
		name = murim_beggar_sect_expansion2.0001.c
		add_character_modifier = { modifier = intelligence_specialist years = 15 }
		add_intrigue = 2
		add_diplomacy = 1
		set_character_flag = beggar_spy_path
		ai_chance = { base = 20 modifier = { add = 15 intrigue >= 12 } }
	}
}
