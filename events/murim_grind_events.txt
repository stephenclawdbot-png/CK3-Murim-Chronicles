# Solo Meditation (repeatable)
murim_grind.0001 = {
	type = character_event
	title = murim_grind.0001.t
	desc = murim_grind.0001.desc
	theme = mental_health
	left_portrait = {
		character = root
		animation = thinking
	}

	trigger = {
		is_alive = yes
		age >= 16
		prowess >= 5
		has_character_flag = murim_initiated
		NOT = { has_character_flag = murim_closed_door_active }
	}

	# Light meditation - safe, low XP
	option = {
		name = murim_grind.0001.a
		change_variable = { name = murim_cultivation_xp add = 1 }
		add_character_modifier = { modifier = murim_meditation_focus days = 30 }
		stress_impact = { base = -5 }
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = patient }
			modifier = { add = 20 has_trait = content }
			modifier = { add = -10 has_trait = ambitious }
		}
	}

	# Deep meditation - moderate risk, good XP
	option = {
		name = murim_grind.0001.b
		random = {
			chance = murim_qi_deviation_chance
			murim_apply_minor_qi_deviation = yes
		}
		change_variable = { name = murim_cultivation_xp add = 3 }
		stress_impact = { base = 5 }
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = diligent }
			modifier = { add = 20 has_trait = ambitious }
			modifier = { add = -15 has_trait = craven }
		}
	}

	# Dangerous push - high risk, high reward
	option = {
		name = murim_grind.0001.c
		random = {
			chance = 45
			murim_apply_minor_qi_deviation = yes
		}
		change_variable = { name = murim_cultivation_xp add = 6 }
		add_stress = 15
		ai_chance = {
			base = 10
			modifier = { add = 40 has_trait = ambitious }
			modifier = { add = 20 has_trait = brave }
			modifier = { add = -30 has_trait = craven }
			modifier = { add = -20 has_trait = patient }
		}
	}
}

# Forbidden Manual Study (repeatable, dangerous)
murim_grind.0030 = {
	type = character_event

	desc = murim_grind.0030.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = scheme
	}

	trigger = {
		murim_can_study_forbidden_trigger = yes
	}

	# Read cautiously
	option = {
		name = murim_grind.0030.a
		change_variable = { name = murim_cultivation_xp add = 2 }
		add_character_modifier = { modifier = murim_studying_manual days = 30 }
		ai_chance = {
			base = 20
			modifier = { add = 10 has_trait = patient }
			modifier = { add = 10 has_trait = diligent }
		}
	}

	# Study deeply
	option = {
		name = murim_grind.0030.b
		murim_apply_forbidden_study_effect = yes
		ai_chance = {
			base = 15
			modifier = { add = 20 has_trait = ambitious }
			modifier = { add = 10 has_trait = deceitful }
		}
	}

	# Absorb fully - guaranteed corruption advance
	option = {
		name = murim_grind.0030.c
		change_variable = { name = murim_cultivation_xp add = 10 }
		# Forced corruption advance
		if = {
			limit = { NOT = { murim_has_forbidden_mark_trigger = yes } }
			add_trait = murim_forbidden_mark_1
		}
		else_if = {
			limit = { has_trait = murim_forbidden_mark_1 }
			remove_trait = murim_forbidden_mark_1
			add_trait = murim_forbidden_mark_2
		}
		else_if = {
			limit = { has_trait = murim_forbidden_mark_2 }
			remove_trait = murim_forbidden_mark_2
			add_trait = murim_forbidden_mark_3
		}
		random = {
			chance = 15
			add_trait = lunatic
		}
		add_character_flag = { flag = murim_forbidden_study_cooldown days = 90 }
		ai_chance = {
			base = 5
			modifier = { add = 30 has_trait = ambitious }
			modifier = { add = 20 has_trait = sadistic }
			modifier = { add = -30 has_trait = compassionate }
			modifier = { add = -20 has_trait = zealous }
		}
	}
}

# Breakthrough Attempt
murim_grind.0040 = {
	type = character_event
	title = murim_grind.0040.t
	desc = murim_grind.0040.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		OR = {
			murim_can_attempt_breakthrough_trigger = yes
			murim_can_attempt_tribulation_trigger = yes
		}
	}

	# Steady breakthrough
	option = {
		name = murim_grind.0040.a
		murim_attempt_breakthrough_effect = yes
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = patient }
			modifier = { add = 10 has_trait = diligent }
		}
	}

	# Force breakthrough - higher chance but higher deviation risk
	option = {
		name = murim_grind.0040.b
		add_character_flag = { flag = murim_studying_forbidden days = 1 }
		murim_attempt_breakthrough_effect = yes
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = ambitious }
			modifier = { add = 10 has_trait = brave }
			modifier = { add = -15 has_trait = patient }
		}
	}

	# Heavenly Tribulation - extreme, only if rank >= 4
	option = {
		name = murim_grind.0040.c
		trigger = {
			murim_can_attempt_tribulation_trigger = yes
			exists = var:murim_cultivation_rank
			var:murim_cultivation_rank >= 4
		}
		add_character_flag = { flag = murim_attempting_breakthrough days = 1 }
		random = {
			chance = murim_tribulation_survival_chance
			# SURVIVED
			change_variable = { name = murim_cultivation_rank add = 1 }
			set_variable = { name = murim_cultivation_xp value = 0 }
			add_trait = murim_tribulation_survivor
			add_prestige = 500
			send_interface_toast = {
				title = murim_tribulation_survived_toast
				left_icon = root
			}
		}
		modifier = {
			factor = 0
		}
		fallback = {
			# FAILED TRIBULATION
			random = {
				chance = 40
				death = { death_reason = death_accident }
			}
			if = {
				limit = { is_alive = yes }
				add_trait = wounded_3
				murim_apply_minor_qi_deviation = yes
			}
		}
		add_character_modifier = { modifier = murim_tribulation_aftermath days = 365 }
		ai_chance = {
			base = 10
			modifier = { add = 40 has_trait = ambitious }
			modifier = { add = 20 has_trait = brave }
			modifier = { add = -50 has_trait = craven }
			modifier = { add = -30 has_trait = content }
		}
	}

	# Abandon
	option = {
		name = murim_grind.0040.d
		add_stress = 15
		ai_chance = {
			base = 5
			modifier = { add = 20 has_trait = craven }
		}
	}
}
