# === EVENT 0002: FIRST ROUND - BRACKET FIGHTS ===
murim_tourney_exp2.0002 = {
	type = character_event
	title = murim_tourney_exp2.0002.t
	desc = murim_tourney_exp2.0002.desc
	theme = martial_arts
	
	trigger = {
		OR = {
			has_character_modifier = murim_tournament_champion_entry
			has_character_modifier = murim_tournament_masked_fighter
		}
	}
	
	immediate = {
		random_ruler = {
			limit = {
				murim_is_martial_artist_trigger = yes
				NOT = { this = root }
				OR = {
					has_character_modifier = murim_tournament_champion_entry
					has_character_modifier = murim_tournament_masked_fighter
				}
			}
			save_scope_as = tournament_opponent
		}
	}
	
	# Option A: Fight with full power from the start
	option = {
		name = murim_tourney_exp2.0002.a
		duel = {
			skill = prowess
			target = scope:tournament_opponent
		}
		random_list = {
			60 = {
				add_prestige = 500
				trigger_event = { id = murim_tourney_exp2.0003 days = 7 }
			}
			40 = {
				add_prestige = 100
				add_character_modifier = { modifier = murim_moderate_injury years = 1 }
				remove_character_modifier = murim_tournament_champion_entry
			}
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Hold back - conserve energy for later rounds
	option = {
		name = murim_tourney_exp2.0002.b
		random_list = {
			50 = {
				add_prestige = 200
				trigger_event = { id = murim_tourney_exp2.0003 days = 7 }
			}
			50 = {
				# Underestimated opponent
				add_prestige = -100
				add_character_modifier = { modifier = murim_moderate_injury years = 1 }
			}
		}
		ai_chance = { base = 30 }
	}
	
	# Option C: Use a secret technique to end it instantly
	option = {
		name = murim_tourney_exp2.0002.c
		trigger = { has_trait = murim_rank_peak }
		add_prestige = 800
		add_dread = 20
		trigger_event = { id = murim_tourney_exp2.0003 days = 7 }
		# Everyone now fears you
		every_ruler = {
			limit = { murim_is_martial_artist_trigger = yes }
			add_opinion = { target = root modifier = murim_tournament_feared opinion = -10 }
		}
		ai_chance = { base = 20 }
	}
}

		create_character = {

		name = murim_tourney_exp2.0003.a

			skill = prowess

			40 = {

# === EVENT 0010: LEGENDARY TOURNAMENT - ONCE IN A CENTURY ===
murim_tourney_exp2.0010 = {
	type = character_event
	title = murim_tourney_exp2.0010.t
	desc = murim_tourney_exp2.0010.desc
	theme = martial_arts
	
	trigger = {
		murim_is_martial_artist_trigger = yes
		OR = {
			has_trait = murim_rank_absolute
			has_trait = murim_rank_transcendent
		}
	}
	
	# Option A: Enter the Legendary Tournament - only absolute masters
	option = {
		name = murim_tourney_exp2.0010.a
		random_list = {
			30 = {
				# Win - become Murim Lord
				add_trait = murim_lord_of_murim
				add_prestige = 10000
				add_prowess = 10
				add_character_modifier = {
					modifier = murim_lord_aura
					years = -1
				}
			}
			40 = {
				# Honorable showing
				add_prestige = 3000
				add_prowess = 4
			}
			30 = {
				# Defeated - near death (youth get lighter injuries)
				if = {
					limit = { age < 16 }
					add_character_modifier = { modifier = murim_severe_injury years = 2 }

				}
				if = {
					limit = { age >= 16 }
					add_character_modifier = { modifier = murim_near_fatal_injury years = 3 }
				}
