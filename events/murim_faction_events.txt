# Murim Chronicles - Orthodox vs Unorthodox Faction Events
# THE GREAT DIVIDE: Righteous Path vs Demonic Path
# ORTHODOX (6 clans): Mount Hua, Shaolin, Wudang, Tang, Beggar Sect, Iron Fist Gate
#   Led by elected MURIM ALLIANCE LEADER. Values honor, code of conduct.
# UNORTHODOX (3 clans): Heavenly Demon, Poison Flower Palace, Phantom Blade Sect
#   Led by HEAVENLY DEMON SECT LEADER automatically. Values power, no rules.
# INTEGRATION: Uses murim_clan_triggers.txt (all 9 clan + faction triggers)
# INTEGRATION: Uses murim_opinion_modifiers.txt (martial-layer opinion only)
# INTEGRATION: Feeds murim_shadow_politics_events.txt (HD infiltration)
# VANILLA SEPARATION: Faction wars are MARTIAL conflicts, NOT political wars.
#   No titles change. No realm boundaries shift. No armies march (unless
#   Heavenly Demon has infiltrated a government — see shadow_politics).
# CHINA-CENTRIC: All major faction events occur in China. Regional echoes abroad.
# GENDER: Fully neutral. Women lead factions, fight in wars, defect, romance.

namespace = murim_faction

# ═══════════════════════════════════════════
# 0001 - FACTION WAR ERUPTS (every 20-30 years)
# Orthodox Alliance vs Unorthodox Coalition
# ═══════════════════════════════════════════
murim_faction.0001 = {
	type = character_event
	title = murim_faction.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { murim_is_in_orthodox_sect_trigger = yes }
				desc = murim_faction.0001.desc_orthodox
				# "The Demonic Path has gone too far. Raids on orthodox schools,
				#  poisonings of our elders, assassinations of promising disciples.
				#  The Murim Alliance Leader has called a GREAT CONVERGENCE —
				#  all righteous sects must unite against the darkness.
				#  The war will be fought in the shadows, in dueling grounds,
				#  and in the hidden valleys of China. Not with armies — with fists."
			}
			triggered_desc = {
				trigger = { murim_is_in_unorthodox_sect_trigger = yes }
				desc = murim_faction.0001.desc_unorthodox
				# "The Heavenly Demon has spoken: the righteous hypocrites have
				#  suppressed us long enough. They call us 'demonic' for refusing
				#  their hollow code. Now we strike back. The Poison Flower Palace
				#  will cripple their supply lines. The Phantom Blade will remove
				#  their leaders. And the Heavenly Demon Sect will crush what remains.
				#  This is not war — this is liberation."
			}
			desc = murim_faction.0001.desc_neutral
			# "The martial world trembles. Orthodox and Unorthodox have declared
			#  open conflict. Every martial artist must choose a side — or hide."
		}
	}
	theme = martial
	
	trigger = {
		murim_is_clan_member_of_any_trigger = yes
		NOT = { has_character_flag = murim_faction_war_active }
		# Only fires every 20-30 years
		NOT = { has_global_variable = murim_faction_war_cooldown }
	}
	
	weight_multiplier = {
		base = 1
		# More likely when tensions are high
		modifier = {
			factor = 2
			has_global_variable = murim_faction_tension_high
		}
	}
	
	# Option A: Rally your clan to the cause
	option = {
		name = murim_faction.0001.a
		# "Sound the war drums. [Clan name] stands ready."
		trigger = { murim_is_in_orthodox_sect_trigger = yes }
		add_character_modifier = {
			modifier = murim_faction_war_veteran
			years = 20
		}
		set_character_flag = {
			flag = murim_faction_war_active
			years = 5
		}
		set_global_variable = {
			name = murim_faction_war_cooldown
			value = yes
			years = 25
		}
		# Series of war events over 1-5 years
		trigger_event = {
			id = murim_faction.0010
			days = { 30 180 }
		}
		stress_impact = {
			brave = minor_stress_loss
			craven = major_stress_gain
		}
		ai_chance = {
			base = 70
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}
	
	# Option B: Advocate for peace / negotiation
	option = {
		name = murim_faction.0001.b
		# "Enough blood. I will seek a truce before more die."
		add_character_modifier = {
			modifier = murim_peacemaker
			years = 10
		}
		trigger_event = {
			id = murim_faction.0005
			days = { 14 60 }
		}
		stress_impact = {
			compassionate = minor_stress_loss
			wrathful = major_stress_gain
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = compassionate
			}
		}
	}
	
	# Option C: Stay neutral / protect only your own
	option = {
		name = murim_faction.0001.c
		# "This is not our fight. Guard our gates and wait for the storm to pass."
		trigger = {
			OR = {
				murim_is_in_neutral_sect_trigger = yes
				murim_is_sect_leader_trigger = yes
			}
		}
		add_character_modifier = {
			modifier = murim_faction_neutral_stance
			years = 5
		}
		# Both sides lose some respect for you
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = patient
			}
		}
	}
}

	type = character_event

	#  dueling ground in central China. The Orthodox champion: [NAME], chosen
	#  by the Alliance Leader. The Unorthodox champion: [NAME], chosen by
	#  the Heavenly Demon. The entire martial world watches."
	theme = martial
	
	# Option A: Volunteer as your faction's champion
	option = {
		name = murim_faction.0006.a
		# "I will fight for the righteous/demonic path."
		trigger = {
			prowess >= 20
			OR = {
				murim_is_sect_leader_trigger = yes

			}
		}
		# Duel mechanics — high stakes
		duel = {
			skill = prowess
			target = scope:faction_champion_opponent
			50 = {
				# Victory — your faction wins the war

					value = scope:duel_value

					modifier = murim_faction_champion_victor

					type = murim_decided_faction_war
				}
			}
			50 = {
				# Defeat — your faction loses, you're wounded
				compare_modifier = {
					value = scope:duel_value

		}

			type = murim_witnessed_faction_duel

	type = character_event
	title = murim_faction.0020.t

		first_valid = {
