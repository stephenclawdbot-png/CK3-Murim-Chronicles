# Murim Chronicles - MASTER SHARED TRIGGERS FILE
# INTEGRATION RULE: ALL event files MUST reference these triggers.
# Do NOT duplicate trigger logic inline in event files.
# Every system uses these same base triggers for consistency.
#
# TRIGGER INDEX:
#   1. murim_is_martial_artist_trigger - has ANY martial trait (core + global + clan)
#   2. murim_is_cultivator_trigger - has cultivation tier trait
#   3. murim_is_ranked_fighter_trigger - has ranking trait or top 50 flag
#   4. murim_is_clan_member_trigger - member of any named clan/sect
#   5. murim_is_eligible_for_tournament_trigger - age >= 16, is martial artist
#   6. murim_is_youth_eligible_trigger - age 8-15, has martial trait
#   7. murim_is_wandering_master_trigger - landless + martial artist
#   8. murim_is_foreign_martial_artist_trigger - non-Korean/Chinese culture + martial
#   9. murim_is_righteous_path_trigger - righteous faction alignment
#  10. murim_is_demonic_path_trigger - demonic faction alignment
#  11. murim_is_neutral_path_trigger - neutral faction (Tang, independents)
#  12. murim_has_fight_memory_with_trigger - has fought target before
#  13. murim_can_duel_trigger - alive, not imprisoned, not incapable
#  14. murim_is_tournament_champion_trigger - holds any tournament champion trait
#  15. murim_is_supreme_champion_trigger - holds supreme martial champion trait

# ============================================================================
# TRIGGER 2: IS CULTIVATOR
# Returns yes if character has any cultivation tier trait

	OR = {
		has_trait = qi_sensing
		has_trait = qi_condensation
		has_trait = foundation_establishment
		has_trait = core_formation
		has_trait = nascent_soul
		has_trait = tribulation_transcendence
	}
}

# ============================================================================
# TRIGGER 3: IS RANKED FIGHTER
# Returns yes if character holds a ranking trait or has top 50 flag
# Used by: ranking events, tournament seeding, ranker death, challenges
# ============================================================================
murim_is_ranked_fighter_trigger = {
	OR = {
		has_trait = ranked_martial_artist
		has_trait = top_10_martial_artist
		has_trait = martial_legend
		has_character_flag = murim_top_50_ranked
		has_character_flag = murim_top_10_ranked
		has_character_flag = murim_rank_1
	}
}

# ============================================================================
# TRIGGER 4: IS CLAN MEMBER
# Returns yes if character belongs to any named murim clan/sect
# Used by: clan events, clan war, sect politics, recruitment
# ============================================================================
murim_is_clan_member_trigger = {
	OR = {
		# Righteous sects
		has_trait = mount_hua_disciple
		has_trait = wudang_disciple
		has_trait = shaolin_monk
		has_trait = beggar_sect_member
		has_trait = emei_swordswoman
		has_trait = kunlun_hermit
		has_trait = kongtong_fist_master
		# Neutral clans
		has_trait = tang_clan_member
		# Demonic sects
		has_trait = heavenly_demon_follower
		# Flags for membership without trait
		has_character_flag = mount_hua_sect_member
		has_character_flag = tang_clan_affiliated
		has_character_flag = heavenly_demon_sect_member
		has_character_flag = wudang_sect_member
		has_character_flag = shaolin_temple_member
		has_character_flag = beggar_sect_affiliated
	}
}

# ============================================================================
# TRIGGER 5: IS ELIGIBLE FOR TOURNAMENT (Grand Tournament, age 16+)
# HARD AGE GATE: Must be 16 or older AND a martial artist
# Used by: tournament qualifier, grand tournament, champions battle
# ============================================================================
murim_is_eligible_for_tournament_trigger = {
	is_alive = yes
	age >= 16
	is_imprisoned = no
	NOT = { has_trait = incapable }
	murim_is_martial_artist_trigger = yes
	NOT = { has_character_flag = tournament_banned }
	NOT = { has_character_flag = tournament_disqualified }
}

# ============================================================================
# TRIGGER 6: IS YOUTH ELIGIBLE (Young Dragons Tournament, age 8-15)
# Age gate: 8-15 ONLY. Must have at least basic martial training.
# Used by: youth tournament events exclusively
# ============================================================================
murim_is_youth_eligible_trigger = {
	is_alive = yes
	age >= 8
	age < 16
	is_imprisoned = no
	NOT = { has_trait = incapable }
	OR = {
		murim_is_martial_artist_trigger = yes
		prowess >= 5
	}
	NOT = { has_character_flag = banned_from_youth_tournament }
}

# ============================================================================
# TRIGGER 7: IS WANDERING MASTER
# Landless character who is a martial artist — uses CK3 adventurer system
# Used by: wandering master events, travel encounters, court arrival events
# ============================================================================
murim_is_wandering_master_trigger = {
	is_alive = yes
	is_landed = no
	murim_is_martial_artist_trigger = yes
	OR = {
		prowess >= 12
		has_trait = wandering_martial_master
		has_trait = disguised_ranker
		has_trait = jianghu_wanderer
	}
}

# ============================================================================
# TRIGGER 11: IS NEUTRAL PATH
# Not strongly aligned with either side — Tang Clan, independents, wanderers
# Used by: clan politics, neutral mediator events, trade events
# ============================================================================
murim_is_neutral_path_trigger = {
	NOR = {
		murim_is_righteous_path_trigger = yes
		murim_is_demonic_path_trigger = yes
	}
	OR = {
		has_trait = tang_clan_member
		has_trait = jianghu_wanderer
		has_trait = wandering_martial_master
		murim_is_foreign_martial_artist_trigger = yes
		# Fallback: martial artist with no faction
		AND = {
			murim_is_martial_artist_trigger = yes
			NOR = {
				has_character_flag = righteous_path_aligned
				has_character_flag = demonic_path_aligned
			}
		}
	}
}

# ============================================================================
# TRIGGER 12: HAS FIGHT MEMORY WITH TARGET
# Checks if root has previously fought scope:target
# Used by: duel events (rematch), revenge arcs, rivalry system
# ============================================================================
murim_has_fight_memory_with_trigger = {
	OR = {
		has_character_flag = fought_$target.id$
		any_memory = {
			has_memory_type = duel_victory_memory
			has_memory_participant = {
				character = scope:target
			}
		}
		any_memory = {
			has_memory_type = duel_defeat_memory
			has_memory_participant = {
				character = scope:target
			}
		}
		any_memory = {
			has_memory_type = youth_tournament_win_memory
			has_memory_participant = {
				character = scope:target
			}
		}
	}
}

# ============================================================================
# TRIGGER 13: CAN DUEL
# Basic eligibility for any martial duel
# Used by: duel events, tournament, honor challenge, assassination defense
# ============================================================================
murim_can_duel_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	NOT = { has_character_flag = duel_cooldown }
	age >= 8
}

# ============================================================================
# TRIGGER 16: IS ELIGIBLE FOR CHAMPIONS BATTLE (10-year event)
# Must be a past Grand Tournament champion (or valid replacement)
# Used by: champions battle events exclusively
# ============================================================================
murim_is_eligible_for_champions_battle_trigger = {
	is_alive = yes
	age >= 16
	is_imprisoned = no
	NOT = { has_trait = incapable }
	OR = {
		has_trait = grand_tournament_champion
		has_trait = supreme_martial_champion
		has_character_flag = grand_tournament_winner
		# Valid replacement: highest ranked non-champion
		AND = {
			murim_is_ranked_fighter_trigger = yes
			has_character_flag = champions_battle_replacement
		}
	}
}

# ============================================================================
# TRIGGER 17: HAS COUNTER STYLE ADVANTAGE
# Checks if root's martial style counters scope:opponent's style
# Used by: duel events (combat advantage/disadvantage modifiers)
# Counter map:
#   Tang Poison -> countered by Wudang/Shaolin
#   Heavenly Demon -> countered by Mount Hua/Beggar
#   Berserker -> countered by Tai Chi/Wudang
#   Shaolin -> countered by Kalari/Tang
#   Speed (Iaijutsu/Engolo) -> countered by Defensive (Wudang/Shaolin)
#   Mounted (Mongol) -> countered by terrain holders
# ============================================================================
murim_has_counter_style_advantage_trigger = {
	OR = {
		# Wudang/Shaolin counter Tang Poison
		AND = {
			OR = { has_trait = wudang_disciple has_trait = shaolin_monk has_trait = iron_body_practitioner }
			scope:opponent = { OR = { has_trait = tang_clan_member has_trait = poison_hand_master } }
		}
		# Mount Hua/Beggar counter Heavenly Demon
		AND = {
			OR = { has_trait = mount_hua_disciple has_trait = beggar_sect_member }
			scope:opponent = { OR = { has_trait = heavenly_demon_follower has_trait = demonic_art_practitioner } }
		}
		# Tai Chi/Wudang counter Berserker
		AND = {
			OR = { has_trait = wudang_disciple has_trait = tai_chi_practitioner }
			scope:opponent = { has_trait = berserker_champion }
		}
		# Kalari/Tang counter Shaolin tank
		AND = {
			OR = { has_trait = kalari_warrior has_trait = tang_clan_member has_trait = poison_hand_master }
			scope:opponent = { OR = { has_trait = shaolin_monk has_trait = iron_body_practitioner has_trait = iron_shirt_master } }
		}
		# Defensive counters Speed
		AND = {
			OR = { has_trait = wudang_disciple has_trait = shaolin_monk has_trait = tai_chi_practitioner }
			scope:opponent = { OR = { has_trait = iaijutsu_expert has_trait = engolo_dancer has_trait = lightfoot_master } }
		}
		# Mount Hua precision counters Berserker rage
		AND = {
			has_trait = mount_hua_disciple
			scope:opponent = { has_trait = berserker_champion }
		}
	}
}
