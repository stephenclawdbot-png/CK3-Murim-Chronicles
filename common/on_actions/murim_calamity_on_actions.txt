# === YEARLY PULSE: CHECK CALAMITY CONDITIONS ===
on_yearly_pulse = {
	effect = {
		# Check if 150-200 years have passed since last calamity or game start
		if = {
			limit = {
				is_ruler = yes
				has_global_variable = murim_world_active
				NOT = { has_global_variable = murim_calamity_active }
				OR = {
					NOT = { has_global_variable = murim_last_calamity_year }
					current_date >= murim_calamity_next_trigger_date_value
				}
				current_date >= 1216.1.1  # Minimum 150 years from typical start
			}
			# Roll for calamity trigger - escalating chance
			random = {
				chance = murim_calamity_yearly_chance_value
				set_global_variable = { name = murim_calamity_active value = yes }
				set_global_variable = { name = murim_calamity_phase value = 1 }
				set_global_variable = { name = murim_calamity_year value = current_year }
				# Phase 1: Omens and warnings
				trigger_event = { id = murim_great_calamity.0001 days = { 1 30 } }
				every_ruler = {
					limit = { murim_is_martial_artist_trigger = yes }
					trigger_event = { id = murim_great_calamity.0002 days = { 5 60 } }
				}
			}
		}
		# Advance calamity phases automatically
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 1 }
				# Phase 1 lasts 1-2 years
				years_from_global_variable = { name = murim_calamity_year value >= 1 }
			}
			set_global_variable = { name = murim_calamity_phase value = 2 }
			# Phase 2: Clan wars erupt - AI declares wars autonomously
			every_ruler = {
				limit = {
					has_trait = murim_clan_leader
					is_ai = yes
				}
				random = {
					chance = 70
					trigger_event = { id = murim_calamity_ai.0001 days = { 1 90 } }
				}
			}
		}
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 2 }
				years_from_global_variable = { name = murim_calamity_year value >= 3 }
			}
			set_global_variable = { name = murim_calamity_phase value = 3 }
			# Phase 3: Mass death - 70-90% kill rate
			every_ruler = {
				limit = {
					murim_is_martial_artist_trigger = yes
					is_ai = yes
				}
				random = {
					chance = 80
					trigger_event = { id = murim_calamity_ai.0003 days = { 1 30 } }
				}
			}
			# Emperor assassination
			random_ruler = {
				limit = {
					has_title = title:e_murim_emperor
				}
				trigger_event = { id = murim_calamity_ai.0004 days = { 30 120 } }
			}
		}
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 3 }
				years_from_global_variable = { name = murim_calamity_year value >= 5 }
			}
			set_global_variable = { name = murim_calamity_phase value = 4 }
			# Phase 4: Survivor selection - 1-3 per clan
			every_ruler = {
				limit = {
					has_trait = murim_clan_leader
					is_alive = yes
				}
				trigger_event = { id = murim_calamity_ai.0006 days = { 1 30 } }
			}
			# Player dynasty protection
			every_player = {
				trigger_event = { id = murim_calamity_ai.0007 days = 1 }
			}
		}
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 4 }
				years_from_global_variable = { name = murim_calamity_year value >= 7 }
			}
			set_global_variable = { name = murim_calamity_phase value = 5 }
			# Phase 5: Mythology formation and rebuilding
			every_ruler = {
				limit = {
					has_trait = murim_calamity_survivor
				}
				trigger_event = { id = murim_calamity_ai.0008 days = { 1 60 } }
			}
			# Reset cycle
			set_global_variable = { name = murim_last_calamity_year value = current_year }
			remove_global_variable = murim_calamity_active
			remove_global_variable = murim_calamity_phase
		}
	}
}

# === ON CHARACTER DEATH DURING CALAMITY ===
on_death = {
	effect = {
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				murim_is_martial_artist_trigger = yes
			}
			# Record death in calamity ledger
			add_to_global_variable = { name = murim_calamity_death_count value = 1 }
			# If clan leader dies, trigger succession crisis
			if = {
				limit = { has_trait = murim_clan_leader }
				random_vassal = {
					limit = { murim_is_martial_artist_trigger = yes }
					trigger_event = { id = murim_calamity_ai.0009 days = { 1 30 } }
				}
			}
			# Ranker death causes shockwaves
			# PERFORMANCE: Only notify clan leaders + players, not every martial artist
			# During calamity phase 3, many rankers die — notifying all causes cascades
			if = {
				limit = {
					OR = {
						has_trait = murim_rank_absolute
						has_trait = murim_rank_transcendent
						has_trait = murim_rank_peak
					}
				}
				every_player = {
					limit = { NOT = { this = root } }
					trigger_event = { id = murim_calamity_ai.0010 days = { 1 14 } }
				}
				every_ruler = {
					limit = {
						has_trait = murim_clan_leader
						is_ai = yes
						NOT = { this = root }
					}
					trigger_event = { id = murim_calamity_ai.0010 days = { 1 14 } }
				}
			}
		}
	}
}

# === AI AUTONOMOUS WAR DECLARATION DURING CALAMITY ===
on_war_started = {
	effect = {
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value >= 2 }
			}
			# Escalate: other clans join wars
			every_ruler = {
				limit = {
					has_trait = murim_clan_leader
					is_ai = yes
					NOT = { is_at_war = yes }
				}
				random = {
					chance = 50
					# AI decides to join or betray
					trigger_event = { id = murim_calamity_ai.0002 days = { 30 180 } }
				}
			}
		}
	}
}

# === QUARTERLY PULSE DURING CALAMITY - CIVILIAN DEATHS ===
# Changed from monthly to quarterly for performance
on_quarterly_pulse = {
	effect = {
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value >= 3 }
				is_ruler = yes
				any_held_title = { tier >= tier_county }
			}
			# Civilian massacre events
			random = {
				chance = 30
				trigger_event = { id = murim_calamity_ai.0005 days = { 1 15 } }
			}
			# Province devastation
			every_held_title = {
				limit = { tier = tier_county }
				random = {
					chance = 25
					change_county_control = -20
					change_development_level = -2
				}
			}
		}
	}
}

# === AI BETRAYAL CHAIN DURING CALAMITY ===
on_alliance_broken = {
	effect = {
		if = {
			limit = {
				has_global_variable = murim_calamity_active
			}
			# Former ally becomes enemy - betray chain
			scope:second = {
				trigger_event = { id = murim_calamity_ai.0002 days = { 1 30 } }
			}
			# Witnesses lose trust
			every_ruler = {
				limit = {
					murim_is_martial_artist_trigger = yes
					has_relation_friend = scope:first
				}
				add_opinion = {
					target = scope:first
					modifier = murim_calamity_betrayer_opinion
					opinion = -50
				}
			}
		}
	}
}

# === POST-CALAMITY REBUILDING TRIGGERS ===
on_title_gain = {
	effect = {
		if = {
			limit = {
				has_trait = murim_calamity_survivor
				scope:title = { tier >= tier_duchy }
			}
			# Survivor founding a new sect/clan
			trigger_event = { id = murim_great_calamity_expansion.0003 days = { 30 90 } }
		}
	}
}

# === CALAMITY CYCLE DETECTOR - PREVENTS BACK-TO-BACK ===
on_game_start = {
	effect = {
		# Initialize calamity tracking
		if = {
			limit = {
				NOT = { has_global_variable = murim_calamity_cycle_count }
			}
			set_global_variable = { name = murim_calamity_cycle_count value = 0 }
			set_global_variable = { name = murim_world_active value = yes }
		}
	}
}

# === FIVE-YEAR PULSE: CALAMITY AFTERMATH CHECK ===
on_five_year_pulse = {
	effect = {
		# After calamity ends, check if world has recovered enough
		if = {
			limit = {
				NOT = { has_global_variable = murim_calamity_active }
				has_global_variable = murim_last_calamity_year
				years_from_global_variable = { name = murim_last_calamity_year value >= 20 }
				years_from_global_variable = { name = murim_last_calamity_year value < 50 }
			}
			# Rebuilding era events
			every_ruler = {
				limit = {
					has_trait = murim_calamity_survivor
					OR = {
						has_trait = murim_calamity_rebuilder
						has_trait = murim_clan_leader
					}
				}
				random = {
					chance = 40
					trigger_event = { id = murim_great_calamity_expansion.0005 days = { 30 365 } }
				}
			}
		}
		# Mythology solidification - 50+ years after calamity
		if = {
			limit = {
				NOT = { has_global_variable = murim_calamity_active }
				has_global_variable = murim_last_calamity_year
				years_from_global_variable = { name = murim_last_calamity_year value >= 50 }
			}
			every_ruler = {
				limit = {
					murim_is_martial_artist_trigger = yes
					NOT = { has_trait = murim_calamity_survivor }
				}
				random = {
					chance = 20
					trigger_event = { id = murim_great_calamity_expansion.0004 days = { 30 365 } }
				}
			}
		}
	}
}

# === CALAMITY INTENSITY ESCALATION ===
on_bi_yearly_pulse = {
	effect = {
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 2 }
			}
			# Escalate war intensity every 2 years
			every_ruler = {
				limit = {
					has_trait = murim_clan_leader
					is_ai = yes
					is_at_war = yes
				}
				# AI gets more aggressive
				add_character_modifier = {
					modifier = murim_calamity_war_frenzy
					years = 2
				}
				# Random betrayals
				random = {
					chance = 30
					random_ally = {
						limit = { murim_is_martial_artist_trigger = yes }
						reverse_add_opinion = {
							target = prev
							modifier = murim_calamity_betrayer_opinion
							opinion = -100
						}
						trigger_event = { id = murim_calamity_ai.0002 days = { 1 30 } }
					}
				}
			}
		}
		# During phase 3, kill stragglers
		if = {
			limit = {
				has_global_variable = murim_calamity_active
				has_global_variable_value = { name = murim_calamity_phase value = 3 }
			}
			every_ruler = {
				limit = {
					murim_is_martial_artist_trigger = yes
					is_ai = yes
					NOT = { has_trait = murim_rank_absolute }
					NOT = { has_trait = murim_rank_transcendent }
				}
				random = {
					chance = 40
					# Weak martial artists die
					death = { death_reason = death_battle }
				}
			}
		}
	}
}

# === SURVIVOR TRAIT ASSIGNMENT POST-CALAMITY ===
on_birthday = {
	effect = {
		# Characters born during/after calamity get special traits
		if = {
			limit = {
				has_global_variable = murim_last_calamity_year
				birth_year >= global_var:murim_calamity_year
				NOT = { has_trait = murim_calamity_born_after }
				age >= 5
				murim_is_martial_artist_trigger = yes
			}
			random = {
				chance = 50
				add_trait = murim_calamity_born_after
				trigger_event = { id = murim_great_calamity_expansion.0008 days = { 1 30 } }
			}
		}
		# Survivors who reach old age become living legends
		if = {
			limit = {
				has_trait = murim_calamity_survivor
				age >= 60
				NOT = { has_trait = murim_calamity_living_legend }
			}
			add_trait = murim_calamity_living_legend
			every_vassal = {
				add_opinion = {
					target = prev
					modifier = murim_living_legend_opinion
					opinion = 30
				}
			}
		}
	}
}
