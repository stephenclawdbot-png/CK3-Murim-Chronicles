# Shaolin 72 Arts Trial - Choosing Your Path
####################################
murim_shaolin_expansion2.0001 = {
	type = character_event
	title = murim_shaolin_expansion2.0001.t
	desc = murim_shaolin_expansion2.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	right_portrait = {
		character = scope:shaolin_abbot
		animation = personality_compassionate
	}
	
	trigger = {
		has_trait = shaolin_disciple
		NOT = { has_character_flag = seventy_two_arts_chosen }
		age >= 16
		prowess >= 10
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = devoted
		}
	}
	
	immediate = {
		random_relation = {
			type = mentor
			save_scope_as = shaolin_abbot
		}
		if = {
			limit = { NOT = { exists = scope:shaolin_abbot } }
			random_courtier = {
				limit = { has_trait = shaolin_elder age >= 50 }
				save_scope_as = shaolin_abbot
			}
		}
		set_character_flag = seventy_two_arts_chosen
	}
	
	# Option A: Iron Body Arts - external strength path
	option = {
		name = murim_shaolin_expansion2.0001.a
		add_character_modifier = { modifier = iron_body_training years = 15 }
		change_prowess = 2
		add_prestige = 100
		set_character_flag = shaolin_iron_body_path
		stress_impact = { brave = medium_stress_loss craven = medium_stress_gain }
		ai_chance = { base = 40 modifier = { add = 20 prowess >= 15 } }
	}
	
	# Option B: Finger Arts - precision combat path
	option = {
		name = murim_shaolin_expansion2.0001.b
		add_character_modifier = { modifier = finger_technique_training years = 15 }
		change_prowess = 1
		add_intrigue = 1
		add_prestige = 100
		set_character_flag = shaolin_finger_path
		ai_chance = { base = 30 modifier = { add = 20 intrigue >= 12 } }
	}
	
	# Option C: Internal Arts - qi and meditation path
	option = {
		name = murim_shaolin_expansion2.0001.c
		add_character_modifier = { modifier = internal_qi_training years = 15 }
		add_learning = 2
		add_prestige = 100
		set_character_flag = shaolin_internal_path
		ai_chance = { base = 30 modifier = { add = 20 learning >= 15 } }
	}
}
