######################################
# EVENT 0001 — Vote Path: Clan Assembly Called
# Democratic clan creation — gather martial artists to vote on leadership
######################################
murim_custom_clan_exp2.0001 = {
	type = character_event
	title = murim_custom_clan_exp2.0001.t
	desc = murim_custom_clan_exp2.0001.desc
	theme = diplomacy_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:first_supporter
		animation = personality_compassionate
	}
	lower_left_portrait = {
		character = scope:second_supporter
		animation = idle
	}

	trigger = {
		NOT = { has_variable = murim_clan_id }
		has_variable = murim_cultivation_level
		var:murim_cultivation_level >= 4
		diplomacy >= 14
		NOT = { has_character_flag = custom_clan_vote_started }
		any_living_character = {
			count >= 5
			NOT = { has_variable = murim_clan_id }
			has_variable = murim_cultivation_level
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = gregarious
		}
		modifier = {
			add = 2
			diplomacy >= 18
		}
	}

	immediate = {
		set_character_flag = { flag = custom_clan_vote_started }
		random_living_character = {
			limit = {
				NOT = { has_variable = murim_clan_id }
				has_variable = murim_cultivation_level
			}
			save_scope_as = first_supporter
		}
		random_living_character = {
			limit = {
				NOT = { has_variable = murim_clan_id }
				has_variable = murim_cultivation_level
				NOT = { this = scope:first_supporter }
			}
			save_scope_as = second_supporter
		}
	}

	# Call the assembly — seek democratic mandate
	option = {
		name = murim_custom_clan_exp2.0001.a
		add_character_modifier = {
			modifier = clan_assembly_organizer
			years = 3
		}
		add_prestige = 200
		add_gold = -100
		hidden_effect = {
			trigger_event = {
				id = murim_custom_clan_exp2.0002
				days = { 30 60 }
			}
		}
		ai_chance = { base = 60 }
	}

	# Skip the vote — just recruit directly
	option = {
		name = murim_custom_clan_exp2.0001.b
		add_prestige = -50
		hidden_effect = {
			trigger_event = {
				id = murim_custom_clan_exp2.0005
				days = { 14 30 }
			}
		}
		ai_chance = { base = 30 }
	}

	# Abandon the idea
	option = {
		name = murim_custom_clan_exp2.0001.c
		add_stress = 10
		ai_chance = { base = 10 }
	}
}

######################################
# EVENT 0002 — Vote Path: The Election
# Martial artists vote on who leads the new clan
######################################
murim_custom_clan_exp2.0002 = {
	type = character_event
	title = murim_custom_clan_exp2.0002.t
	desc = murim_custom_clan_exp2.0002.desc
	theme = diplomacy_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:vote_rival
		animation = personality_vengeful
	}

	trigger = {
		has_character_flag = custom_clan_vote_started
	}

	immediate = {
		if = {
			limit = {
				any_living_character = {
					NOT = { has_variable = murim_clan_id }
					has_variable = murim_cultivation_level
					NOT = { this = root }
					diplomacy >= 12
				}
			}
			random_living_character = {
				limit = {
					NOT = { has_variable = murim_clan_id }
					has_variable = murim_cultivation_level
					NOT = { this = root }
					diplomacy >= 12
				}
				save_scope_as = vote_rival
			}
		}
	}

	# Win the vote through charisma and speeches
	option = {
		name = murim_custom_clan_exp2.0002.a
		duel = {
			skill = diplomacy
			target = scope:vote_rival
			55 = {
				desc = murim_custom_clan_exp2.0002.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				send_interface_toast = {
					title = murim_custom_clan_exp2.0002.a.success.toast
					left_icon = root
					add_prestige = 500
					set_variable = {
						name = murim_clan_id
						value = 10
					}
					set_variable = {
						name = murim_custom_clan_name
						value = flag:vote_clan
					}
					add_trait = vote_elected_leader
					add_character_modifier = {
						modifier = democratically_elected_clan_leader
						years = 20
					}
					scope:vote_rival = {
						add_opinion = {
							target = root
							modifier = lost_election_gracefully
							opinion = -10
						}
						set_variable = {
							name = murim_clan_id
							value = 10
						}
					}
				}
			}
			45 = {
				desc = murim_custom_clan_exp2.0002.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				send_interface_toast = {
					title = murim_custom_clan_exp2.0002.a.failure.toast
					left_icon = root
					add_prestige = -200
					scope:vote_rival = {
						add_trait = vote_elected_leader
						set_variable = {
							name = murim_clan_id
							value = 10
						}
					}
					set_variable = {
						name = murim_clan_id
						value = 10
					}
					add_character_modifier = {
						modifier = lost_clan_election
						years = 5
					}
				}
			}
		}
		ai_chance = { base = 60 }
	}

	# Rig the vote through bribery
	option = {
		name = murim_custom_clan_exp2.0002.b
		trigger = {
			gold >= 200
			intrigue >= 12
		}
		add_gold = -200
		add_prestige = 300
		set_variable = {
			name = murim_clan_id
			value = 10
		}
		add_trait = vote_elected_leader
		add_secret = {
			type = secret_rigged_election
			target = root
		}
		add_character_modifier = {
			modifier = rigged_clan_election
			years = 10
		}
		ai_chance = { base = 25 }
	}

	# Concede before the vote — let the rival lead
	option = {
		name = murim_custom_clan_exp2.0002.c
		trigger = {
			exists = scope:vote_rival
		}
		scope:vote_rival = {
			add_opinion = {
				target = root
				modifier = graceful_concession
				opinion = 30
			}
			add_trait = vote_elected_leader
		}
		set_variable = {
			name = murim_clan_id
			value = 10
		}
		add_prestige = 100
		add_piety = 100
		ai_chance = { base = 15 }
	}
}
