murim_shadow_politics_expansion6.0001 = {
	type = character_event
	title = murim_shadow_pol_exp6.0001.t
	desc = murim_shadow_pol_exp6.0001.desc
	theme = intrigue
	left_portrait = { character = root animation = scheme }

	trigger = {
		has_trait = murim_heavenly_demon_sect
		has_global_variable = murim_puppet_emperor_installed
		NOT = { has_character_flag = shadow_endgame_begun }
	}

	weight_multiplier = {
		base = 30
		modifier = { add = 20 intrigue >= 20 }
	}

	immediate = {
		set_character_flag = shadow_endgame_begun
		random_living_character = {
			limit = { has_character_modifier = murim_puppet_emperor }
			save_scope_as = puppet_emperor
		}
		random_living_character = {
			limit = { has_trait = murim_martial_artist is_ruler = yes NOT = { has_trait = murim_heavenly_demon_sect } prestige >= 3000 }
			save_scope_as = orthodox_leader
		}
	}

	option = {
		name = murim_shadow_pol_exp6.0001.a
		add_character_modifier = { modifier = murim_shadow_emperor_endgame years = 10 }
		add_intrigue = 5
		trigger_event = { id = murim_shadow_politics_expansion6.0002 days = { 30 60 } }
	}
	option = {
		name = murim_shadow_pol_exp6.0001.b
		add_character_modifier = { modifier = murim_shadow_consolidation years = 15 }
		add_prestige = 500
		if = { limit = { exists = scope:puppet_emperor }
			scope:puppet_emperor = { add_character_modifier = { modifier = murim_tightened_control years = 10 } }
		}
	}
	option = {
		name = murim_shadow_pol_exp6.0001.c
		add_character_modifier = { modifier = murim_shadow_withdrawal years = 10 }
		add_gold = 500
		remove_global_variable = murim_puppet_emperor_installed
	}
}

					add_character_modifier = { modifier = murim_demon_emperor_victorious years = 30 }

				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }

		}
	}
