######################################
# EVENT 0001 — Orthodox Alliance War Council
# Orthodox faction leaders gather to plan campaign against unorthodox
######################################
murim_faction_war_exp.0001 = {
	type = character_event
	title = murim_faction_war_exp.0001.t
	desc = murim_faction_war_exp.0001.desc
	theme = war_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:shaolin_leader
		animation = personality_compassionate
	}
	lower_left_portrait = {
		character = scope:wudang_leader
		animation = personality_bold
	}

	trigger = {
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_monk
			has_trait = wudang_disciple
			has_trait = iron_fist_disciple
			has_trait = tang_clan_member
		}
		has_variable = murim_clan_id
		NOT = { has_character_flag = orthodox_war_council_attended }
		any_ruler = {
			has_trait = heavenly_demon_member
			is_at_war = no
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = zealous
		}
		modifier = {
			add = 2
			martial >= 15
		}
	}

	immediate = {
		set_character_flag = { flag = orthodox_war_council_attended days = 730 }
		if = {
			limit = {
				any_ruler = { has_trait = shaolin_monk }
			}
			random_ruler = {
				limit = { has_trait = shaolin_monk }
				save_scope_as = shaolin_leader
			}
		}
		if = {
			limit = {
				any_ruler = { has_trait = wudang_disciple }
			}
			random_ruler = {
				limit = { has_trait = wudang_disciple }
				save_scope_as = wudang_leader
			}
		}
	}

	# Lead the alliance offensive
	option = {
		name = murim_faction_war_exp.0001.a
		add_character_modifier = {
			modifier = orthodox_war_commander
			years = 5
		}
		add_prestige = 500
		add_martial = 2
		every_ruler = {
			limit = {
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = iron_fist_disciple
					has_trait = tang_clan_member
				}
			}
			add_opinion = {
				target = root
				modifier = orthodox_war_leader
				opinion = 20
			}
		}
		hidden_effect = {
			trigger_event = {
				id = murim_faction_war_exp.0002
				days = { 30 90 }
			}
		}
		ai_chance = { base = 50 }
	}

	# Advocate for defensive strategy
	option = {
		name = murim_faction_war_exp.0001.b
		add_character_modifier = {
			modifier = orthodox_defensive_strategist
			years = 5
		}
		add_prestige = 200
		add_martial = 1
		ai_chance = { base = 30 }
	}

	# Propose peace negotiations instead
	option = {
		name = murim_faction_war_exp.0001.c
		trigger = {
			diplomacy >= 16
		}
		add_piety = 300
		add_character_modifier = {
			modifier = orthodox_peace_advocate
			years = 5
		}
		every_ruler = {
			limit = {
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
				}
			}
			add_opinion = {
				target = root
				modifier = proposed_peace_instead
				opinion = -10
			}
		}
		ai_chance = { base = 20 }
	}
}

murim_faction_war_exp.0002 = {
	type = character_event
	title = murim_faction_war_exp.0002.t
	desc = murim_faction_war_exp.0002.desc
	theme = war_theme
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = scope:unorthodox_commander
		animation = personality_vengeful
	}

	trigger = {
		has_character_modifier = orthodox_war_commander
	}

	immediate = {
		random_ruler = {
			limit = {
				OR = {
					has_trait = heavenly_demon_member
					has_trait = phantom_blade_disciple
				}
			}
			save_scope_as = unorthodox_commander
		}
	}

	# Lead the charge personally
	option = {
		name = murim_faction_war_exp.0002.a
		duel = {
			skill = prowess
			target = scope:unorthodox_commander
			50 = {
				desc = murim_faction_war_exp.0002.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 4.0
				}
				send_interface_toast = {
					title = murim_faction_war_exp.0002.a.success.toast
					left_icon = root
					add_prestige = 800
					add_character_modifier = { modifier = faction_war_victory years = 10 }
					add_trait = faction_war_hero
					scope:unorthodox_commander = {
						health = -3
						add_prestige = -500
					}
				}
			}
			50 = {
				desc = murim_faction_war_exp.0002.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -4.0
				}
				send_interface_toast = {
					title = murim_faction_war_exp.0002.a.failure.toast
					left_icon = root
					health = -3
					add_prestige = -300
					add_character_modifier = { modifier = faction_war_wounded years = 3 }
				}
			}
		}
		ai_chance = { base = 60 }
	}

	# Command from the rear — strategic approach
	option = {
		name = murim_faction_war_exp.0002.b
		add_martial = 2
		random = {
			chance = 60
			add_prestige = 400
			add_character_modifier = { modifier = strategic_war_victory years = 8 }
		}
		random = {
			chance = 40
			add_prestige = -100
			add_character_modifier = { modifier = strategic_withdrawal years = 3 }
		}
		ai_chance = { base = 30 }
	}

	# Send champions to fight one-on-one instead of mass battle
	option = {
		name = murim_faction_war_exp.0002.c
		trigger = {
			diplomacy >= 14
		}
		add_prestige = 200
		add_character_modifier = {
			modifier = champion_duel_proposal
			years = 3
		}
		hidden_effect = {
			trigger_event = {
				id = murim_faction_war_exp.0003
				days = { 7 14 }
			}
		}
		ai_chance = { base = 10 }
	}
}

murim_faction_war_exp.0003 = {
	type = character_event
	title = murim_faction_war_exp.0003.t
	desc = murim_faction_war_exp.0003.desc
	theme = martial_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:unorthodox_champion
		animation = personality_vengeful
	}

	trigger = {
		has_character_modifier = champion_duel_proposal
	}

	immediate = {
		random_ruler = {
			limit = {
				OR = {
					has_trait = heavenly_demon_member
					has_trait = phantom_blade_disciple
				}
				prowess >= 15
			}
			save_scope_as = unorthodox_champion
		}
	}

	# Fight as the orthodox champion yourself
	option = {
		name = murim_faction_war_exp.0003.a
		duel = {
			skill = prowess
			target = scope:unorthodox_champion
			50 = {
				desc = murim_faction_war_exp.0003.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 5.0
				}
				send_interface_toast = {
					title = murim_faction_war_exp.0003.a.success.toast
					left_icon = root
					add_prestige = 1500
					add_trait = faction_champion
					add_character_modifier = { modifier = champion_of_orthodoxy years = 20 }
					scope:unorthodox_champion = {
						murim_combat_death_or_injury_with_killer_effect = { REASON = death_battle KILLER = root }
					}
				}
			}
			50 = {
				desc = murim_faction_war_exp.0003.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -5.0
				}
				send_interface_toast = {
					title = murim_faction_war_exp.0003.a.failure.toast
					left_icon = root
					murim_combat_death_or_injury_with_killer_effect = { REASON = death_battle KILLER = scope:unorthodox_champion }
				}
			}
		}
		ai_chance = { base = 50 }
	}

	# Nominate another fighter
	option = {
		name = murim_faction_war_exp.0003.b
		random_courtier = {
			limit = {
				prowess >= 15
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
				}
			}
			weight = {
				base = 1
				modifier = { add = 5 prowess >= 20 }
			}
			duel = {
				skill = prowess
				target = scope:unorthodox_champion
				50 = {
					add_prestige = 500
					root = { add_prestige = 300 }
				}
				50 = {
					murim_combat_death_or_injury_with_killer_effect = { REASON = death_battle KILLER = scope:unorthodox_champion }
					root = { add_prestige = -200 }
				}
			}
		}
		ai_chance = { base = 40 }
	}

	# Cancel and return to full war
	option = {
		name = murim_faction_war_exp.0003.c
		add_prestige = -200
		remove_character_modifier = champion_duel_proposal
		ai_chance = { base = 10 }
	}
}
