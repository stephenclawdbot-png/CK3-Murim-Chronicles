######################################
# 0008 - REPUTATION CHALLENGE
# Someone publicly questions your ranking
######################################

murim_rank.0008 = {
	type = character_event
	title = murim_rank.0008.t
	desc = murim_rank.0008.desc
	theme = martial
	left_portrait = {
		character = scope:reputation_challenger
		animation = personality_bold
	}
	right_portrait = {
		character = root
		animation = anger
	}
	override_background = {
		reference = market
	}

	trigger = {
		is_alive = yes
		has_variable = murim_martial_rank_position
		var:murim_martial_rank_position <= 50
		any_living_character = {
			has_character_flag = murim_initiated
			prowess >= 15
			NOT = { this = root }
			OR = {
				has_trait = arrogant
				has_trait = ambitious
			}
		}
		NOT = { has_character_flag = murim_reputation_challenge_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			var:murim_martial_rank_position <= 10
		}
	}

	immediate = {
		random_living_character = {
			limit = {
				has_character_flag = murim_initiated
				prowess >= 15
				NOT = { this = root }
				OR = {
					has_trait = arrogant
					has_trait = ambitious
				}
			}
			save_scope_as = reputation_challenger
		}
		create_character_memory = {
			type = murim_reputation_challenged_memory
			participants = {
				challenger = scope:reputation_challenger
			}
		}
	}

	# Silence them with a public duel
	option = {
		name = murim_rank.0008.a
		custom_tooltip = murim_rank_silence_duel_tt
		duel = {
			skill = prowess
			target = scope:reputation_challenger
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				desc = murim_rank_silenced_doubter_tt
				add_prestige = 500
				scope:reputation_challenger = {
					add_trait = wounded_1
					add_prestige = -300
					add_opinion = {
						target = root
						modifier = murim_defeated_by_ranked_opinion
					}
				}
				create_character_memory = {
					type = murim_defended_ranking_memory
					participants = {
						defeated = scope:reputation_challenger
					}
				}
			}
			20 = {
				desc = murim_rank_inconclusive_tt
				add_prestige = 100
				scope:reputation_challenger = {
					add_opinion = {
						target = root
						modifier = murim_grudging_respect_opinion
					}
				}
			}
			30 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				desc = murim_rank_reputation_damaged_tt
				add_trait = wounded_1
				add_prestige = -500
				add_character_modifier = {
					modifier = murim_ranking_questioned_modifier
					years = 3
				}
				random = {
					chance = 20
					add_trait = craven
				}
			}
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_reputation_challenge_cooldown
				years = 2
			}
		}
		ai_chance = {
			base = 50
			modifier = { add = 25 has_trait = brave }
		}
	}

	# Ignore the insult - actions speak louder
	option = {
		name = murim_rank.0008.b
		add_prestige = -100
		stress_impact = {
			proud = minor_stress_impact_gain
			arrogant = major_stress_impact_gain
		}
		add_character_modifier = {
			modifier = murim_above_petty_challenges_modifier
			years = 2
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_reputation_challenge_cooldown
				years = 2
			}
		}
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = patient }
		}
	}

	# Have your disciples deal with them
	option = {
		name = murim_rank.0008.c
		trigger = {
			has_character_flag = murim_sect_leader
			any_courtier = {
				has_character_flag = murim_sect_member
				prowess >= 12
			}
		}
		custom_tooltip = murim_rank_disciples_handle_tt
		scope:reputation_challenger = {
			add_trait = wounded_1
			add_opinion = {
				target = root
				modifier = murim_sent_lackeys_opinion
			}
		}
		add_tyranny = 5
		hidden_effect = {
			set_character_flag = {
				flag = murim_reputation_challenge_cooldown
				years = 2
			}
		}
		ai_chance = {
			base = 20
			modifier = { add = 20 has_trait = callous }
		}
	}
}

######################################
# 0009 - RANK MANIPULATION
# Someone tries to artificially boost/lower rankings
######################################

murim_rank.0009 = {
	type = character_event
	title = murim_rank.0009.t
	desc = murim_rank.0009.desc
	theme = intrigue
	left_portrait = {
		character = scope:rank_manipulator
		animation = scheme
	}
	override_background = {
		reference = corridor_night
	}

	trigger = {
		is_alive = yes
		has_variable = murim_martial_rank_position
		var:murim_martial_rank_position <= 50
		any_living_character = {
			has_character_flag = murim_initiated
			intrigue >= 14
			OR = {
				has_trait = deceitful
				has_trait = ambitious
			}
			NOT = { this = root }
		}
		NOT = { has_character_flag = murim_rank_manipulation_cooldown }
	}

	immediate = {
		random_living_character = {
			limit = {
				has_character_flag = murim_initiated
				intrigue >= 14
				OR = {
					has_trait = deceitful
					has_trait = ambitious
				}
				NOT = { this = root }
			}
			save_scope_as = rank_manipulator
		}
		create_character_memory = {
			type = murim_rank_manipulation_discovered_memory
			participants = {
				manipulator = scope:rank_manipulator
			}
		}
	}

	# Expose the manipulation publicly
	option = {
		name = murim_rank.0009.a
		custom_tooltip = murim_rank_expose_manipulation_tt
		add_prestige = 300
		scope:rank_manipulator = {
			add_prestige = -500
			add_character_modifier = {
				modifier = murim_exposed_cheater_modifier
				years = 5
			}
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_rank_manipulation_cooldown
				years = 5
			}
		}
		ai_chance = {
			base = 40
			modifier = { add = 25 has_trait = just }
			modifier = { add = 15 has_trait = honest }
		}
	}

	# Blackmail the manipulator for personal gain
	option = {
		name = murim_rank.0009.b
		trigger = { intrigue >= 12 }
		custom_tooltip = murim_rank_blackmail_manipulator_tt
		add_gold = 200
		scope:rank_manipulator = {
			add_opinion = {
				target = root
				modifier = murim_blackmailed_opinion
			}
		}
		add_character_modifier = {
			modifier = murim_blackmail_leverage
			years = 3
		}
		stress_impact = {
			honest = major_stress_impact_gain
			just = major_stress_impact_gain
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_rank_manipulation_cooldown
				years = 5
			}
			trigger_event = { id = murim_jianghu.0001 days = { 30 60 } }
		}
		ai_chance = {
			base = 30
			modifier = { add = 25 has_trait = deceitful }
		}
	}

	# Ignore it - rankings sort themselves out
	option = {
		name = murim_rank.0009.c
		add_piety = 50
		hidden_effect = {
			set_character_flag = {
				flag = murim_rank_manipulation_cooldown
				years = 5
			}
		}
		ai_chance = { base = 30 }
	}
}
