# === EVENT 0001: EIGHTEEN DRAGON-SUBDUING PALMS TRIAL ===
murim_beggar_exp2.0001 = {
	type = character_event
	title = murim_beggar_exp2.0001.t
	desc = murim_beggar_exp2.0001.desc
	theme = martial_arts
	
	trigger = {
		has_trait = murim_clan_beggar
		has_trait_rank = { trait = murim_cultivation_rank value >= 3 }
		NOT = { has_character_flag = beggar_dragon_palm_trial }
	}
	
	immediate = {
		set_character_flag = { flag = beggar_dragon_palm_trial days = 3650 }
	}
	
	# Option A: Train under the waterfall - brute force mastery
	option = {
		name = murim_beggar_exp2.0001.a
		random_list = {
			50 = {
				add_trait = murim_technique_dragon_palm
				add_prowess = 8
				add_martial = 3
				add_prestige = 1000
			}
			30 = {
				# Partial - learn 9 of 18 palms
				add_prowess = 4
				add_martial = 2
				add_character_modifier = {
					modifier = murim_beggar_nine_palms
					years = 5
				}
			}
			20 = {
				# Failure - body can't handle the qi
				add_character_modifier = { modifier = murim_critical_injury years = 2 }
				add_trait = murim_combat_trauma_shattered_arms
			}
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Meditative approach - understand the dragon's spirit
	option = {
		name = murim_beggar_exp2.0001.b
		add_learning = 3
		add_prowess = 4
		add_character_modifier = {
			modifier = murim_beggar_dragon_spirit
			years = -1
		}
		ai_chance = { base = 35 }
	}
	
	# Option C: Modify the technique - create Beggar's Dragon Fist
	option = {
		name = murim_beggar_exp2.0001.c
		add_trait = murim_technique_beggar_dragon_fist
		add_prowess = 5
		add_martial = 2
		add_prestige = 500
		ai_chance = { base = 15 }
	}
}
