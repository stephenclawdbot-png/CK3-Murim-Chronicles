# Murim Chronicles - Clan/Sect Hierarchy Triggers (ALL 9 GREAT CLANS)
# INTEGRATION: Used by murim_sect_politics_events.txt, murim_clan_membership_effects.txt,
#              murim_game_start_effects.txt, murim_tang_birthright_effects.txt,
#              murim_faction_events.txt, murim_shadow_politics_events.txt,

# Cross-refs: murim_triggers.txt (base murim_is_clan_member_trigger),
#             murim_harmony_values.txt (harmony checks),
#             murim_faction_effects.txt (faction alignment)
# RULE: Clans are ORGANIZATIONS managed through traits, NOT CK3 dynasties.
#       A character's CK3 dynasty is separate from their sect membership.
#       EXCEPTION: Tang Clan is bloodline-based (see murim_tang_birthright_effects.txt).
# GENDER: NO gender restrictions on ANY rank. Women can hold ALL positions.
# FACTIONS: Orthodox (righteous) vs Unorthodox (demonic). Affects interactions, wars, alliances.
# VANILLA SEPARATION: Clan faction does NOT affect feudal politics, succession, or wars.
# INDEX:
#   1-6. Rank checks (leader, vice, elder, inner, outer, associate) — ALL 9 CLANS
#   7. murim_get_sect_rank_value - numeric rank for comparison
#   8. murim_is_higher_rank_than_prev_trigger - promotion eligibility
#   9. murim_is_same_sect_as_prev_trigger - same sect check
#   10. murim_is_tang_bloodline_trigger - Tang Clan blood check
#   11. murim_is_eligible_for_promotion_trigger - can be promoted
#   12. murim_is_succession_candidate_trigger - eligible to become sect leader
#   13. murim_sect_has_leader_trigger - does the sect have a leader
#   14. murim_sect_harmony_is_high_trigger - harmony >= 70
#   15. murim_sect_harmony_is_low_trigger - harmony <= 30
#   16. murim_is_in_orthodox_sect_trigger - Mount Hua, Shaolin, Wudang, Tang, Beggar, Iron Fist
#   17. murim_is_in_unorthodox_sect_trigger - Heavenly Demon, Poison Flower, Phantom Blade
#   18. murim_is_in_neutral_sect_trigger - custom clans that haven't declared
#   19-27. Individual clan membership triggers (all 9)
#   28. murim_is_faction_rival_trigger - target is in opposing faction
#   29. murim_is_custom_clan_member_trigger - member of player/AI-created clan
#   30. murim_can_create_custom_clan_trigger - eligible for clan creation
#   31. murim_is_clan_leader_of_any_trigger - leader of any of the 9+ clans
#   32. murim_is_poison_specialist_trigger - Poison Flower or Tang poison user
#   33. murim_is_assassin_type_trigger - Phantom Blade or assassination-trained

# ═══════════════════════════════════════════
#  7. NUMERIC RANK VALUE (for comparison)
# ═══════════════════════════════════════════
murim_get_sect_rank_value = {
	# Returns: leader=6, vice=5, elder=4, inner=3, outer=2, associate=1, none=0
	value = 0
	if = {
		limit = { murim_is_sect_leader_trigger = yes }
		add = 6
	}
	else_if = {
		limit = { murim_is_sect_vice_leader_trigger = yes }
		add = 5
	}
	else_if = {
		limit = { murim_is_sect_elder_trigger = yes }
		add = 4
	}
	else_if = {
		limit = { murim_is_inner_disciple_trigger = yes }
		add = 3
	}
	else_if = {
		limit = { murim_is_outer_disciple_trigger = yes }
		add = 2
	}
	else_if = {
		limit = { murim_is_sect_associate_trigger = yes }
		add = 1
	}
}

# ═══════════════════════════════════════════
#  8. HIGHER RANK THAN PREV (for events)
# ═══════════════════════════════════════════
murim_is_higher_rank_than_prev_trigger = {
	# Used in event scopes where prev = another character
	# Compare root's rank to prev's rank
	custom_description = {
		text = "Holds a higher sect rank"
		murim_get_sect_rank_value > prev.murim_get_sect_rank_value
	}
}

# ═══════════════════════════════════════════
#  9. SAME SECT CHECK
# ═══════════════════════════════════════════
murim_is_same_sect_as_prev_trigger = {
	OR = {
		# Mount Hua
		AND = { murim_is_mount_hua_disciple_trigger = yes prev = { murim_is_mount_hua_disciple_trigger = yes } }
		# Tang Clan
		AND = { murim_is_tang_clan_member_trigger = yes prev = { murim_is_tang_clan_member_trigger = yes } }
		# Heavenly Demon
		AND = { murim_is_heavenly_demon_member_trigger = yes prev = { murim_is_heavenly_demon_member_trigger = yes } }
		# Wudang
		AND = { murim_is_wudang_disciple_trigger = yes prev = { murim_is_wudang_disciple_trigger = yes } }
		# Shaolin
		AND = { murim_is_shaolin_monk_trigger = yes prev = { murim_is_shaolin_monk_trigger = yes } }
		# Beggar Sect
		AND = { murim_is_beggar_sect_member_trigger = yes prev = { murim_is_beggar_sect_member_trigger = yes } }
		# Poison Flower Palace
		AND = { murim_is_poison_flower_disciple_trigger = yes prev = { murim_is_poison_flower_disciple_trigger = yes } }
		# Iron Fist Gate
		AND = { murim_is_iron_fist_disciple_trigger = yes prev = { murim_is_iron_fist_disciple_trigger = yes } }
		# Phantom Blade Sect
		AND = { murim_is_phantom_blade_disciple_trigger = yes prev = { murim_is_phantom_blade_disciple_trigger = yes } }
		# Custom clan (same flag value)
		AND = {
			has_character_flag = custom_clan_member
			prev = { has_character_flag = custom_clan_member }
			# Additional check: same clan ID variable
			custom_description = {
				text = "Same custom clan"
				var:custom_clan_id = prev.var:custom_clan_id
			}
		}
	}
}

# ═══════════════════════════════════════════
#  11. PROMOTION ELIGIBILITY
# ═══════════════════════════════════════════
murim_is_eligible_for_promotion_trigger = {
	is_alive = yes
	is_imprisoned = no
	murim_is_clan_member_of_any_trigger = yes
	NOR = {
		murim_is_sect_leader_trigger = yes
		has_character_flag = murim_recently_promoted # 2-year cooldown
		has_character_flag = murim_promotion_denied # 1-year cooldown
	}
	# Rank-specific requirements
	OR = {
		# Associate -> Outer: 1+ year membership
		AND = {
			murim_is_sect_associate_trigger = yes
			has_character_flag = murim_membership_1yr
		}
		# Outer -> Inner: 3+ years, proven in combat or training
		AND = {
			murim_is_outer_disciple_trigger = yes
			has_character_flag = murim_membership_3yr
			OR = {
				prowess >= 10
				has_trait = murim_tier_2_martial_artist
				has_trait = murim_tier_3_martial_artist
			}
		}
		# Inner -> Elder: 8+ years, high prowess, respected
		AND = {
			murim_is_inner_disciple_trigger = yes
			has_character_flag = murim_membership_8yr
			prowess >= 15
			OR = {
				has_trait = murim_tier_4_martial_artist
				has_trait = murim_tier_5_martial_artist
				age >= 40
			}
		}
		# Elder -> Vice Leader: 15+ years, top tier, leader endorsement
		AND = {
			murim_is_sect_elder_trigger = yes
			has_character_flag = murim_membership_15yr
			prowess >= 20
		}
	}
}

# ═══════════════════════════════════════════
#  12. SUCCESSION CANDIDATE
# ═══════════════════════════════════════════
murim_is_succession_candidate_trigger = {
	is_alive = yes
	is_imprisoned = no

		murim_is_sect_elder_trigger = yes

				has_trait = murim_tier_6_martial_artist
			}
		}
	}
}

# ═══════════════════════════════════════════
#  30. CAN CREATE CUSTOM CLAN
#  PATH 1: Hard Way — tier 5+, 10yr exp, 5 followers, 500 gold
#  PATH 2: Supreme Way — #1 ranked OR defeated 3 clan leaders
# ═══════════════════════════════════════════
murim_can_create_custom_clan_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOR = {
		murim_is_clan_member_of_any_trigger = yes
		has_character_flag = custom_clan_creation_cooldown # 10-year on rejection
	}
	OR = {
		# PATH 1: Hard Way
		AND = {
			OR = {
				has_trait = murim_tier_5_martial_artist
				has_trait = murim_tier_6_martial_artist
				has_trait = murim_tier_7_martial_artist
				has_trait = murim_tier_8_martial_artist
			}
			has_character_flag = murim_10yr_experience
			# 5+ followers (characters with martial_master = ROOT)
			any_relation = {
				type = mentor
				count >= 5
			}
			gold >= 500
		}
		# PATH 2: Supreme Way
		AND = {
			OR = {
				has_character_modifier = murim_supreme_champion # #1 ranked
				has_character_flag = defeated_three_clan_leaders # beat 3 leaders in succession
			}
		}
	}
}

	OR = {
		murim_is_poison_flower_disciple_trigger = yes
		AND = {

			OR = {
				has_character_flag = tang_poison_stage_1
				has_character_flag = tang_poison_stage_2
				has_character_flag = tang_poison_stage_3
			}
		}
		has_trait = murim_poison_master # Independent poison users
	}
}

# ═══════════════════════════════════════════
#  35. CLAN RIVALRY MATRIX
#  Specific clan-vs-clan rivalries beyond faction lines
# ═══════════════════════════════════════════
murim_is_clan_rival_of_prev_trigger = {
	OR = {
		# Tang vs Poison Flower (both poison, opposite factions)
		AND = { murim_is_tang_clan_member_trigger = yes prev = { murim_is_poison_flower_disciple_trigger = yes } }
		AND = { murim_is_poison_flower_disciple_trigger = yes prev = { murim_is_tang_clan_member_trigger = yes } }
		# Mount Hua vs Iron Fist (sword vs fist philosophy)
		AND = { murim_is_mount_hua_disciple_trigger = yes prev = { murim_is_iron_fist_disciple_trigger = yes } }
		AND = { murim_is_iron_fist_disciple_trigger = yes prev = { murim_is_mount_hua_disciple_trigger = yes } }
		# Mount Hua vs Heavenly Demon (classic murim rivalry)
		AND = { murim_is_mount_hua_disciple_trigger = yes prev = { murim_is_heavenly_demon_member_trigger = yes } }
		AND = { murim_is_heavenly_demon_member_trigger = yes prev = { murim_is_mount_hua_disciple_trigger = yes } }
		# Shaolin vs Phantom Blade (honor vs assassination)
		AND = { murim_is_shaolin_monk_trigger = yes prev = { murim_is_phantom_blade_disciple_trigger = yes } }
		AND = { murim_is_phantom_blade_disciple_trigger = yes prev = { murim_is_shaolin_monk_trigger = yes } }
	}
}

# ═══════════════════════════════════════════
#  36. HEAVENLY DEMON SHADOW POLITICS ELIGIBILITY
#  Can the HD leader attempt to infiltrate a ruler?
# ═══════════════════════════════════════════
murim_hd_can_infiltrate_ruler_trigger = {
	# Target (ROOT) must be a ruler in China/Korea
	is_ruler = yes
	OR = {
		murim_is_in_china_trigger = yes
		murim_is_in_goryeo_trigger = yes
	}
	# Must have exploitable personality
	OR = {
		has_trait = ambitious
		has_trait = deceitful
		has_trait = arbitrary
		has_trait = murim_cruel
		has_trait = sadistic
		has_trait = greedy
	}
	# Not already infiltrated or aware
	NOR = {
		has_character_modifier = heavenly_demon_puppet
		has_character_flag = hd_infiltration_exposed
		has_character_flag = hd_infiltration_rejected
		# Orthodox clan members are immune
		murim_is_in_orthodox_sect_trigger = yes
	}
}
