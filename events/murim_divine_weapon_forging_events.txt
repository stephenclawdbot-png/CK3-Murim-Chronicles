# EVENT 0001 - Discovery of a Master Blacksmith
# Triggered: on_action yearly pulse, martial artists seeking power
# AI: Full weapon quest initiation
murim_weapon.0001 = {
    type = character_event
    title = murim_weapon.0001.t
    desc = murim_weapon.0001.desc
    theme = martial
    
    left_portrait = {
        character = root
        animation = thinking
    }
    right_portrait = {
        character = scope:master_smith
        animation = personality_bold
    }
    
    trigger = {
        is_alive = yes
        age >= 18
        prowess >= 14
        gold >= 200
        NOT = { has_character_flag = seeking_divine_weapon }
        NOT = { has_character_modifier = murim_divine_weapon_wielder }
        OR = {
            murim_is_martial_artist_trigger = yes
            has_trait = murim_sect_leader
            has_trait = murim_wandering_cultivator
        }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 2.0
            has_trait = ambitious
        }
        modifier = {
            factor = 1.5
            prowess >= 20
        }
        modifier = {
            factor = 2.0
            has_trait = murim_sect_leader
        }
    }
    
    immediate = {
        random_known_character = {
            limit = {
                is_alive = yes
                OR = {
                    has_trait = murim_master_blacksmith
                    stewardship >= 16
                }
                NOT = { this = root }
            }
            save_scope_as = master_smith
            alternative = {
                create_character = {
                    location = root.capital_province
                    age = { 45 70 }
                    gender_female_chance = 50
                    trait = murim_master_blacksmith
                    trait = patient
                    trait = diligent
                    stewardship = { 16 22 }
                    martial = { 12 18 }
                    culture = root.culture
                    faith = root.faith
                    save_scope_as = master_smith
                }
            }
        }
    }
    
    # Option A: Commission a divine weapon
    option = {
        name = murim_weapon.0001.a
        trigger = {
            exists = scope:master_smith
            gold >= 500
        }
        add_gold = -500
        add_character_flag = {
            flag = seeking_divine_weapon
            days = 1825
        }
        reverse_add_opinion = {
            target = scope:master_smith
            modifier = murim_patron_of_smithing
            opinion = 25
        }
        # Begin the forging quest
        trigger_event = {
            id = murim_weapon.0002
            days = { 30 90 }
        }
        ai_chance = {
            base = 45
            modifier = {
                factor = 2.0
                gold >= 800
            }
            modifier = {
                factor = 2.0
                has_trait = ambitious
            }
            modifier = {
                factor = 0.5
                gold < 600
            }
        }
    }
    
    # Option B: Seek an ancient weapon instead
    option = {
        name = murim_weapon.0001.b
        add_character_flag = {
            flag = seeking_divine_weapon
            days = 1825
        }
        add_character_modifier = {
            modifier = murim_ancient_weapon_seeker
            years = 3
        }
        trigger_event = {
            id = murim_weapon.0005
            days = { 60 180 }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                learning >= 14
            }
            modifier = {
                factor = 1.5
                has_trait = adventurer
            }
        }
    }
    
    # Option C: Strength comes from within, not weapons
    option = {
        name = murim_weapon.0001.c
        add_piety = 50
        add_prowess_skill = 1
        add_character_modifier = {
            modifier = murim_bare_fist_philosophy
            years = 5
        }
        ai_chance = {
            base = 25
            modifier = {
                factor = 3.0
                has_trait = murim_shaolin_elder
            }
            modifier = {
                factor = 2.0
                has_trait = murim_iron_fist_elder
            }
            modifier = {
                factor = 2.0
                has_trait = content
            }
        }
    }
}
