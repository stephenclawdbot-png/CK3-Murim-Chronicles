# === FACTION BALANCE INTEGRATION ===
murim_integration_update_faction_balance_effect = {
	# Recalculate orthodox vs unorthodox power balance
	# Based on total ranks of all living martial artists in each faction
	
	set_variable = { name = orthodox_total_power value = 0 }
	set_variable = { name = unorthodox_total_power value = 0 }
	
	every_living_character = {
		limit = {
			has_variable = murim_rank
			OR = {
				has_trait = shaolin_disciple
				has_trait = wudang_disciple
				has_trait = mount_hua_disciple
				has_trait = beggar_sect_member
				has_trait = iron_fist_gate_member
			}
		}
		root = {
			change_variable = { name = orthodox_total_power add = prev.var:murim_rank }
		}
	}
	
	every_living_character = {
		limit = {
			has_variable = murim_rank
			OR = {
				has_trait = heavenly_demon_cultist
				has_trait = tang_clan_member
				has_trait = poison_flower_member
				has_trait = phantom_blade_member
			}
		}
		root = {
			change_variable = { name = unorthodox_total_power add = prev.var:murim_rank }
		}
	}
	
	# Determine dominant faction
	if = {
		limit = {
			var:orthodox_total_power > var:unorthodox_total_power
			# Check if difference is significant (>20%)
		}
		set_global_variable = { name = murim_dominant_faction value = flag:orthodox }
		if = {
			limit = { NOT = { has_global_variable = murim_orthodox_dominant_notified } }
			set_global_variable = { name = murim_orthodox_dominant_notified value = yes }
			trigger_event = { id = murim_faction_war_integration.0001 days = 30 }
		}
	}
	else_if = {
		limit = {
			var:unorthodox_total_power > var:orthodox_total_power
		}
		set_global_variable = { name = murim_dominant_faction value = flag:unorthodox }
		if = {
			limit = { NOT = { has_global_variable = murim_unorthodox_dominant_notified } }
			set_global_variable = { name = murim_unorthodox_dominant_notified value = yes }
			trigger_event = { id = murim_faction_war_integration.0002 days = 30 }
		}
	}
	else = {
		set_global_variable = { name = murim_dominant_faction value = flag:balanced }
	}
}

# === TOURNAMENT-TO-RANKING INTEGRATION ===
murim_integration_tournament_rank_effect = {
	# After winning a tournament, check if rank should increase
	
	if = {
		limit = {
			has_variable = murim_tournament_victories
			var:murim_tournament_victories >= 3
			has_variable = murim_rank
			var:murim_rank < 5
		}
		# Three tournament wins = guaranteed rank 5 minimum
		set_variable = { name = murim_rank value = 5 }
		trigger_event = { id = murim_ranking_integration.0001 days = 1 }
	}
	
	if = {
		limit = {
			has_variable = murim_tournament_victories
			var:murim_tournament_victories >= 5
			has_variable = murim_rank
			var:murim_rank < 7
		}
		set_variable = { name = murim_rank value = 7 }
		trigger_event = { id = murim_ranking_integration.0002 days = 1 }
	}
	
	if = {
		limit = {
			has_variable = murim_grand_tournament_champion
			has_variable = murim_rank
			var:murim_rank < 8
		}
		set_variable = { name = murim_rank value = 8 }
		trigger_event = { id = murim_ranking_integration.0003 days = 1 }
		add_character_modifier = {
			modifier = grand_tournament_champion_fame
			years = 20
		}
	}
}

# === CALAMITY-TO-ALL-SYSTEMS INTEGRATION ===
murim_integration_calamity_start_effect = {
	# When Great Calamity begins, affect ALL systems
	
	# Disable tournaments during calamity
	set_global_variable = { name = murim_tournaments_suspended value = yes }
	
	# Force all faction wars to escalate
	set_global_variable = { name = murim_faction_war_escalated value = yes }
	
	# All shadow politics schemes accelerate
	set_global_variable = { name = murim_shadow_politics_overdrive value = yes }
	
	# Ranking system frozen (no peaceful advancement during calamity)
	set_global_variable = { name = murim_rankings_frozen value = yes }
	
	# All clans enter survival mode
	every_living_character = {
		limit = {
			has_variable = murim_clan_id
			has_variable = murim_rank
			var:murim_rank >= 5
		}
		add_character_modifier = {
			modifier = calamity_survival_instinct
			years = 50
		}
		trigger_event = { id = murim_calamity_ai_integration.0001 days = { 1 30 } }
	}
	
	# Clan leaders must choose: fight, hide, or betray
	every_living_character = {
		limit = {
			has_trait = sect_leader
			has_variable = murim_clan_id
		}
		trigger_event = { id = murim_calamity_ai_integration.0002 days = { 5 15 } }
	}
}

murim_integration_calamity_end_effect = {
	# When Great Calamity ends, restore and rebuild
	
	# Re-enable tournaments
	remove_global_variable = murim_tournaments_suspended
	
	# De-escalate faction wars
	remove_global_variable = murim_faction_war_escalated
	
	# Shadow politics return to normal
	remove_global_variable = murim_shadow_politics_overdrive
	
	# Unfreeze rankings
	remove_global_variable = murim_rankings_frozen
	
	# Survivors get special traits and bonuses
	every_living_character = {
		limit = {
			has_trait = calamity_survivor
			has_variable = murim_rank
		}
		change_variable = { name = murim_rank add = 1 }
		add_character_modifier = {
			modifier = calamity_hardened_veteran
			years = 30
		}
	}
	
	# Trigger rebuilding phase
	trigger_event = { id = murim_great_calamity_rebuild.0001 days = 30 }
	
	# Previous era becomes mythology
	set_global_variable = { name = murim_era_count value = 1 }
	if = {
		limit = { has_global_variable = murim_era_count }
		change_global_variable = { name = murim_era_count add = 1 }
	}
}

# === SHADOW POLITICS TO FACTION WAR INTEGRATION ===
murim_integration_shadow_to_faction_effect = {
	# Shadow politics outcomes can trigger faction wars
	
	if = {
		limit = {
			has_global_variable = heavenly_demon_puppet_emperor
			NOT = { has_global_variable = murim_puppet_war_triggered }
		}
		set_global_variable = { name = murim_puppet_war_triggered value = yes }
		
		# Orthodox alliance discovers the puppet emperor
		random_living_character = {
			limit = {
				has_trait = shaolin_disciple
				has_trait = sect_leader
			}
			trigger_event = { id = murim_faction_war_shadow.0001 days = { 30 90 } }
		}
		
		# Wudang seer has vision of corruption
		random_living_character = {
			limit = {
				has_trait = wudang_disciple
				var:murim_rank >= 7
			}
			trigger_event = { id = murim_faction_war_shadow.0002 days = { 15 60 } }
		}
	}
	
	# Custom clan political influence
	if = {
		limit = {
			has_global_variable = murim_custom_clan_exists
			has_global_variable = murim_custom_clan_faction
		}
		# Custom clan's faction allegiance affects balance
		if = {
			limit = { global_var:murim_custom_clan_faction = flag:orthodox }
			change_global_variable = { name = orthodox_political_power add = 5 }
		}
		else = {
			change_global_variable = { name = unorthodox_political_power add = 5 }
		}
	}
}

# === ROMANCE TO CLAN INTEGRATION ===
murim_integration_romance_clan_effect = {
	# Inter-clan romance creates political implications
	
	if = {
		limit = {
			exists = scope:romance_target
			has_variable = murim_clan_id
			scope:romance_target = { has_variable = murim_clan_id }
			NOT = { var:murim_clan_id = scope:romance_target.var:murim_clan_id }
		}
		
		# Different clan romance - political tension or alliance opportunity
		random_list = {
			40 = {
				# Clan leaders disapprove - forbidden love
				trigger_event = { id = murim_romance_integration.0001 days = 30 }
			}
			30 = {
				# Potential alliance through marriage
				trigger_event = { id = murim_romance_integration.0002 days = 30 }
			}
			20 = {
				# Spy accusation
				trigger_event = { id = murim_romance_integration.0003 days = 30 }
			}
			10 = {
				# Romeo and Juliet - clan war trigger
				trigger_event = { id = murim_romance_integration.0004 days = 30 }
			}
		}
		
		# Check if orthodox-unorthodox cross-faction romance
		if = {
			limit = {
				murim_is_orthodox_clan_trigger = yes
				scope:romance_target = { murim_is_unorthodox_clan_trigger = yes }
			}
			# Cross-faction forbidden love is MUCH more dangerous

				modifier = forbidden_cross_faction_love
				years = 5
			}
			trigger_event = { id = murim_romance_integration.0005 days = 15 }
		}
	}
}

# === MENTOR TO CULTIVATION INTEGRATION ===
murim_integration_mentor_cultivation_effect = {
	# Mentor relationship affects cultivation speed and technique learning
	
	if = {
		limit = {
			exists = scope:mentor
			has_variable = murim_rank
		}
		
		# Mentor's rank determines training effectiveness
		if = {
			limit = { scope:mentor = { var:murim_rank >= 8 } }
			add_character_modifier = {
				modifier = legendary_mentor_guidance
				years = 5
			}
			# Can learn mentor's clan techniques
			if = {
				limit = {
					scope:mentor = { has_trait = shaolin_disciple }
					NOT = { has_trait = shaolin_disciple }
				}
				trigger_event = { id = murim_mentor_integration.0001 days = 90 }
				# Learn external Shaolin technique
			}
			if = {
				limit = {
					scope:mentor = { has_trait = wudang_disciple }
					NOT = { has_trait = wudang_disciple }
				}
				trigger_event = { id = murim_mentor_integration.0002 days = 90 }
			}
			if = {
				limit = {
					scope:mentor = { has_trait = heavenly_demon_cultist }
					NOT = { has_trait = heavenly_demon_cultist }
				}
				trigger_event = { id = murim_mentor_integration.0003 days = 90 }
				# DANGER: learning demonic arts
			}
		}
		
		# Mentor death triggers revenge/inheritance arc
		if = {
			limit = {
				scope:mentor = { is_alive = no }
				NOT = { has_variable = murim_mentor_death_processed }
			}
			set_variable = { name = murim_mentor_death_processed value = yes }
			trigger_event = { id = murim_mentor_integration.0010 days = 1 }
			
			# Inherit mentor's technique mastery partially
			if = {
				limit = { scope:mentor = { has_variable = murim_technique_mastery } }
				set_variable = {
					name = murim_inherited_technique
					value = scope:mentor.var:murim_technique_mastery
				}
				change_variable = { name = murim_inherited_technique multiply = 0.5 }
			}
		}
	}
}

# === DAILY LIFE TO CULTIVATION INTEGRATION ===
murim_integration_daily_cultivation_effect = {
	# Daily activities affect cultivation progress
	
	if = {
		limit = { has_variable = murim_cultivation_progress }
		
		# Meditation increases progress
		if = {
			limit = { has_character_modifier = daily_meditation_routine }
			change_variable = { name = murim_cultivation_progress add = 2 }
		}
		
		# Training increases progress
		if = {
			limit = { has_character_modifier = daily_training_routine }
			change_variable = { name = murim_cultivation_progress add = 3 }
		}
		
		# Studying manuals
		if = {
			limit = { has_character_modifier = studying_martial_manual }
			change_variable = { name = murim_cultivation_progress add = 4 }
		}
		
		# Drinking/debauchery reduces progress
		if = {
			limit = { has_trait = drunkard }
			change_variable = { name = murim_cultivation_progress add = -2 }
		}
		
		# Check for rank up
		if = {
			limit = {
				var:murim_cultivation_progress >= 100
				has_variable = murim_rank
				var:murim_rank < 10
			}
			change_variable = { name = murim_rank add = 1 }
			set_variable = { name = murim_cultivation_progress value = 0 }
			trigger_event = { id = murim_ranking_integration.0010 days = 1 }
			murim_integration_clan_rank_update_effect = yes
		}
	}
}
