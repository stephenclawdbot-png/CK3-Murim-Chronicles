# Murim Chronicles - Sect/Clan Internal Politics Events
# 14 events covering promotions, succession, betrayal, harmony, discord
# INTEGRATION: Uses shared triggers from murim_triggers.txt:
#   murim_is_clan_member_trigger, murim_is_martial_artist_trigger
# INTEGRATION: Uses shared triggers from murim_clan_triggers.txt:
#   is_sect_leader, is_sect_elder, is_inner_disciple, is_outer_disciple
# INTEGRATION: Uses shared effects from murim_clan_membership_effects.txt:
#   promote_to_inner_disciple_effect, promote_to_elder_effect, etc.
# INTEGRATION: Uses harmony values from murim_harmony_values.txt
# Cross-refs:
#   murim_duel_events (formal leadership challenge duels)
#   murim_clan_war_events (sect splits cause wars)
#   murim_cultivation_events (betrayer gains demonic techniques)
#   murim_ranking_events (leader change affects rankings)
#   murim_politics_events (sect politics ripple to murim politics)
#   murim_romance_events (romantic entanglement betrayal)
#   murim_training_events (harmony bonus = training speed)
#   murim_mentor_events (elder trains inner disciples)
# GENDER: NO gender restrictions. All ranks open to all genders.
# AI BEHAVIOR: AI actively participates in all events. See ai_chance blocks.

namespace = murim_sect_politics

# murim_sect_politics.0001 - Outer Disciple Promotion Trial
# Triggered: yearly pulse, 15% chance for outer disciples with 2+ years tenure
# AI: Auto-attempts at 15% yearly if prowess >= 8
murim_sect_politics.0001 = {
	type = character_event
	title = murim_sect_politics.0001.t
	desc = murim_sect_politics.0001.desc
	theme = martial
	
	# NO gender restriction - open to all
	trigger = {
		murim_is_clan_member_trigger = yes
		has_character_modifier = sect_outer_disciple
		NOT = { has_character_modifier = sect_inner_disciple }
		NOT = { has_character_modifier = sect_elder }
		NOT = { has_character_modifier = sect_leader }
		age >= 14
		# Must have been outer disciple for at least 2 years
		has_character_modifier = sect_outer_disciple
		prowess >= 5
		is_alive = yes
		is_imprisoned = no
	}
	
	# Option A: Take the formal trial (prowess + martial skill check)
	option = {
		name = murim_sect_politics.0001.a
		# Prowess duel check against threshold
		duel = {
			skill = prowess
			value = 10
			# Win: promoted to inner disciple
			50 = {
				desc = murim_sect_politics.0001.a.success
				trigger = {
					prowess >= 10
				}
				send_interface_toast = {
					title = murim_sect_politics.0001.a.success.toast
				}
				remove_character_modifier = sect_outer_disciple
				add_character_modifier = {
					modifier = sect_inner_disciple
				}
				add_prestige = 200
				# Cross-ref: notify sect leader
				if = {
					limit = {
						exists = scope:sect_leader
					}
					scope:sect_leader = {
						trigger_event = {
							id = murim_sect_politics.0001
							days = { 1 3 }
						}
					}
				}
				stress_impact = {
					ambitious = medium_stress_loss
					humble = minor_stress_gain
				}
			}
			# Fail: stay outer disciple, cooldown
			50 = {
				desc = murim_sect_politics.0001.a.fail
				trigger = {
					prowess < 10
				}
				add_character_modifier = {
					modifier = promotion_trial_cooldown
					years = 2
				}
				add_prestige = -50
				stress_impact = {
					ambitious = major_stress_gain
					content = minor_stress_loss
				}
			}
		}
		ai_chance = {
			base = 70
			modifier = {
				add = 30
				has_trait = ambitious
			}
			modifier = {
				add = -40
				has_trait = content
			}
			modifier = {
				add = 20
				prowess >= 12
			}
		}
	}
	
	# Option B: Ask a senior to vouch for you (diplomacy check)
	option = {
		name = murim_sect_politics.0001.b
		# Diplomacy check - social promotion
		random_list = {
			60 = {
				trigger = { diplomacy >= 10 }
				remove_character_modifier = sect_outer_disciple
				add_character_modifier = {
					modifier = sect_inner_disciple
				}
				add_prestige = 150
				# Some inner disciples resent political promotion
				random_list = {
					40 = {
						# Resentment from peers
						add_character_modifier = {
							modifier = politically_promoted_stigma
							years = 3
						}
					}
					60 = { }
				}
			}
			40 = {
				trigger = { diplomacy < 10 }
				add_prestige = -100
				add_character_modifier = {
					modifier = promotion_trial_cooldown
					years = 3
				}
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 40
				diplomacy >= 12
			}
			modifier = {
				add = -20
				has_trait = honest
			}
		}
	}
	
	# Option C: Decline - not ready yet (patient/humble response)
	option = {
		name = murim_sect_politics.0001.c
		add_prestige = 25
		# Earns respect from elders for humility
		add_character_modifier = {
			modifier = humble_disciple_reputation
			years = 2
		}
		stress_impact = {
			ambitious = major_stress_gain
			patient = medium_stress_loss
			humble = medium_stress_loss
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 30
				has_trait = patient
			}
			modifier = {
				add = 20
				has_trait = humble
			}
		}
	}
	
	# Option D: Challenge the strongest inner disciple directly (risky, bold)
	option = {
		name = murim_sect_politics.0001.d
		# Must have high prowess to even attempt
		trigger = { prowess >= 12 }
		# Cross-ref: fires a duel event
		save_scope_as = challenger
		# Find strongest inner disciple
		random_list = {
			50 = {
				trigger = { prowess >= 15 }
				# Decisive victory - promoted AND gain reputation
				remove_character_modifier = sect_outer_disciple
				add_character_modifier = {
					modifier = sect_inner_disciple
				}
				add_prestige = 300
				add_character_modifier = {
					modifier = bold_promotion_fame
					years = 5
				}
				# Cross-ref: duel event creates fight memory
				# Uses shared effect: create_fight_memory_effect
			}
			30 = {
				# Narrow victory - promoted but made an enemy
				remove_character_modifier = sect_outer_disciple
				add_character_modifier = {
					modifier = sect_inner_disciple
				}
				add_prestige = 200
				# The defeated inner disciple becomes a rival
			}
			20 = {
				# Defeat - humiliation, demotion risk
				add_prestige = -200
				add_character_modifier = {
					modifier = promotion_trial_cooldown
					years = 5
				}
				# Cross-ref: injury roll
				random_list = {
					30 = {
						add_trait = wounded_1
					}
					70 = { }
				}
			}
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 35
				has_trait = ambitious
			}
			modifier = {
				add = 20
				has_trait = brave
			}
			modifier = {
				add = -30
				has_trait = craven
			}
		}
	}
}

# murim_sect_politics.0002 - Elder Nomination

	type = character_event
	title = murim_sect_politics.0002.t
	desc = murim_sect_politics.0002.desc
	theme = martial
	
	trigger = {
		murim_is_clan_member_trigger = yes
		has_character_modifier = sect_inner_disciple
		NOT = { has_character_modifier = sect_elder }
		age >= 30
		prowess >= 15
		martial >= 10
		is_alive = yes
		is_imprisoned = no
	}
	
	# Option A: Accept nomination graciously
	option = {
		name = murim_sect_politics.0002.a

		random_list = {
			60 = {
				desc = murim_sect_politics.0002.a.approved

					modifier = sect_elder
				}
				add_prestige = 500

		name = murim_sect_politics.0002.b

		add_piety = 50
		add_character_modifier = {
			modifier = selfless_disciple_reputation
			years = 5
		}
		stress_impact = {
			ambitious = major_stress_gain
			humble = medium_stress_loss
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 25
				has_trait = humble
			}
		}
	}
	
	# Option C: Demand immediate promotion - you've earned it (aggressive)
	option = {
		name = murim_sect_politics.0002.c
		trigger = { has_trait = ambitious }
		random_list = {
			40 = {
				# Forceful but effective - promoted with some resentment
				remove_character_modifier = sect_inner_disciple
				add_character_modifier = {
					modifier = sect_elder
				}
				add_prestige = 400
				add_character_modifier = {
					modifier = forced_promotion_resentment
					years = 5
				}
			}
			60 = {
				# Backfires - seen as arrogant
				add_prestige = -200
				add_character_modifier = {
					modifier = arrogant_disciple_mark
					years = 3
				}
			}
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = arrogant
			}
		}
	}
}
