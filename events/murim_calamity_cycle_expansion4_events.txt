murim_calamity_cycle_expansion4.0001 = {
	type = character_event
	title = murim_calamity_cycle_expansion4.0001.t
	desc = murim_calamity_cycle_expansion4.0001.desc
	theme = death
	
	# THE BLOOD TIDE RISES — All 9 clans begin autonomous war declarations
	# Triggered by on_action when calamity intensity > 80
	
	trigger = {
		has_variable = murim_great_calamity_active
		var:murim_calamity_intensity >= 80
		NOT = { has_character_flag = calamity_blood_tide_triggered }
	}
	
	immediate = {
		set_character_flag = {
			flag = calamity_blood_tide_triggered
			days = 3650
		}
		# Every clan leader declares war on nearest rival
		every_ruler = {
			limit = {
				has_trait = clan_leader
				is_alive = yes
				NOT = { has_character_flag = calamity_war_declared }
			}
			set_character_flag = {
				flag = calamity_war_declared
				days = 1825
			}
			# Find nearest rival clan leader
			random_living_character = {
				limit = {
					has_trait = clan_leader
					is_alive = yes
					NOT = { this = prev }
					NOT = { has_opinion_modifier = { modifier = murim_sworn_brother_opinion target = prev } }
				}
				weight = {
					base = 10
					modifier = {
						add = 20
						has_relation_rival = prev
					}
					modifier = {
						add = 15
						has_trait = unorthodox_faction
						prev = { has_trait = orthodox_faction }
					}
					modifier = {
						add = 15
						has_trait = orthodox_faction
						prev = { has_trait = unorthodox_faction }
					}
				}
				save_scope_as = war_target_leader
			}
			trigger_event = {
				id = murim_calamity_cycle_expansion4.0002
				days = { 3 15 }
			}
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0001.a
		# The blood tide is upon us — all clans go to war
		custom_tooltip = murim_blood_tide_all_clans_war_tt
		add_character_modifier = {
			modifier = murim_blood_tide_witness
			years = 20
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0001.b
		# Try to unite clans against the calamity
		trigger = {
			has_trait = clan_leader
			diplomacy >= 20
		}
		custom_tooltip = murim_unite_against_calamity_tt
		trigger_event = {
			id = murim_calamity_cycle_expansion4.0005
			days = { 10 30 }
		}
	}
}

murim_calamity_cycle_expansion4.0002 = {
	type = character_event
	title = murim_calamity_cycle_expansion4.0002.t
	desc = murim_calamity_cycle_expansion4.0002.desc
	theme = war
	
	# AI AUTONOMOUS WAR DECLARATION — Clan leader attacks rival
	# This fires for each clan leader individually
	
	trigger = {
		is_alive = yes
		has_trait = clan_leader
		exists = scope:war_target_leader
	}
	
	immediate = {
		scope:war_target_leader = {
			save_scope_as = enemy_leader
		}
		# Calculate war outcome based on martial + clan strength
		set_variable = {
			name = calamity_war_power
			value = {
				value = martial
				add = prowess
				if = {
					limit = { has_trait = grandmaster_rank }
					add = 30
				}
				if = {
					limit = { has_trait = master_rank }
					add = 20
				}
				if = {
					limit = { has_trait = expert_rank }
					add = 10
				}
			}
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0002.a
		# Launch full assault on enemy clan
		custom_tooltip = murim_calamity_war_assault_tt
		duel = {
			skill = martial
			target = scope:enemy_leader
			50 = {
				desc = murim_calamity_war_victory
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2
				}
				# Victory — enemy clan devastated
				scope:enemy_leader = {
					add_character_modifier = {
						modifier = murim_clan_devastated
						years = 10
					}
					# Kill 70-90% of enemy clan members
					every_courtier = {
						limit = {
							has_any_murim_clan_trait_trigger = yes
						}
						random = {
							chance = 80
							death = {
								death_reason = death_battle
								killer = prev.prev
							}
						}
					}
				}
				add_prestige = 500
				add_character_modifier = {
					modifier = murim_calamity_war_victor
					years = 5
				}
			}
			50 = {
				desc = murim_calamity_war_defeat
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2
				}
				# Defeat — your clan devastated
				add_character_modifier = {
					modifier = murim_clan_devastated
					years = 10
				}
				every_courtier = {
					limit = {
						has_any_murim_clan_trait_trigger = yes
					}
					random = {
						chance = 75
						death = {
							death_reason = death_battle
							killer = scope:enemy_leader
						}
					}
				}
			}
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0002.b
		# Challenge enemy leader to single combat instead of clan war
		trigger = {
			prowess >= 25
		}
		custom_tooltip = murim_calamity_single_combat_tt
		trigger_event = {
			id = murim_calamity_cycle_expansion4.0003
			days = { 1 5 }
		}
	}
}

murim_calamity_cycle_expansion4.0004 = {
	type = character_event
	title = murim_calamity_cycle_expansion4.0004.t
	desc = murim_calamity_cycle_expansion4.0004.desc
	theme = death
	
	# EMPEROR ASSASSINATION — Heavenly Demon or unorthodox faction kills the emperor
	# AI autonomous — fires when calamity intensity hits 90+
	
	trigger = {
		has_variable = murim_great_calamity_active
		var:murim_calamity_intensity >= 90
		NOT = { has_global_variable = calamity_emperor_fallen }
	}
	
	immediate = {
		set_global_variable = {
			name = calamity_emperor_fallen
			value = yes
		}
		# Find the current emperor/murim overlord
		random_living_character = {
			limit = {
				OR = {
					has_trait = murim_emperor
					has_title = title:e_china_murim
				}
			}
			save_scope_as = doomed_emperor
		}
		# Find the assassin — Heavenly Demon preferred
		random_living_character = {
			limit = {
				OR = {
					has_trait = heavenly_demon
					AND = {
						has_trait = unorthodox_faction
						has_trait = grandmaster_rank
					}
				}
				is_alive = yes
			}
			weight = {
				base = 10
				modifier = {
					add = 50
					has_trait = heavenly_demon
				}
			}
			save_scope_as = emperor_assassin
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0004.a
		# The emperor falls — murim descends into total chaos
		custom_tooltip = murim_emperor_assassination_tt
		if = {
			limit = {
				exists = scope:doomed_emperor
				exists = scope:emperor_assassin
			}
			scope:doomed_emperor = {
				death = {
					death_reason = death_murder
					killer = scope:emperor_assassin
				}
			}
			scope:emperor_assassin = {
				add_prestige = 2000
				add_character_modifier = {
					modifier = murim_emperor_slayer
					years = 50
				}
			}
		}
		# All martial world loses stability
		every_ruler = {
			limit = {
				OR = {
					has_trait = orthodox_faction
					has_trait = unorthodox_faction
				}
			}
			add_stress = 30
			add_character_modifier = {
				modifier = murim_no_emperor_chaos
				years = 10
			}
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0004.b
		# Try to protect the emperor (orthodox leaders only)
		trigger = {
			has_trait = orthodox_faction
			has_trait = clan_leader
		}
		custom_tooltip = murim_protect_emperor_tt
		trigger_event = {
			id = murim_calamity_cycle_expansion4.0006
			days = { 1 3 }
		}
	}
}

			}
			weight = {

				}

					has_trait = deceitful
				}
				modifier = {
					add = intrigue
				}
			}
			save_scope_as = ceasefire_betrayer
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0008.a
		# The ceasefire is shattered
		custom_tooltip = murim_ceasefire_betrayed_tt
		remove_global_variable = calamity_ceasefire_active

			limit = {
				exists = scope:ceasefire_betrayer
			}
			scope:ceasefire_betrayer = {
				add_trait = murim_great_betrayer
				add_prestige = 500
				# The betrayer attacks during the summit
				every_ruler = {
					limit = {
						has_trait = clan_leader
						has_character_modifier = murim_ceasefire_participant
						NOT = { this = scope:ceasefire_betrayer }
					}
					random = {
						chance = 30
						death = {
							death_reason = death_murder
							killer = scope:ceasefire_betrayer
						}
					}
				}
			}
		}
		# Calamity intensifies further
		change_global_variable = {
			name = murim_calamity_intensity
			add = 15
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion4.0008.b
		# Expose the betrayer before they strike
		trigger = {
			intrigue >= 18
		}
		custom_tooltip = murim_expose_betrayer_tt
