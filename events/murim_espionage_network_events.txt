# EVENT 0001 - Establish Spy Network in Rival Clan
# Triggered: on_action yearly pulse, sect leaders with intrigue >= 12
# AI: Full spy placement with weighted decisions
murim_espionage.0001 = {
    type = character_event
    title = murim_espionage.0001.t
    desc = murim_espionage.0001.desc
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = scheme
    }
    right_portrait = {
        character = scope:spy_candidate
        animation = personality_rational
    }
    
    trigger = {
        is_alive = yes
        intrigue >= 12
        OR = {
            has_trait = murim_sect_leader
            has_trait = murim_shadow_operative
            has_trait = murim_unorthodox_elder
        }
        NOT = { has_character_flag = placed_spy_recently }
        any_living_character = {
            is_alive = yes
            NOT = { this = root }
            NOT = { is_allied_to = root }
            OR = {
                has_trait = murim_sect_leader
                murim_is_martial_artist_trigger = yes
            }
        }
        gold >= 100
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 2.0
            has_trait = paranoid
        }
        modifier = {
            factor = 1.5
            intrigue >= 16
        }
        modifier = {
            factor = 2.0
            has_trait = murim_shadow_operative
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = placed_spy_recently
            days = 365
        }
        # Find a spy candidate from courtiers
        random_courtier = {
            limit = {
                intrigue >= 10
                NOT = { has_trait = honest }
                opinion = { target = root value >= 0 }
            }
            save_scope_as = spy_candidate
            alternative = {
                create_character = {
                    location = root.capital_province
                    age = { 22 35 }
                    gender_female_chance = 50
                    trait = murim_shadow_operative
                    trait = deceitful
                    trait = patient
                    intrigue = { 12 20 }
                    culture = root.culture
                    faith = root.faith
                    save_scope_as = spy_candidate
                }
            }
        }
        # Find target clan/character
        random_living_character = {
            limit = {
                is_alive = yes
                NOT = { this = root }
                NOT = { is_allied_to = root }
                OR = {
                    has_trait = murim_sect_leader
                    has_trait = murim_shaolin_elder
                    has_trait = murim_wudang_elder
                    has_trait = murim_mount_hua_elder
                    has_trait = murim_tang_clan_elder
                    has_trait = murim_heavens_sword_elder
                    has_trait = murim_namgung_elder
                    has_trait = murim_iron_fist_elder
                    has_trait = murim_poison_flower_elder
                    has_trait = murim_phantom_blade_elder
                }
            }
            save_scope_as = espionage_target
        }
    }
    
    # Option A: Plant deep cover spy
    option = {
        name = murim_espionage.0001.a
        trigger = {
            exists = scope:spy_candidate
            exists = scope:espionage_target
        }
        add_gold = -200
        hidden_effect = {
            scope:spy_candidate = {
                add_character_flag = {
                    flag = murim_deep_cover_spy
                    days = 3650
                }
                set_variable = {
                    name = spy_master
                    value = root
                }
            }
        }
        add_character_modifier = {
            modifier = murim_spy_network_active
            years = 10
        }
        # Delayed intelligence reports
        trigger_event = {
            id = murim_espionage.0002
            days = { 90 365 }
        }
        stress_impact = {
            honest = medium_stress_impact_gain
        }
        ai_chance = {
            base = 50
            modifier = {
                factor = 2.0
                has_trait = paranoid
            }
            modifier = {
                factor = 2.0
                intrigue >= 18
            }
            modifier = {
                factor = 0.3
                has_trait = honest
            }
            modifier = {
                factor = 1.5
                has_trait = ambitious
            }
        }
    }
    
    # Option B: Plant saboteur instead
    option = {
        name = murim_espionage.0001.b
        trigger = {
            exists = scope:spy_candidate
            exists = scope:espionage_target
        }
        add_gold = -150
        hidden_effect = {
            scope:spy_candidate = {
                add_character_flag = {
                    flag = murim_saboteur_agent
                    days = 1825
                }
                set_variable = {
                    name = spy_master
                    value = root
                }
            }
        }
        add_character_modifier = {
            modifier = murim_saboteur_deployed
            years = 5
        }
        trigger_event = {
            id = murim_espionage.0004
            days = { 60 240 }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                has_trait = sadistic
            }
            modifier = {
                factor = 1.5
                has_trait = wrathful
            }
        }
    }
    
    # Option C: Use diplomatic channels instead
    option = {
        name = murim_espionage.0001.c
        add_diplomacy_skill = 1
        add_prestige = 50
        stress_impact = {
            deceitful = minor_stress_impact_gain
        }
        ai_chance = {
            base = 20
            modifier = {
                factor = 3.0
                has_trait = honest
            }
            modifier = {
                factor = 2.0
                diplomacy >= 16
            }
        }
    }
}

# EVENT 0003 - Spy Discovered! Mole Hunt Begins
# Triggered: on_action yearly for characters with spies in their court
# AI: Full counter-espionage response
murim_espionage.0003 = {
    type = character_event
    title = murim_espionage.0003.t
    desc = murim_espionage.0003.desc
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = anger
    }
    right_portrait = {
        character = scope:suspected_spy
        animation = fear
    }
    
    trigger = {
        is_alive = yes
        OR = {
            has_trait = murim_sect_leader
            murim_is_martial_artist_trigger = yes
        }
        any_courtier = {
            has_character_flag = murim_deep_cover_spy
        }
        intrigue >= 10
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 3.0
            has_trait = paranoid
        }
        modifier = {
            factor = 2.0
            intrigue >= 18
        }
    }
    
    immediate = {
        random_courtier = {
            limit = {
                has_character_flag = murim_deep_cover_spy
            }
            save_scope_as = actual_spy
        }
        # May or may not correctly identify the spy
        random_list = {
            50 = {
                modifier = {
                    factor = 1.5
                    intrigue >= 16
                }
                # Correctly identified
                if = {
                    limit = { exists = scope:actual_spy }
                    scope:actual_spy = {
                        save_scope_as = suspected_spy
                    }
                }
                set_variable = { name = correct_identification value = 1 }
            }
            50 = {
                # Wrong person suspected
                random_courtier = {
                    limit = {
                        NOT = { has_character_flag = murim_deep_cover_spy }
                        is_alive = yes
                    }
                    save_scope_as = suspected_spy
                }
                set_variable = { name = correct_identification value = 0 }
            }
        }
    }
    
    # Option A: Interrogate the suspect
    option = {
        name = murim_espionage.0003.a
        trigger = {
            exists = scope:suspected_spy
        }
        # Interrogation skill check
        random_list = {
            50 = {
                modifier = {
                    factor = 1.5
                    intrigue >= 16
                }
                modifier = {
                    factor = 1.5
                    has_trait = sadistic
                }
                desc = murim_espionage.0003.a.confession
                if = {
                    limit = {
                        has_variable = correct_identification
                    }
                    # If correct spy, get useful info
                    add_prestige = 200
                    add_character_modifier = {
                        modifier = murim_counter_spy_success
                        years = 3
                    }
                    scope:suspected_spy = {
                        remove_character_flag = murim_deep_cover_spy
                        imprison = {
                            target = scope:suspected_spy
                            type = dungeon
                        }
                    }
                }
                if = {
                    limit = {
                        NOT = { has_variable = correct_identification }
                    }
                    # False confession from innocent
                    scope:suspected_spy = {
                        add_opinion = {
                            target = root
                            modifier = murim_falsely_accused
                            opinion = -60
                        }
                        imprison = {
                            target = scope:suspected_spy
                            type = dungeon
                        }
                    }
                    # Real spy still active
                }
            }
            50 = {
                desc = murim_espionage.0003.a.silence
                add_stress = 15
                add_dread = 10
                # Suspect reveals nothing
            }
        }
        ai_chance = {
            base = 50
            modifier = {
                factor = 2.0
                has_trait = sadistic
            }
            modifier = {
                factor = 1.5
                intrigue >= 16
            }
        }
    }
    
    # Option B: Set a trap with false information
    option = {
        name = murim_espionage.0003.b
        trigger = {
            intrigue >= 14
        }
        add_character_modifier = {
            modifier = murim_counter_intelligence_trap
            years = 2
        }
        # Feed false intel through suspected channel
        trigger_event = {
            id = murim_espionage.0005
            days = { 30 90 }
        }
        ai_chance = {
            base = 35
            modifier = {
                factor = 2.0
                intrigue >= 18
            }
            modifier = {
                factor = 2.0
                has_trait = patient
            }
        }
    }
    
    # Option C: Execute suspect immediately
    option = {
        name = murim_espionage.0003.c
        trigger = {
            exists = scope:suspected_spy
        }
        scope:suspected_spy = {
            death = {
                death_reason = death_execution
                killer = root
            }
        }
        add_dread = 30
        add_tyranny = 10
        every_courtier = {
            limit = { is_alive = yes }
            add_opinion = {
                target = root
                modifier = murim_ruthless_execution
                opinion = -15
            }
        }
        # If wrong person, real spy gets scared
        if = {
            limit = {
                exists = scope:actual_spy
                NOT = { scope:actual_spy = scope:suspected_spy }
            }
            scope:actual_spy = {
                add_character_flag = {
                    flag = murim_spy_frightened
                    days = 365
                }
            }
        }
        ai_chance = {
            base = 15
            modifier = {
                factor = 3.0
                has_trait = sadistic
            }
            modifier = {
                factor = 2.0
                has_trait = wrathful
            }
            modifier = {
                factor = 0.2
                has_trait = compassionate
            }
        }
    }
    
    # Option D: Ignore suspicions
    option = {
        name = murim_espionage.0003.d
        add_stress = 10
        stress_impact = {
            paranoid = major_stress_impact_gain
        }
        ai_chance = {
            base = 10
            modifier = {
                factor = 2.0
                has_trait = trusting
            }
            modifier = {
                factor = 0.1
                has_trait = paranoid
            }
        }
    }
}
