############################################
# DARK QI CORRUPTION SPREAD
# Heavenly Demon practitioner's dark qi
# begins corrupting those around them.
############################################
murim_demon_exp.0001 = {
	type = character_event
	title = murim_demon_exp.0001.t
	desc = murim_demon_exp.0001.desc
	theme = mystic
	left_portrait = {
		character = root
		animation = personality_vengeful
	}

	trigger = {
		has_trait = heavenly_demon_disciple
		murim_is_martial_artist_trigger = yes
		has_variable = murim_clan_rank
		var:murim_clan_rank >= 3
		prowess >= 16
		NOT = { has_character_modifier = murim_dark_qi_spread_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = cultivation_stage_4 }
		modifier = { add = 2 prowess >= 20 }
		modifier = { add = 1 dread >= 40 }
	}

	immediate = {
		add_character_modifier = { modifier = murim_dark_qi_spread_cooldown years = 5 }
		# Find someone nearby being corrupted
		random_living_character = {
			limit = {
				has_trait = martial_artist
				is_alive = yes
				NOT = { has_trait = heavenly_demon_disciple }
				NOT = { this = root }
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = tang_clan_member
					has_trait = beggar_sect_member
					has_trait = iron_fist_disciple
					has_trait = poison_flower_disciple
					has_trait = phantom_blade_disciple
				}
			}
			weight = {
				base = 10
				modifier = { add = 10 age <= 25 }
				modifier = { add = 5 has_trait = ambitious }
				modifier = { add = 5 has_trait = wrathful }
			}
			save_scope_as = corruption_target
		}
	}

	# Actively corrupt them - recruit to the dark path
	option = {
		name = murim_demon_exp.0001.a
		trigger = { exists = scope:corruption_target }
		random_list = {
			40 = {
				modifier = { add = 15 intrigue >= 16 }
				modifier = { add = 10 scope:corruption_target = { has_trait = ambitious } }
				desc = murim_demon_exp.0001.a.success
				# Target is corrupted - joins Heavenly Demon
				scope:corruption_target = {
					# Remove old clan trait
					if = { limit = { has_trait = mount_hua_disciple } remove_trait = mount_hua_disciple }
					if = { limit = { has_trait = shaolin_monk } remove_trait = shaolin_monk }
					if = { limit = { has_trait = wudang_disciple } remove_trait = wudang_disciple }
					if = { limit = { has_trait = tang_clan_member } remove_trait = tang_clan_member }
					if = { limit = { has_trait = beggar_sect_member } remove_trait = beggar_sect_member }
					if = { limit = { has_trait = iron_fist_disciple } remove_trait = iron_fist_disciple }
					if = { limit = { has_trait = poison_flower_disciple } remove_trait = poison_flower_disciple }
					if = { limit = { has_trait = phantom_blade_disciple } remove_trait = phantom_blade_disciple }
					add_trait = heavenly_demon_disciple
					set_variable = { name = murim_faction value = flag:unorthodox }
					set_variable = { name = murim_clan_rank value = 1 }
					add_opinion = { target = root modifier = respect_martial_prowess opinion = 20 }
				}
				add_prestige = 200
				add_dread = 15
				# Cross-ref: faction war escalation
				trigger_event = { id = murim_faction_war.0002 days = { 30 60 } }
			}
			35 = {
				desc = murim_demon_exp.0001.a.resist
				# Target resists but is weakened
				scope:corruption_target = {
					add_stress = 30
					add_character_modifier = { modifier = murim_dark_qi_tainted years = 3 }
					add_opinion = { target = root modifier = insult_opinion opinion = -30 }
				}
				add_prestige = 50
			}
			25 = {
				desc = murim_demon_exp.0001.a.backfire
				# Backfire - dark qi rebounds
				add_stress = 30
				add_trait = wounded_1
				add_prestige = -100
				scope:corruption_target = {
					add_opinion = { target = root modifier = insult_opinion opinion = -50 }
					# They report you to their clan
					trigger_event = { id = murim_faction_war.0004 days = { 5 15 } }
				}
			}
		}
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:dark_qi_corruption_attempted }
		}
		stress_impact = { compassionate = major_stress_impact_gain sadistic = minor_stress_impact_loss }
		ai_chance = { base = 40 modifier = { add = 20 has_trait = sadistic } }
	}

	# Contain the dark qi - prevent uncontrolled spread
	option = {
		name = murim_demon_exp.0001.b
		add_prowess_skill = 1
		add_stress = 15
		add_character_modifier = { modifier = murim_dark_qi_contained years = 5 }
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:dark_qi_contained }
		}
		ai_chance = { base = 40 modifier = { add = 20 has_trait = patient } }
	}

	# Embrace the chaos - let it spread freely
	option = {
		name = murim_demon_exp.0001.c
		add_dread = 25
		add_prowess_skill = 2
		add_stress = -10
		add_character_modifier = { modifier = murim_dark_qi_unleashed years = 3 }
		# Multiple people tainted
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:dark_qi_chaos_unleashed }
		}
		stress_impact = { compassionate = major_stress_impact_gain sadistic = minor_stress_impact_loss }
		ai_chance = { base = 20 modifier = { add = 30 has_trait = sadistic } }
	}
}

				base = 10
				modifier = { add = 20 prowess >= 20 }

			save_scope_as = demon_rival
		}
	}

	# Challenge for the throne through ritual combat
	option = {
		name = murim_demon_exp.0002.a
		trigger = { exists = scope:demon_rival }
		duel = {
			skill = prowess
			target = scope:demon_rival
			50 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_demon_exp.0002.a.crowned
				add_prestige = 1000
				add_dread = 30
				add_prowess_skill = 3
				set_variable = { name = murim_clan_rank value = 5 }
				add_character_modifier = { modifier = murim_heavenly_demon_lord years = -1 }
				scope:demon_rival = {
					random_list = {
						40 = { add_trait = wounded_2 }
						30 = { add_trait = maimed }
						30 = { death = { death_reason = death_duel killer = root } }
					}
				}
				# Cross-ref: shadow politics - new demon lord claims puppet emperor
				trigger_event = { id = murim_shadow.0001 days = { 30 60 } }
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:demon_lord_coronation_victory }
				}
			}
			50 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:demon_rival }
				desc = murim_demon_exp.0002.a.defeated
				add_prestige = -300
				random_list = {
					30 = { add_trait = wounded_2 }
					30 = { add_trait = maimed }
					20 = { death = { death_reason = death_duel killer = scope:demon_rival } }
					20 = {
						add_character_modifier = { modifier = murim_demon_lord_exile years = 10 }
						change_variable = { name = murim_clan_rank add = -2 }
					}
				}
				scope:demon_rival = {
					set_variable = { name = murim_clan_rank value = 5 }
					add_character_modifier = { modifier = murim_heavenly_demon_lord years = -1 }
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:demon_lord_coronation_defeat }
				}
			}
		}
		stress_impact = { brave = minor_stress_impact_loss craven = major_stress_impact_gain }
		ai_chance = { base = 60 modifier = { add = 20 has_trait = ambitious } }
	}

	# Political maneuvering - secure votes from elders
	option = {
		name = murim_demon_exp.0002.b
		random_list = {
			50 = {
				modifier = { add = 20 diplomacy >= 16 }
				modifier = { add = 15 intrigue >= 16 }
				desc = murim_demon_exp.0002.b.political_win
				add_prestige = 700
				add_dread = 20
				set_variable = { name = murim_clan_rank value = 5 }
				add_character_modifier = { modifier = murim_heavenly_demon_lord years = -1 }
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:demon_lord_political_coronation }
				}
			}

				modifier = { add = 15 scope:demon_rival = { diplomacy >= 14 } }

				add_prestige = -200

					set_variable = { name = murim_clan_rank value = 5 }

				}

					add_to_variable_list = { name = murim_combat_memories target = flag:demon_lord_political_defeat }

			}
		}
		stress_impact = { ambitious = minor_stress_impact_loss content = medium_stress_impact_gain }
		ai_chance = { base = 30 modifier = { add = 20 diplomacy >= 14 } }

			40 = {
				desc = murim_demon_exp.0005.b.failed
				add_prestige = -150
				# Betrayer warns Orthodox - cross-ref faction war
				trigger_event = { id = murim_faction_war.0007 days = { 10 30 } }
			}
		}
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:betrayer_assassination_ordered }
		}
		stress_impact = { compassionate = major_stress_impact_gain sadistic = minor_stress_impact_loss }
		ai_chance = { base = 30 modifier = { add = 20 intrigue >= 14 } }
	}

	# Let them go - the sect is stronger without traitors
	option = {
		name = murim_demon_exp.0005.c
		add_prestige = -50
		add_stress = 20
		if = {
			limit = { exists = scope:sect_betrayer }
			scope:sect_betrayer = {
				remove_trait = heavenly_demon_disciple
				add_trait = shaolin_monk
				set_variable = { name = murim_faction value = flag:orthodox }
			}
		}
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:betrayer_released }
		}
		ai_chance = { base = 20 modifier = { add = 20 has_trait = compassionate } }
	}
}
