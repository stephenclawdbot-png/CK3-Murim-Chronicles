### EVENT 0001 — The Heavenly Demon's Imperial Ambition
# The Heavenly Demon Sect plots to install a puppet on the imperial throne
murim_shadow_politics_expansion2.0001 = {
	type = character_event
	title = murim_shadow_politics_expansion2.0001.t
	desc = murim_shadow_politics_expansion2.0001.desc
	theme = intrigue
	
	left_portrait = {
		character = root
		animation = scheme

		has_trait = murim_heavenly_demon_clan

			id = murim_shadow_politics_expansion2.0003
			days = { 365 730 }
		}
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = patient }
		}
	}
	
	# Option C: Focus on martial power — the throne is irrelevant
	option = {
		name = murim_shadow_politics_expansion2.0001.c

### EVENT 0002 — Selecting the Puppet Candidate
# Choose which imperial candidate to control
murim_shadow_politics_expansion2.0002 = {
	type = character_event
	title = murim_shadow_politics_expansion2.0002.t
	desc = murim_shadow_politics_expansion2.0002.desc
	theme = intrigue
	
	left_portrait = {
		character = root
		animation = thinking
	}
	right_portrait = {
		character = scope:puppet_candidate
		animation = personality_compassionate
	}
	
	trigger = {
		has_character_flag = murim_shadow_throne_active
		is_alive = yes
	}
	
	immediate = {
		# Find a weak-willed ruler or heir
		random_living_character = {
			limit = {
				OR = {
					is_ruler = yes
					any_parent = { is_ruler = yes }
				}
				NOT = { this = root }
				NOT = { has_trait = murim_heavenly_demon_clan }
				age >= 16
			}
			weight = {
				base = 1
				modifier = { add = 10 has_trait = craven }
				modifier = { add = 5 has_trait = lazy }
				modifier = { add = 5 diplomacy < 8 }
				modifier = { add = -5 has_trait = brave }
				modifier = { add = -5 has_trait = ambitious }
			}
			save_scope_as = puppet_candidate
		}
	}
	
	# Option A: Approach through blackmail
	option = {
		name = murim_shadow_politics_expansion2.0002.a
		add_intrigue_skill = 2
		random_list = {
			60 = {
				scope:puppet_candidate = {
					add_opinion = {
						target = root
						modifier = murim_blackmailed_into_submission
						opinion = -40
					}
					set_character_flag = murim_demon_puppet
				}
				add_character_modifier = {
					modifier = murim_puppet_master
					years = 30
				}
				custom_tooltip = murim_puppet_secured_through_blackmail_tt
				trigger_event = {
					id = murim_shadow_politics_expansion2.0004
					days = { 180 365 }
				}
			}
			40 = {
				add_stress = 30
				custom_tooltip = murim_blackmail_material_insufficient_tt
			}
		}
		ai_chance = { base = 40 }
	}
	
	# Option B: Approach through seduction and gifts
	option = {
		name = murim_shadow_politics_expansion2.0002.b
		add_gold = -300
		add_diplomacy_skill = 2
		random_list = {
			70 = {
				scope:puppet_candidate = {
					add_opinion = {
						target = root
						modifier = murim_generous_benefactor
						opinion = 30
					}
					set_character_flag = murim_demon_puppet
				}
				add_character_modifier = {
					modifier = murim_puppet_master
					years = 30
				}
				trigger_event = {
					id = murim_shadow_politics_expansion2.0004
					days = { 180 365 }
				}
			}
			30 = {
				add_gold = -200
				custom_tooltip = murim_gifts_rejected_suspicion_tt
			}
		}
		ai_chance = { base = 35 }
	}
	
	# Option C: Approach through force — abduct and condition
	option = {
		name = murim_shadow_politics_expansion2.0002.c
		add_dread = 20
		random_list = {
			50 = {
				scope:puppet_candidate = {
					set_character_flag = murim_demon_puppet
					add_trait = craven
				}
				add_character_modifier = {
					modifier = murim_puppet_master
					years = 30
				}
				trigger_event = {
					id = murim_shadow_politics_expansion2.0004
					days = { 90 180 }
				}
			}
			50 = {
				add_prestige = -300
				add_stress = 40
				custom_tooltip = murim_abduction_failed_scandal_tt
			}
		}
		ai_chance = {
			base = 25
			modifier = { add = 20 has_trait = sadistic }
		}
	}
}
