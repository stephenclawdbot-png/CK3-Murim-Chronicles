######################################
# EVENT 0001 — Orphan Path: Street Fight Recruitment
# An orphan catches the eye of a clan elder during a street brawl
######################################
murim_entry_path_exp2.0001 = {
	type = character_event
	title = murim_entry_path_exp2.0001.t
	desc = murim_entry_path_exp2.0001.desc
	theme = martial_theme
	left_portrait = {
		character = root
		animation = anger
	}
	right_portrait = {
		character = scope:recruiting_elder
		animation = personality_bold
	}

	trigger = {
		NOT = { has_variable = murim_clan_id }
		NOT = { has_variable = murim_cultivation_level }
		prowess >= 8
		age >= 14
		age <= 25
		NOT = { has_character_flag = orphan_path_started }
		is_lowborn = yes
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			prowess >= 12
		}
		modifier = {
			add = 2
			has_trait = brave
		}
	}

	immediate = {
		set_character_flag = orphan_path_started
		random_ruler = {
			limit = {
				has_variable = murim_clan_id
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = beggar_sect_member
					has_trait = iron_fist_disciple
				}
			}
			save_scope_as = recruiting_elder
		}
	}

	# Accept the elder's offer — join the clan
	option = {
		name = murim_entry_path_exp2.0001.a
		set_variable = {
			name = murim_cultivation_level
			value = 1
		}
		set_variable = {
			name = murim_clan_id
			value = scope:recruiting_elder.var:murim_clan_id
		}
		add_trait = orphan_recruit
		add_character_modifier = {
			modifier = street_fighter_potential
			years = 10
		}
		scope:recruiting_elder = {
			add_opinion = {
				target = root
				modifier = recruited_promising_orphan
				opinion = 20
			}
		}
		add_prestige = 100
		hidden_effect = {
			trigger_event = {
				id = murim_entry_path_exp2.0002
				days = { 90 180 }
			}
		}
		ai_chance = { base = 70 }
	}

	# Refuse — stay independent
	option = {
		name = murim_entry_path_exp2.0001.b
		add_prestige = -50
		add_character_modifier = {
			modifier = refused_clan_recruitment
			years = 3
		}
		ai_chance = { base = 20 }
	}

	# Ask to be tested first
	option = {
		name = murim_entry_path_exp2.0001.c
		duel = {
			skill = prowess
			target = scope:recruiting_elder
			30 = {
				desc = murim_entry_path_exp2.0001.c.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2.0
				}
				send_interface_toast = {
					title = murim_entry_path_exp2.0001.c.success.toast
					left_icon = root
					set_variable = { name = murim_cultivation_level value = 1 }
					set_variable = { name = murim_clan_id value = scope:recruiting_elder.var:murim_clan_id }
					add_trait = orphan_recruit
					add_trait = tested_by_combat
					add_prestige = 200
				}
			}
			70 = {
				desc = murim_entry_path_exp2.0001.c.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2.0
				}
				send_interface_toast = {
					title = murim_entry_path_exp2.0001.c.failure.toast
					left_icon = root
					set_variable = { name = murim_cultivation_level value = 1 }
					set_variable = { name = murim_clan_id value = scope:recruiting_elder.var:murim_clan_id }
					add_trait = orphan_recruit
					add_character_modifier = { modifier = humbled_by_test years = 2 }
				}
			}
		}
		ai_chance = { base = 10 }
	}
}

		NOT = { has_variable = murim_cultivation_level }

		NOT = { has_character_flag = noble_blood_awakened }

			any_parent = {
				has_variable = murim_clan_id
			}
			any_grandparent = {
				has_variable = murim_clan_id
			}
		}

		base = 1

			add = 5

		name = murim_entry_path_exp2.0003.a

			modifier = bloodline_awakening_power

		add_prowess_skill = 3
		add_prestige = 300

			limit = {
				any_parent = { has_variable = murim_clan_id }
			}
			random_parent = {
				limit = { has_variable = murim_clan_id }
				root = {
					set_variable = {
						name = murim_clan_id

			trigger_event = {
				id = murim_entry_path_exp2.0004

		add_prestige = 100
		add_character_modifier = {
			modifier = rejected_martial_bloodline
			years = 10
		}
		ai_chance = { base = 20 }
	}

	# Seek out the parent who hid this from you
	option = {
		name = murim_entry_path_exp2.0003.c
		set_variable = { name = murim_cultivation_level value = 1 }
		add_character_modifier = {
			modifier = searching_for_martial_parent
			years = 5
		}
		add_stress = 20
		ai_chance = { base = 10 }
	}

######################################
# EVENT 0004 — Noble Blood Path: Clan Inheritance Challenge
# Noble heir must prove worthy of clan birthright
######################################
murim_entry_path_exp2.0004 = {
	type = character_event
	title = murim_entry_path_exp2.0004.t
	desc = murim_entry_path_exp2.0004.desc
	theme = martial_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:rival_heir
		animation = personality_vengeful
	}

	trigger = {
		has_trait = noble_blood_martial
		has_character_flag = noble_blood_awakened
	}

	immediate = {
		if = {
			limit = {
				any_courtier = {
					has_variable = murim_clan_id
					var:murim_clan_id = root.var:murim_clan_id
					NOT = { this = root }
					prowess >= 12
				}
			}
			random_courtier = {
				limit = {
					has_variable = murim_clan_id
					var:murim_clan_id = root.var:murim_clan_id
					NOT = { this = root }
					prowess >= 12
				}
				save_scope_as = rival_heir
			}
		}
	}

	# Prove your worth through combat
	option = {
		name = murim_entry_path_exp2.0004.a
		trigger = {
			exists = scope:rival_heir
		}
		duel = {
			skill = prowess
			target = scope:rival_heir
			50 = {
				desc = murim_entry_path_exp2.0004.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				send_interface_toast = {
					title = murim_entry_path_exp2.0004.a.success.toast
					left_icon = root
					add_prestige = 500
					add_trait = proven_heir
					add_character_modifier = { modifier = proven_clan_heir years = 20 }
					change_variable = { name = murim_cultivation_level add = 1 }
				}
			}
			50 = {
				desc = murim_entry_path_exp2.0004.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				send_interface_toast = {
					title = murim_entry_path_exp2.0004.a.failure.toast
					left_icon = root
					add_prestige = -200
					add_character_modifier = { modifier = failed_heir_challenge years = 5 }
					health = -1
				}
			}
		}
		ai_chance = { base = 60 }
	}

	# Prove through scholarship instead
	option = {
		name = murim_entry_path_exp2.0004.b
		trigger = {
			learning >= 14
		}
		add_learning_skill = 2
		add_trait = scholarly_heir
		add_character_modifier = {
			modifier = heir_through_knowledge
			years = 15
		}
		add_prestige = 300
		ai_chance = { base = 25 }
	}

	# Renounce the inheritance
	option = {
		name = murim_entry_path_exp2.0004.c
		add_prestige = -300
		add_piety = 200
		remove_trait = noble_blood_martial
		add_trait = renounced_heir
		ai_chance = { base = 15 }
	}
}
