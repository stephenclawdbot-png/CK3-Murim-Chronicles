# 0001 — Omens of the Cycle: Heavenly Signs
##############################
murim_calamity_cycle.0001 = {
	type = character_event
	title = murim_calamity_cycle.0001.t
	desc = murim_calamity_cycle.0001.desc
	theme = skull
	
	left_portrait = {
		character = root
		animation = worry
	}
	
	trigger = {
		is_ruler = yes
		has_global_variable = murim_world_active
		NOT = { has_global_variable = calamity_active }
		NOT = { has_character_flag = calamity_omens_seen }
		current_date >= 1250.1.1
		OR = {
			has_trait_rank = { trait = martial_cultivation value >= 5 }
			has_trait = celestial_sage
			learning >= 20
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 10
			has_global_variable = previous_calamity_occurred
		}
	}
	
	immediate = {
		set_character_flag = { flag = calamity_omens_seen }
	}
	
	# Warn the Murim Alliance of impending doom
	option = {
		name = murim_calamity_cycle.0001.a
		
		add_prestige = 200
		add_character_modifier = {
			modifier = calamity_prophet
			years = 10
		}
		
		hidden_effect = {
			set_global_variable = { name = calamity_warning_issued value = yes }
			every_ruler = {
				limit = {
					is_ai = yes
					OR = {
						has_trait = mount_hua_disciple
						has_trait = shaolin_disciple
						has_trait = wudang_disciple
						has_trait = beggar_sect_member
						has_trait = tang_clan_member
						has_trait = iron_fist_disciple
						has_trait = poison_flower_disciple
						has_trait = phantom_blade_disciple
						has_trait = heavenly_demon_disciple
					}
				}
				random = {
					chance = 30
					trigger_event = { id = murim_calamity_cycle.0002 days = { 30 90 } }
				}
			}
		}
		
		ai_chance = { base = 70 }
	}
	
	# Keep the knowledge secret — prepare alone
	option = {
		name = murim_calamity_cycle.0001.b
		
		add_character_modifier = {
			modifier = secret_calamity_preparation
			years = 10
		}
		add_stress = 30
		
		ai_chance = { base = 20 }
	}
	
	# Dismiss it as superstition
	option = {
		name = murim_calamity_cycle.0001.c
		
		add_prestige = -50
		add_stress = -10
		
		ai_chance = { base = 10 }
	}
}

					chance = 80

						limit = {
							is_ai = yes

								has_trait = shaolin_disciple
								has_trait = wudang_disciple
							}
						}

							start_war = {

			}
		}

		name = murim_calamity_cycle.0003.b

			modifier = sealed_meditation
