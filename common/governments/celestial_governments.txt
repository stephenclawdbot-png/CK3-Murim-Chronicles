################################################################
# MURIM CHRONICLES -- GOVERNMENT TYPES
# Path: common/governments/celestial_governments.txt
# Version: 2.0 -- Expanded from 3 to 8 government types
#
# Defines distinct governance structures for the martial world,
# each reflecting a different relationship between power,
# authority, cultivation philosophy, and social organization.
#
# 1. orthodox_sect_government      -- Righteous sects (hierarchy + virtue)
# 2. unorthodox_sect_government    -- Demonic/heterodox sects (fear + power)
# 3. free_cultivator_government    -- Independent cultivators (freedom)
# 4. hidden_valley_government      -- Isolationist hidden sects (secrecy)
# 5. murim_alliance_government     -- Coalition of sects (diplomacy)
# 6. imperial_cultivation_court    -- State-sponsored cultivation (bureaucracy)
# 7. demonic_cult_government       -- Sacrifice-based dark sects (corruption)
# 8. wandering_sword_government    -- Mercenary pavilions (reputation)
#
# CK3 government_type syntax reference:
#   create_cadet_branches, valid_holdings, can_get_government,
#   character_modifier, ai_priority, flag
################################################################

################################################################
# 1. ORTHODOX SECT GOVERNMENT
# Righteous sects follow the Dao through virtue and hierarchy.
# Elder councils guide decisions. Succession by cultivation rank.
# Strong bonuses to learning, piety, and vassal loyalty.
################################################################
orthodox_sect_government = {
	create_cadet_branches = yes

	valid_holdings = {
		castle
		city
		church
	}

	can_get_government = {
		OR = {
			has_character_flag = murim_orthodox_cultivator
			has_character_flag = murim_sect_master
			AND = {
				has_character_flag = murim_initiated
				NOT = { has_character_flag = murim_demonic_cultivator }
				NOT = { has_character_flag = murim_unorthodox_path }
			}
		}
	}

	character_modifier = {
		# Sect master bonuses -- structured hierarchy and virtue
		monthly_prestige = 0.5
		monthly_piety = 0.5
		domain_limit = 2
		vassal_limit = 12
		learning = 3
		diplomacy = 2
		monthly_dynasty_prestige = 0.3
		# Righteous reputation -- feared less, respected more
		dread_baseline_add = -10
		# Sect unity through shared cultivation
		vassal_opinion = 5
		same_faith_opinion = 10
		# Orthodox cultivation efficiency
		monthly_lifestyle_xp_gain_mult = 0.05
		# Sect treasury management
		domain_tax_mult = 0.05
		# Health from disciplined lifestyle
		health = 0.25
	}

	ai_priority = {
		base = 50
		modifier = {
			add = 30
			has_character_flag = murim_orthodox_cultivator
		}
		modifier = {
			add = 25
			has_character_flag = murim_sect_master
		}
		modifier = {
			add = 15
			learning >= 12
		}
		modifier = {
			add = 10
			diplomacy >= 10
		}
		modifier = {
			add = 10
			has_trait = just
		}
		modifier = {
			add = 10
			has_trait = zealous
		}
		modifier = {
			add = -50
			has_character_flag = murim_demonic_cultivator
		}
		modifier = {
			add = -30
			has_trait = sadistic
		}
	}

	flag = government_is_sect
	flag = government_is_orthodox
	flag = government_has_elder_council
	flag = government_allows_sect_succession
}

			add = 10

		}
		modifier = {
			add = -20

		}
		modifier = {

################################################################
# 4. HIDDEN VALLEY SECT GOVERNMENT
# Isolationist sects hidden in remote valleys and mountains.
# Hoards ancient knowledge, hostile to outsiders.
# Massive learning/cultivation bonuses but terrible diplomacy.
# Secret techniques passed only within the sect.
################################################################
hidden_valley_government = {
	create_cadet_branches = yes

	valid_holdings = {
		castle
		church
	}

	can_get_government = {
		OR = {
			has_character_flag = murim_hidden_valley_master
			AND = {
				has_character_flag = murim_initiated
				has_character_flag = murim_isolationist
				NOT = { has_character_flag = murim_demonic_cultivator }
			}
			AND = {
				has_character_flag = murim_sect_master
				has_character_flag = murim_isolationist
			}
		}
	}

	character_modifier = {
		# Hidden valley -- secrecy and ancient knowledge
		monthly_prestige = 0.2
		monthly_piety = 0.8
		domain_limit = 2
		vassal_limit = 5
		# Ancient knowledge repository
		learning = 5
		# Master cultivators in isolation
		prowess = 2
		martial = 1
		# Terrible at politics -- isolationist penalty
		diplomacy = -3
		# Hidden from the world
		dread_baseline_add = -15
		# Secret technique preservation
		monthly_lifestyle_xp_gain_mult = 0.2
		# Suspicious of outsiders
		different_faith_opinion = -15
		# Protected valley -- defensive bonus
		defender_advantage = 5
		# Long-lived through perfect cultivation environment
		health = 0.75
		# Self-sufficient economy
		domain_tax_mult = -0.1
		# Inner peace
		stress_loss_mult = 0.2
	}

	ai_priority = {
		base = 15
		modifier = {
			add = 50
			has_character_flag = murim_hidden_valley_master
		}
		modifier = {
			add = 30
			has_character_flag = murim_isolationist
		}
		modifier = {
			add = 20
			learning >= 18
		}
		modifier = {
			add = 15
			has_trait = shy
		}
		modifier = {
			add = 10
			has_trait = patient
		}
		modifier = {
			add = -40
			has_trait = gregarious
		}
		modifier = {
			add = -30
			has_trait = ambitious
		}
	}

	flag = government_is_sect
	flag = government_is_hidden
	flag = government_is_isolationist
	flag = government_has_secret_archive
}

################################################################

################################################################
# 6. IMPERIAL CULTIVATION COURT
# State-sponsored cultivation backed by imperial authority.
# Bureaucratic system with official ranks and titles.
# Best resource generation, strong army, but rigid hierarchy.
# Cultivators serve the throne in exchange for resources.
################################################################
imperial_cultivation_court = {
	create_cadet_branches = yes

	valid_holdings = {
		castle
		city
		church
	}

	can_get_government = {
		OR = {
			has_character_flag = murim_imperial_cultivator
			AND = {
				highest_held_title_tier >= tier_kingdom
				has_character_flag = murim_initiated
				NOT = { has_character_flag = murim_demonic_cultivator }
			}
			AND = {
				has_character_flag = murim_sect_master
				is_vassal_of = {
					highest_held_title_tier >= tier_empire
				}
			}
		}
	}

	character_modifier = {
		# Imperial court -- bureaucratic cultivation
		monthly_prestige = 1.0
		monthly_piety = 0.3
		domain_limit = 5
		vassal_limit = 15
		# Imperial bureaucracy bonuses
		stewardship = 3
		diplomacy = 2
		martial = 2
		# Imperial authority
		dread_baseline_add = 10
		# State resources for cultivation
		monthly_lifestyle_xp_gain_mult = 0.05
		# Imperial treasury
		domain_tax_mult = 0.15
		# Mandate of Heaven legitimacy
		monthly_dynasty_prestige = 0.8
		# Imperial army bonuses
		knight_effectiveness_mult = 0.1
		men_at_arms_maintenance = -0.05
		# Rigid hierarchy drawbacks
		vassal_opinion = -5
		stress_gain_mult = 0.1
		# Court cultivation environment
		health = 0.25
	}

	ai_priority = {
		base = 40
		modifier = {
			add = 40
			has_character_flag = murim_imperial_cultivator
		}
		modifier = {
			add = 30
			highest_held_title_tier >= tier_kingdom
		}
		modifier = {
			add = 20
			stewardship >= 12
		}
		modifier = {
			add = 15
			has_trait = ambitious
		}
		modifier = {
			add = -30
			has_trait = content
		}
		modifier = {
			add = -40
			has_character_flag = murim_demonic_cultivator
		}
	}

	flag = government_is_imperial
	flag = government_is_bureaucratic
	flag = government_has_cultivation_ministry
	flag = government_has_mandate_of_heaven
}
