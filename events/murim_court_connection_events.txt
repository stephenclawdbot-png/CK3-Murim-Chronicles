# Murim Chronicles - Martial Court Connection Events
# Links martial world leaders to vanilla rulers without overriding vanilla mechanics
# The martial world operates BEHIND vanilla — shadow advisors, not replacements
# Lightweight: ~8 events, fun interactions, AI-friendly

namespace = murim_court

########################################
# murim_court.0001 - A Martial Advisor Arrives at Court
# Triggered: yearly pulse for rulers with martial courtiers
# A skilled martial artist offers service as an informal advisor
########################################
murim_court.0001 = {
	type = character_event
	title = murim_court.0001.t
	desc = murim_court.0001.desc
	theme = martial

	left_portrait = {
		character = root
		animation = personality_rational
	}
	right_portrait = {
		character = scope:martial_advisor
		animation = personality_bold
	}

	trigger = {
		is_ruler = yes
		NOT = { has_character_flag = murim_has_martial_advisor }
		NOT = { has_character_flag = murim_court_advisor_cooldown }
		any_courtier = {
			has_variable = murim_martial_artist
			prowess >= 12
			is_ai = yes
			NOT = { has_character_flag = murim_is_court_advisor }
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			any_courtier = {
				has_variable = murim_martial_artist
				prowess >= 18
			}
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				has_variable = murim_martial_artist
				prowess >= 12
				is_ai = yes
				NOT = { has_character_flag = murim_is_court_advisor }
			}
			weight = {
				base = 1
				modifier = {
					factor = 3
					prowess >= 18
				}
				modifier = {
					factor = 2
					has_character_flag = murim_sect_leader
				}
			}
			save_scope_as = martial_advisor
		}
	}

	# Accept as martial advisor
	option = {
		name = murim_court.0001.a
		add_character_flag = murim_has_martial_advisor
		scope:martial_advisor = {
			add_character_flag = murim_is_court_advisor
			add_opinion = {
				target = root
				modifier = murim_grateful_acceptance
			}
		}
		add_character_modifier = {
			modifier = murim_martial_advisor_at_court
			years = 10
		}
		stress_impact = {
			base = -10
		}
		ai_chance = {
			base = 70
		}
	}

	# Politely decline
	option = {
		name = murim_court.0001.b
		add_character_flag = {
			flag = murim_court_advisor_cooldown
			years = 3
		}
		scope:martial_advisor = {
			add_opinion = {
				target = root
				modifier = murim_passed_over
			}
		}
		ai_chance = {
			base = 30
		}
	}
}

########################################
# murim_court.0004 - The Emperor's Martial Tournament
# A ruler hosts a tournament to attract martial world talent
# Draws in powerful fighters, builds prestige
########################################
murim_court.0004 = {
	type = character_event
	title = murim_court.0004.t
	desc = murim_court.0004.desc
	theme = martial

	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		is_ruler = yes
		highest_held_title_tier >= tier_kingdom
		NOT = { has_character_flag = murim_hosted_court_tournament }
		gold >= 150
	}

	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			highest_held_title_tier >= tier_empire
		}
	}

	immediate = {
		add_character_flag = {
			flag = murim_hosted_court_tournament
			years = 10
		}
	}

	# Grand tournament — expensive but massive prestige
	option = {
		name = murim_court.0004.a
		remove_short_term_gold = 150
		add_prestige = 500
		add_character_modifier = {
			modifier = murim_imperial_tournament_host
			years = 5
		}
		# Attract a martial courtier
		create_character = {
			age = { 25 35 }
			gender_female_chance = 50
			trait = brave
			prowess = { 14 22 }
			martial = { 8 14 }
			diplomacy = { 4 8 }
			faith = root.faith
			culture = root.culture
			dynasty = none
		}
		new_character = {
			add_trait = martial_artist
			set_variable = murim_martial_artist
			add_character_flag = murim_tournament_recruit
			add_character_modifier = {
				modifier = murim_imperial_tournament_champion
				years = 10
			}
			add_opinion = {
				target = root
				modifier = murim_grateful_acceptance
			}
		}
		ai_chance = {
			base = 50
		}
	}

	# Modest exhibition — cheaper, still draws interest
	option = {
		name = murim_court.0004.b
		remove_short_term_gold = 50
		add_prestige = 200
		add_character_modifier = {
			modifier = murim_court_exhibition_host
			years = 3
		}
		ai_chance = {
			base = 35
		}
	}

	# Not worth the trouble
	option = {
		name = murim_court.0004.c
		add_prestige = -25
		ai_chance = {
			base = 15
		}
	}
}

########################################
# murim_court.0007 - AI: Martial Leader Approaches a Ruler
# AI-only event: martial leaders autonomously seek court connections
# Creates the shadow influence pipeline for AI
########################################
murim_court.0007 = {
	type = character_event
	title = murim_court.0007.t
	desc = murim_court.0007.desc
	theme = diplomacy

	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		is_ai = yes
		has_variable = murim_martial_artist
		prowess >= 14
		NOT = { has_character_flag = murim_seeking_court_influence }
		NOT = { has_character_flag = murim_court_influence_cooldown }
		OR = {
			has_character_flag = murim_sect_leader
			has_character_flag = murim_sect_elder
			prowess >= 20
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			has_character_flag = murim_sect_leader
		}
		modifier = {
			factor = 1.5
			intrigue >= 12
		}
	}

	immediate = {
		add_character_flag = {
			flag = murim_court_influence_cooldown
			years = 5
		}
	}

	# Offer service openly
	option = {
		name = murim_court.0007.a
		add_character_flag = {
			flag = murim_seeking_court_influence
			years = 5
		}
		add_character_modifier = {
			modifier = murim_court_influence_seeker
			years = 5
		}
		if = {
			limit = { is_ruler = yes }
			liege = {
				add_opinion = {
					target = root
					modifier = murim_respect_opinion

		}

			modifier = murim_shadow_court_member
			years = 5

		ai_chance = {
			base = 30
		}
