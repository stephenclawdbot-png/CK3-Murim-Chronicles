### EVENT 1: Installing the Puppet Emperor ###
murim_shadow_pol_exp4.0001 = {
	type = character_event
	title = murim_shadow_pol_exp4.0001.t
	desc = murim_shadow_pol_exp4.0001.desc
	theme = intrigue
	
	trigger = {
		has_trait = heavenly_demon_heir
		intrigue >= 20
		NOT = { has_character_flag = puppet_emperor_installed }
		any_ruler = {
			is_ruler = yes
			has_primary_title = yes
			NOT = { has_trait = heavenly_demon_heir }
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 5 has_trait = schemer }
		modifier = { add = 3 intrigue >= 25 }
	}
	
	immediate = {
		random_ruler = {
			limit = {
				is_ruler = yes
				has_primary_title = yes
				NOT = { has_trait = heavenly_demon_heir }
				martial <= 14
			}
			save_scope_as = target_emperor
		}
	}
	
	# Option A: Blackmail and install puppet
	option = {
		name = murim_shadow_pol_exp4.0001.a
		add_character_flag = { flag = puppet_emperor_installed days = -1 }
		add_prestige = 1000
		add_intrigue = 3
		add_dread = 40
		reverse_add_opinion = {
			target = scope:target_emperor
			modifier = puppet_master_opinion
			opinion = -60
		}
		hidden_effect = {
			scope:target_emperor = {
				add_character_flag = { flag = is_puppet_emperor days = -1 }
				set_variable = { name = puppet_master value = root }
			}
			trigger_event = { id = murim_shadow_pol_exp4.0002 days = { 30 90 } }
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Assassinate and replace with your own candidate
	option = {
		name = murim_shadow_pol_exp4.0001.b
		trigger = { intrigue >= 22 }
		hidden_effect = {
			random_list = {
				60 = {
					modifier = { add = 20 intrigue >= 25 }
					scope:target_emperor = {
						death = { death_reason = death_murder_unknown killer = root }
					}
					root = {
						add_prestige = 1500
						add_character_flag = { flag = puppet_emperor_installed days = -1 }
						add_dread = 30
					}
				}
				40 = {
					root = {
						add_prestige = -500
						add_character_flag = { flag = assassination_attempt_exposed days = 365 }
					}
				}
			}
		}
		ai_chance = { base = 30 }
	}
	
	# Option C: Marry into the imperial family
	option = {
		name = murim_shadow_pol_exp4.0001.c
		trigger = { diplomacy >= 16 is_married = no }
		add_character_flag = { flag = puppet_emperor_installed days = -1 }
		add_character_flag = { flag = imperial_marriage days = -1 }
		add_prestige = 800
		add_diplomacy = 2
		ai_chance = { base = 15 }
	}
	
	# Option D: Rule from the shadows without a puppet
	option = {
		name = murim_shadow_pol_exp4.0001.d
		add_character_flag = { flag = shadow_ruler days = -1 }
		add_intrigue = 4
		add_prestige = 600
		add_dread = 20
		ai_chance = { base = 20 }
	}
}

			limit = {
