# === SEEK TRIBUNAL NOMINATION ===
murim_seek_tribunal_nomination_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	title = murim_seek_tribunal_nomination_decision
	desc = murim_seek_tribunal_nomination_decision_desc
	selection_tooltip = murim_seek_tribunal_nomination_decision_tooltip

	is_shown = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 5
		NOT = { has_trait = murim_tribunal_judge }
		NOT = { has_trait = murim_tribunal_exile }
	}

	is_valid = {
		murim_can_join_tribunal_trigger = yes
		NOT = { has_character_flag = murim_tribunal_nomination_cooldown }
	}

	effect = {
		set_character_flag = { flag = murim_tribunal_nomination_cooldown years = 5 }
		trigger_event = { id = murim_tribunal_system.0001 days = 30 }
		custom_tooltip = murim_seek_tribunal_nomination_tt
	}

	ai_check_interval = 360
	ai_will_do = {
		base = 10
		modifier = { add = 20 has_trait = ambitious }
		modifier = { add = 15 diplomacy >= 15 }
		modifier = { add = -15 has_trait = shy }
	}
}

	ai_will_do = {
		base = 10
		modifier = { add = 15 diplomacy >= 14 }
		modifier = { add = 10 has_trait = greedy }
		modifier = { add = -20 murim_is_unorthodox_faction_trigger = yes }
	}
}
