# EVENT 0001 - Discovery of the Demonic Path
# Triggered: on_action yearly pulse, martial artists with dark inclinations
# AI: Full dark cultivation entry with weighted personality
murim_demonic.0001 = {
    type = character_event
    title = murim_demonic.0001.t
    desc = murim_demonic.0001.desc
    theme = skulduggery
    
    left_portrait = {
        character = root
        animation = personality_callous
    }
    right_portrait = {
        character = scope:demonic_tempter
        animation = scheme
    }
    
    trigger = {
        is_alive = yes
        age >= 16
        NOT = { has_trait = murim_demon_cultivator }
        NOT = { has_character_flag = refused_demonic_path }
        OR = {
            murim_is_martial_artist_trigger = yes
            has_trait = murim_wandering_cultivator
            has_trait = murim_sect_leader
            has_trait = murim_unorthodox_elder
        }
        OR = {
            has_trait = ambitious
            has_trait = sadistic
            has_trait = wrathful
            has_trait = callous
            prowess >= 16
            AND = {
                has_trait = murim_qi_deviation_survivor
                stress >= 50
            }
            AND = {
                has_character_modifier = murim_devastating_heartbreak
                has_trait = wrathful
            }
        }
        NOT = { has_trait = murim_righteous_elder }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 3.0
            has_trait = sadistic
        }
        modifier = {
            factor = 2.0
            has_trait = ambitious
        }
        modifier = {
            factor = 2.0
            has_trait = wrathful
        }
        modifier = {
            factor = 2.5
            has_character_modifier = murim_devastating_heartbreak
        }
        modifier = {
            factor = 0.3
            has_trait = compassionate
        }
        modifier = {
            factor = 0.1
            has_trait = just
        }
    }
    
    immediate = {
        # Find or create the demonic tempter
        random_known_character = {
            limit = {
                is_alive = yes
                has_trait = murim_demon_cultivator
                NOT = { this = root }
            }
            save_scope_as = demonic_tempter
            alternative = {
                create_character = {
                    age = { 40 70 }
                    gender_female_chance = 30
                    trait = murim_demon_cultivator
                    trait = deceitful
                    trait = sadistic
                    trait = callous
                    prowess = { 18 28 }
                    intrigue = { 16 22 }
                    culture = root.culture
                    faith = root.faith
                    save_scope_as = demonic_tempter
                }
            }
        }
    }
    
    # Option A: Embrace the demonic path
    option = {
        name = murim_demonic.0001.a
        add_trait = murim_demon_cultivator
        add_prowess_skill = 3
        add_character_modifier = {
            modifier = murim_demonic_qi_awakening
            years = 5
        }
        add_dread = 20
        # Begin corruption
        set_variable = {
            name = demonic_corruption
            value = 10
        }
        reverse_add_opinion = {
            target = scope:demonic_tempter
            modifier = murim_demonic_disciple
            opinion = 30
        }
        # Orthodox characters react
        every_known_character = {
            limit = {
                is_alive = yes
                has_trait = murim_orthodox_disciple
            }
            add_opinion = {
                target = root
                modifier = murim_fallen_to_darkness
                opinion = -30
            }
        }
        stress_impact = {
            just = major_stress_impact_gain
            compassionate = major_stress_impact_gain
            honest = medium_stress_impact_gain
        }
        trigger_event = {
            id = murim_demonic.0002
            days = { 60 180 }
        }
        ai_chance = {
            base = 40
            modifier = {
                factor = 3.0
                has_trait = sadistic
            }
            modifier = {
                factor = 2.5
                has_trait = ambitious
            }
            modifier = {
                factor = 2.0
                has_trait = wrathful
            }
            modifier = {
                factor = 0.1
                has_trait = compassionate
            }
            modifier = {
                factor = 0.1
                has_trait = just
            }
        }
    }
    
    # Option B: Take the knowledge but resist corruption
    option = {
        name = murim_demonic.0001.b
        trigger = {
            learning >= 14
        }
        add_learning_skill = 2
        add_character_modifier = {
            modifier = murim_dark_knowledge_seeker
            years = 5
        }
        add_character_flag = {
            flag = murim_studied_demonic_arts
            days = 3650
        }
        add_stress = 20
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                learning >= 18
            }
            modifier = {
                factor = 1.5
                has_trait = patient
            }
        }
    }
    
    # Option C: Reject the demonic path completely
    option = {
        name = murim_demonic.0001.c
        add_piety = 100
        add_prestige = 50
        add_character_flag = {
            flag = refused_demonic_path
            days = 3650
        }
        add_character_modifier = {
            modifier = murim_righteous_resolve
            years = 3
        }
        if = {
            limit = { exists = scope:demonic_tempter }
            reverse_add_opinion = {
                target = scope:demonic_tempter
                modifier = murim_rejected_dark_offer
                opinion = -30
            }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 4.0
                has_trait = just
            }
            modifier = {
                factor = 3.0
                has_trait = compassionate
            }
            modifier = {
                factor = 2.0
                has_trait = zealous
            }
        }
    }
}
