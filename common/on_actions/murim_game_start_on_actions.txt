# Murim Chronicles - Game Start On Actions
# Hooks on_game_start to seed the martial world ONCE at game load
# Bookmark-agnostic: fires on 867, 1066, or any custom start
# Cross-refs: murim_game_start_effects.txt (all seeding logic)
#             murim_on_actions.txt (yearly pulses for ongoing generation)
# INTEGRATION: This is the ENTRY POINT for world initialization.
#              All other systems (tournaments, rankings, duels, sects) depend on this running first.

on_game_start = {
	effect = {
		# ============================
		# SEED THE ENTIRE MARTIAL WORLD
		# ============================
		# Phase 1: 80-100 martial artists in Goryeo/China (weighted distribution)
		# Phase 2: 6 major clans populated (10-15 members each with hierarchy)
		# Phase 3: Top 50 rankings filled from power scores
		# Phase 4: 8-10 wandering masters scattered
		# Phase 5: Foreign masters in 7 global regions
		# Phase 6: Clan relationships set (righteous vs demonic)
		murim_seed_martial_world_effect = yes
		
		# Set global flag so seeding never runs twice
		set_global_variable = {
			name = murim_world_seeded
			value = yes
		}
		
		# Log completion for debugging
		debug_log = "Murim Chronicles: Martial world seeding complete."
	}
}
