### EVENT 1: The Last Elder's Testament ###
murim_calamity_cycle_exp2.0001 = {
	type = character_event
	title = murim_calamity_cycle_exp2.0001.t
	desc = murim_calamity_cycle_exp2.0001.desc
	theme = martial_arts
	
	trigger = {
		has_trait = calamity_survivor
		age >= 60
		NOT = { has_character_flag = testament_given }
		any_child = { is_alive = yes age >= 14 }
	}
	
	weight_multiplier = {
		base = 2
		modifier = { add = 5 health <= 2 }
	}
	
	immediate = {
		random_child = {
			limit = { is_alive = yes age >= 14 }
			save_scope_as = testament_heir
		}
	}
	
	# Option A: Pass down the full truth of the Calamity
	option = {
		name = murim_calamity_cycle_exp2.0001.a
		add_character_flag = { flag = testament_given days = -1 }
		scope:testament_heir = {
			add_trait = calamity_witness_heir
			add_learning = 3
			add_martial = 2
			add_prestige = 500
			add_character_flag = { flag = knows_calamity_truth days = -1 }
		}
		add_prestige = 800
		add_piety = 300
		ai_chance = { base = 45 }
	}
	
	# Option B: Mythologize it - turn horror into legend
	option = {
		name = murim_calamity_cycle_exp2.0001.b
		add_character_flag = { flag = testament_given days = -1 }
		scope:testament_heir = {
			add_trait = mythology_keeper
			add_learning = 4
			add_prestige = 400
		}
		add_prestige = 600
		add_piety = 500
		ai_chance = { base = 30 }
	}
	
	# Option C: Suppress the memory - let the next generation be free
	option = {
		name = murim_calamity_cycle_exp2.0001.c
		add_character_flag = { flag = testament_given days = -1 }
		add_stress = -30
		add_piety = 200
		scope:testament_heir = {
			add_prestige = 200
		}
		ai_chance = { base = 20 }
	}
	
	# Option D: Weaponize the memory - train them to prevent it
	option = {
		name = murim_calamity_cycle_exp2.0001.d
		add_character_flag = { flag = testament_given days = -1 }
		scope:testament_heir = {
			add_trait = calamity_preventer
			add_martial = 4
			add_prowess = 5
			add_prestige = 600
			add_stress = 20
		}
		add_prestige = 500
		ai_chance = { base = 15 }
	}
}
