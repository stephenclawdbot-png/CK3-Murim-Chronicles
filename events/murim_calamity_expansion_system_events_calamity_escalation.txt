# Incident 1: Mysterious deaths among cultivators
murim_calamity_escalation.0001 = {
	type = character_event
	title = murim_calamity_escalation.0001.t
	desc = murim_calamity_escalation.0001.desc
	theme = intrigue

	trigger = {
		has_variable = murim_martial_artist
		NOT = { has_global_variable = murim_great_calamity_active }
		NOT = { has_character_flag = murim_escalation_witnessed }
	}

	immediate = {
		set_character_flag = { flag = murim_escalation_witnessed years = 3 }
		change_global_variable = { name = murim_calamity_tension add = 1 }
	}

	# Investigate the deaths
	option = {
		name = murim_calamity_escalation.0001.a
		add_intrigue = 1
		add_learning = 1
		ai_chance = { base = 45 }
	}

	# Warn the sects
	option = {
		name = murim_calamity_escalation.0001.b
		add_diplomacy = 1
		add_prestige = 50
		ai_chance = { base = 35 }
	}

	# It's not your problem
	option = {
		name = murim_calamity_escalation.0001.c
		add_stress = -10
		ai_chance = { base = 20 }
	}
}

# Threshold crossed — calamity begins
murim_calamity_escalation.0030 = {
	type = character_event
	title = murim_calamity_escalation.0030.t
	desc = murim_calamity_escalation.0030.desc
	theme = skull
	left_portrait = { character = root animation = personality_zealous }

	trigger = {
		has_variable = murim_martial_artist
		NOT = { has_global_variable = murim_great_calamity_active }
	}

	immediate = {
		set_global_variable = { name = murim_great_calamity_active value = yes }
		set_global_variable = { name = murim_calamity_phase value = 1 }
	}

	# Prepare for the worst
	option = {
		name = murim_calamity_escalation.0030.a
		add_martial = 2
		add_stress = 30
		hidden_effect = {
			trigger_event = { id = murim_demonic_invasion.0001 days = 90 }
		}
		ai_chance = { base = 50 }
	}

	# This was always coming
	option = {
		name = murim_calamity_escalation.0030.b
		add_prestige = 100
		add_learning = 1
		ai_chance = { base = 30 }
	}

	# Flee while you still can
	option = {
		name = murim_calamity_escalation.0030.c
		add_stress = -15
		add_prestige = -100
		ai_chance = { base = 20 }
	}
}
