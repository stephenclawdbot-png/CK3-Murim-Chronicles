# Murim Chronicles Phase 2 - Grind Loop Scripted Effects
# Reusable effects called by events and AI on_actions

murim_apply_body_tempering_effect = {
	# Remove current body temper trait and add next level
	if = {
		limit = { has_trait = murim_body_temper_8_diamond_flesh }
		remove_trait = murim_body_temper_8_diamond_flesh
		add_trait = murim_body_temper_9_void_body
		set_variable = { name = murim_body_tempering_level value = 9 }
		random = {
			chance = 15
			add_trait = wounded_1
		}
		add_character_flag = { flag = murim_tempering_cooldown days = 365 }
	}
	else_if = {
		limit = { has_trait = murim_body_temper_7_obsidian_core }
		remove_trait = murim_body_temper_7_obsidian_core
		add_trait = murim_body_temper_8_diamond_flesh
		set_variable = { name = murim_body_tempering_level value = 8 }
		random = {
			chance = 15
			add_trait = wounded_1
		}
		add_character_flag = { flag = murim_tempering_cooldown days = 365 }
	}
	else_if = {
		limit = { has_trait = murim_body_temper_6_jade_sinew }
		remove_trait = murim_body_temper_6_jade_sinew
		add_trait = murim_body_temper_7_obsidian_core
		set_variable = { name = murim_body_tempering_level value = 7 }
		random = {
			chance = 15
			add_trait = wounded_1
		}
		add_character_flag = { flag = murim_tempering_cooldown days = 365 }
	}
	else_if = {
		limit = { has_trait = murim_body_temper_5_gold_marrow }
		remove_trait = murim_body_temper_5_gold_marrow
		add_trait = murim_body_temper_6_jade_sinew
		set_variable = { name = murim_body_tempering_level value = 6 }
		add_character_flag = { flag = murim_tempering_cooldown days = 180 }

		limit = { has_trait = murim_body_temper_4_silver_blood }
		remove_trait = murim_body_temper_4_silver_blood
		add_trait = murim_body_temper_5_gold_marrow
		set_variable = { name = murim_body_tempering_level value = 5 }
		add_character_flag = { flag = murim_tempering_cooldown days = 180 }
	}

		limit = { has_trait = murim_body_temper_3_steel_meridians }
		remove_trait = murim_body_temper_3_steel_meridians
		add_trait = murim_body_temper_4_silver_blood
		set_variable = { name = murim_body_tempering_level value = 4 }
		add_character_flag = { flag = murim_tempering_cooldown days = 180 }

		limit = { has_trait = murim_body_temper_2_copper_bones }

		add_character_flag = { flag = murim_tempering_cooldown days = 180 }
	}
	else_if = {
		limit = { has_trait = murim_body_temper_1_iron_skin }
		remove_trait = murim_body_temper_1_iron_skin
		add_trait = murim_body_temper_2_copper_bones
		set_variable = { name = murim_body_tempering_level value = 2 }
		add_character_flag = { flag = murim_tempering_cooldown days = 180 }
	}
	else = {
		# No body temper trait yet - start at level 1
		add_trait = murim_body_temper_1_iron_skin
		set_variable = { name = murim_body_tempering_level value = 1 }
		add_character_flag = { flag = murim_tempering_cooldown days = 180 }
	}
	send_interface_toast = {
		title = murim_body_tempering_success_toast
		left_icon = root
	}
}

murim_apply_tempering_failure_effect = {
	# Failure consequences scale with current level
	if = {
		limit = {
			exists = var:murim_body_tempering_level
			var:murim_body_tempering_level >= 7
		}
		# Levels 7-9: Severe - death possible
		add_stress = 60
		add_trait = wounded_2
		add_character_modifier = { modifier = murim_body_tempering_recovery months = 6 }
		random = {
			chance = 30
			murim_apply_minor_qi_deviation = yes
		}
		random = {
			chance = 15
			death = { death_reason = death_accident }
		}
	}
	else_if = {
		limit = {
			exists = var:murim_body_tempering_level
			var:murim_body_tempering_level >= 4
		}
		# Levels 4-6: Major damage
		add_stress = 40
		add_trait = wounded_1
		add_character_modifier = { modifier = murim_body_tempering_recovery months = 3 }
		random = {
			chance = 25
			murim_apply_minor_qi_deviation = yes
		}
	}
	else = {
		# Levels 1-3: Minor damage
		add_stress = 20
		random = {
			chance = 20
			add_trait = wounded_1
		}
	}
}

murim_apply_minor_qi_deviation = {
	# Escalating qi deviation
	if = {
		limit = { has_trait = murim_qi_deviation_crippling }
		# Already crippling - chance of death
		random = {
			chance = 50
			death = { death_reason = death_accident }
		}
	}
	else_if = {
		limit = { has_trait = murim_qi_deviation_major }
		remove_trait = murim_qi_deviation_major
		add_trait = murim_qi_deviation_crippling
	}
	else_if = {
		limit = { has_trait = murim_qi_deviation_minor }
		remove_trait = murim_qi_deviation_minor
		add_trait = murim_qi_deviation_major
	}
	else = {
		add_trait = murim_qi_deviation_minor
	}
	trigger_event = { id = murim_grind.0060 days = 3 }
}

murim_attempt_breakthrough_effect = {
	add_character_flag = { flag = murim_attempting_breakthrough days = 1 }
	random = {
		chance = murim_breakthrough_success_chance
		# SUCCESS
		change_variable = { name = murim_cultivation_rank add = 1 }
		set_variable = { name = murim_cultivation_xp value = 0 }
		# Swap cultivation trait
		if = {
			limit = { has_trait = murim_outer_qi }
			remove_trait = murim_outer_qi
			add_trait = murim_qi_gathering
		}
		else_if = {
			limit = { has_trait = murim_qi_gathering }
			remove_trait = murim_qi_gathering
			add_trait = murim_foundation
		}
		else_if = {
			limit = { has_trait = murim_foundation }
			remove_trait = murim_foundation
			add_trait = murim_golden_core
		}
		else_if = {
			limit = { has_trait = murim_golden_core }
			remove_trait = murim_golden_core
			add_trait = murim_nascent_soul
		}
		else_if = {
			limit = { has_trait = murim_nascent_soul }
			remove_trait = murim_nascent_soul
			add_trait = murim_spirit_severing
		}
		add_prestige = 200
		send_interface_toast = {
			title = murim_breakthrough_success_toast
			left_icon = root
		}
	}
	modifier = {
		# FAILURE
		factor = 0
	}
	fallback = {
		add_trait = murim_breakthrough_failure
		add_stress = 50
		random = {
			chance = 20
			murim_apply_minor_qi_deviation = yes
		}
		add_character_flag = { flag = murim_breakthrough_cooldown days = 730 }
		send_interface_toast = {
			title = murim_breakthrough_failure_toast
			left_icon = root
		}
	}
}

murim_grant_sparring_xp_effect = {
	change_variable = { name = murim_cultivation_xp add = murim_sparring_xp_gain }
	if = {
		limit = { NOT = { exists = var:murim_duels_won } }
		set_variable = { name = murim_duels_won value = 0 }
	}
	change_variable = { name = murim_duels_won add = 1 }
	if = {
		limit = {
			var:murim_duels_won >= 20
			NOT = { has_trait = murim_sparring_addict }
			NOT = { has_trait = murim_hermit_cultivator }
		}
		random = {
			chance = 40
			add_trait = murim_sparring_addict
		}
	}
	add_character_flag = { flag = murim_sparring_cooldown days = 30 }
}

murim_apply_forbidden_study_effect = {
	change_variable = { name = murim_cultivation_xp add = 5 }
	add_character_flag = { flag = murim_studying_forbidden days = 1 }
	if = {
		limit = { NOT = { murim_has_forbidden_mark_trigger = yes } }
		add_trait = murim_forbidden_mark_1
	}
	else_if = {
		limit = { has_trait = murim_forbidden_mark_1 }
		random = {
			chance = 30
			remove_trait = murim_forbidden_mark_1
			add_trait = murim_forbidden_mark_2
		}
	}
	else_if = {
		limit = { has_trait = murim_forbidden_mark_2 }
		random = {
			chance = 40
			remove_trait = murim_forbidden_mark_2
			add_trait = murim_forbidden_mark_3
		}
	}
	else_if = {
		limit = { has_trait = murim_forbidden_mark_3 }
		random = {
			chance = 25
			add_trait = lunatic
		}
	}
	add_character_flag = { flag = murim_forbidden_study_cooldown days = 90 }
}

murim_apply_disciple_training_effect = {
	scope:disciple = {
		change_variable = { name = murim_cultivation_xp add = murim_disciple_training_xp }
		add_character_flag = { flag = murim_has_master days = -1 }
	}
	if = {
		limit = { NOT = { exists = var:murim_disciples_trained } }
		set_variable = { name = murim_disciples_trained value = 0 }
	}
	change_variable = { name = murim_disciples_trained add = 1 }
	if = {
		limit = {
			var:murim_disciples_trained >= 3
			NOT = { has_trait = murim_master_teacher }
		}
		add_trait = murim_master_teacher
	}
	add_character_flag = { flag = murim_training_disciple days = 180 }
	scope:disciple = {
		add_character_flag = { flag = murim_training_disciple days = 180 }
	}
}

murim_apply_qi_exchange_effect = {
	change_variable = { name = murim_qi_reserves add = -20 }
	add_character_modifier = { modifier = murim_qi_exchange_given months = 3 }
	scope:target = {
		if = {
			limit = { has_trait = murim_qi_deviation_minor }
			random = {
				chance = 30
				remove_trait = murim_qi_deviation_minor
			}
		}
		add_character_modifier = { modifier = murim_qi_exchange_received months = 6 }
		add_opinion = {
			modifier = murim_qi_exchange_gratitude
			target = root
		}
	}
}

murim_remove_breakthrough_failure_effect = {
	if = {
		limit = { has_trait = murim_breakthrough_failure }
		remove_trait = murim_breakthrough_failure
	}
	remove_character_flag = murim_breakthrough_cooldown
}
