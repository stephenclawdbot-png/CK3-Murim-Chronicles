### EVENT 1: Grand Tournament Opening Ceremony ###
murim_grand_tourn_exp2.0001 = {
	type = character_event
	title = murim_grand_tourn_exp2.0001.t
	desc = murim_grand_tourn_exp2.0001.desc
	theme = martial_arts
	
	trigger = {
		martial >= 14
		prowess >= 12
		NOT = { has_character_flag = grand_tournament_entered }
		age >= 16

		base = 1
		modifier = { add = 3 has_trait = tournament_champion }
		modifier = { add = 2 prowess >= 18 }

	option = {

			set_variable = { name = tournament_wins value = 0 }
			set_variable = { name = tournament_bracket value = 1 }
			trigger_event = { id = murim_grand_tourn_exp2.0002 days = { 3 7 } }
		}
		ai_chance = { base = 30 }

	# Option C: Enter in disguise

		name = murim_grand_tourn_exp2.0001.c

### EVENT 2: First Round Match ###
murim_grand_tourn_exp2.0002 = {
	type = character_event
	title = murim_grand_tourn_exp2.0002.t
	desc = murim_grand_tourn_exp2.0002.desc
	theme = martial_arts
	
	trigger = {
		has_character_flag = grand_tournament_entered
		has_variable = tournament_bracket
		var:tournament_bracket == 1
	}
	
	immediate = {
		random_ruler = {
			limit = {
				NOT = { this = root }
				prowess >= 10
				martial >= 10
				is_alive = yes
			}
			save_scope_as = first_round_opponent
		}
	}
	
	# Option A: Fight with full power from the start
	option = {
		name = murim_grand_tourn_exp2.0002.a
		duel = {
			skill = prowess
			target = scope:first_round_opponent
			60 = {
				desc = murim_grand_tourn_exp2.0002.a.success
				compare_modifier = { value = scope:duel_value multiplier = 3.0 min = -49 }
				add_prestige = 400
				change_variable = { name = tournament_wins add = 1 }
				set_variable = { name = tournament_bracket value = 2 }
				hidden_effect = {
					trigger_event = { id = murim_grand_tourn_exp2.0003 days = { 3 7 } }
				}
			}
			40 = {
				desc = murim_grand_tourn_exp2.0002.a.failure
				compare_modifier = { value = scope:duel_value multiplier = -3.0 min = -49 }
				add_prestige = -200
				increase_wounds_effect = { reason = tournament }
				remove_character_flag = grand_tournament_entered
			}
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Test the opponent first - defensive strategy
	option = {
		name = murim_grand_tourn_exp2.0002.b
		duel = {
			skill = martial
			target = scope:first_round_opponent

				desc = murim_grand_tourn_exp2.0002.b.success
				compare_modifier = { value = scope:duel_value multiplier = 2.5 min = -49 }
				add_prestige = 300
				add_martial_skill = 1
				change_variable = { name = tournament_wins add = 1 }
				set_variable = { name = tournament_bracket value = 2 }
				hidden_effect = {
					trigger_event = { id = murim_grand_tourn_exp2.0003 days = { 3 7 } }
				}
			}
			45 = {
				desc = murim_grand_tourn_exp2.0002.b.failure
				compare_modifier = { value = scope:duel_value multiplier = -2.5 min = -49 }
				add_prestige = -100
				remove_character_flag = grand_tournament_entered
			}
		}
		ai_chance = { base = 35 }
	}
	
	# Option C: Use a secret technique to end it quickly
	option = {
		name = murim_grand_tourn_exp2.0002.c
		trigger = {
			OR = {
				has_trait = technique_master
				has_trait = custom_offensive_art
				has_trait = void_blade_master
			}
		}
		duel = {
			skill = prowess
			target = scope:first_round_opponent
			75 = {
				desc = murim_grand_tourn_exp2.0002.c.success
				add_prestige = 600
				add_dread = 15
				change_variable = { name = tournament_wins add = 1 }
				set_variable = { name = tournament_bracket value = 2 }
				hidden_effect = {
					trigger_event = { id = murim_grand_tourn_exp2.0003 days = { 3 7 } }
				}
			}
			25 = {
				desc = murim_grand_tourn_exp2.0002.c.failure
				add_prestige = -300
				remove_character_flag = grand_tournament_entered
			}
		}

### EVENT 8: Young Dragon vs Grand Master - Legendary Matchup ###
murim_grand_tourn_exp2.0008 = {
	type = character_event
	title = murim_grand_tourn_exp2.0008.t
	desc = murim_grand_tourn_exp2.0008.desc
	theme = martial_arts
	
	trigger = {
		OR = {
			AND = { age <= 25 prowess >= 18 }
			AND = { age >= 45 prowess >= 20 }
		}
		NOT = { has_character_flag = legendary_matchup_done }
	}
	
	weight_multiplier = { base = 1 modifier = { add = 5 has_trait = grand_tournament_champion } }
	
	immediate = {
		if = {
			limit = { age <= 25 }
			random_ruler = {
				limit = { age >= 45 prowess >= 18 martial >= 18 }
				save_scope_as = legendary_opponent
			}
		}
		else = {
			random_ruler = {
				limit = { age <= 25 prowess >= 16 martial >= 14 }
				save_scope_as = legendary_opponent
			}
		}
	}
	
	# Option A: Fight the legendary battle
	option = {
		name = murim_grand_tourn_exp2.0008.a
		add_character_flag = { flag = legendary_matchup_done days = -1 }
		duel = {
			skill = prowess
			target = scope:legendary_opponent
			50 = {
				desc = murim_grand_tourn_exp2.0008.a.success
				compare_modifier = { value = scope:duel_value multiplier = 4.0 min = -49 }
				add_prestige = 2000
				add_trait = legendary_duelist
				add_prowess = 5
			}
			50 = {
				desc = murim_grand_tourn_exp2.0008.a.failure
				compare_modifier = { value = scope:duel_value multiplier = -4.0 min = -49 }
				add_prestige = 500
				add_martial_skill = 2
			}
		}
		ai_chance = { base = 60 }
	}
	
	# Option B: Decline - mutual respect without combat
	option = {
		name = murim_grand_tourn_exp2.0008.b
		add_character_flag = { flag = legendary_matchup_done days = -1 }
		add_prestige = 800
		reverse_add_opinion = {
			target = scope:legendary_opponent
			modifier = worthy_rival_opinion
			opinion = 30
		}
		ai_chance = { base = 25 }
	}
	
	# Option C: Train together instead of fighting
	option = {
		name = murim_grand_tourn_exp2.0008.c
		add_character_flag = { flag = legendary_matchup_done days = -1 }
		add_martial_skill = 3
		add_prowess = 3
		add_prestige = 600
		scope:legendary_opponent = {
			add_martial_skill = 2
			add_prowess = 2
		}
		ai_chance = { base = 15 }
	}
}
