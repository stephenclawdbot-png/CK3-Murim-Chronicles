######################################
# 3050 - AMBUSH ATTACK
# Attacked while traveling between locations
######################################

murim_duel.3050 = {
	type = character_event
	title = murim_duel.3050.t
	desc = murim_duel.3050.desc
	theme = martial
	left_portrait = {
		character = scope:ambusher
		animation = personality_bold
	}
	right_portrait = {
		character = root
		animation = shock
	}
	override_background = {
		reference = wilderness
	}

	trigger = {
		is_alive = yes
		has_character_flag = murim_initiated
		any_living_character = {
			has_character_flag = murim_initiated
			NOT = { this = root }
			OR = {
				has_relation_rival = root
				has_opinion_modifier = { modifier = murim_duel_humiliation target = root }
				has_opinion_modifier = { modifier = murim_crippled_by_enemy target = root }
			}
		}
		NOT = { has_character_flag = murim_ambush_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			any_relation = {
				type = rival
				has_trait = deceitful
			}
		}
	}

	immediate = {
		random_living_character = {
			limit = {
				has_character_flag = murim_initiated
				NOT = { this = root }
				OR = {
					has_relation_rival = root
					has_opinion_modifier = { modifier = murim_duel_humiliation target = root }
				}
			}
			save_scope_as = ambusher
		}
		create_character_memory = {
			type = murim_ambushed_memory
			participants = {
				attacker = scope:ambusher
			}
		}
	}

	# Fight back with full force
	option = {
		name = murim_duel.3050.a
		custom_tooltip = murim_duel_fight_ambush_tt
		duel = {
			skill = prowess
			target = scope:ambusher
			40 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				desc = murim_duel_survived_ambush_tt
				add_prestige = 500
				add_prowess = 1
				scope:ambusher = {
					add_trait = wounded_2
					add_prestige = -400
				}
				create_character_memory = {
					type = murim_survived_ambush_victory_memory
					participants = {
						defeated_ambusher = scope:ambusher
					}
				}
				hidden_effect = {
					trigger_event = { id = murim_rank.0001 days = { 7 14 } }
				}
			}
			25 = {
				desc = murim_duel_ambush_escaped_tt
				add_trait = wounded_1
				add_prestige = 100
			}
			35 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				desc = murim_duel_ambush_defeated_tt
				add_trait = wounded_3
				add_prestige = -500
				random = {
					chance = 20
					add_trait = incapable
				}
				random = {
					chance = 15
					add_trait = craven
				}
				scope:ambusher = {
					add_prestige = 300
				}
				hidden_effect = {
					trigger_event = { id = murim_training.0001 days = { 60 120 } }
				}
			}
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_ambush_cooldown
				years = 2
			}
		}
		ai_chance = {
			base = 60
			modifier = { add = 20 has_trait = brave }
		}
	}

	# Try to flee
	option = {
		name = murim_duel.3050.b
		custom_tooltip = murim_duel_flee_ambush_tt
		hidden_effect = {
			random_list = {
				40 = {
					modifier = { add = 20 prowess >= 16 }
					add_prestige = -100
					add_character_modifier = {
						modifier = murim_narrow_escape
						years = 1
					}
				}
				60 = {
					modifier = { add = -15 prowess >= 16 }
					add_trait = wounded_2
					add_prestige = -300
					random = {
						chance = 30
						add_trait = craven
					}
				}
			}
		}
		hidden_effect = {
			set_character_flag = {
				flag = murim_ambush_cooldown
				years = 2
			}
		}
		ai_chance = {
			base = 40
			modifier = { add = 30 has_trait = craven }
		}
	}
}
