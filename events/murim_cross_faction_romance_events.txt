# EVENT 0001 - Chance Encounter: Love at First Sight
# Triggered: on_action yearly pulse, any martial artist age 16-40
# AI: Full romantic pursuit with weighted personality
murim_romance.0101 = {
    type = character_event
    title = murim_romance.0101.t
    desc = murim_romance.0101.desc
    theme = seduction
    
    left_portrait = {
        character = root
        animation = flirtation
    }
    right_portrait = {
        character = scope:romantic_interest
        animation = personality_compassionate
    }
    
    trigger = {
        is_alive = yes
        age >= 16
        age <= 45
        is_married = no
        NOT = { has_character_flag = forbidden_romance_active }
        OR = {
            murim_is_martial_artist_trigger = yes
            has_trait = murim_sect_leader
            has_trait = murim_wandering_cultivator
        }
        any_living_character = {
            is_alive = yes
            age >= 16
            age <= 45
            NOT = { this = root }
            is_married = no
            # Must be from opposing faction
            OR = {
                AND = {
                    root = { has_trait = murim_orthodox_disciple }
                    has_trait = murim_unorthodox_elder
                }
                AND = {
                    root = { has_trait = murim_unorthodox_elder }
                    has_trait = murim_orthodox_disciple
                }
                AND = {
                    root = { has_trait = murim_shaolin_elder }
                    OR = {
                        has_trait = murim_phantom_blade_elder
                        has_trait = murim_poison_flower_elder
                    }
                }
                AND = {
                    root = { has_trait = murim_heavens_sword_elder }
                    has_trait = murim_namgung_elder
                }
                AND = {
                    root = { has_trait = murim_tang_clan_elder }
                    has_trait = murim_iron_fist_elder
                }
                # Any two different clans for rival romance
                NOT = {
                    any_relation = {
                        type = friend
                        this = prev
                    }
                }
            }
        }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 2.0
            has_trait = lustful
        }
        modifier = {
            factor = 1.5
            has_trait = compassionate
        }
        modifier = {
            factor = 0.3
            has_trait = chaste
        }
        modifier = {
            factor = 0.5
            age >= 35
        }
    }
    
    immediate = {
        random_living_character = {
            limit = {
                is_alive = yes
                age >= 16
                age <= 45
                NOT = { this = root }
                is_married = no
                NOT = { is_allied_to = root }
            }
            save_scope_as = romantic_interest
        }
    }
    
    # Option A: Pursue the forbidden romance
    option = {
        name = murim_romance.0101.a
        trigger = {
            exists = scope:romantic_interest
        }
        add_character_flag = {
            flag = forbidden_romance_active
            days = 3650
        }
        set_relation_potential_lover = scope:romantic_interest
        add_character_modifier = {
            modifier = murim_forbidden_love_bloom
            years = 3
        }
        reverse_add_opinion = {
            target = scope:romantic_interest
            modifier = murim_romantic_attraction
            opinion = 30
        }
        # Chain to secret meetings
        trigger_event = {
            id = murim_romance.0102
            days = { 30 90 }
        }
        stress_impact = {
            shy = medium_stress_impact_gain
            chaste = major_stress_impact_gain
        }
        ai_chance = {
            base = 40
            modifier = {
                factor = 3.0
                has_trait = lustful
            }
            modifier = {
                factor = 2.0
                has_trait = brave
            }
            modifier = {
                factor = 2.0
                has_trait = compassionate
            }
            modifier = {
                factor = 0.2
                has_trait = chaste
            }
            modifier = {
                factor = 0.3
                has_trait = cynical
            }
        }
    }
    
    # Option B: Admire from afar but do not pursue
    option = {
        name = murim_romance.0101.b
        add_stress = 15
        add_character_modifier = {
            modifier = murim_unrequited_longing
            years = 2
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                has_trait = shy
            }
            modifier = {
                factor = 2.0
                has_trait = content
            }
        }
    }
    
    # Option C: Reject feelings - duty over love
    option = {
        name = murim_romance.0101.c
        add_stress = 25
        add_prestige = 50
        add_piety = 50
        add_character_modifier = {
            modifier = murim_iron_heart_discipline
            years = 3
        }
        stress_impact = {
            compassionate = medium_stress_impact_gain
            lustful = major_stress_impact_gain
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 3.0
                has_trait = chaste
            }
            modifier = {
                factor = 2.0
                has_trait = zealous
            }
            modifier = {
                factor = 2.0
                has_trait = murim_orthodox_disciple
            }
        }
    }
}
