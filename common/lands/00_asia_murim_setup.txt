# MURIM CHRONICLES -- ASIA MAP SETUP

# Defines the geographical framework for the Murim world
# using real vanilla CK3 title IDs from the East Asia map.

# This file establishes:
#   1. Murim Region Definitions (4 zones)

# ==============================================================
# 1. MURIM REGION DEFINITIONS
# Four concentric zones radiating outward from the cultural
# heartland of the Central Plains (Zhongyuan).
# Each zone defines which duchies/kingdoms belong to it and
# what baseline modifiers characters in that zone receive.
# ==============================================================

# --- HEARTLAND (Zhongyuan / Central Plains) ---
# The cradle of martial arts. Highest sect density,
# fastest cultivation, most political intrigue.
murim_heartland = {
	color = { 200 160 60 }

	provinces = {
		# Song Dynasty core -- the cultural and martial heart
		d_kaifeng			# Kaifeng (historical Song capital)
		d_luoyang			# Luoyang (ancient capital, Buddhist temples)
		d_xiangyang			# Xiangyang (legendary wuxia battleground)
		d_nanjing			# Nanjing (southern capital)
		d_hangzhou			# Hangzhou (Song southern capital, Lin'an)
		d_chengdu			# Chengdu (Sichuan, Emei Sect territory)
		d_jiankang			# Jiankang (Six Dynasties capital)
		d_yangzhou			# Yangzhou (Grand Canal hub, merchant sects)
	}

	modifier = {
		murim_cultivation_speed_mult = 0.25	# +25% cultivation XP gain
		murim_sect_influence_mult = 0.20		# +20% sect political power
		murim_herb_discovery_mult = 0.15		# +15% chance to find rare herbs
		monthly_prestige = 0.3					# Prestige from living in the heartland
	}
}

# --- OUTER MARCHES (Frontier Provinces) ---

murim_outer_marches = {

		# Eastern coast
		d_qingzhou			# Qingzhou (Shandong, Mount Tai nearby)

		# Southern expansion
		d_changsha			# Changsha (Hunan, southern orthodox sects)
		d_fuzhou			# Fuzhou (Fujian, coastal martial arts)
		d_guangzhou			# Guangzhou (Canton, southern kung fu)

		# Western reaches
		d_lanzhou			# Lanzhou (Gansu corridor)
		d_liangzhou			# Liangzhou (Silk Road gateway)
	}

	modifier = {
		murim_cultivation_speed_mult = 0.10	# +10% cultivation XP
		murim_sect_influence_mult = 0.10		# +10% sect power
		murim_herb_discovery_mult = 0.20		# +20% herb discovery (wilder terrain)
		monthly_prestige = 0.1
	}
}

		d_lhasa				# Lhasa (Tibet, tantric cultivation)

		# Southeast Asian marches
		d_thanglong			# Thang Long (Dai Viet, Vietnamese martial arts)

		d_kaesong			# Kaesong (Goryeo capital)

	}

	}

	heartland_imperial_petition = {
		region = murim_heartland
		type = political
		description = "Petition the Imperial Court for sect recognition"

			prestige >= 200
		}

			murim_sect_influence_mult = 0.15

		}
	}

	outer_marches_border_defense = {
		region = murim_outer_marches
		type = military

		}
		effects = {
			monthly_gold = 0.5
			murim_combat_xp_mult = 0.20
			murim_jianghu_standing_gain = 2
		}
	}

	# Frontier: Ancient tomb exploration
	frontier_ancient_tomb_expedition = {
		region = murim_frontier
		type = exploration
		description = "Mount an expedition to explore ancient cultivation tombs"
