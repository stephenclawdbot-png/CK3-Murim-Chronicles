######################################
# SHATTERED MERIDIANS
# Cannot cultivate above current rank ever again
# Caused by: devastating qi explosion, critical breakthrough failure during combat
######################################

shattered_meridians = {
	index = 1
	category = personality

	# Permanent cultivation ceiling
	character_modifier = {
		monthly_piety = -0.5
		stress_gain_mult = 0.15
		health = -0.25
		learning = -3
	}

	# Cannot be removed by normal means
	can_have = {
		is_alive = yes
	}

	# Incompatible with qi_crippled (meridians must exist to be shattered vs fully gone)
	opposites = {
		qi_crippled
	}

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_trait = shattered_meridians
				}
				desc = trait_shattered_meridians_desc
			}
		}
	}

	# AI impact: will never attempt breakthrough, avoids cultivation decisions
	ai_energy = -25
	ai_boldness = -20
	ai_honor = 0
	ai_greed = 0
	ai_rationality = 15

	flag = murim_combat_trauma
	flag = murim_permanent_injury
	flag = cannot_advance_cultivation
}

######################################

#   + wounded/severely_wounded existing conditions
#   - has_trait = robust (reduces severity)
#
# murim_roll_combat_trauma:
#   Called after every duel resolution
#   Rolls for: physical injury, mental trauma, technique phobia
#   Scales with severity value above
#   Winners roll at 50% of loser's chance (winners can still get hurt)
