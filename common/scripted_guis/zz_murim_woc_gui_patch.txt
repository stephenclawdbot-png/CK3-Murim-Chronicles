# Murim Chronicles - WoC GUI Patch
# Overrides Ways of China scripted GUI triggers that crash on landless characters
# The vanilla WoC GUI checks realm/county data every frame on open character windows
# but has no guards against landless adventurers, causing tens of thousands of null scope
# errors and eventual access violation crash at 0x00007FF6825B3A49
#
# Fix: Add is_landed = yes guard so landless characters skip these checks entirely
# NOTE: zz_ prefix ensures this loads AFTER WoC's own files alphabetically

has_additional_my_realm_window_woc = {
	scope = character
	is_shown = {
		is_landed = yes
		has_realm_law = realm_law_woc_enabled
	}
}

can_has_councillor_woc_debug = {
	scope = character
	is_valid = {
		is_landed = yes
		has_realm_law = realm_law_woc_enabled
	}
}
