# Murim Chronicles - Martial Awareness Trait
# The GATEWAY trait: you know the martial world exists, but aren't trained yet.
# Prerequisite for most martial training events. Some discovery events grant this first,
# others skip straight to training (master teaches you directly).
# INTEGRATION: Checked by murim_discovery_triggers.txt (has_martial_awareness),
#              murim_discovery_events.txt (granted by several events),
#              murim_clan_recruitment_events.txt (recruitment preference for aware characters),
#              murim_training_events.txt (some training requires awareness first)
# Cross-refs: murim_triggers.txt (murim_is_martial_artist_trigger excludes awareness-only),
#             murim_youth_tournament_events.txt (spectators gain awareness)
# GENDER: No restrictions. Available to all characters regardless of sex.

# martial_awareness = you understand qi, sects, techniques, the jianghu.
# You are NOT a martial artist. You are AWARE that martial artists exist.
# Think of it as: you've seen behind the curtain but haven't stepped through.

martial_awareness = {
	index = 2000
	
	# Category: special/lifestyle
	category = lifestyle
	
	# Not genetic, not inherited — this is learned through experience
	genetic = no
	inherit_chance = 0
	
	# Can be both good and bad — it opens doors but also dangers
	good = yes
	
	# Visible to the character and others
	shown_in_encyclopedia = yes
	
	# Compatibility: does NOT conflict with martial traits.
	# You can have martial_awareness AND then gain a martial trait.
	# Once you gain a martial trait, awareness is redundant but harmless.
	# We do NOT remove it when gaining martial training — it stays as backstory.
	
	# Modifiers: subtle benefits from KNOWING about the martial world
	# You can't fight, but you can recognize fighters, avoid danger, and make connections.
	diplomacy = 1          # You understand a hidden social network
	intrigue = 1           # You know secrets most people don't
	martial = 0            # Knowing about martial arts doesn't make you a fighter
	stewardship = 0
	learning = 1           # You've absorbed forbidden knowledge
	prowess = 0            # Zero combat benefit — you're aware, not trained
	
	# Opinion modifiers
	# Martial artists respect someone who understands their world
	# Non-martial people think you're a bit odd for believing in "qi warriors"
	same_opinion = 5       # Other aware characters understand each other
	opposite_opinion = 0
	
	# AI weight: AI characters value this trait moderately
	# It's a stepping stone, not a destination
	ai_rationality = 5
	ai_boldness = 5
	ai_energy = 5
	
	# Ruler designer cost: low — it's just awareness
	ruler_designer_cost = 5
	
	# Description key for localization
	desc = martial_awareness_desc
	
	# Icon
	icon = murim_dao_heart
}

# IMPLEMENTATION NOTES:
# 1. martial_awareness is granted by:
#    - murim_discover.1004 (tea house conversation, option A: listen carefully)
#    - murim_discover.2001 (martial demonstration, option A: attend personally)
#    - murim_discover.3002 (tournament invitation, option B: watch and learn)
#    - murim_discover.4001 (crossing sect territory, various options)
#    - murim_youth_tournament_events.txt (spectators at Young Dragon Tournament)
#    - Any event where the character WITNESSES martial arts without training
#
# 2. martial_awareness is NOT required for:
#    - Direct training events (a master teaches you — skips awareness)
#    - murim_discover.1002 (injured stranger teaches you directly)
#    - murim_discover.1006 (ancient cave self-study)
#    - murim_discover.2002 (bodyguard offers training)
#    - Being recruited by a clan (recruitment skips to training)
#
# 3. martial_awareness BOOSTS recruitment chances:
#    - murim_recruit.0020 (normal recruitment) gives +2 weight to aware characters
#    - murim_recruit.0002 (emergency recruitment) gives +2 weight
#    - Aware characters are more likely to be noticed by scouting elders
#
# 4. martial_awareness is PERMANENT once gained:
#    - You can't un-know the martial world
#    - Even if you never train, you carry this knowledge
#    - It affects dialogue options in other events (aware characters get extra choices)
#
# 5. AI behavior with martial_awareness:
#    - AI characters with awareness are 3x more likely to seek martial training
#    - AI parents with awareness are 2x more likely to accept clan recruitment for children
#    - AI rulers with awareness are more likely to hire martial bodyguards
#    - AI courtiers with awareness actively seek out martial training opportunities
