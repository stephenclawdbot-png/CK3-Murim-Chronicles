# Celestial Court Summons — emperor demands martial service
murim_celestial.2001 = {
	type = character_event
	title = murim_celestial.2001.t
	desc = murim_celestial.2001.desc
	theme = diplomacy
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 4
		NOT = { has_character_flag = murim_celestial_court_cooldown }
	}

	immediate = {
		set_character_flag = { flag = murim_celestial_court_cooldown years = 5 }
	}

	# Serve the emperor — gain political favor
	option = {
		name = murim_celestial.2001.a
		add_prestige = 300
		add_diplomacy = 2
		hidden_effect = {
			add_character_modifier = { modifier = murim_imperial_favor years = 5 }
		}
		ai_chance = {
			base = 40
			modifier = { add = 20 diplomacy >= 14 }
		}
	}

	# Refuse — the martial world stands apart
	option = {
		name = murim_celestial.2001.b
		add_prestige = -100
		add_dread = 15
		add_martial = 1
		ai_chance = {
			base = 35
			modifier = { add = 20 has_trait = brave }
		}
	}

	# Negotiate terms — serve but on your conditions
	option = {
		name = murim_celestial.2001.c
		add_intrigue = 2
		add_gold = 200
		ai_chance = {
			base = 25
			modifier = { add = 20 intrigue >= 14 }
		}
	}
}

# Immortal Ascension Trial begins
murim_immortal_ascension.0001 = {
	type = character_event
	title = murim_immortal_ascension.0001.t
	desc = murim_immortal_ascension.0001.desc
	theme = faith
	left_portrait = { character = root animation = personality_zealous }

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 6
		NOT = { has_character_flag = murim_ascension_attempted }
	}

	immediate = {
		set_character_flag = murim_ascension_attempted
		set_character_flag = murim_ascension_trial_declared
	}

	# Begin the ascension — shed mortality
	option = {
		name = murim_immortal_ascension.0001.a
		random_list = {
			40 = {
				modifier = { add = prowess factor = 0.3 }
				modifier = { add = learning factor = 0.3 }
				add_prestige = 2000
				add_prowess = 10
				change_variable = { name = murim_cultivation_xp add = 500 }
				hidden_effect = {
					add_character_modifier = { modifier = murim_immortal_cultivator }
				}
			}
			35 = {
				add_prestige = 500
				add_prowess = 3
				change_variable = { name = murim_cultivation_xp add = 200 }
				add_stress = 40
			}
			25 = {
				add_health = -3
				add_stress = 60
				add_prestige = -200
			}
		}
		ai_chance = {
			base = 50
			modifier = { add = 30 prowess >= 30 }
		}
	}

	# Postpone — continue cultivating in the mortal realm
	option = {
		name = murim_immortal_ascension.0001.b
		add_stress = 10
		change_variable = { name = murim_cultivation_xp add = 50 }
		remove_character_flag = murim_ascension_attempted
		ai_chance = { base = 30 }
	}

	# Forge a new path — neither mortal nor immortal
	option = {
		name = murim_immortal_ascension.0001.c
		add_prowess = 5
		add_learning = 3
		change_variable = { name = murim_cultivation_xp add = 300 }
		add_prestige = 750
		ai_chance = { base = 20 }
	}
}
