murim_custom_clan_expansion4.0001 = {
	type = character_event
	title = murim_custom_clan_exp4.0001.t
	desc = murim_custom_clan_exp4.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		has_variable = murim_custom_clan_founder
		has_trait = murim_cultivation_realm_5
		prowess >= 25
		NOT = { has_character_flag = supreme_path_offered }
	}

	weight_multiplier = {
		base = 15
		modifier = { add = 20 has_trait = ambitious }
		modifier = { add = 10 prowess >= 30 }
	}

	immediate = {
		set_character_flag = supreme_path_offered

	option = {
		name = murim_custom_clan_exp4.0001.a

		add_prowess = 10
		add_martial = 5

	}

		name = murim_custom_clan_exp4.0001.b

		add_character_modifier = { modifier = murim_supreme_path_shadow years = 30 }
		add_intrigue = 10

}

	type = character_event

	theme = martial
	left_portrait = { character = root animation = war_over_win }

	trigger = { has_character_modifier = murim_supreme_path_martial }

	immediate = {
		random_living_character = {
			limit = { murim_is_martial_artist_trigger = yes prowess >= 28 NOT = { this = root } }

		name = murim_custom_clan_exp4.0002.a
		trigger = { exists = scope:supreme_trial_opponent }
		duel = {
			skill = prowess
			target = scope:supreme_trial_opponent
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -49 }
				desc = murim_custom_clan_exp4.0002.a.win
				root = {
					add_trait = murim_supreme_martial_lord
					add_prowess = 15
					add_prestige = 5000
					add_character_modifier = { modifier = murim_supreme_martial_aura years = 100 }
				}
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }
				desc = murim_custom_clan_exp4.0002.a.lose
				root = { add_trait = wounded_3 add_prestige = -1000 }

		}
	}
	option = {
		name = murim_custom_clan_exp4.0002.b

	}

murim_custom_clan_expansion4.0009 = {
	type = character_event
	title = murim_custom_clan_exp4.0009.t
	desc = murim_custom_clan_exp4.0009.desc
	theme = intrigue
	left_portrait = { character = root animation = worry }

	trigger = {
		has_variable = murim_custom_clan_founder
		is_ruler = yes
		age >= 55
		any_child = { age >= 20 }
	}

	immediate = {
		random_child = {
			limit = { age >= 20 }
			save_scope_as = clan_heir
			weight = { base = 1 modifier = { add = 5 murim_is_martial_artist_trigger = yes } modifier = { add = 3 prowess >= 15 } }
		}
		random_vassal = {
			limit = { murim_is_martial_artist_trigger = yes prowess >= 18 NOT = { is_child_of = root } }
			save_scope_as = rival_heir
		}
	}

	option = {
		name = murim_custom_clan_exp4.0009.a
		trigger = { exists = scope:clan_heir }
		scope:clan_heir = {
			add_prestige = 1000
			add_character_modifier = { modifier = murim_custom_clan_heir years = 30 }
		}
		add_character_modifier = { modifier = murim_legacy_secured years = 20 }
		if = { limit = { exists = scope:rival_heir }
			scope:rival_heir = { add_opinion = { modifier = murim_passed_over target = root opinion = -40 } }
		}
	}
	option = {
		name = murim_custom_clan_exp4.0009.b
		add_character_modifier = { modifier = murim_meritocratic_succession years = 10 }
		trigger_event = { id = murim_custom_clan_expansion4.0010 days = { 30 60 } }
	}
	option = {
		name = murim_custom_clan_exp4.0009.c
		add_character_modifier = { modifier = murim_no_succession_plan years = 10 }
		add_stress = 20
	}
}

murim_custom_clan_expansion4.0010 = {
	type = character_event
	title = murim_custom_clan_exp4.0010.t
	desc = murim_custom_clan_exp4.0010.desc
	theme = martial
	left_portrait = { character = root animation = war_over_win }

	trigger = { has_character_modifier = murim_meritocratic_succession }

	immediate = {
		every_vassal = {
			limit = { murim_is_martial_artist_trigger = yes prowess >= 12 age >= 18 }
			add_to_temporary_list = succession_candidates
		}
		ordered_in_list = {
			list = succession_candidates
			order_by = prowess
			max = 1
			save_scope_as = strongest_candidate
		}
	}

	option = {
		name = murim_custom_clan_exp4.0010.a
		if = { limit = { exists = scope:strongest_candidate }
			scope:strongest_candidate = {
				add_prestige = 2000
				add_character_modifier = { modifier = murim_custom_clan_new_leader years = 30 }
			}
		}
		add_character_modifier = { modifier = murim_fair_succession_complete years = 20 }
	}
	option = {
		name = murim_custom_clan_exp4.0010.b
		trigger = { prowess >= 25 exists = scope:strongest_candidate }
		duel = {
			skill = prowess
			target = scope:strongest_candidate
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -49 }
				desc = murim_custom_clan_exp4.0010.b.win
				root = {
					add_prestige = 3000
					add_character_modifier = { modifier = murim_eternal_clan_founder years = 50 }
				}
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }
				desc = murim_custom_clan_exp4.0010.b.lose
				root = { add_trait = wounded_2 add_prestige = -500 }
			}
		}
	}
	option = {
		name = murim_custom_clan_exp4.0010.c
		add_character_modifier = { modifier = murim_council_of_elders_chosen years = 30 }
		add_diplomacy = 3
		every_in_list = {
			list = succession_candidates
			limit = { age >= 40 }
			add_character_modifier = { modifier = murim_elder_council_member years = 20 }
		}
	}
}
