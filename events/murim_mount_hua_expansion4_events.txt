murim_mount_hua_expansion4.0001 = {
	type = character_event
	title = murim_mount_hua_exp4.0001.t
	desc = murim_mount_hua_exp4.0001.desc
	theme = war
	left_portrait = { character = root animation = personality_bold }
	right_portrait = { character = scope:rival_sect_leader animation = anger }

	trigger = {
		has_trait = murim_clan_mount_hua
		is_ruler = yes
		any_living_character = {
			murim_is_martial_artist_trigger = yes
			NOT = { has_trait = murim_clan_mount_hua }
			is_ruler = yes
			opinion = { target = root value <= -30 }
		}
		NOT = { has_character_flag = mount_hua_great_war_started }
	}

	weight_multiplier = {
		base = 20
		modifier = { add = 30 has_trait = ambitious }
	}

	immediate = {
		set_character_flag = mount_hua_great_war_started
		random_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
				NOT = { has_trait = murim_clan_mount_hua }
				is_ruler = yes
				opinion = { target = root value <= -30 }
			}
			save_scope_as = rival_sect_leader
		}
	}

	option = {
		name = murim_mount_hua_exp4.0001.a
		trigger = { exists = scope:rival_sect_leader }
		add_prestige = 300
		add_character_modifier = { modifier = murim_mount_hua_war_leader years = 10 }
		trigger_event = { id = murim_mount_hua_expansion4.0002 days = { 30 60 } }
	}
	option = {
		name = murim_mount_hua_exp4.0001.b
		add_diplomacy = 2
		if = { limit = { exists = scope:rival_sect_leader }
			scope:rival_sect_leader = { add_opinion = { modifier = murim_peace_proposal target = root opinion = 20 } }
		}
		add_character_modifier = { modifier = murim_mount_hua_diplomat years = 10 }
	}
	option = {
		name = murim_mount_hua_exp4.0001.c
		add_character_modifier = { modifier = murim_mount_hua_fortified years = 15 }
		add_gold = -200
		add_martial = 3
	}
}

	type = character_event
	title = murim_mount_hua_exp4.0002.t
	desc = murim_mount_hua_exp4.0002.desc
	theme = war
	left_portrait = { character = root animation = war_over_win }

	trigger = { has_character_modifier = murim_mount_hua_war_leader }

	immediate = {
		random_living_character = {
			limit = { murim_is_martial_artist_trigger = yes NOT = { has_trait = murim_clan_mount_hua } is_ruler = yes }
			save_scope_as = war_target
		}
	}

	option = {
		name = murim_mount_hua_exp4.0002.a
		trigger = { exists = scope:war_target prowess >= 20 }
		duel = {
			skill = prowess
			target = scope:war_target
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -49 }
				desc = murim_mount_hua_exp4.0002.a.win
				root = {
					add_prestige = 2000
					add_character_modifier = { modifier = murim_plum_blossom_conqueror years = 20 }
				}
				scope:war_target = { add_prestige = -500 add_trait = wounded_2 }
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }
				desc = murim_mount_hua_exp4.0002.a.lose
				root = { add_prestige = -500 add_trait = wounded_3 }
			}
		}
	}
	option = {
		name = murim_mount_hua_exp4.0002.b
		add_gold = -300
		add_character_modifier = { modifier = murim_hired_mercenaries years = 5 }
		add_prestige = 500
		trigger_event = { id = murim_mount_hua_expansion4.0003 days = { 30 60 } }
	}
	option = {
		name = murim_mount_hua_exp4.0002.c
		add_prestige = -200
		add_piety = 300
		remove_character_modifier = murim_mount_hua_war_leader
		add_character_modifier = { modifier = murim_mount_hua_retreat years = 10 }
	}
}
