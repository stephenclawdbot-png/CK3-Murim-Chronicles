# 1. Wulin Congress Emergency Vote
# The martial alliance calls an emergency congress.
# A critical vote determines the fate of the entire murim world.
##############################
murim_politics_exp2.0001 = {
	type = character_event
	title = murim_politics_exp2.0001.t
	desc = murim_politics_exp2.0001.desc
	theme = diplomacy
	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		is_ruler = yes
		is_alive = yes
		diplomacy >= 8
		NOT = { has_character_modifier = wulin_congress_vote_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			diplomacy >= 15
		}
		modifier = {
			add = 0.3
			has_trait = ambitious
		}
	}

	immediate = {
		add_character_modifier = {
			modifier = wulin_congress_vote_cooldown
			years = 5
		}
	}

	# Vote to mobilize the entire orthodox alliance against the unorthodox threat
	option = {
		name = murim_politics_exp2.0001.a
		add_prestige = 200
		add_diplomacy_skill = 1
		add_piety = 100
		add_character_modifier = {
			modifier = orthodox_war_mobilization
			years = 3
		}
		every_vassal = {
			limit = { is_alive = yes }
			add_opinion = {
				target = root
				modifier = murim_righteous_leader_opinion
				opinion = 15
			}
		}
		create_character_memory = {
			type = murim_congress_war_vote_memory
		}
		hidden_effect = {
			random_list = {
				60 = {
					# Vote passes - massive war preparation
					trigger_event = { id = murim_faction_war.0001 days = { 30 60 } }
				}
				40 = {
					# Vote fails - political humiliation
					add_prestige = -300
					add_stress = 40
					trigger_event = { id = murim_politics.0001 days = { 14 30 } }
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = zealous }
			modifier = { add = 20 has_trait = brave }
		}
	}

	# Vote for diplomatic resolution - propose a grand tournament to settle disputes
	option = {
		name = murim_politics_exp2.0001.b
		add_prestige = 100
		add_diplomacy_skill = 2
		add_character_modifier = {
			modifier = peace_broker_fame
			years = 5
		}
		create_character_memory = {
			type = murim_congress_peace_vote_memory
		}
		# Chain: grand tournament organized
		trigger_event = { id = murim_duel.0005 days = { 60 120 } }
		ai_chance = {
			base = 35
			modifier = { add = 20 has_trait = just }
			modifier = { add = 15 has_trait = patient }
		}
	}

	# Secretly vote to protect your own sect's interests above all
	option = {
		name = murim_politics_exp2.0001.c
		add_intrigue_skill = 2
		add_gold = 200
		add_piety = -100
		add_character_modifier = {
			modifier = political_schemer_reputation
			years = 3
		}
		create_character_memory = {
			type = murim_congress_self_interest_vote_memory
		}
		hidden_effect = {
			random_list = {

					# Scheme discovered - reputation damaged

							target = root

			}

# 2. Martial Law Declaration
# A sect leader declares martial law over their territory
# after a devastating attack or internal crisis.
##############################
murim_politics_exp2.0002 = {
	type = character_event
	title = murim_politics_exp2.0002.t
	desc = murim_politics_exp2.0002.desc
	theme = war
	left_portrait = {
		character = root
		animation = anger
	}

	trigger = {
		is_ruler = yes
		is_alive = yes
		martial >= 10
		OR = {
			any_vassal = {
				opinion = { target = root value <= -30 }
			}
			has_trait = paranoid
			has_character_modifier = sect_under_attack
		}
		NOT = { has_character_modifier = martial_law_cooldown }
	}

		base = 0.8
		modifier = {
			add = 0.5

		}
		modifier = {
			add = 0.3
			martial >= 15
		}
	}

	immediate = {
		add_character_modifier = {
			modifier = martial_law_cooldown
			years = 5
		}
	}

	# Full martial law - lock down all sect territory
	option = {
		name = murim_politics_exp2.0002.a
		add_prestige = 100
		add_martial_skill = 1
		add_tyranny = 20
		add_character_modifier = {
			modifier = martial_law_enforced
			years = 2
		}
		every_vassal = {
			limit = { is_alive = yes }
			add_opinion = {
				target = root
				modifier = murim_martial_law_opinion
				opinion = -20
			}
		}
		create_character_memory = {
			type = murim_martial_law_declared_memory
		}
		hidden_effect = {
			# Some may rebel
			random_vassal = {
				limit = {
					is_alive = yes
					opinion = { target = root value <= -10 }
				}
				weight = {
					base = 1
					modifier = {
						add = 2
						has_trait = brave
					}
				}
				save_scope_as = potential_rebel
				random_list = {
					40 = {
						# Vassal rebels openly
						trigger_event = { id = murim_sect.0003 days = { 14 30 } }
					}
					60 = { }
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = paranoid }
		}
	}

	# Selective curfew - target only suspected dissidents
	option = {
		name = murim_politics_exp2.0002.b
		add_intrigue_skill = 1
		add_tyranny = 5
		add_character_modifier = {
			modifier = selective_crackdown
			years = 2
		}
		create_character_memory = {
			type = murim_selective_crackdown_memory
		}
		random_list = {
			60 = {
				# Successfully root out traitors
				add_prestige = 200
				random_vassal = {
					limit = {
						is_alive = yes
						opinion = { target = root value <= -20 }
					}
					save_scope_as = caught_dissident
					imprison = {
						target = scope:caught_dissident
						type = dungeon
					}
				}
			}
			40 = {
				# Wrong people targeted - unrest grows
				add_stress = 30
				every_vassal = {
					limit = { is_alive = yes }
					add_opinion = {
						target = root
						modifier = murim_false_accusation_opinion
						opinion = -15
					}
				}
			}
		}
		ai_chance = {
			base = 35
			modifier = { add = 15 has_trait = deceitful }
		}
	}

	# Show strength without martial law - hold a public demonstration
	option = {
		name = murim_politics_exp2.0002.c
		add_prestige = 50
		add_stress = 20
		duel = {
			skill = prowess
			target = root
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				min = 5
				desc = murim_politics_exp2.0002.c.success
				add_prestige = 300
				every_vassal = {
					limit = { is_alive = yes }
					add_opinion = {
						target = root
						modifier = murim_strength_demonstration_opinion
						opinion = 20
					}
				}
				create_character_memory = {
					type = murim_power_demonstration_memory
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min = 5
				desc = murim_politics_exp2.0002.c.failure
				add_prestige = -200
				add_trait = wounded_1
				create_character_memory = {
					type = murim_failed_demonstration_memory
				}
				# Chain: emboldened rebels
				trigger_event = { id = murim_sect.0003 days = { 7 14 } }
			}
		}
		ai_chance = {
			base = 25
			modifier = { add = 20 has_trait = brave }
		}
	}
