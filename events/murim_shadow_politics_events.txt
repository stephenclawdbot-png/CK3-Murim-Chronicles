# Murim Chronicles - Heavenly Demon Shadow Politics Events
# The Heavenly Demon Sect infiltrates GOVERNMENTS.
# DESIGN: If any Chinese/Korean ruler has aggressive/deceitful/ambitious traits,
#   15% yearly chance Heavenly Demon approaches secretly.
# VANILLA SEPARATION EXCEPTION: This is the ONE system where martial and political
#   worlds INTENTIONALLY overlap. The HD uses martial power to influence politics.
#   But the overlap is CHOSEN by event options, not automatic.
# CHINA-CENTRIC: Only Chinese/Korean rulers can be infiltrated.
# INTEGRATION: Uses murim_clan_triggers.txt (murim_hd_can_infiltrate_ruler_trigger)
# INTEGRATION: Feeds murim_faction_events.txt (exposure triggers faction crisis)
# AI BEHAVIOR: AI Heavenly Demon leaders ALWAYS attempt infiltration. Player can choose.

namespace = murim_shadow

# ═══════════════════════════════════════════
# 0001 - HD LEADER INITIATES INFILTRATION (yearly pulse)
# AI always attempts. Player chooses.
# ═══════════════════════════════════════════
murim_shadow.0001 = {
	type = character_event
	title = murim_shadow.0001.t
	desc = murim_shadow.0001.desc
	# "As leader of the Heavenly Demon Sect, you survey the political
	#  landscape. The Emperor of [realm] is [trait-based description].
	#  Such weakness is an invitation. A puppet on the throne would give
	#  the Demonic Path access to armies, treasuries, and the legal
	#  authority to crush the orthodox sects. Do you send your Shadow Envoy?"
	theme = intrigue
	
	trigger = {
		has_character_modifier = heavenly_demon_sect_leader
		NOT = { has_character_flag = hd_infiltration_cooldown }
	}
	
	# Option A: Send the Shadow Envoy
	option = {
		name = murim_shadow.0001.a
		# "Find me a ruler who can be... persuaded."
		# Find eligible ruler
		random_living_character = {
			limit = {
				murim_hd_can_infiltrate_ruler_trigger = yes
				NOT = { this = root }
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					has_trait = ambitious
				}
				modifier = {
					add = 3
					has_trait = deceitful
				}
				modifier = {
					add = 2
					has_trait = greedy
				}
				modifier = {
					add = 5
					highest_held_title_tier >= tier_kingdom
				}
			}
			save_scope_as = infiltration_target
		}
		# Send envoy to target
		scope:infiltration_target = {
			trigger_event = {
				id = murim_shadow.0010
				days = { 30 90 }
			}
		}
		set_character_flag = {
			flag = hd_infiltration_cooldown
			years = 3
		}
		ai_chance = { base = 100 } # AI ALWAYS attempts
	}
	
	# Option B: Wait for better opportunity
	option = {
		name = murim_shadow.0001.b
		# "Not yet. The time is not right."
		set_character_flag = {
			flag = hd_infiltration_cooldown
			years = 1
		}
		ai_chance = { base = 0 } # AI never waits
	}
}

# 0010 - RULER RECEIVES HD APPROACH
# The ruler is contacted by the Heavenly Demon's agent
# ═══════════════════════════════════════════
murim_shadow.0010 = {
	type = character_event
	title = murim_shadow.0010.t
	desc = murim_shadow.0010.desc
	# "A figure appears in your chamber — impossible, given the guards.
	#  They move like smoke, bow with predatory grace, and speak with
	#  silk-wrapped menace: 'The Heavenly Demon Sect extends its hand
	#  to you, great ruler. We offer: protection from ALL martial threats,
	#  elimination of your political rivals through... accidents, intelligence
	#  from our networks spanning all of China, and the most powerful
	#  bodyguards the martial world has ever produced. In return, we ask
	#  only... accommodation. A seat at your court. Freedom to operate
	#  within your borders. And occasionally, your authority directed
	#  against those who oppose us.' 15% base acceptance."
	theme = intrigue
	
	trigger = {
		murim_hd_can_infiltrate_ruler_trigger = yes
	}
	
	# Option A: Accept the Heavenly Demon's offer
	option = {
		name = murim_shadow.0010.a
		# "Your... services are welcome at my court."
		add_character_modifier = {
			modifier = heavenly_demon_puppet
			years = -1 # permanent until exposed
		}
		# HD agent becomes court physician/advisor (hidden role)
		# Ruler gains benefits
		add_intrigue_skill = 2
		add_character_modifier = {
			modifier = hd_shadow_protection
			years = -1
			# +10 scheme power defense, +5 dread, -10 piety
		}
		set_character_flag = hd_puppet_ruler
		# Persecution of orthodox sects begins
		trigger_event = {
			id = murim_shadow.0020
			days = { 90 365 }
		}
		# Risk of discovery starts
		trigger_event = {
			id = murim_shadow.0030
			days = { 365 1825 }
		}
		create_character_memory = {
			type = murim_accepted_demon_pact
		}
		stress_impact = {
			honest = major_stress_gain
			paranoid = minor_stress_gain
		}
		ai_chance = {
			base = 15 # 15% base
			modifier = {
				add = 10
				has_trait = ambitious
			}
			modifier = {
				add = 10
				has_trait = deceitful
			}
			modifier = {
				add = 10
				has_trait = greedy
			}
			modifier = {
				add = -20
				has_trait = just
			}
			modifier = {
				add = -20
				has_trait = honest
			}
		}
	}
	
	# Option B: Reject firmly
	option = {
		name = murim_shadow.0010.b
		# "Leave before I have you arrested."
		set_character_flag = {
			flag = hd_infiltration_rejected
			years = 10
		}
		add_character_modifier = {
			modifier = murim_rejected_demonic_offer
			years = 5
		}
		# HD may retaliate
		random_list = {
			70 = { } # Nothing happens
			30 = {
				# HD sends assassin as punishment
				trigger_event = {
					id = murim_shadow.0015
					days = { 30 180 }
				}
			}
		}
		ai_chance = { base = 60 }
	}
	
	# Option C: Pretend to accept, then alert orthodox sects
	option = {
		name = murim_shadow.0010.c
		# "Of course... (Guards, send word to the Murim Alliance immediately.)"
		trigger = {
			OR = {
				has_trait = deceitful
				has_trait = just
				intrigue >= 12
			}
		}
		set_character_flag = {
			flag = hd_infiltration_rejected
			years = 20
		}
		add_character_modifier = {
			modifier = murim_defender_of_righteousness
			years = 10
		}
		# Alert the orthodox faction — major intelligence coup
		# Orthodox clans gain opinion of this ruler
		ai_chance = {
			base = 15
			modifier = {
				add = 15
				has_trait = just
			}
		}
	}
	
	# Option D: Negotiate better terms
	option = {
		name = murim_shadow.0010.d
		# "Your offer is... interesting. But I want more."
		trigger = { has_trait = greedy }
		# Extended negotiation — better deal but deeper entanglement
		add_gold = 500
		add_character_modifier = {
			modifier = heavenly_demon_puppet
			years = -1
		}
		add_character_modifier = {
			modifier = hd_deep_entanglement
			years = -1
		}
		set_character_flag = hd_puppet_ruler
		trigger_event = {
			id = murim_shadow.0020
			days = { 60 180 }
		}
		trigger_event = {
			id = murim_shadow.0030
			days = { 180 1095 } # Discovered faster due to greed
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = greedy
			}
		}
	}
}
