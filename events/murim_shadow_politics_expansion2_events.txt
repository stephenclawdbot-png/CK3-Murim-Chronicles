# === EVENT 0001: HEAVENLY DEMON APPROACHES THE EMPEROR ===
murim_shadow_pol2.0001 = {
	type = character_event
	title = murim_shadow_pol2.0001.t
	desc = murim_shadow_pol2.0001.desc
	theme = intrigue
	
	trigger = {
		has_trait = murim_clan_heavenly_demon
		has_trait = murim_clan_leader
		NOT = { has_character_flag = heavenly_demon_puppet_plan }
		any_ruler = {
			has_title = title:e_murim_emperor
		}
	}
	
	immediate = {
		set_character_flag = { flag = heavenly_demon_puppet_plan days = -1 }
		random_ruler = {
			limit = { has_title = title:e_murim_emperor }
			save_scope_as = murim_emperor
		}
	}
	
	# Option A: Begin the puppet emperor scheme
	option = {
		name = murim_shadow_pol2.0001.a
		add_character_modifier = {
			modifier = murim_puppet_emperor_scheme
			years = -1
		}
		add_intrigue_skill = 3
		trigger_event = { id = murim_shadow_pol2.0002 days = { 90 365 } }
		ai_chance = { base = 60 }
	}
	
	# Option B: Assassinate and replace
	option = {
		name = murim_shadow_pol2.0001.b
		add_dread = 40
		random_list = {
			40 = {
				scope:murim_emperor = {
					death = { death_reason = death_murder killer = root }
				}
				trigger_event = { id = murim_shadow_pol2.0005 days = 30 }
			}
			60 = {
				# Failed assassination - emperor is protected
				add_prestige = -500
				scope:murim_emperor = {
					add_opinion = { target = root modifier = murim_assassination_attempt_opinion opinion = -100 }
				}
			}
		}
		ai_chance = { base = 20 }
	}
	
	# Option C: Ignore the imperial court entirely
	option = {
		name = murim_shadow_pol2.0001.c
		add_prestige = 200
		ai_chance = { base = 20 }
	}
}
