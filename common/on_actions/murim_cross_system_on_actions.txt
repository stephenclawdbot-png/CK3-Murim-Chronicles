# Murim Cross-System On Actions
# Links espionage, black market, tournaments, calamity, and shadow networks
# AI martial artists trigger these the same as players

# === YEARLY PULSE: CROSS-SYSTEM CHECKS ===
on_yearly_pulse = {
	# Espionage networks can discover black market contacts
	random_events = {
		chance_to_happen = 15
		trigger = {
			has_character_flag = murim_spy_network_active
			NOT = { has_character_flag = murim_black_market_contact }
		}
		# Spy network discovers underground market
		100 = murim_cross_system.0001
	}

	# Black market dealers get drawn into espionage
	random_events = {
		chance_to_happen = 12
		trigger = {
			has_character_flag = murim_black_market_contact
			NOT = { has_character_flag = murim_spy_network_active }
			intrigue >= 12
		}
		100 = murim_cross_system.0002
	}

	# Tournament champions attract shadow sect recruitment
	random_events = {
		chance_to_happen = 20
		trigger = {
			has_character_flag = murim_tournament_champion
			NOT = { has_character_flag = murim_shadow_sect_member }
			has_trait = ambitious
		}
		100 = murim_cross_system.0003
	}

	# Calamity survivors with espionage skills become resistance leaders
	random_events = {
		chance_to_happen = 18
		trigger = {
			has_character_flag = murim_calamity_survivor
			has_character_flag = murim_spy_network_active
			NOT = { has_character_flag = murim_resistance_leader }
		}
		100 = murim_cross_system.0004
	}

	# Clan succession wars trigger espionage escalation
	random_events = {
		chance_to_happen = 25
		trigger = {
			has_character_flag = murim_succession_war_active
			any_courtier = {
				has_character_flag = murim_spy_network_active
			}
		}
		100 = murim_cross_system.0005
	}

	# Blood oath allies share intelligence networks
	random_events = {
		chance_to_happen = 15
		trigger = {
			has_character_flag = murim_blood_oath_bound
			has_character_flag = murim_spy_network_active
			any_ally = {
				has_character_flag = murim_blood_oath_bound
				NOT = { has_character_flag = murim_spy_network_active }
			}
		}
		100 = murim_cross_system.0006
	}

	# Forbidden technique users attract tribunal attention
	random_events = {
		chance_to_happen = 30
		trigger = {
			has_character_flag = murim_forbidden_technique_user
			NOT = { has_character_flag = murim_tribunal_warrant }
			any_ruler = {
				has_character_flag = murim_tribunal_judge
			}
		}
		100 = murim_cross_system.0007
	}

	# Immortal ascension seekers discover hidden vaults
	random_events = {
		chance_to_happen = 10
		trigger = {
			has_character_flag = murim_immortal_path_active
			NOT = { has_character_flag = murim_vault_discovered }
			learning >= 15
		}
		100 = murim_cross_system.0008
	}
}

# === QUARTERLY PULSE: ACTIVE SYSTEM MAINTENANCE ===
# Changed from monthly to quarterly for performance
on_quarterly_pulse = {
	# Active spy networks require gold maintenance
	random_events = {
		chance_to_happen = 30
		trigger = {
			has_character_flag = murim_spy_network_active
			gold < 50
		}
		100 = murim_cross_system.0060
	}

	# Calamity phase advancement check
	random_events = {
		chance_to_happen = 20
		trigger = {
			has_character_flag = murim_calamity_phase_active
		}
		100 = murim_cross_system.0061
	}

	# Shadow sect influence spread
	random_events = {
		chance_to_happen = 15
		trigger = {
			has_character_flag = murim_shadow_sect_member
			any_neighboring_ruler = {
				NOT = { has_character_flag = murim_shadow_sect_member }
			}
		}
		100 = murim_cross_system.0062
	}
}
