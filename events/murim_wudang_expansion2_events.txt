# === EVENT 0001: TAIJI GRAND MASTERY TRIAL ===
murim_wudang_exp2.0001 = {
	type = character_event
	title = murim_wudang_exp2.0001.t
	desc = murim_wudang_exp2.0001.desc
	theme = martial_arts
	
	trigger = {
		has_trait = murim_clan_wudang
