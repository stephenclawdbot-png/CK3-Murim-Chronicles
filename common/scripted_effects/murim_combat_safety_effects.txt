# Murim Combat Safety Effects
# Protects minors (under 18) from death in martial combat
# Kids can be injured in tournaments/duels but cannot die
# Also handles civilian casualty mechanics

# Use instead of direct death = {} in combat/duel events
# Minors get severe injury instead of death
# $REASON$ = death_battle, death_duel, death_murder etc.
murim_combat_death_or_injury_effect = {
	if = {
		limit = { age < 18 }
		# Minors survive but are severely injured
		add_trait = wounded_3
		add_character_modifier = {
			modifier = murim_near_death_injury
			years = 3
		}
		add_stress = 50
	}
	if = {
		limit = { age >= 18 }
		death = { death_reason = $REASON$ }
	}
}

# Version with killer specified
# $REASON$ = death_battle, death_duel, death_murder etc.
# $KILLER$ = the character who caused the death
murim_combat_death_or_injury_with_killer_effect = {
	if = {
		limit = { age < 18 }
		add_trait = wounded_3
		add_character_modifier = {
			modifier = murim_near_death_injury
			years = 3
		}
		add_stress = 50
	}
	if = {
		limit = { age >= 18 }
		death = { death_reason = $REASON$ killer = $KILLER$ }
	}
}

# Civilian casualty effect - martial artist fights can harm bystanders
# Called on the martial artist who caused collateral damage
murim_civilian_casualty_effect = {
	random_list = {
		40 = {
			# Minor collateral - property damage only
			custom_tooltip = murim_civilian_damage_minor_tt
			add_gold = -25
			add_prestige = -50
		}
		35 = {
			# Civilians injured
			custom_tooltip = murim_civilian_damage_injured_tt
			add_gold = -50
			add_prestige = -150
			add_dread = 5
			add_piety = -50
			add_character_modifier = {
				modifier = murim_civilian_harm_guilt
				years = 3
			}
		}
		25 = {
			# Civilian death - serious consequences
			custom_tooltip = murim_civilian_damage_death_tt
			add_gold = -100
			add_prestige = -300
			add_dread = 15
			add_piety = -150
			add_tyranny = 10
			add_character_modifier = {
				modifier = murim_civilian_killer_shame
				years = 10
			}
			stress_impact = {
				compassionate = massive_stress_impact_gain
				just = major_stress_impact_gain
			}
			# Orthodox martial artists lose standing
			if = {
				limit = { has_character_flag = murim_orthodox_member }
				change_variable = { name = murim_reputation add = -20 }
			}
		}
	}
}
