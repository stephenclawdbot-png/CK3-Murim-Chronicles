namespace = murim_succession

# Claim leadership — initiate succession
murim_succession.0100 = {
	type = character_event
	title = murim_succession.0100.t
	desc = murim_succession.0100.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		has_variable = murim_martial_artist
		has_character_flag = murim_succession_claimant
	}

	# Challenge through combat
	option = {
		name = murim_succession.0100.a
		trigger = { prowess >= 18 }
		random_list = {
			50 = {
				modifier = { add = prowess factor = 0.3 }
				hidden_effect = { murim_win_succession_effect = yes }
				set_character_flag = { flag = murim_is_clan_leader years = -1 }
			}
			50 = {
				hidden_effect = { murim_lose_succession_effect = yes }
			}
		}
		ai_chance = { base = 45 modifier = { add = 25 has_trait = brave } }
	}

	# Challenge through politics — gather support
	option = {
		name = murim_succession.0100.b
		trigger = { diplomacy >= 14 }
		random_list = {
			50 = {
				modifier = { add = diplomacy factor = 0.3 }
				hidden_effect = { murim_win_succession_effect = yes }
				set_character_flag = { flag = murim_is_clan_leader years = -1 }
			}
			50 = {
				hidden_effect = { murim_lose_succession_effect = yes }
			}
		}
		ai_chance = { base = 35 modifier = { add = 20 diplomacy >= 16 } }
	}

	# Withdraw claim
	option = {
		name = murim_succession.0100.c
		remove_character_flag = murim_succession_claimant
		add_stress = 15
		ai_chance = { base = 20 }
	}
}

# Succession triggered by leader's death (AI)
murim_succession.0110 = {
	type = character_event
	title = murim_succession.0110.t
	desc = murim_succession.0110.desc
	theme = martial

	trigger = {
		has_variable = murim_martial_artist
		murim_succession_eligible_trigger = yes
	}

	# Step forward as candidate
	option = {
		name = murim_succession.0110.a
		hidden_effect = { murim_claim_clan_leadership_effect = yes }
		trigger_event = { id = murim_succession.0100 days = 14 }
		ai_chance = { base = 50 modifier = { add = 30 has_trait = ambitious } }
	}

	# Support another candidate
	option = {
		name = murim_succession.0110.b
		add_prestige = 75
		add_diplomacy = 1
		ai_chance = { base = 30 }
	}

	# Remain neutral
	option = {
		name = murim_succession.0110.c
		add_stress = -10
		ai_chance = { base = 20 }
	}
}
