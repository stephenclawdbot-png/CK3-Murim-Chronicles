# === SEEK FORBIDDEN KNOWLEDGE ===
murim_seek_forbidden_knowledge_decision = {
	title = murim_seek_forbidden_knowledge_decision
	desc = murim_seek_forbidden_knowledge_decision_desc
	selection_tooltip = murim_seek_forbidden_knowledge_decision_tooltip

	is_shown = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 3
		NOT = { has_trait = murim_corruption_stage_4 }
	}

	is_valid = {
		murim_can_learn_forbidden_technique_trigger = yes
		NOT = { has_character_flag = murim_seeking_forbidden_cooldown }
	}

	effect = {
		set_character_flag = { flag = murim_seeking_forbidden_cooldown years = 3 }
		trigger_event = { id = murim_forbidden_technique.0001 days = 7 }
		custom_tooltip = murim_seek_forbidden_knowledge_tt
	}

	ai_check_interval = 60
	ai_will_do = {
		base = 5
		modifier = { add = 15 has_trait = ambitious }
		modifier = { add = 10 has_trait = sadistic }
		modifier = { add = -20 has_trait = just }
		modifier = { add = -15 has_trait = compassionate }
		modifier = { add = 10 intrigue >= 15 }
	}
}
