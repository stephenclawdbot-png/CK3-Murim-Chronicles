# Murim Chronicles - Missing Scripted Triggers Patch
# Stub definitions for triggers referenced but not previously defined
# These ensure all trigger calls resolve without CK3 errors

# Can be exposed as a secret cultivator
murim_can_be_exposed_as_cultivator_trigger = {
	has_variable = murim_martial_artist
	is_ruler = yes
	NOT = { has_character_flag = murim_cultivation_exposed }
}

# Can become a shadow broker
murim_can_become_shadow_broker_trigger = {
	intrigue >= 12
	has_variable = murim_martial_artist
	NOT = { has_character_flag = murim_is_shadow_broker }
}

# Technique tier triggers
murim_can_unlock_any_tier1_trigger = {
	has_variable = murim_martial_artist
	prowess >= 10
}

murim_can_unlock_any_tier2_trigger = {
	has_variable = murim_martial_artist
	prowess >= 15
}

murim_can_unlock_any_tier3_trigger = {
	has_variable = murim_martial_artist
	prowess >= 20
}

murim_has_tier1_technique_trigger = {
	OR = {
		has_character_flag = murim_has_tier1_technique
		has_trait = murim_outer_qi
		has_trait = murim_qi_gathering
	}
}

murim_has_tier2_technique_trigger = {
	OR = {
		has_character_flag = murim_has_tier2_technique
		has_trait = murim_foundation
		has_trait = murim_golden_core
	}
}

murim_has_tier3_technique_trigger = {
	OR = {
		has_character_flag = murim_has_tier3_technique
		has_trait = murim_nascent_soul
		has_trait = murim_spirit_severing
		has_trait = murim_heavenly_tribulation
	}
}

# Clan membership checks
murim_is_any_clan_member_trigger = {
	OR = {
		has_trait = mount_hua_disciple
		has_trait = wudang_disciple
		has_trait = shaolin_monk
		has_trait = heavenly_demon_disciple
		has_trait = beggar_sect_member
		has_trait = iron_fist_disciple
		has_trait = poison_flower_disciple
		has_trait = phantom_blade_disciple
		has_trait = tang_clan_member
		has_trait = tang_clan_associate
	}
}

# Region checks

	murim_is_china_region_trigger = yes
}

murim_is_goryeo_trigger = {
	murim_is_goryeo_region_trigger = yes
}

# Rank checks
murim_is_clan_elder_trigger = {
	has_variable = murim_clan_rank
	var:murim_clan_rank >= 4
}

murim_is_clan_leader_trigger = {
	has_variable = murim_clan_rank
	var:murim_clan_rank >= 6
}

murim_is_sect_leadership_trigger = {
	has_variable = murim_clan_rank
	var:murim_clan_rank >= 5
}

murim_is_grandmaster_or_above_trigger = {
	OR = {
		has_trait = murim_nascent_soul
		has_trait = murim_spirit_severing
		has_trait = murim_heavenly_tribulation
		prowess >= 25
	}
}

# Orthodox clan check
murim_is_orthodox_clan_trigger = {
	OR = {
		has_trait = mount_hua_disciple
		has_trait = wudang_disciple
		has_trait = shaolin_monk
		has_trait = beggar_sect_member
		has_trait = iron_fist_disciple
		has_trait = tang_clan_member
		has_trait = tang_clan_associate
	}
}

# Recruitment checks
murim_is_recruitable_candidate_trigger = {
	NOT = { murim_is_any_clan_member_trigger = yes }
	NOT = { has_trait = martial_artist }
	age >= 14
	age <= 45
	prowess >= 8
}

murim_is_valid_jianghu_recruit_trigger = {
	has_variable = murim_martial_artist
	NOT = { has_character_flag = murim_jianghu_recruit_cooldown }
	prowess >= 10
}

# Same sect/faction checks
murim_is_same_sect_trigger = {
	exists = var:murim_sect_id
	root = {
		exists = var:murim_sect_id
	}
}

murim_same_faction_as_root_trigger = {
	exists = var:murim_faction_alignment
	root = {
		exists = var:murim_faction_alignment
	}
}

# Wandering master check
murim_is_wandering_master_trigger = {
	has_variable = murim_martial_artist
	is_landed = no
	prowess >= 15
	OR = {
		has_character_flag = murim_wandering_martial_artist
		has_character_flag = murim_wandering_master
		NOT = { murim_is_any_clan_member_trigger = yes }
	}
}
