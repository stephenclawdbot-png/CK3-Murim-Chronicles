# Ranking announcement — a new ranker rises
murim_ranking.0001 = {
	type = character_event
	title = murim_ranking.0001.t
	desc = murim_ranking.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 3
		NOT = { has_character_flag = murim_ranking_announced }
	}

	immediate = {
		set_character_flag = { flag = murim_ranking_announced years = 2 }
	}

	# Accept the recognition with humility
	option = {
		name = murim_ranking.0001.a
		add_prestige = 200
		add_stress = -10
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = humble }
		}
	}

	# Use this fame to attract disciples
	option = {
		name = murim_ranking.0001.b
		add_prestige = 100
		add_diplomacy = 1
		ai_chance = {
			base = 35
			modifier = { add = 20 has_trait = ambitious }
		}
	}

	# Challenge the ranker above you
	option = {
		name = murim_ranking.0001.c
		add_prowess = 2
		add_stress = 15
		hidden_effect = {
			trigger_event = { id = murim_ranking.0003 days = 30 }
		}
		ai_chance = {
			base = 25
			modifier = { add = 30 prowess >= 20 }
		}
	}
}
