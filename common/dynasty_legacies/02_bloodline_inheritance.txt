# Bloodline Purity Mechanics
bloodline_purity = {
	pure_bloodline = {
		name = "Pure Bloodline"
		desc = "Unbroken lineage of martial cultivators."
		
		modifier = {
			inherit_chance_for_martial_traits = 0.3
			martial_skill_inheritance = 0.2
			prowess_inheritance = 0.2
		}
		
		trigger = {
			dynasty_prestige >= 1000
			every_dynasty_member = {
				OR = {
					has_trait = qi_refinement
					has_trait = foundation
					has_trait = core_formation
				}
			}
		}
	}
	
	bloodline_diluted = {
		name = "Diluted Bloodline"
		desc = "Mixed with mortal blood. Potential decreased."
		
		modifier = {
			inherit_chance_for_martial_traits = -0.2
			time_to_cultivate = 0.5
		}
		
		trigger = {
			OR = {
				has_trait = lowborn
				has_marriage_with = { NOT = { has_trait = qi_refinement } }
			}
		}
	}
}
