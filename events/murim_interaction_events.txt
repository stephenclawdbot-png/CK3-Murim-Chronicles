murim_interact.0001 = {
	type = character_event
	title = murim_interact.0001.t
	desc = murim_interact.0001.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:partner
		animation = personality_bold
	}

	trigger = {
		exists = scope:partner
	}

	# Light sparring
	option = {
		name = murim_interact.0001.a
		if = {
			limit = { has_variable = murim_cultivation_xp }
			change_variable = { name = murim_cultivation_xp add = 2 }
		}
		else = {
			set_variable = { name = murim_cultivation_xp value = 2 }
		}
		scope:partner = {
			if = {
				limit = { has_variable = murim_cultivation_xp }
				change_variable = { name = murim_cultivation_xp add = 2 }
			}
			else = {
				set_variable = { name = murim_cultivation_xp value = 2 }
			}
		}
		add_opinion = {
			modifier = murim_sparring_respect
			target = scope:partner
		}
		scope:partner = {
			add_opinion = {
				modifier = murim_sparring_respect
				target = root
			}
		}
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = patient }
		}
	}

	# Intense sparring
	option = {
		name = murim_interact.0001.b
		if = {
			limit = { has_variable = murim_cultivation_xp }
			change_variable = { name = murim_cultivation_xp add = 3 }
		}
		else = {
			set_variable = { name = murim_cultivation_xp value = 3 }
		}
		scope:partner = {
			if = {
				limit = { has_variable = murim_cultivation_xp }
				change_variable = { name = murim_cultivation_xp add = 3 }
			}
			else = {
				set_variable = { name = murim_cultivation_xp value = 3 }
			}
		}
		random = {
			chance = 15
			random_list = {
				50 = {
					add_trait = wounded_1
				}
				50 = {
					scope:partner = { add_trait = wounded_1 }
				}
			}
		}
		add_character_modifier = { modifier = murim_sparring_fatigue days = 30 }
		scope:partner = {
			add_character_modifier = { modifier = murim_sparring_fatigue days = 30 }
		}
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = brave }
			modifier = { add = 20 has_trait = wrathful }
			modifier = { add = -15 has_trait = craven }
		}
	}

	# Death match escalation (only with rivals)
	option = {
		name = murim_interact.0001.c
		trigger = {
			has_relation_rival = scope:partner
		}
		if = {
			limit = { prowess >= scope:partner.prowess }
			if = {
				limit = { has_variable = murim_cultivation_xp }
				change_variable = { name = murim_cultivation_xp add = 6 }
			}
			else = {
				set_variable = { name = murim_cultivation_xp value = 6 }
			}
			add_prestige = 100
			scope:partner = {
				random = {
					chance = 30
					death = { death_reason = death_duel killer = root }
				}
				if = {
					limit = { is_alive = yes }
					add_trait = wounded_2
				}
			}
		}
		else = {
			random = {
				chance = 30
				death = { death_reason = death_duel killer = scope:partner }
			}
			if = {
				limit = { is_alive = yes }
				add_trait = wounded_2
			}
			scope:partner = {
				if = {
					limit = { has_variable = murim_cultivation_xp }
					change_variable = { name = murim_cultivation_xp add = 6 }
				}
				else = {
					set_variable = { name = murim_cultivation_xp value = 6 }
				}
				add_prestige = 100
			}
		}
		ai_chance = {
			base = 5
			modifier = { add = 40 has_trait = wrathful }
			modifier = { add = -30 has_trait = craven }
		}
	}
}

# The death match begins
murim_interact.1001 = {
	type = character_event
	title = murim_interact.1001.t
	desc = murim_interact.1001.desc
	theme = martial
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = scope:defender
		animation = war_over_lose
	}
	override_background = {
		reference = wilderness
	}

	trigger = {
		is_alive = yes
		exists = scope:defender
		scope:defender = { is_alive = yes }
	}

	immediate = {
		# Calculate fight outcome based on prowess + cultivation + luck
		save_temporary_scope_value_as = {
			name = challenger_power
			value = {
				value = prowess
				# Cultivation bonus
				if = {
					limit = { has_trait = murim_spirit_severing }
					add = 15
				}
				else_if = {
					limit = { has_trait = murim_nascent_soul }
					add = 12
				}
				else_if = {
					limit = { has_trait = murim_golden_core }
					add = 8
				}
				else_if = {
					limit = { has_trait = murim_foundation }
					add = 5
				}
				# Technique bonus
				if = {
					limit = { has_trait = murim_heavenly_sword }
					add = 4
				}
				if = {
					limit = { has_trait = murim_divine_fist }
					add = 4
				}
				if = {
					limit = { has_trait = murim_void_strike }
					add = 3
				}
				# Combat experience bonus
				if = {
					limit = {
						has_variable = murim_duel_wins
						var:murim_duel_wins >= 5
					}
					add = 3
				}
				# Wound penalty
				if = {
					limit = { has_trait = wounded_1 }
					add = -3
				}
				if = {
					limit = { has_trait = wounded_2 }
					add = -6
				}
			}
		}
	}

	# All-out lethal assault
	option = {
		name = murim_interact.1001.a
		duel = {
			skill = prowess
			target = scope:defender

			10 = {
				desc = murim_interact.1001.a.crit_win
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2
				}
				# Devastating victory
				add_prestige = 300
				if = {
					limit = { has_variable = murim_duel_wins }
					change_variable = { name = murim_duel_wins add = 1 }
				}
				else = {
					set_variable = { name = murim_duel_wins value = 1 }
				}
				if = {
					limit = { has_variable = murim_cultivation_xp }
					change_variable = { name = murim_cultivation_xp add = 8 }
				}
				else = {
					set_variable = { name = murim_cultivation_xp value = 8 }
				}
				scope:defender = {
					# High chance of death on critical loss
					random_list = {
						40 = {
							death = { death_reason = death_duel killer = root }
						}
						30 = {
							add_trait = wounded_3
							if = {
								limit = { NOT = { has_trait = incapable } }
								random = {
									chance = 40
									add_trait = incapable
								}
							}
						}
						30 = {
							add_trait = wounded_2
							# Qi deviation from devastating defeat
							random = {
								chance = 50
								if = {
									limit = {
										NOR = {
											has_trait = murim_qi_deviation_minor
											has_trait = murim_qi_deviation_severe
											has_trait = murim_qi_deviation_crippling
										}
									}
									add_trait = murim_qi_deviation_severe
								}
							}
						}
					}
					if = {
						limit = {
							is_alive = yes
							has_variable = murim_duel_losses
						}
						change_variable = { name = murim_duel_losses add = 1 }
					}
					else_if = {
						limit = { is_alive = yes }
						set_variable = { name = murim_duel_losses value = 1 }
					}
				}
				# Trigger spectator reactions
				trigger_event = { id = murim_interact.1004 days = 3 }
			}

			6 = {
				desc = murim_interact.1001.a.win
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1
				}
				# Normal victory
				add_prestige = 150
				if = {
					limit = { has_variable = murim_duel_wins }
					change_variable = { name = murim_duel_wins add = 1 }
				}
				else = {
					set_variable = { name = murim_duel_wins value = 1 }
				}
				if = {
					limit = { has_variable = murim_cultivation_xp }
					change_variable = { name = murim_cultivation_xp add = 5 }
				}
				else = {
					set_variable = { name = murim_cultivation_xp value = 5 }
				}
				scope:defender = {
					random_list = {
						25 = {
							death = { death_reason = death_duel killer = root }
						}
						40 = {
							add_trait = wounded_2
						}
						35 = {
							add_trait = wounded_1
							# Combat trauma from losing a death match
							random = {
								chance = 35
								if = {
									limit = {
										NOT = { has_trait = murim_combat_trauma_shell_shock }
									}
									add_trait = murim_combat_trauma_shell_shock
								}
							}
						}
					}
					if = {
						limit = {
							is_alive = yes
							has_variable = murim_duel_losses
						}
						change_variable = { name = murim_duel_losses add = 1 }
					}
					else_if = {
						limit = { is_alive = yes }
						set_variable = { name = murim_duel_losses value = 1 }
					}
				}
				trigger_event = { id = murim_interact.1004 days = 3 }
			}

			6 = {
				desc = murim_interact.1001.a.lose
				compare_modifier = {
					value = scope:duel_value
					multiplier = -1
				}
				# Defeat
				add_prestige = -100
				random_list = {
					25 = {
						death = { death_reason = death_duel killer = scope:defender }
					}
					40 = {
						add_trait = wounded_2
					}
					35 = {
						add_trait = wounded_1
						random = {
							chance = 35
							if = {
								limit = {
									NOT = { has_trait = murim_combat_trauma_shell_shock }
								}
								add_trait = murim_combat_trauma_shell_shock
							}
						}
					}
				}
				if = {
					limit = {
						is_alive = yes
						has_variable = murim_duel_losses
					}
					change_variable = { name = murim_duel_losses add = 1 }
				}
				else_if = {
					limit = { is_alive = yes }
					set_variable = { name = murim_duel_losses value = 1 }
				}
				scope:defender = {
					add_prestige = 150
					if = {
						limit = { has_variable = murim_duel_wins }
						change_variable = { name = murim_duel_wins add = 1 }
					}
					else = {
						set_variable = { name = murim_duel_wins value = 1 }
					}
					if = {
						limit = { has_variable = murim_cultivation_xp }
						change_variable = { name = murim_cultivation_xp add = 5 }
					}
					else = {
						set_variable = { name = murim_cultivation_xp value = 5 }
					}
				}
			}

			3 = {
				desc = murim_interact.1001.a.crit_lose
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2
				}
				# Devastating defeat
				random_list = {
					50 = {
						death = { death_reason = death_duel killer = scope:defender }
					}
					25 = {
						add_trait = wounded_3
						if = {
							limit = { NOT = { has_trait = incapable } }
							random = {
								chance = 40
								add_trait = incapable
							}
						}
					}
					25 = {
						add_trait = wounded_2
						# Qi deviation from devastating loss
						if = {
							limit = {
								NOR = {
									has_trait = murim_qi_deviation_minor
									has_trait = murim_qi_deviation_severe
									has_trait = murim_qi_deviation_crippling
								}
							}
							add_trait = murim_qi_deviation_severe
						}
					}
				}
				if = {
					limit = {
						is_alive = yes
						has_variable = murim_duel_losses
					}
					change_variable = { name = murim_duel_losses add = 1 }
				}
				else_if = {
					limit = { is_alive = yes }
					set_variable = { name = murim_duel_losses value = 1 }
				}
				scope:defender = {
					add_prestige = 300
					if = {
						limit = { has_variable = murim_duel_wins }
						change_variable = { name = murim_duel_wins add = 1 }
					}
					else = {
						set_variable = { name = murim_duel_wins value = 1 }
					}
					if = {
						limit = { has_variable = murim_cultivation_xp }
						change_variable = { name = murim_cultivation_xp add = 8 }
					}
					else = {
						set_variable = { name = murim_cultivation_xp value = 8 }
					}
				}
			}
		}
		ai_chance = {
			base = 60
			modifier = { add = 30 has_trait = brave }
			modifier = { add = 20 has_trait = wrathful }
		}
	}

	# Try to negotiate mercy mid-fight
	option = {
		name = murim_interact.1001.b
		add_prestige = -50
		# Attempt to de-escalate -- diplomacy check
		random = {
			chance = {
				value = diplomacy
				multiply = 3
				max = 60
			}
			# Successful de-escalation -- turns into heavy spar
			add_trait = wounded_1
			scope:defender = {
				add_trait = wounded_1
			}
			if = {
				limit = { has_variable = murim_cultivation_xp }
				change_variable = { name = murim_cultivation_xp add = 3 }
			}
			else = {
				set_variable = { name = murim_cultivation_xp value = 3 }
			}
		}
		random = {
			chance = {
				value = 60
				subtract = {
					value = diplomacy
					multiply = 3
				}
				min = 20
			}
			# Failed -- defender attacks while you hesitate
			add_trait = wounded_2
			add_prestige = -100
			random = {
				chance = 20
				if = {
					limit = { NOT = { has_trait = craven } }
					add_trait = craven
				}
			}
		}
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = craven }
			modifier = { add = 20 has_trait = compassionate }
			modifier = { add = -20 has_trait = brave }
		}
	}
}

						if = {
							limit = { exists = scope:murim_new_follower }

				}
