# === SPY NETWORK DISCOVERS BLACK MARKET ===
murim_cross_system.0001 = {
	type = character_event
	title = murim_cross_system.0001.t
	desc = murim_cross_system.0001.desc
	theme = intrigue
	
	trigger = {
		has_character_flag = murim_spy_network_active
		NOT = { has_character_flag = murim_black_market_contact }
	}
	
	# Your spies discovered an underground market
	option = {
		name = murim_cross_system.0001.a
		# Establish contact
		add_character_flag = murim_black_market_contact
		add_prestige = -100
		add_gold = -50
		add_character_modifier = {
			modifier = murim_underground_connections
			years = 5
		}
	}
	option = {
		name = murim_cross_system.0001.b
		# Report to orthodox authorities
		add_piety = 200
		add_character_flag = murim_orthodox_informant
		remove_character_flag = murim_spy_network_active
	}
	option = {
		name = murim_cross_system.0001.c
		# Exploit both sides
		trigger = { intrigue >= 18 }
		add_character_flag = murim_black_market_contact
		add_character_flag = murim_double_agent
		add_gold = 100
		add_stress = 30
	}
}

# === TOURNAMENT CHAMPION ATTRACTS SHADOW SECT ===
murim_cross_system.0003 = {
	type = character_event
	title = murim_cross_system.0003.t
	desc = murim_cross_system.0003.desc
	theme = skulduggery
	
	trigger = {
		has_character_flag = murim_tournament_champion
		NOT = { has_character_flag = murim_shadow_sect_member }
	}
	
	immediate = {
		random_realm_county = {
			save_scope_as = shadow_meeting_location
		}
	}
	
	option = {
		name = murim_cross_system.0003.a
		# Join the shadow sect
		add_character_flag = murim_shadow_sect_member
		add_prowess = 3
		add_tyranny = 10
		add_character_modifier = {
			modifier = murim_shadow_sect_initiate
			years = -1
		}
	}
	option = {
		name = murim_cross_system.0003.b
		# Refuse and report them
		add_piety = 300
		add_prestige = 200
		add_character_flag = murim_shadow_sect_enemy
	}
	option = {
		name = murim_cross_system.0003.c
		# Infiltrate as double agent
		trigger = { intrigue >= 15 }
		add_character_flag = murim_shadow_sect_member
		add_character_flag = murim_shadow_infiltrator
		add_intrigue_skill = 2
		add_stress = 25
	}
}

# === IMMORTAL SEEKER DISCOVERS VAULT ===
murim_cross_system.0008 = {
	type = character_event
	title = murim_cross_system.0008.t
	desc = murim_cross_system.0008.desc
	theme = learning
	
	trigger = {
		has_character_flag = murim_immortal_path_active
		NOT = { has_character_flag = murim_vault_discovered }
		learning >= 15
	}
	
	option = {
		name = murim_cross_system.0008.a
		# Enter the vault alone
		add_character_flag = murim_vault_discovered
		add_learning_skill = 3
		random_list = {
			50 = {
				add_character_modifier = {
					modifier = murim_ancient_knowledge
					years = -1
				}
			}
			30 = {
				add_character_flag = murim_forbidden_technique_user
				add_prowess = 5
			}
			20 = {
				add_stress = 50
				add_trait = wounded
			}
		}
	}
	option = {
		name = murim_cross_system.0008.b
		# Bring your disciples
		add_character_flag = murim_vault_discovered
		every_courtier = {
			limit = {
				has_character_flag = murim_disciple
				count <= 3
			}
			add_learning_skill = 1
		}
	}
	option = {
		name = murim_cross_system.0008.c
		# Seal it and report to elders
		add_piety = 300
		add_character_flag = murim_vault_sealed
	}
}
