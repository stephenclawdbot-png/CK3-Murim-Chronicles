murim_breakthrough.0001 = {
	type = character_event

murim_breakthrough.0100 = {
	type = character_event
	title = murim_breakthrough.0100.t
	desc = murim_breakthrough.0100.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_honorable
	}

	trigger = { is_alive = yes }

	immediate = {
		remove_character_flag = murim_breakthrough_in_progress
		set_character_flag = {
			flag = murim_breakthrough_cooldown
			years = 3
		}
		# Advance cultivation rank
		if = {
			limit = { has_trait = cultivation_rank_qi_condensation }
			remove_trait = cultivation_rank_qi_condensation
			add_trait = cultivation_rank_foundation
		}
		else_if = {
			limit = { has_trait = cultivation_rank_foundation }
			remove_trait = cultivation_rank_foundation
			add_trait = cultivation_rank_core_formation
		}
		else_if = {
			limit = { has_trait = cultivation_rank_core_formation }
			remove_trait = cultivation_rank_core_formation
			add_trait = cultivation_rank_nascent_soul
		}
		else_if = {
			limit = { has_trait = cultivation_rank_nascent_soul }
			remove_trait = cultivation_rank_nascent_soul
			add_trait = cultivation_rank_spirit_severing
		}
		else_if = {
			limit = { has_trait = cultivation_rank_spirit_severing }
			remove_trait = cultivation_rank_spirit_severing
			add_trait = cultivation_rank_dao_seeking
		}
		else_if = {
			limit = { has_trait = cultivation_rank_dao_seeking }
			remove_trait = cultivation_rank_dao_seeking
			add_trait = cultivation_rank_tribulation
		}
		else_if = {
			limit = { has_trait = cultivation_rank_tribulation }
			remove_trait = cultivation_rank_tribulation
			add_trait = cultivation_rank_immortal_ascension
			trigger_event = { id = murim_breakthrough.0500 days = 7 }
		}
	}

	# Celebrate the breakthrough
	option = {
		name = murim_breakthrough.0100.a
		add_prestige = 500
		add_piety = 200
		stress_impact = { base = -30 }
		if = {
			limit = { var:murim_breakthrough_approach = 3 }
			add_prowess_skill = 3
		}
		else_if = {
			limit = { var:murim_breakthrough_approach = 2 }
			add_prowess_skill = 2
		}
		else = {
			add_prowess_skill = 1
		}
	}

	# Share insights with disciples
	option = {
		name = murim_breakthrough.0100.b
		trigger = {
			any_relation = {
				type = mentor
				count >= 1
			}
		}
		add_prestige = 300
		add_piety = 400
		every_relation = {
			type = ward
			limit = { has_character_flag = murim_initiated }
			add_opinion = {
				target = root
				modifier = murim_master_breakthrough_inspiration
			}
			change_variable = {
				name = murim_cultivation_xp
				add = 50
			}
		}
	}
}
