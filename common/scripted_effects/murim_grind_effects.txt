murim_apply_tempering_failure_effect = {
	# Failure consequences scale with current level
	if = {
		limit = {
			exists = var:murim_body_tempering_level
			var:murim_body_tempering_level >= 7
		}
		# Levels 7-9: Severe - death possible
		add_stress = 60
		add_trait = wounded_2
		add_character_modifier = { modifier = murim_body_tempering_recovery months = 6 }
		random = {
			chance = 30
			murim_apply_minor_qi_deviation = yes
		}
		random = {
			chance = 15
			death = { death_reason = death_accident }
		}
	}
	else_if = {
		limit = {
			exists = var:murim_body_tempering_level
			var:murim_body_tempering_level >= 4
		}
		# Levels 4-6: Major damage
		add_stress = 40
		add_trait = wounded_1
		add_character_modifier = { modifier = murim_body_tempering_recovery months = 3 }
		random = {
			chance = 25
			murim_apply_minor_qi_deviation = yes
		}
	}
	else = {
		# Levels 1-3: Minor setback
		add_stress = 20
		add_character_modifier = { modifier = murim_body_tempering_recovery months = 1 }
		random = {
			chance = 10
			add_trait = wounded_1
		}
	}
	send_interface_toast = {
		title = murim_body_tempering_failure_toast
		left_icon = root
	}
}

murim_apply_meditation_session_effect = {
	# Standard meditation session -- accumulates meditation XP
	if = {
		limit = {
			NOT = { exists = var:murim_meditation_xp }
		}
		set_variable = { name = murim_meditation_xp value = 0 }
	}
	# XP gain scales with learning stat
	if = {
		limit = { learning >= 15 }
		change_variable = { name = murim_meditation_xp add = 5 }
	}
	else_if = {
		limit = { learning >= 10 }
		change_variable = { name = murim_meditation_xp add = 3 }
	}
	else = {
		change_variable = { name = murim_meditation_xp add = 2 }
	}
	# Bonus for cultivation traits
	if = {
		limit = {
			OR = {
				has_trait = murim_cultivation_nascent_soul
				has_trait = murim_cultivation_soul_formation
				has_trait = murim_cultivation_dao_seeking
			}
		}
		change_variable = { name = murim_meditation_xp add = 2 }
	}
	# Stress relief from meditation
	add_stress = -10
	# Small chance of enlightenment
	random = {
		chance = 5
		add_character_modifier = { modifier = murim_bottleneck_enlightenment_near months = 3 }
	}
	add_character_flag = { flag = murim_meditation_cooldown days = 30 }
}

murim_apply_deep_meditation_effect = {
	# Extended meditation -- higher rewards but longer cooldown
	if = {
		limit = {
			NOT = { exists = var:murim_meditation_xp }
		}
		set_variable = { name = murim_meditation_xp value = 0 }
	}
	if = {
		limit = { learning >= 18 }
		change_variable = { name = murim_meditation_xp add = 10 }
	}
	else_if = {
		limit = { learning >= 12 }
		change_variable = { name = murim_meditation_xp add = 7 }
	}
	else = {
		change_variable = { name = murim_meditation_xp add = 4 }
	}
	add_stress = -25
	add_piety = 25
	# Risk of heart demon during deep meditation
	random = {
		chance = 10
		add_character_modifier = { modifier = murim_bottleneck_inner_demons months = 6 }
		add_stress = 30
	}
	# Higher enlightenment chance
	random = {
		chance = 12
		add_character_modifier = { modifier = murim_bottleneck_enlightenment_near months = 6 }
		add_learning_skill = 1
	}
	add_character_flag = { flag = murim_meditation_cooldown days = 90 }
}

murim_apply_qi_gathering_effect = {
	# Gather ambient qi to build cultivation progress
	if = {
		limit = {
			NOT = { exists = var:murim_qi_gathered }
		}
		set_variable = { name = murim_qi_gathered value = 0 }
	}
	# Gathering efficiency based on prowess and martial
	if = {
		limit = { prowess >= 20 }
		change_variable = { name = murim_qi_gathered add = 6 }
	}
	else_if = {
		limit = { prowess >= 12 }
		change_variable = { name = murim_qi_gathered add = 4 }
	}
	else = {
		change_variable = { name = murim_qi_gathered add = 2 }
	}
	# Location bonus -- check for sacred ground
	if = {
		limit = {
			exists = var:murim_at_sacred_ground
			var:murim_at_sacred_ground >= 1
		}
		change_variable = { name = murim_qi_gathered add = 3 }
	}
	# Threshold check for cultivation progress
	if = {
		limit = {
			exists = var:murim_qi_gathered
			var:murim_qi_gathered >= 30
		}
		set_variable = { name = murim_qi_gathered value = 0 }
		if = {
			limit = {
				NOT = { exists = var:murim_cultivation_progress }
			}
			set_variable = { name = murim_cultivation_progress value = 0 }
		}
		change_variable = { name = murim_cultivation_progress add = 1 }
		send_interface_toast = {
			title = murim_qi_gathering_milestone_toast
			left_icon = root
		}
	}
	add_character_flag = { flag = murim_qi_gathering_cooldown days = 15 }
}

murim_apply_cultivation_cycling_effect = {
	# Standard cultivation cycle -- the core grind loop
	if = {
		limit = {
			NOT = { exists = var:murim_cultivation_progress }
		}
		set_variable = { name = murim_cultivation_progress value = 0 }
	}
	# Progress gain based on current realm
	if = {
		limit = {
			OR = {
				has_trait = murim_cultivation_nascent_soul
				has_trait = murim_cultivation_soul_formation
				has_trait = murim_cultivation_dao_seeking
			}
		}
		# High realm -- slow but steady
		change_variable = { name = murim_cultivation_progress add = 1 }
	}
	else_if = {
		limit = {
			OR = {
				has_trait = murim_cultivation_core_formation
				has_trait = murim_cultivation_golden_core
			}
		}
		# Mid realm -- moderate pace
		change_variable = { name = murim_cultivation_progress add = 2 }
	}
	else = {
		# Low realm -- fast early progress
		change_variable = { name = murim_cultivation_progress add = 3 }
	}
	# Bonus from high learning
	if = {
		limit = { learning >= 15 }
		change_variable = { name = murim_cultivation_progress add = 1 }
	}
	# Risk of qi deviation during cycling
	random = {
		chance = 3
		add_character_modifier = { modifier = murim_qi_deviation_mild months = 2 }
		add_stress = 15
	}
	# Breakthrough check
	if = {
		limit = {
			exists = var:murim_cultivation_progress
			var:murim_cultivation_progress >= 50
		}
		set_variable = { name = murim_cultivation_progress value = 0 }

			title = murim_cultivation_breakthrough_ready_toast
