murim_integration_bridge.0001 = {
	type = character_event
	title = murim_integration_bridge.0001.t
	desc = murim_integration_bridge.0001.desc
	theme = martial_arts
	
	# Fires when calamity phase reaches critical - forces clan response
	trigger = {
		has_global_variable = murim_great_calamity_active
		is_clan_leader = yes
		NOT = { has_character_flag = clan_calamity_response_chosen }
	}
	
	immediate = {
		set_character_flag = clan_calamity_response_chosen
		# Determine clan survival odds based on clan type
		if = {
			limit = { has_trait = orthodox_clan_member }
			set_variable = { name = clan_survival_base value = 40 }
		}
		else_if = {
			limit = { has_trait = unorthodox_clan_member }
			set_variable = { name = clan_survival_base value = 30 }
		}
		else = {
			set_variable = { name = clan_survival_base value = 50 }
		}
	}
	
	# Option A: Fortify clan compound - defensive survival
	option = {
		name = murim_integration_bridge.0001.a
		add_character_modifier = {
			modifier = clan_fortification_active
			years = 10
		}
		change_variable = { name = clan_survival_base add = 20 }
		every_vassal = {
			limit = { has_trait_group = murim_clan }
			add_opinion = { modifier = respected_clan_defense target = root }
		}
		trigger_event = { id = murim_integration_bridge.0010 days = 30 }
	}
	
	# Option B: Evacuate clan members - flee and rebuild
	option = {
		name = murim_integration_bridge.0001.b
		add_character_modifier = {
			modifier = clan_evacuation_underway
			years = 5
		}
		change_variable = { name = clan_survival_base add = 10 }
		add_prestige = -500
		every_vassal = {
			limit = { has_trait_group = murim_clan }
			random = {
				chance = 70
				add_trait = calamity_refugee
			}
		}
		trigger_event = { id = murim_integration_bridge.0011 days = 60 }
	}
	
	# Option C: Alliance with rival clan - desperate pact
	option = {
		name = murim_integration_bridge.0001.c
		# Find strongest rival clan leader
		random_ruler = {
			limit = {
				is_clan_leader = yes
				NOT = { this = root }
				NOT = { alliance_with = root }
			}
			weight = {
				base = 1
				modifier = {
					add = 10
					martial >= 20
				}
			}
			save_scope_as = desperate_ally
		}
		if = {
			limit = { exists = scope:desperate_ally }
			add_alliance = { target = scope:desperate_ally }
			reverse_add_opinion = { modifier = calamity_desperate_alliance target = scope:desperate_ally }
			change_variable = { name = clan_survival_base add = 15 }
		}
		trigger_event = { id = murim_integration_bridge.0012 days = 45 }
	}
	
	# Option D: Sacrifice clan elders - blood ritual protection
	option = {
		name = murim_integration_bridge.0001.d
		trigger = { has_trait = unorthodox_clan_member }
		add_piety = -1000
		add_dread = 50
		change_variable = { name = clan_survival_base add = 35 }
		# Kill oldest 3 vassals
		ordered_vassal = {
			limit = { age >= 50 }
			order_by = age
			max = 3
			death = { death_reason = death_execution }
		}
		trigger_event = { id = murim_integration_bridge.0013 days = 15 }
	}
}

murim_integration_bridge.0002 = {
	type = character_event
	title = murim_integration_bridge.0002.t
	desc = murim_integration_bridge.0002.desc
	theme = martial_arts
	
	# Post-calamity clan rebuilding - connects to custom clan creation
	trigger = {
		has_global_variable = murim_post_calamity_era
		is_clan_leader = yes
		has_trait = calamity_survivor
	}
	
	immediate = {
		# Calculate rebuild potential
		set_variable = {
			name = rebuild_potential
			value = {
				value = martial
				add = diplomacy
				multiply = 2
			}
		}
	}
	
	# Option A: Rebuild original clan from ashes
	option = {
		name = murim_integration_bridge.0002.a
		add_character_modifier = {
			modifier = rebuilding_ancestral_clan
			years = 20
		}
		add_prestige = 200
		# Spawn new disciples
		create_character = {
			location = root.capital_province
			age = 16
			trait = martial_prodigy
			culture = root.culture
			faith = root.faith
			employer = root
		}
		create_character = {
			location = root.capital_province
			age = 18
			trait = qi_sensing
			culture = root.culture
			faith = root.faith
			employer = root
		}
		trigger_event = { id = murim_integration_bridge.0020 days = 180 }
	}
	
	# Option B: Found entirely new clan - links to custom clan creation
	option = {
		name = murim_integration_bridge.0002.b
		trigger_event = { id = murim_custom_clan.0001 days = 30 }
		add_character_modifier = {
			modifier = new_clan_founder_post_calamity
			years = 30
		}
		add_prestige = 500
	}
	
	# Option C: Merge with another survivor clan
	option = {
		name = murim_integration_bridge.0002.c
		random_ruler = {
			limit = {
				is_clan_leader = yes
				has_trait = calamity_survivor
				NOT = { this = root }
			}
			save_scope_as = merge_target
		}
		if = {
			limit = { exists = scope:merge_target }
			# Determine who leads merged clan
			if = {
				limit = { martial >= scope:merge_target.martial }
				scope:merge_target = { set_employer = root }
				add_trait = merged_clan_patriarch
			}
			else = {
				set_employer = scope:merge_target
				scope:merge_target = { add_trait = merged_clan_patriarch }
			}
		}
		trigger_event = { id = murim_integration_bridge.0021 days = 90 }
	}
}

murim_integration_bridge.0003 = {
	type = character_event
	title = murim_integration_bridge.0003.t
	desc = murim_integration_bridge.0003.desc
	theme = martial_arts
	
	# Top rankers automatically qualify for Grand Tournament
	trigger = {
		OR = {
			has_character_modifier = murim_rank_1
			has_character_modifier = murim_rank_2
			has_character_modifier = murim_rank_3
			has_character_modifier = murim_top_10
		}
		has_global_variable = grand_tournament_announced
		NOT = { has_character_flag = tournament_qualification_processed }
	}
	
	immediate = {
		set_character_flag = tournament_qualification_processed
		set_variable = {
			name = tournament_seed
			value = {
				value = 0
				if = {
					limit = { has_character_modifier = murim_rank_1 }
					add = 100
				}
				else_if = {
					limit = { has_character_modifier = murim_rank_2 }
					add = 90
				}
				else_if = {
					limit = { has_character_modifier = murim_rank_3 }
					add = 80
				}
				else = {
					add = 60
				}
				add = martial
				add = prowess
			}
		}
	}
	
	# Option A: Accept seeded position - enter as favorite
	option = {
		name = murim_integration_bridge.0003.a
		add_character_flag = tournament_seeded_entry
		add_prestige = 300
		add_character_modifier = {
			modifier = tournament_pressure_of_rank
			years = 1
		}
		trigger_event = { id = murim_tournament.0001 days = 14 }
	}
	
	# Option B: Decline and send disciple instead
	option = {
		name = murim_integration_bridge.0003.b
		random_vassal = {
			limit = {
				has_trait_group = murim_cultivation
				age >= 16
				age <= 30
			}
			weight = {
				base = 1
				modifier = { add = 5 prowess >= 15 }
			}
			save_scope_as = tournament_proxy
			add_character_flag = tournament_proxy_entry
		}
		if = {
			limit = { exists = scope:tournament_proxy }
			scope:tournament_proxy = {
				trigger_event = { id = murim_tournament.0001 days = 14 }
			}
			add_opinion = { modifier = entrusted_with_honor target = scope:tournament_proxy }
		}
	}
	
	# Option C: Challenge the tournament system itself
	option = {
		name = murim_integration_bridge.0003.c
		trigger = { has_trait = unorthodox_clan_member }
		add_character_flag = tournament_challenger
		add_dread = 20
		add_prestige = -200
		trigger_event = { id = murim_integration_bridge.0030 days = 7 }
	}
}

murim_integration_bridge.0006 = {
	type = character_event
	title = murim_integration_bridge.0006.t
	desc = murim_integration_bridge.0006.desc
	theme = martial_arts
	
	# Faction war outcome determines shadow politics power balance
	trigger = {
		has_global_variable = faction_war_concluded
		OR = {
			has_character_flag = orthodox_faction_leader
			has_character_flag = unorthodox_faction_leader
		}
	}
	
	immediate = {
		if = {
			limit = { has_global_variable = orthodox_faction_won }
			set_global_variable = shadow_politics_orthodox_dominant
		}
		else = {
			set_global_variable = shadow_politics_unorthodox_dominant
		}
	}
	
	# Option A: Establish new Murim order
	option = {
		name = murim_integration_bridge.0006.a
		trigger = { has_character_flag = winning_faction_leader }
		add_prestige = 2000
		add_character_modifier = {
			modifier = new_murim_order_founder
			years = 50
		}
		# Dismantle enemy shadow networks
		if = {
			limit = { has_global_variable = orthodox_faction_won }
			remove_global_variable = heavenly_demon_puppet_emperor
			set_global_variable = orthodox_murim_order
		}
		else = {
			set_global_variable = unorthodox_murim_order
			set_global_variable = heavenly_demon_open_rule
		}
		trigger_event = { id = murim_integration_bridge.0060 days = 60 }
	}
	
	# Option B: Mercy - integrate defeated faction
	option = {
		name = murim_integration_bridge.0006.b
		trigger = { has_character_flag = winning_faction_leader }
		add_piety = 500
		add_prestige = 500
		every_ruler = {
			limit = {
				OR = {
					AND = {
						has_global_variable = orthodox_faction_won
						has_trait = unorthodox_clan_member
					}
					AND = {
						has_global_variable = unorthodox_faction_won
						has_trait = orthodox_clan_member
					}
				}
			}
			add_opinion = { modifier = shown_mercy_after_war target = root }
			remove_character_flag = faction_war_enemy
		}
	}
	
	# Option C: Purge defeated faction completely
	option = {
		name = murim_integration_bridge.0006.c
		trigger = {
			has_character_flag = winning_faction_leader
			has_trait = unorthodox_clan_member
		}
		add_dread = 100
		add_tyranny = 50
		every_ruler = {
			limit = { has_trait = orthodox_clan_member }
			random = {
				chance = 50
				death = { death_reason = death_execution }
			}
		}
		trigger_event = { id = murim_integration_bridge.0061 days = 30 }
	}
}

murim_integration_bridge.0007 = {
	type = character_event
	title = murim_integration_bridge.0007.t
	desc = murim_integration_bridge.0007.desc
	theme = martial_arts
	
	# New entrants to murim world get clan recruitment offers
	trigger = {
		has_character_flag = entered_murim_world
		NOT = { has_trait_group = murim_clan }
		NOT = { has_character_flag = clan_recruitment_offers_sent }
		age >= 16
	}
	
	immediate = {
		set_character_flag = clan_recruitment_offers_sent
		# Determine which clans are interested based on entry path
		if = {
			limit = { has_character_flag = entry_path_warrior }
			set_variable = { name = warrior_clan_interest value = 80 }
			set_variable = { name = scholar_clan_interest value = 20 }
		}
		else_if = {
			limit = { has_character_flag = entry_path_scholar }
			set_variable = { name = warrior_clan_interest value = 20 }
			set_variable = { name = scholar_clan_interest value = 80 }
		}
		else_if = {
			limit = { has_character_flag = entry_path_orphan }
			set_variable = { name = warrior_clan_interest value = 50 }
			set_variable = { name = scholar_clan_interest value = 50 }
		}
		else = {
			set_variable = { name = warrior_clan_interest value = 40 }
			set_variable = { name = scholar_clan_interest value = 40 }
		}
	}
	
	# Option A: Join orthodox clan (Shaolin/Wudang/Mount Hua path)
	option = {
		name = murim_integration_bridge.0007.a
		trigger_event = { id = murim_clan_scout.0001 days = 7 }
		add_character_flag = seeking_orthodox_clan
		add_character_modifier = {
			modifier = orthodox_path_seeker
			years = 2
		}
	}
	
	# Option B: Join unorthodox clan (Heavenly Demon/Phantom Blade path)
	option = {
		name = murim_integration_bridge.0007.b
		trigger_event = { id = murim_clan_scout.0010 days = 7 }
		add_character_flag = seeking_unorthodox_clan
		add_character_modifier = {
			modifier = unorthodox_path_seeker
			years = 2
		}
	}
	
	# Option C: Remain clanless - lone wolf path
	option = {
		name = murim_integration_bridge.0007.c
		add_trait = lone_wolf_martial_artist
		add_character_modifier = {
			modifier = clanless_wanderer
			years = 50
		}
		add_prowess = 3
		trigger_event = { id = murim_jianghu.0001 days = 30 }
	}
	
	# Option D: Found own clan immediately (requires high martial)
	option = {
		name = murim_integration_bridge.0007.d
		trigger = {
			martial >= 20
			prestige >= 1000
		}
		trigger_event = { id = murim_custom_clan.0001 days = 14 }
		add_character_flag = ambitious_clan_founder
	}
}

	type = character_event
	title = murim_integration_bridge.0009.t
	desc = murim_integration_bridge.0009.desc
	theme = martial_arts
	
	# Major cultivation breakthrough triggers ranking reassessment
	trigger = {
		has_character_flag = cultivation_breakthrough_recent
		NOT = { has_character_flag = ranking_reassessment_pending }
	}
	
	immediate = {
		set_character_flag = ranking_reassessment_pending
		remove_character_flag = cultivation_breakthrough_recent
		# Calculate new power level
		set_variable = {
			name = power_assessment
			value = {
				value = martial
				add = prowess
				if = {
					limit = { has_trait = qi_master }
					add = 30
				}
				if = {
					limit = { has_trait = transcendent_cultivator }
					add = 50
				}
				if = {
					limit = { has_trait = peak_realm }
					add = 20
				}
			}
		}
	}
	
	# Option A: Demonstrate power publicly - ranking challenge
	option = {
		name = murim_integration_bridge.0009.a
		add_prestige = 100
		trigger_event = { id = murim_ranking.0010 days = 14 }
		remove_character_flag = ranking_reassessment_pending
	}
	
	# Option B: Keep breakthrough secret
	option = {
		name = murim_integration_bridge.0009.b
		add_trait = hidden_power
		add_character_modifier = {
			modifier = concealed_cultivation_level
			years = 10
		}
		remove_character_flag = ranking_reassessment_pending
	}
	
	# Option C: Challenge all top 10 simultaneously
	option = {
		name = murim_integration_bridge.0009.c
		trigger = {
			var:power_assessment >= 80
			has_trait = transcendent_cultivator
		}
		add_character_flag = challenging_all_top_10
		add_prestige = -500
		add_dread = 30
		trigger_event = { id = murim_integration_bridge.0090 days = 7 }
	}
}
