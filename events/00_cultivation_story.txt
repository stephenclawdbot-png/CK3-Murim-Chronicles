# Event 1: The Awakening - Breaking through to Qi Refinement
cultivation.0001 = {
	type = character_event
	title = cultivation.0001.t
	desc = cultivation.0001.desc
	
	trigger = {
		is_adult = yes
		NOT = { has_trait = cultivation_realm }
		age < 50
	}
	
	weight_multiplier = {
		base = 50
		modifier = {
			add = 100
			has_trait = genius
		}
		modifier = {
			add = 50
			has_trait = diligent
		}
	}
	
	immediate = {
		add_trait = qi_refinement
		add_prestige = 100
	}
	
	option = {
		name = cultivation.0001.a
		custom_tooltip = cultivation_awakening
		trigger_event = {
			id = cultivation.0002
			days = 365
		}
	}
	
	option = {
		name = cultivation.0001.b
		add_trait = diligent
		add_stress = 20
	}
}
