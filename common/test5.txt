# Event 1: Grand Tournament Announcement
murim_test_15k.0001 = {
	type = character_event
	title = murim_test_15k.0001.t
	desc = murim_test_15k.0001.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		has_trait = murim_cultivator
		is_ruler = yes
		realm_size >= 5
		NOT = { has_character_flag = hosted_grand_tournament }
		prestige >= 500
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 5
			has_trait = ambitious
		}
		modifier = {
			add = 3
			prestige >= 2000
		}
		modifier = {
			add = 2
			diplomacy >= 16
		}
	}
	
	immediate = {
		save_scope_as = tournament_host
		hidden_effect = {
			add_character_flag = {
				flag = hosted_grand_tournament
				years = 15
			}
			add_character_flag = {
				flag = tournament_in_progress
				years = 1
			}
			every_vassal = {
				limit = {
					has_trait = murim_cultivator
					prowess >= 10
				}
				add_character_flag = {
					flag = tournament_participant
					years = 1
				}
			}
			random_vassal = {
				limit = {
					has_trait = murim_cultivator
					prowess >= 14
				}
				save_scope_as = top_contender
			}
		}
	}
	
	option = {
		name = murim_test_15k.0001.a
		custom_tooltip = announce_grand_tournament_tt
		remove_short_term_gold = 200
		add_prestige = 300
		add_character_modifier = {
			modifier = tournament_host_prestige
			years = 5
		}
		every_vassal = {
			limit = { has_trait = murim_cultivator }
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 10
			}
		}
		trigger_event = {
			id = murim_test_15k.0002
			days = 30
		}
		ai_chance = {
			base = 90
			modifier = {
				factor = 2
				has_trait = ambitious
			}
		}
	}
	
	option = {
		name = murim_test_15k.0001.b
		add_prestige = -50
		ai_chance = {
			base = 10
			modifier = {
				factor = 3
				has_trait = lazy
			}
		}
	}
}
