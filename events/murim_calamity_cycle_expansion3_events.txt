	type = character_event
	title = murim_calamity_cycle_expansion3.0003.t
	desc = murim_calamity_cycle_expansion3.0003.desc
	theme = intrigue
	left_portrait = { character = root animation = shock }
	right_portrait = { character = scope:betrayer animation = personality_vengeful }
	
	trigger = {
		has_global_variable = great_calamity_active
		is_ruler = yes
		NOT = { has_character_flag = calamity_betrayal_event }
	}
	weight_multiplier = { base = 3 }
	
	immediate = {
		set_character_flag = calamity_betrayal_event
		random_courtier = {
			limit = {
				has_trait = deceitful
				opinion = { target = root value < 0 }
			}
			save_scope_as = betrayer
		}
		if = {
			limit = { NOT = { exists = scope:betrayer } }
			random_courtier = {
				limit = { NOT = { this = root } prowess >= 10 }
				save_scope_as = betrayer
			}
		}
	}
	
	# Option A: Execute the betrayer publicly
	option = {
		name = murim_calamity_cycle_expansion3.0003.a
		if = {
			limit = { exists = scope:betrayer }
			scope:betrayer = { death = { death_reason = death_execution killer = root } }
		}
		add_dread = 25
		add_character_modifier = { modifier = ruthless_calamity_leader years = 10 }
		add_stress = 15
		ai_chance = { base = 40 }
	}
	
	# Option B: Imprison and interrogate
	option = {
		name = murim_calamity_cycle_expansion3.0003.b
		if = {
			limit = { exists = scope:betrayer }
			imprison = { target = scope:betrayer type = dungeon }
		}
		add_character_modifier = { modifier = calamity_interrogator years = 5 }
		add_intrigue = 1
		ai_chance = { base = 35 }
	}
	
	# Option C: Show mercy - the calamity turns everyone desperate
	option = {
		name = murim_calamity_cycle_expansion3.0003.c
		add_piety = 100
		add_prestige = 100
		add_character_modifier = { modifier = merciful_during_calamity years = 10 }
		add_stress = -15
		ai_chance = { base = 25 }
	}
}
