###################################
# Install Puppet Emperor
# Heavenly Demon places figurehead on throne
###################################
murim_shadow_exp.0001 = {
	type = character_event
	title = murim_shadow_exp.0001.t
	desc = murim_shadow_exp.0001.desc
	theme = intrigue
	
	trigger = {
		has_trait = trait_heavenly_demon_sect
		has_variable = murim_clan_leader
		intrigue >= 16
		NOT = { has_global_variable = murim_puppet_emperor_installed }
		NOT = { has_character_flag = puppet_emperor_attempt }
		any_living_character = {
			has_title = e_china_emperor
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = puppet_emperor_attempt
			days = 3650
		}
		random_living_character = {
			limit = { has_title = e_china_emperor }
			save_scope_as = current_emperor
		}
	}
	
	option = {
		name = murim_shadow_exp.0001.a
		# Assassinate emperor and install puppet
		duel = {
			skill = intrigue
			target = scope:current_emperor
			55 = {
				comparator = scope:duel_value
				desc = murim_shadow_exp.0001.success
				scope:current_emperor = {
					death = { death_reason = death_murim_shadow_assassination }
				}
				set_global_variable = murim_puppet_emperor_installed
				set_variable = murim_shadow_puppeteer
				# Create puppet emperor
				create_character = {
					location = root.capital_province
					template = murim_puppet_emperor_template
					save_scope_as = puppet_emperor
				}
				scope:puppet_emperor = {
					add_trait = murim_puppet_ruler
					add_character_modifier = {
						modifier = murim_controlled_by_demon_sect
						years = 50
					}
				}
				add_prestige = 1000
				add_character_modifier = {
					modifier = murim_shadow_emperor_maker
					years = 20
				}
			}
			45 = {
				comparator = scope:duel_value
				desc = murim_shadow_exp.0001.failure
				# Plot discovered
				scope:current_emperor = {
					add_opinion = {
						target = root
						modifier = murim_assassination_plot_discovered
						opinion = -100
					}
				}
				add_prestige = -500
				imprison = { target = root type = dungeon }
			}
		}
		ai_chance = { base = 60 }
	}
	
	option = {
		name = murim_shadow_exp.0001.b
		# Corrupt emperor through bribery instead
		add_gold = -500
		scope:current_emperor = {
			add_character_modifier = {
				modifier = murim_demon_sect_corrupted
				years = 10
			}
			add_opinion = {
				target = root
				modifier = murim_bribed_by_demons
				opinion = 30
			}
		}
		ai_chance = { base = 40 }
	}
}
