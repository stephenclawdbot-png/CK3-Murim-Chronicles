# Murim Chronicles Phase 4 - Jianghu Underground Triggers
# Conditions for underground network membership, shadow operations, imperial infiltration

######################################
# JIANGHU MEMBERSHIP CHECKS
######################################

murim_is_jianghu_member_trigger = {
	OR = {
		has_trait = shadow_broker
		has_trait = sect_spy
		has_trait = jianghu_enforcer
		has_trait = wulin_alliance_member
		has_trait = black_market_dealer
		has_trait = forbidden_text_smuggler
		has_trait = underground_tournament_champion
		has_trait = hidden_master
		has_character_flag = murim_jianghu_inducted
	}
}

murim_is_jianghu_leader_trigger = {
	murim_is_jianghu_member_trigger = yes
	OR = {
		has_trait = shadow_broker
		has_trait = hidden_master
		has_trait = wulin_alliance_member
		AND = {
			has_character_flag = murim_jianghu_inducted
			var:murim_jianghu_influence >= 80
		}
	}
}

murim_is_active_jianghu_operative_trigger = {
	murim_is_jianghu_member_trigger = yes
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	age >= 16
}

######################################
# UNDERGROUND CONNECTION CHECKS
######################################

murim_has_underground_connections_trigger = {
	OR = {
		murim_is_jianghu_member_trigger = yes
		has_character_flag = murim_underground_contact
		any_relation = {
			type = friend
			murim_is_jianghu_member_trigger = yes
		}
		any_relation = {
			type = lover
			murim_is_jianghu_member_trigger = yes
		}
		any_vassal = {
			murim_is_jianghu_member_trigger = yes
		}
		any_courtier = {
			murim_is_jianghu_member_trigger = yes
		}
	}
}

murim_has_strong_underground_network_trigger = {
	murim_has_underground_connections_trigger = yes
	OR = {
		intrigue >= 15
		has_trait = shadow_broker
		AND = {
			is_ruler = yes
			any_courtier = {
				count >= 3
				murim_is_jianghu_member_trigger = yes
			}
		}
	}
}

murim_has_underground_informant_trigger = {
	OR = {
		has_character_flag = murim_has_planted_spy
		any_courtier = {
			has_trait = sect_spy
		}
		any_relation = {
			type = friend
			has_trait = imperial_informant
		}
	}
}

######################################
# IMPERIAL CULTIVATOR SECRET CHECKS
######################################

murim_is_imperial_cultivator_secret_trigger = {
	has_character_flag = murim_initiated
	OR = {
		has_trait = court_cultivator_secret
		AND = {
			is_ruler = yes
			highest_held_title_tier >= tier_county
			murim_is_cultivator_trigger = yes
			NOT = { has_character_flag = murim_cultivation_public }
		}
	}
}

murim_is_known_cultivator_at_court_trigger = {
	murim_is_cultivator_trigger = yes
	OR = {
		has_character_flag = murim_cultivation_public
		has_trait = retired_legend
		any_secret = {
			secret_type = secret_murim_cultivator
			is_known_by = scope:target
		}
	}
}

murim_is_hidden_cultivator_trigger = {
	murim_is_cultivator_trigger = yes
	NOT = { has_character_flag = murim_cultivation_public }
	NOT = { has_trait = retired_legend }
}

murim_can_be_exposed_as_cultivator_trigger = {
	murim_is_hidden_cultivator_trigger = yes
	NOT = { has_character_modifier = murim_identity_perfectly_concealed }
	OR = {
		has_trait_rank = {
			trait = cultivation_rank
			rank >= 3
		}
		has_character_flag = murim_used_technique_publicly
		any_secret = {
			secret_type = secret_murim_cultivator
			is_known_by >= 3
		}
	}
}

######################################
# COURT INFLUENCE CHECKS
######################################

murim_can_influence_court_trigger = {
	is_alive = yes
	age >= 16
	is_imprisoned = no
	OR = {
		is_ruler = yes
		is_powerful_vassal = yes
		is_councillor = yes
		has_trait = shadow_broker
		has_trait = puppet_official
		AND = {
			murim_is_jianghu_member_trigger = yes
			intrigue >= 12
		}
		AND = {
			murim_has_underground_connections_trigger = yes
			diplomacy >= 14
		}
	}
}

murim_can_manipulate_succession_trigger = {
	murim_can_influence_court_trigger = yes
	OR = {
		has_trait = shadow_broker
		intrigue >= 18
		AND = {
			murim_has_strong_underground_network_trigger = yes
			diplomacy >= 14
		}
	}
	NOT = { has_character_modifier = murim_recently_manipulated_succession }
}

murim_can_influence_title_grant_trigger = {
	murim_can_influence_court_trigger = yes
	OR = {
		is_powerful_vassal = yes
		has_trait = shadow_broker
		has_trait = puppet_official
		AND = {
			murim_is_jianghu_leader_trigger = yes
			intrigue >= 14
		}
	}
}

######################################
# SHADOW BROKER CHECKS
######################################

murim_is_shadow_broker_trigger = {
	has_trait = shadow_broker
	is_alive = yes
	is_imprisoned = no
}

murim_is_active_shadow_broker_trigger = {
	murim_is_shadow_broker_trigger = yes
	intrigue >= 12
	gold >= 50
	NOT = { has_character_modifier = murim_network_compromised }
}

murim_can_become_shadow_broker_trigger = {
	NOT = { has_trait = shadow_broker }
	murim_is_jianghu_member_trigger = yes
	intrigue >= 16
	gold >= 200
	age >= 25
	OR = {
		has_trait = deceitful
		has_trait = schemer
		has_trait = ambitious
		has_character_flag = murim_ran_black_market
	}
}

murim_has_broker_intelligence_trigger = {
	murim_is_shadow_broker_trigger = yes
	OR = {
		has_character_flag = murim_spy_network_active
		any_courtier = {
			count >= 2
			has_trait = sect_spy
		}
		var:murim_intelligence_network >= 5
	}
}

		}
	}
}

		}

			type = friend

		}
	}
}

######################################
# TRADE ROUTE CONTROL CHECKS
######################################

murim_controls_trade_route_trigger = {
	is_ruler = yes
	OR = {
		has_character_flag = murim_controls_silk_road_segment
		has_character_flag = murim_controls_tea_route
		has_character_flag = murim_controls_salt_route
		has_character_flag = murim_controls_jade_route
		has_character_modifier = murim_trade_route_controller
	}
}

murim_can_seize_trade_route_trigger = {
	is_ruler = yes
	is_at_war = no
	OR = {
		murim_is_jianghu_leader_trigger = yes
		AND = {
			murim_has_strong_underground_network_trigger = yes
			gold >= 300
		}
		AND = {
			martial >= 14
			has_character_flag = murim_jianghu_inducted
			num_of_knights >= 5
		}
	}
	NOT = { has_character_modifier = murim_recently_seized_trade_route }
}

murim_controls_multiple_trade_routes_trigger = {
	custom_description = {
		text = "murim_controls_two_or_more_routes"
		calc_true_if = {
			amount >= 2
			has_character_flag = murim_controls_silk_road_segment
			has_character_flag = murim_controls_tea_route
			has_character_flag = murim_controls_salt_route
			has_character_flag = murim_controls_jade_route
		}
	}
}
