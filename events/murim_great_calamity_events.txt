# Murim Chronicles - THE GREAT CALAMITY
# The cyclical apocalypse of the martial world. Heaven's judgment on mortal hubris.
# 4 PHASES: Gathering Storm (5yr) -> Great War (2-3yr) -> Silence (1yr) -> Renewal
# INTEGRATION: murim_calamity_triggers.txt, murim_calamity_on_actions.txt, murim_calamity_traits.txt
# DESIGN: Player sees omens 10-20 years before. Can attempt to prevent/delay/accelerate.
# DEVASTATION: 70-90% character death. 1-3 survivors per clan. Some clans wiped entirely.
# NAMESPACE: murim_calamity.XXXX

# OMEN 1: Strange Qi Disturbance
murim_calamity.0001 = {
	type = character_event
	title = murim_calamity.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_trait = murim_omen_seer }
				desc = murim_calamity.0001.desc_seer
			}
			desc = murim_calamity.0001.desc
		}
	}
	theme = death
	
	trigger = {
		murim_is_martial_artist_trigger = yes
		NOT = { has_character_flag = murim_received_qi_omen }
		murim_calamity_omens_should_start_trigger = yes
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = murim_omen_seer
		}
		modifier = {
			add = 1
			murim_is_grandmaster_or_above_trigger = yes
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = murim_received_qi_omen
			days = 1825  # 5 years cooldown
		}
	}
	
	# Option A: Meditate on the disturbance
	option = {
		name = murim_calamity.0001.a
		add_stress = 10
		add_character_modifier = {
			modifier = murim_qi_disturbance_awareness
			years = 5
		}
		random = {
			chance = 40
			add_trait = murim_omen_seer
		}
		create_character_memory = {
			type = murim_felt_qi_omen
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 30
				has_trait = patient
			}
		}
	}
	
	# Option B: Dismiss it as nothing
	option = {
		name = murim_calamity.0001.b
		add_prestige = 50
		random = {
			chance = 30
			add_trait = murim_omen_denier
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = arrogant
			}
		}
	}
	
	# Option C: Warn others (diplomacy check)
	option = {
		name = murim_calamity.0001.c
		trigger = { diplomacy >= 12 }
		add_prestige = -100
		add_piety = 50
		stress_impact = {
			paranoid = medium_stress_impact_loss
		}
		every_courtier = {
			limit = {
				murim_is_martial_artist_trigger = yes
			}
			random = {
				chance = 25
				add_trait = murim_omen_seer
			}
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = compassionate
			}
		}
	}
}

	type = character_event
	title = murim_calamity.1010.t
	desc = murim_calamity.1010.desc
	theme = war
	
	trigger = {
		murim_calamity_in_gathering_storm_trigger = yes
		murim_is_martial_artist_trigger = yes
		NOT = { has_character_flag = murim_calamity_faction_choice }
	}
	
	immediate = {
		add_character_flag = {
			flag = murim_calamity_faction_choice
		}
	}
	
	# Option A: Stand with the Orthodox Alliance
	option = {
		name = murim_calamity.1010.a
		trigger = {
			NOT = {
				OR = {
					has_trait = heavenly_demon_disciple
					has_trait = poison_flower_member
					has_trait = phantom_blade_member
				}
			}
		}
		add_character_flag = murim_calamity_orthodox_side
		add_prestige = 100
		add_piety = 100

			limit = {

					has_trait = shaolin_disciple

					has_trait = mount_hua_disciple
					has_trait = beggar_sect_member
					has_trait = iron_fist_member
				}
			}
			add_opinion = {
				target = root
				modifier = murim_stood_with_orthodox

			}
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Join the Unorthodox Coalition
	option = {
		name = murim_calamity.1010.b
		trigger = {
			NOT = {
				OR = {
					has_trait = shaolin_disciple
