# Murim Chronicles - Clan Scout Character Effects
# Spawns and manages NAMED TRAVELING RECRUITER characters for each clan.
# Scouts are REAL characters on the map with traits, travel routes, and lifespans.
# CHINA-CENTRIC: All scouts DEPART FROM and RETURN TO China (Song Dynasty region).
# VANILLA SEPARATION: Scout travel is martial-layer only. Does NOT trigger political events,
#   wars, or diplomatic incidents. Scouts use martial_travel modifier, not vanilla movement.
# INTEGRATION: Referenced by murim_clan_recruitment_events.txt (dual-mode recruitment),
#              murim_clan_scout_events.txt (travel events),
#              murim_game_start_effects.txt (initial scout spawning),
#              murim_on_actions.txt (yearly scout movement pulse)
# GENDER: All scouts can be male or female. No restrictions.

# spawn_clan_scouts_effect — called on game start and when a scout dies (1-2 year delay)
# Spawns 1-2 scout characters per clan with clan-specific traits and travel preferences.
# All scouts start in CHINA (Song Dynasty region) at their clan's HQ province.

murim_spawn_clan_scouts_effect = {

    # Route: China -> Goryeo -> Japan -> India -> Persia -> back to China (5-6 year cycle)
    if = {
        limit = { has_variable = mount_hua_needs_scout }
        create_character = {
            name = "Mount Hua Scout"
            location = title:c_luoyang.title_province    # Mount Hua HQ region
            age = { 30 40 }
            trait = brave
            trait = just
            trait = temperate
            trait = outer_disciple    # Senior enough to represent the clan
            dynasty = none
            culture = culture:han
            religion = religion:taoism
            prowess = { 12 18 }
            martial = { 10 15 }
            diplomacy = { 12 18 }    # Scouts need diplomacy for recruitment pitches
            save_scope_as = mount_hua_scout
        }
        scope:mount_hua_scout = {
            add_trait = martial_artist
            set_variable = { name = scout_clan value = flag:mount_hua }
            set_variable = { name = scout_route_stage value = 0 }
            set_variable = { name = scout_years_traveling value = 0 }
            set_variable = { name = scout_recruits_found value = 0 }
            set_variable = { name = scout_max_cycle_years value = 5 }
            add_character_modifier = {
                modifier = murim_clan_scout_modifier
                years = 100
            }
            # Mount Hua preference flags
            set_variable = { name = scout_prefers_disciplined value = yes }
            set_variable = { name = scout_prefers_honorable value = yes }
            set_variable = { name = scout_approach_style value = flag:polite_demonstration }
        }
        remove_variable = mount_hua_needs_scout
    }

    # Shaolin Scouts — wander monasteries and temples worldwide, seek spiritual talent
    # Route: China -> Tibet -> India -> Southeast Asia -> Goryeo -> back to China
    if = {
        limit = { has_variable = shaolin_needs_scout }
        create_character = {
            name = "Shaolin Scout"
            location = title:c_kaifeng.title_province    # Near Shaolin Temple
            age = { 35 50 }
            trait = calm
            trait = compassionate
            trait = temperate
            trait = inner_disciple    # Shaolin sends senior monks
            dynasty = none
            culture = culture:han
            religion = religion:buddhism
            prowess = { 14 20 }
            martial = { 8 12 }
            diplomacy = { 14 18 }
            learning = { 14 20 }    # Monks are learned
            save_scope_as = shaolin_scout
        }
        scope:shaolin_scout = {
            add_trait = martial_artist
            set_variable = { name = scout_clan value = flag:shaolin }
            set_variable = { name = scout_route_stage value = 0 }
            set_variable = { name = scout_years_traveling value = 0 }
            set_variable = { name = scout_recruits_found value = 0 }
            set_variable = { name = scout_max_cycle_years value = 6 }
            add_character_modifier = {
                modifier = murim_clan_scout_modifier
                years = 100

            set_variable = { name = scout_prefers_spiritual value = yes }
            set_variable = { name = scout_prefers_orphans value = yes }
            set_variable = { name = scout_approach_style value = flag:peaceful_invitation }

    if = {
        limit = { has_variable = heavenly_demon_needs_scout }

            name = "Heavenly Demon Scout"
            location = title:c_chengdu.title_province    # Heavenly Demon remote HQ
            age = { 25 40 }
            trait = ambitious
            trait = deceitful
            trait = sadistic
            trait = inner_disciple
            dynasty = none
            culture = culture:han
            religion = religion:taoism

            intrigue = { 16 22 }    # Shadow operatives

        }

# murim_advance_scout_route_effect — called yearly on each scout
# Moves scout to next stage of their route. Triggers court visit events.
# VANILLA SEPARATION: Uses martial_travel flag, NOT vanilla travel/movement system.

murim_advance_scout_route_effect = {
    # Increment travel counter
    change_variable = { name = scout_years_traveling add = 1 }
    
    # Check if cycle complete (return to China)
    if = {
        limit = {
            var:scout_years_traveling >= var:scout_max_cycle_years
        }
        # Return to China with recruits
        trigger_event = { id = murim_scout.0010 days = { 15 30 } }    # Return event
        set_variable = { name = scout_years_traveling value = 0 }
        set_variable = { name = scout_route_stage value = 0 }
    }
    # Otherwise advance route
    else = {
        change_variable = { name = scout_route_stage add = 1 }
        
        # Each route stage triggers a court visit event based on region
        # The scout_events file handles the actual interactions
        trigger_event = { id = murim_scout.0001 days = { 30 90 } }    # Arrive at new region
    }
}

# murim_scout_evaluate_candidate_effect — evaluates a character for martial potential
# Called when scout encounters a potential recruit during travel
# Returns weighted score used by recruitment decision events

murim_scout_evaluate_candidate_effect = {
    # Base score from prowess relative to age
    # Young characters with high prowess = huge bonus
    if = {
        limit = {
            age < 16
            prowess >= 8
        }
        change_variable = { name = candidate_score add = 10 }    # Youth prodigy
    }
    if = {
        limit = {
            age >= 16
            age < 25
            prowess >= 12
        }
        change_variable = { name = candidate_score add = 7 }    # Young talent
    }
    if = {
        limit = {
            age >= 25
            prowess >= 16
        }
        change_variable = { name = candidate_score add = 5 }    # Adult fighter
    }
    
    # Trait bonuses
    if = { limit = { has_trait = genius } change_variable = { name = candidate_score add = 8 } }
    if = { limit = { has_trait = quick } change_variable = { name = candidate_score add = 5 } }
    if = { limit = { has_trait = intelligent } change_variable = { name = candidate_score add = 4 } }
    if = { limit = { has_trait = brave } change_variable = { name = candidate_score add = 4 } }
    if = { limit = { has_trait = strong } change_variable = { name = candidate_score add = 4 } }
    if = { limit = { has_trait = robust } change_variable = { name = candidate_score add = 3 } }
    if = { limit = { has_trait = ambitious } change_variable = { name = candidate_score add = 3 } }
    if = { limit = { has_trait = diligent } change_variable = { name = candidate_score add = 2 } }
    if = { limit = { has_trait = shrewd } change_variable = { name = candidate_score add = 3 } }
    
    # Martial awareness = already knows the murim exists
    if = {
        limit = { has_trait = martial_awareness }
        change_variable = { name = candidate_score add = 5 }
    }
    
    # Tournament veteran = showed talent publicly
    if = {
        limit = { has_character_modifier = murim_tournament_participant }
        change_variable = { name = candidate_score add = 6 }
    }
    if = {
        limit = { has_character_modifier = murim_tournament_winner }
        change_variable = { name = candidate_score add = 10 }
    }
    
    # Clan-specific preference bonuses
    # Mount Hua: disciplined + honorable
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:mount_hua }
            OR = {
                has_trait = just
                has_trait = temperate
                has_trait = honest
            }
        }
        change_variable = { name = candidate_score add = 4 }
    }
    
    # Shaolin: spiritual + calm
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:shaolin }
            OR = {
                has_trait = calm
                has_trait = compassionate
                has_trait = content
                religion = { has_doctrine = tenet_monasticism }
            }
        }
        change_variable = { name = candidate_score add = 4 }
    }
    
    # Heavenly Demon: ambitious + ruthless + desperate
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:heavenly_demon }
            OR = {
                has_trait = ambitious
                has_trait = sadistic
                has_trait = callous
                has_trait = deceitful
                is_imprisoned = yes
                gold < 0
            }
        }
        change_variable = { name = candidate_score add = 5 }
    }
    
    # Tang Clan: blood Tang members get MASSIVE bonus, others evaluated as associates
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:tang_clan }
            has_trait = tang_clan_member
        }
        change_variable = { name = candidate_score add = 20 }    # Lost bloodline found!
    }
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:tang_clan }
            NOT = { has_trait = tang_clan_member }
            OR = {
                learning >= 14    # Alchemist/poison knowledge
                has_trait = shrewd
                has_trait = physician_3    # Medical knowledge useful for poisons
            }
        }
        change_variable = { name = candidate_score add = 3 }    # Associate potential
    }
    
    # Wudang: intelligent + scholarly
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:wudang }
            OR = {
                learning >= 14
                has_trait = scholar
                has_trait = theologian
                has_trait = patient
            }
        }
        change_variable = { name = candidate_score add = 4 }
    }
    
    # Beggar Sect: poor + abandoned + information network potential
    if = {
        limit = {
            scope:scout = { var:scout_clan = flag:beggar_sect }
            OR = {
                gold < 10
                is_landed = no
                has_trait = compassionate
                has_trait = gregarious
                intrigue >= 12    # Information broker potential
            }
        }
        change_variable = { name = candidate_score add = 4 }
    }
    
    # 5% pure random "gut feeling" pick — scout just SEES something
    random = {
        chance = 5
        change_variable = { name = candidate_score add = 8 }
        set_variable = { name = gut_feeling_pick value = yes }
    }
}

murim_scout_killed_effect = {
    # Scout was killed during travel. Clan learns after delay.
    # Set flag for replacement spawn after 1-2 years
    if = {
        limit = { var:scout_clan = flag:mount_hua }
        set_global_variable = { name = mount_hua_needs_scout value = yes }
    }
    if = {
        limit = { var:scout_clan = flag:shaolin }
        set_global_variable = { name = shaolin_needs_scout value = yes }
    }
    if = {
        limit = { var:scout_clan = flag:heavenly_demon }
        set_global_variable = { name = heavenly_demon_needs_scout value = yes }
    }
    if = {
        limit = { var:scout_clan = flag:tang_clan }
        set_global_variable = { name = tang_clan_needs_scout value = yes }
    }
    if = {
        limit = { var:scout_clan = flag:wudang }
        set_global_variable = { name = wudang_needs_scout value = yes }
    }
    if = {
        limit = { var:scout_clan = flag:beggar_sect }
        set_global_variable = { name = beggar_sect_needs_scout value = yes }
    }
}

# murim_scout_captured_effect — scout imprisoned by hostile ruler
murim_scout_captured_effect = {
    # Triggers rescue/ransom event chain for the clan
    # Clan has options: pay ransom, send rescue party, abandon scout
    save_scope_as = captured_scout
    scope:captured_scout.host = { save_scope_as = captor_ruler }
    trigger_event = { id = murim_scout.0020 days = { 30 60 } }    # Clan learns of capture
}

# murim_scout_expelled_effect — scout kicked out of a realm
murim_scout_expelled_effect = {
    # Realm is blacklisted for 10 years
    scope:expelling_ruler.primary_title = {
        set_variable = {
            name = murim_scout_blacklisted
            value = yes
        }
        # Remove blacklist after 10 years
        # (handled by on_actions yearly check)
    }
    set_variable = { name = scout_route_stage value = 0 }    # Reset route, skip this region
}

murim_clan_scout_modifier = {
    # Identifies character as an active clan scout
    # VANILLA SEPARATION: This modifier gives NO political benefits
    # Only affects martial-layer interactions
    monthly_prestige = 0    # No political prestige from scouting
    prowess = 2    # Slight combat edge from travel experience
    # Custom tooltip: "This character is an active clan scout traveling the world to recruit new disciples."
}

# ===========================
# SCOUT RETURN WITH RECRUITS EFFECT
# ===========================

murim_scout_return_with_recruits_effect = {
    # Called when scout cycle completes and they return to China
    # Creates 2-5 recruit characters based on scout success
    
    # Number of recruits based on scout's evaluation scores during travel
    # Minimum 2, maximum 5
    set_local_variable = { name = num_recruits value = 2 }
    if = {
        limit = { var:scout_recruits_found >= 3 }
        set_local_variable = { name = num_recruits value = 3 }
    }
    if = {
        limit = { var:scout_recruits_found >= 5 }
        set_local_variable = { name = num_recruits value = 4 }
    }
    if = {
        limit = { var:scout_recruits_found >= 7 }
        set_local_variable = { name = num_recruits value = 5 }
    }
    
    # Reset scout counters for next cycle
    set_variable = { name = scout_recruits_found value = 0 }
    set_variable = { name = scout_years_traveling value = 0 }
    
    # Each recruit arrives with:
    # - outer_disciple trait
    # - Culture from their origin region (creates diversity in clans)
    # - 5% chance of being a generational talent (fast-tracked inner disciple)
    # - Possible homesick modifier if from far away (Europe/Africa = guaranteed)
    
    # Trigger the "Scout Returns" event which processes arrivals
    trigger_event = { id = murim_scout.0010 days = 5 }
}
