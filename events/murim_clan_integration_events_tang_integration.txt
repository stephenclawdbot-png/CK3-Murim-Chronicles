	title = murim_tang_integration.0001.t

		change_variable = { name = murim_cultivation_xp add = 60 }
		ai_chance = {

			modifier = { add = 20 intrigue >= 14 }
		}
	}

	# Strengthen blood ties — the clan is family above all
	option = {
		name = murim_tang_integration.0001.c
		add_dynasty_prestige = 200

		ai_chance = { base = 25 }
	}
