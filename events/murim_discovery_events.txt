# Murim Chronicles - Martial World Discovery Events
# HOW non-martial characters discover the murim. The growth engine for the AI world.
# 20 events across 4 paths: Landless Adventurer, Landed Noble, Courtier, Exploration
# TRIGGER: yearly pulse + travel actions. Age 8+. No martial trait. ~3-5% Korea/China, ~1% global.
# Landless characters get DOUBLE chance (6-10%) — they wander and stumble into things.
# INTEGRATION: Uses murim_discovery_triggers.txt (eligibility, region checks)
# INTEGRATION: Grants martial_awareness trait (from murim_awareness_trait.txt) as gateway
# INTEGRATION: Calls murim_clan_membership_effects.txt (join sect path)
# INTEGRATION: Calls murim_training_events.txt (training follow-ups)
# INTEGRATION: References murim_triggers.txt (murim_is_martial_artist_trigger for exclusion)
# INTEGRATION: Hooks into murim_on_actions.txt (yearly pulse) and travel on_actions
# GENDER: ALL events fully gender-neutral. No sex checks. Men and women equally.
# AI: All events fire for AI characters with same weights. AI decisions based on personality traits.

namespace = murim_discover

# 1001: Strange Sounds in the Mountains
murim_discover.1001 = {
	type = character_event
	title = murim_discover.1001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_trait = brave }
				desc = murim_discover.1001.desc_brave
			}
			triggered_desc = {
				trigger = { has_trait = craven }
				desc = murim_discover.1001.desc_craven
			}
			desc = murim_discover.1001.desc
		}
	}
	theme = martial
	
	trigger = {
		murim_is_eligible_for_discovery_trigger = yes
		is_landed = no
		# Wandering or adventuring — no court to anchor you
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_trait = adventurer
		}
		modifier = {
			add = 0.3
			has_trait = brave
		}
		modifier = {
			add = -0.5
			has_trait = craven
		}
	}
	
	immediate = {
		# Set up the scene: two martial masters dueling in a mountain clearing
		random_pool_character = {
			province = root.location
			limit = {
				murim_is_martial_artist_trigger = yes
				age >= 25
			}
			alternative = global
			save_scope_as = dueling_master_1
		}
		random_pool_character = {
			province = root.location
			limit = {
				murim_is_martial_artist_trigger = yes
				age >= 25
				NOT = { this = scope:dueling_master_1 }
			}
			alternative = global
			save_scope_as = dueling_master_2
		}
	}
	
	# Option A: Investigate — witness the duel
	option = {
		name = murim_discover.1001.a
		# "The sounds draw you closer..."
		add_trait = martial_awareness
		add_character_modifier = {
			modifier = witnessed_martial_combat_modifier
			years = 3
		}
		# Follow-up: awe event, potential training path
		trigger_event = {
			id = murim_discover.1001_followup
			days = { 3 7 }
		}
		stress_impact = {
			brave = minor_stress_impact_loss
			craven = medium_stress_impact_gain
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = brave
			}
			modifier = {
				add = 20
				has_trait = curious  # If trait exists
			}
		}
	}
	
	# Option B: Avoid — safe but miss opportunity
	option = {
		name = murim_discover.1001.b
		# "Best not to get involved with such people..."
		add_prestige = -10
		# Small chance you still gain awareness from the sounds alone
		random = {
			chance = 20
			add_trait = martial_awareness
		}
		stress_impact = {
			craven = minor_stress_impact_loss
			brave = minor_stress_impact_gain
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 40
				has_trait = craven
			}
		}
	}
	
	# Option C: Rush in to help — reckless but memorable
	option = {
		name = murim_discover.1001.c
		# "Someone could be hurt! You charge toward the sounds..."
		add_trait = martial_awareness
		# Risk: you might get injured
		random = {
			chance = 40
			add_trait = wounded_1
		}
		# Reward: the master you tried to help remembers you
		random = {
			chance = 60
			scope:dueling_master_1 = {
				add_opinion = {
					modifier = grateful_opinion
					target = root
					opinion = 30
				}
			}
			# Potential mentor event
			trigger_event = {
				id = murim_discover.1001_mentor
				days = { 14 30 }
			}
		}
		stress_impact = {
			brave = medium_stress_impact_loss
			calm = minor_stress_impact_gain
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 40
				has_trait = brave
			}
			modifier = {
				add = 20
				has_trait = compassionate
			}
		}
	}
}
