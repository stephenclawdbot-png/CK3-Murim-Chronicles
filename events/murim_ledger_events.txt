# Murim Chronicles - Murim Ledger Events (Player UI)
# Event-based martial world overview. Player-only. AI never fires these.
# Triggered by open_murim_ledger_decision (common/decisions/murim_ledger_decisions.txt)
# INTEGRATION: Reads murim_ranking_values.txt (top 50 rankings)
# INTEGRATION: Reads murim_harmony_values.txt (clan stability)
# INTEGRATION: Reads murim_clan_triggers.txt (clan membership checks)
# INTEGRATION: Reads murim_tournament_values.txt (tournament schedule)
# DESIGN: VANILLA SEPARATION — pure martial layer UI. No feudal titles, no political data.
# GENDER: No gender restrictions. All martial characters access equally.

namespace = murim_ledger

# ============================================================
# MAIN MENU (0001)
# ============================================================

murim_ledger.0001 = {
	type = character_event
	title = murim_ledger.0001.t
	# "The Murim Ledger"
	
	desc = murim_ledger.0001.desc
	# "Before you lies the Murim Ledger — a compilation of records maintained 
	#  by the information networks of the six great clans, updated by wandering 
	#  scribes who track the deeds of martial artists across the known world.
	#  
	#  Its pages contain the rankings of the mightiest fighters, the state of 
	#  each great clan, your own martial record, recent events that have shaken 
	#  the martial world, and the schedule of upcoming tournaments.
	#  
	#  What would you like to review?"
	
	theme = diplomacy
	
	# Tab 1: Rankings
	option = {
		name = murim_ledger.0001.a

		trigger_event = {
			id = murim_ledger.0010

		name = murim_ledger.0001.c

			id = murim_ledger.0030
		}
	}
	
	# Tab 4: Recent Events
	option = {
		name = murim_ledger.0001.d
		# "View Recent Martial Events"
		trigger_event = {
			id = murim_ledger.0040
		}
	}
	
	# Tab 5: Tournaments
	option = {
		name = murim_ledger.0001.e
		# "View the Tournament Schedule"
		trigger_event = {
			id = murim_ledger.0050
		}
	}
	
	# Close
	option = {
		name = murim_ledger.0001.f
		# "Close the Ledger"

		custom_tooltip = murim_vanilla_separation_verified_tt
	}
