# 1. BRIBE IMPERIAL OFFICIAL
######################################

murim_bribe_imperial_official_interaction = {
	category = interaction_category_hostile
	icon = icon_gold
	desc = murim_bribe_imperial_official_interaction_desc
	
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = { murim_has_underground_connections_trigger = yes }
		scope:recipient = {
			is_ruler = yes
			highest_held_title_tier >= tier_county
			NOT = { murim_is_jianghu_member_trigger = yes }
		}
	}
	
	is_valid_showing_failures_only = {
		scope:actor = {
			gold >= 150
			NOT = { has_character_modifier = murim_bribe_cooldown }
		}
	}
	
	on_accept = {
		scope:actor = {
			add_gold = -150
			add_character_modifier = {
				modifier = murim_bribe_cooldown
				years = 2
			}
			change_variable = { name = murim_jianghu_influence add = 8 }
			add_opinion = { target = scope:recipient modifier = murim_bribed_official_opinion }
		}
		scope:recipient = {
			add_gold = 150
			add_opinion = { target = scope:actor modifier = murim_accepted_bribe_opinion }
			add_character_modifier = {
				modifier = murim_bribed_official
				years = 5
			}
		}
	}
	
	on_decline = {
		scope:actor = {
			add_prestige = -50
			random = {
				chance = 30
				add_character_modifier = { modifier = murim_bribe_attempt_known years = 3 }
			}
		}
	}
	
	ai_accept = {
		base = -20
		modifier = { add = 30 scope:recipient = { ai_greed >= 50 } }
		modifier = { add = 20 scope:recipient = { gold < 50 } }
		modifier = { add = 15 scope:recipient = { has_trait = greedy } }
		modifier = { add = -30 scope:recipient = { has_trait = just } }
		modifier = { add = -20 scope:recipient = { has_trait = honest } }
		modifier = { add = 10 scope:recipient = { has_trait = cynical } }
		modifier = { add = 15 opinion = { target = scope:actor value >= 30 } }
	}
	
	auto_accept = no
	ai_frequency = 24
	ai_potential = {
		murim_has_underground_connections_trigger = yes
		gold >= 200
	}
}

murim_plant_sect_spy_interaction = {
	category = interaction_category_hostile
	icon = icon_intrigue
	desc = murim_plant_sect_spy_interaction_desc
	
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			murim_is_jianghu_member_trigger = yes
			intrigue >= 10
		}
		scope:recipient = {
			is_ruler = yes
			NOT = { has_character_flag = murim_spy_planted_here }
		}
	}
	
	is_valid_showing_failures_only = {
		scope:actor = {
			gold >= 100
			any_courtier = {
				has_trait = sect_spy
				NOT = { has_character_flag = murim_planted_at_court }
			}
		}
	}
	
	on_accept = {
		scope:actor = {
			add_gold = -100
			random_courtier = {
				limit = {
					has_trait = sect_spy
					NOT = { has_character_flag = murim_planted_at_court }
				}
				save_scope_as = spy_agent
			}
			set_character_flag = murim_has_planted_spy
			change_variable = { name = murim_spies_planted add = 1 }
			change_variable = { name = murim_jianghu_influence add = 10 }
		}
		scope:recipient = {
			set_character_flag = murim_spy_planted_here
			add_courtier = scope:spy_agent
		}
		scope:spy_agent = {
			set_character_flag = murim_planted_at_court
			add_character_modifier = { modifier = murim_deep_cover_operative years = 10 }
		}
	}
	
	auto_accept = yes
	ai_frequency = 48
	ai_potential = {
		murim_is_jianghu_member_trigger = yes
		intrigue >= 12
		gold >= 150
	}
}

murim_challenge_for_territory_interaction = {
	category = interaction_category_hostile
	icon = icon_martial
	desc = murim_challenge_for_territory_interaction_desc
	
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			murim_is_jianghu_member_trigger = yes
			is_ruler = yes
		}
		scope:recipient = {
			is_ruler = yes
			is_neighbor_to_realm = scope:actor
			murim_is_jianghu_member_trigger = yes
		}
	}
	
	is_valid_showing_failures_only = {
		scope:actor = {
			prowess >= 10
			NOT = { has_character_modifier = murim_territory_challenge_cooldown }
		}
	}
	
	on_accept = {
		# Prowess duel determines outcome
		scope:actor = {
			duel = {
				skill = prowess
				target = scope:recipient
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = 3.0
					}
					desc = murim_territory_duel_win
					scope:actor = {
						add_prestige = 200
						change_variable = { name = murim_jianghu_influence add = 15 }
						add_character_modifier = { modifier = murim_territory_champion years = 5 }
					}
					scope:recipient = {
						add_prestige = -150
						change_variable = { name = murim_jianghu_influence add = -10 }
					}
				}
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = -3.0
					}
					desc = murim_territory_duel_lose
					scope:actor = {
						add_prestige = -150
						change_variable = { name = murim_jianghu_influence add = -10 }
					}
					scope:recipient = {
						add_prestige = 200
						change_variable = { name = murim_jianghu_influence add = 15 }
					}
				}
			}
			add_character_modifier = { modifier = murim_territory_challenge_cooldown years = 3 }
		}
	}
	
	on_decline = {
		scope:recipient = {
			add_prestige = -100
			add_character_modifier = { modifier = murim_declined_territory_challenge years = 5 }
		}
		scope:actor = {
			add_prestige = 50
		}
	}
	
	ai_accept = {
		base = -10
		modifier = { add = 30 scope:recipient = { has_trait = brave } }
		modifier = { add = 20 scope:recipient = { prowess >= scope:actor.prowess } }
		modifier = { add = -30 scope:recipient = { has_trait = craven } }
		modifier = { add = 15 scope:recipient = { has_trait = wrathful } }
	}
	
	auto_accept = no
	ai_frequency = 72
	ai_potential = {
		murim_is_jianghu_member_trigger = yes
		is_ruler = yes
		prowess >= 14
	}
}
