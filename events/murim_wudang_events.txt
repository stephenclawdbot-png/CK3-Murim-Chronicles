# WUDANG.0010 - The Sect Master Selection Tournament
# Wudang's sect master retires. A tournament decides the successor.
murim_wudang.0010 = {
	type = character_event
	title = murim_wudang.0010.t
	desc = murim_wudang.0010.desc
	theme = martial
	left_portrait = {
		character = root
		animation = war_over_win
	}

	trigger = {
		murim_is_wudang_member_trigger = yes
		OR = {
			has_trait = elder_rank
			has_trait = vice_leader_rank
		}
	}

	# Enter the tournament to become sect master
	option = {
		name = murim_wudang.0010.a
		random_list = {
			30 = {
				add_trait = leader_rank
				add_character_modifier = {
					modifier = wudang_new_sect_master
					years = -1
				}
				add_prestige = 1000
				add_prowess_skill = 3
				create_character_memory = {
					type = wudang_became_sect_master_memory
				}
			}
			40 = {
				add_trait = wounded_1
				add_prowess_skill = 1
				add_character_modifier = {
					modifier = wudang_tournament_runner_up
					years = 5
				}
			}
			30 = {
				add_trait = wounded_2
				add_prestige = -200
				create_character_memory = {
					type = wudang_tournament_eliminated_memory
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = ambitious
			}
		}
	}

	# Judge the tournament instead (kingmaker role)
	option = {
		name = murim_wudang.0010.b
		add_diplomacy_skill = 3
		add_character_modifier = {
			modifier = wudang_tournament_judge
			years = 10
		}
		create_character_memory = {
			type = wudang_judged_tournament_memory
		}
		ai_chance = {
			base = 35
		}
	}

	# Rig the tournament for your preferred candidate
	option = {
		name = murim_wudang.0010.c
		add_intrigue_skill = 4
		random_list = {
			40 = {
				add_character_modifier = {
					modifier = wudang_puppet_master
					years = 10
				}
				add_opinion = {
					target = scope:wudang_puppet
					modifier = grateful_opinion
					opinion = 50
				}
			}
			60 = {
				add_character_modifier = {
					modifier = wudang_rigging_exposed
					years = 5
				}
				add_prestige = -500
			}
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 30
				has_trait = deceitful
			}
		}
	}

	# Propose abolishing the tournament for a council vote
	option = {
		name = murim_wudang.0010.d
		add_diplomacy_skill = 2
		add_learning_skill = 2
		random_list = {
			40 = {
				add_character_modifier = {
					modifier = wudang_reformer
					years = 10
				}
				add_prestige = 300
			}
			60 = {
				add_prestige = -100
			}
		}
		ai_chance = {
			base = 10
		}
	}
}
