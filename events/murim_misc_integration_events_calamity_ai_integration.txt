murim_calamity_ai_integration.0001 = {
	type = character_event
	title = murim_calamity_ai_integration.0001.t
	desc = murim_calamity_ai_integration.0001.desc
	theme = skull
	left_portrait = { character = root animation = personality_callous }

	trigger = {
		has_variable = murim_martial_artist
		is_ai = yes
		has_global_variable = murim_great_calamity_active
	}

	immediate = {
		set_character_flag = { flag = murim_calamity_ai_reacted years = 2 }
	}

	# Fight desperately to survive
	option = {
		name = murim_calamity_ai_integration.0001.a
		add_prowess = 2
		add_stress = 30
		ai_chance = {
			base = 50
			modifier = { add = 25 prowess >= 15 }
		}
	}

	# Flee to the mountains
	option = {
		name = murim_calamity_ai_integration.0001.b
		add_stress = -10
		add_prestige = -50
		ai_chance = { base = 30 }
	}

	# Protect the weak — shield the non-cultivators
	option = {
		name = murim_calamity_ai_integration.0001.c
		add_piety = 150
		add_prestige = 100
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = compassionate }
		}
	}
}
