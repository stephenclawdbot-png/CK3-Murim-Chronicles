# First step on the martial path
murim_cultivation.0001 = {
	type = character_event
	title = murim_cultivation.0001.t
	desc = murim_cultivation.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_zealous }

	trigger = {
		has_variable = murim_martial_artist
		NOT = { has_character_flag = murim_first_cultivation }
	}

	immediate = {
		set_character_flag = murim_first_cultivation
	}

	# Begin orthodox cultivation — slow and steady
	option = {
		name = murim_cultivation.0001.a
		add_martial = 1
		change_variable = { name = murim_cultivation_xp add = 20 }
		add_stress = -10
		ai_chance = {
			base = 50
			modifier = { add = 20 has_trait = patient }
		}
	}

	# Push your body to its limits immediately
	option = {
		name = murim_cultivation.0001.b
		add_prowess = 2
		change_variable = { name = murim_cultivation_xp add = 35 }
		add_stress = 20
		ai_chance = {
			base = 30
			modifier = { add = 25 has_trait = brave }
		}
	}

	# Seek a master to guide you
	option = {
		name = murim_cultivation.0001.c
		add_learning = 1
		change_variable = { name = murim_cultivation_xp add = 15 }
		ai_chance = { base = 20 }
	}
}
