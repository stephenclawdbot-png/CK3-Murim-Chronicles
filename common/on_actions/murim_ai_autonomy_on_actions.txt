######################################
# QUARTERLY AI DUEL PULSE
# Changed from on_monthly to reduce lag; every_ruler instead of every_living
# AI martial artists autonomously challenge each other
######################################

on_quarterly_pulse = {
	effect = {
		# --- AI-vs-AI BACKGROUND DUELS (quarterly, background resolution) ---
		every_ruler = {
			limit = {
				is_alive = yes
				is_ai = yes
				has_variable = murim_martial_artist
				age >= 16
				prowess >= 10
				NOT = { has_character_flag = murim_duel_cooldown }
				# 50% chance to even consider dueling this month (frequency throttle)
				trigger_if = {
					limit = { always = yes }
					random = 50
				}
			}
			random_list = {
				##############################
				# 1. HONOR DUEL - formal challenge between cultivators
				##############################
				25 = {
					modifier = {
						add = 20
						has_trait = just
					}
					modifier = {
						add = 15
						has_trait = brave
					}
					modifier = {
						add = 10
						has_trait = honorable
					}
					modifier = {
						add = -15
						has_trait = craven
					}
					modifier = {
						add = 10
						prowess >= 15
					}
					trigger = {
						any_living_character = {
							is_ai = yes
							prowess >= 8
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
							OR = {
								has_character_flag = murim_initiated
								has_character_flag = murim_cultivation_enabled
								AND = {
									exists = var:murim_cultivation_rank
									var:murim_cultivation_rank >= 1
								}
							}
						}
					}
					# Find a worthy opponent
					random_living_character = {
						limit = {
							is_ai = yes
							prowess >= 8
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
							OR = {
								has_character_flag = murim_initiated
								has_character_flag = murim_cultivation_enabled
								AND = {
									exists = var:murim_cultivation_rank
									var:murim_cultivation_rank >= 1
								}
							}
						}
						save_scope_as = ai_duel_target
					}
					# Set cooldowns
					set_character_flag = {
						flag = murim_duel_cooldown
						months = 3
					}
					scope:ai_duel_target = {
						set_character_flag = {
							flag = murim_duel_cooldown
							months = 3
						}
					}
					# Resolve duel in background
					duel = {
						skill = prowess
						target = scope:ai_duel_target
						50 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = 3.0
							}
							# Winner gains prestige and prowess XP
							add_prestige = 100
							add_character_modifier = {
								modifier = murim_duel_victor_aura
								years = 1
							}
							scope:ai_duel_target = {
								add_prestige = -50
							}
							# Notify the player about nearby duel
							every_player = {
								limit = {
									is_alive = yes
									OR = {
										any_courtier = { this = root }
										any_courtier = { this = scope:ai_duel_target }
										any_vassal = { this = root }
										any_vassal = { this = scope:ai_duel_target }
										any_neighboring_and_across_water_top_liege_realm_owner = {
											OR = {
												this = root
												this = scope:ai_duel_target
											}
										}
									}
								}
								send_interface_message = {
									type = event_martial_good
									title = murim_ai_duel_honor_victory_notification
									desc = murim_ai_duel_honor_victory_notification_desc
									left_icon = root
									right_icon = scope:ai_duel_target
								}
							}
						}
						15 = {
							# Draw - mutual respect
							add_opinion = {
								target = scope:ai_duel_target
								modifier = murim_duel_respect
							}
							scope:ai_duel_target = {
								add_opinion = {
									target = prev
									modifier = murim_duel_respect
								}
							}
						}
						35 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = -3.0
							}
							# Loser
							add_prestige = -50
							scope:ai_duel_target = {
								add_prestige = 100
								add_character_modifier = {
									modifier = murim_duel_victor_aura
									years = 1
								}
							}
							# Notify player
							every_player = {
								limit = {
									is_alive = yes
									OR = {
										any_courtier = { this = root }
										any_courtier = { this = scope:ai_duel_target }
										any_vassal = { this = root }
										any_vassal = { this = scope:ai_duel_target }
									}
								}
								send_interface_message = {
									type = event_martial_neutral
									title = murim_ai_duel_honor_defeat_notification
									desc = murim_ai_duel_honor_defeat_notification_desc
									left_icon = scope:ai_duel_target
									right_icon = root
								}
							}
						}
					}
				}

				##############################
				# 2. DEATH MATCH - lethal duel, rare but dramatic
				##############################
				8 = {
					modifier = {
						add = 15
						has_trait = wrathful
					}
					modifier = {
						add = 15
						has_trait = sadistic
					}
					modifier = {
						add = 10
						has_trait = vengeful
					}
					modifier = {
						add = -20
						has_trait = compassionate
					}
					modifier = {
						add = -15
						has_trait = craven
					}
					trigger = {
						prowess >= 14
						any_living_character = {
							is_ai = yes
							prowess >= 12
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
							OR = {
								has_negative_opinion_of = prev
								AND = {
									exists = prev.var:murim_cultivation_rank
									exists = var:murim_cultivation_rank
								}
							}
						}
					}
					random_living_character = {
						limit = {
							is_ai = yes
							prowess >= 12
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
						}
						save_scope_as = ai_death_target
					}
					set_character_flag = {
						flag = murim_duel_cooldown
						years = 1
					}
					scope:ai_death_target = {
						set_character_flag = {
							flag = murim_duel_cooldown
							years = 1
						}
					}
					duel = {
						skill = prowess
						target = scope:ai_death_target
						50 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = 3.5
							}
							# Kill the opponent
							scope:ai_death_target = {
								death = {
									death_reason = death_duel
									killer = prev
								}
							}
							add_prestige = 300
							add_dread = 25
							add_character_modifier = {
								modifier = murim_duel_killer_reputation
								years = 5
							}
							# CHAIN: Trigger revenge arc from victim's family
							scope:ai_death_target = {
								if = {
									limit = {
										exists = dynasty
										any_dynasty_member = {
											is_alive = yes
											NOT = { this = root }
											prowess >= 10
										}
									}
									random_dynasty_member = {
										limit = {
											is_alive = yes
											NOT = { this = root }
											prowess >= 10
										}
										save_scope_as = revenge_seeker
										set_character_flag = {
											flag = murim_revenge_target_flag
											years = 10
										}
										set_variable = {
											name = murim_revenge_target
											value = root
										}
										# Revenge duel fires within 6 months
										trigger_event = { id = murim_duel.0100 days = { 30 180 } }
									}
								}
							}
							# Notify all players
							every_player = {
								limit = { is_alive = yes }
								send_interface_message = {
									type = event_martial_bad
									title = murim_ai_death_match_notification
									desc = murim_ai_death_match_notification_desc
									left_icon = root
									right_icon = scope:ai_death_target
								}
							}
						}
						10 = {
							# Both severely wounded
							add_trait = wounded_1
							scope:ai_death_target = {
								add_trait = wounded_1
							}
						}
						40 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = -3.5
							}
							# Root dies
							death = {
								death_reason = death_duel
								killer = scope:ai_death_target
							}
							scope:ai_death_target = {
								add_prestige = 300
								add_dread = 25
								add_character_modifier = {
									modifier = murim_duel_killer_reputation
									years = 5
								}
							}
							# Notify players
							every_player = {
								limit = { is_alive = yes }
								send_interface_message = {
									type = event_martial_bad
									title = murim_ai_death_match_notification
									desc = murim_ai_death_match_reverse_notification_desc
									left_icon = scope:ai_death_target
									right_icon = root
								}
							}
						}
					}
				}

				##############################
				# 3. SECT RANKING FIGHT - compete for sect position
				##############################
				20 = {
					modifier = {
						add = 15
						has_trait = ambitious
					}
					modifier = {
						add = 10
						has_trait = arrogant
					}
					modifier = {
						add = -10
						has_trait = content
					}
					trigger = {
						exists = var:murim_sect_id
						any_living_character = {
							is_ai = yes
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
							exists = var:murim_sect_id
							var:murim_sect_id = prev.var:murim_sect_id
							prowess >= 8
						}
					}
					random_living_character = {
						limit = {
							is_ai = yes
							NOT = { this = prev }
							NOT = { has_character_flag = murim_duel_cooldown }
							exists = var:murim_sect_id
							var:murim_sect_id = prev.var:murim_sect_id
							prowess >= 8
						}
						save_scope_as = ai_sect_rival
					}
					set_character_flag = {
						flag = murim_duel_cooldown
						months = 4
					}
					scope:ai_sect_rival = {
						set_character_flag = {
							flag = murim_duel_cooldown
							months = 4
						}
					}
					duel = {
						skill = prowess
						target = scope:ai_sect_rival
						55 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = 3.0
							}
							add_prestige = 150
							# Gain sect standing
							if = {
								limit = { exists = var:murim_sect_rank }
								change_variable = {
									name = murim_sect_rank
									add = 1
								}
							}
							scope:ai_sect_rival = {
								if = {
									limit = { exists = var:murim_sect_rank }
									change_variable = {
										name = murim_sect_rank
										add = -1
									}
								}
								add_prestige = -75
							}
							# Notify player if in same sect
							every_player = {
								limit = {
									is_alive = yes
									exists = var:murim_sect_id
									var:murim_sect_id = root.var:murim_sect_id
								}
								send_interface_message = {
									type = event_martial_good
									title = murim_ai_sect_ranking_fight_notification
									desc = murim_ai_sect_ranking_fight_notification_desc
									left_icon = root
									right_icon = scope:ai_sect_rival
								}
							}
						}
						15 = {
							# Draw
							add_opinion = {
								target = scope:ai_sect_rival
								modifier = murim_sect_rivalry_respect
							}
						}
						30 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = -3.0
							}
							add_prestige = -75
							scope:ai_sect_rival = {
								add_prestige = 150
								if = {
									limit = { exists = var:murim_sect_rank }
									change_variable = {
										name = murim_sect_rank
										add = 1
									}
								}
							}
						}
					}
				}

				##############################
				# 4. AMBUSH ATTACK - intrigue-based surprise attack
				##############################
				10 = {
					modifier = {
						add = 20
						has_trait = deceitful
					}
					modifier = {
						add = 15
						has_trait = sadistic
					}
					modifier = {
						add = 10
						intrigue >= 15
					}
					modifier = {
						add = -20
						has_trait = just
					}
					modifier = {
						add = -15
						has_trait = honorable
					}
					trigger = {
						intrigue >= 12
						any_living_character = {
							is_ai = yes
							NOT = { this = prev }
							NOT = { has_character_flag = murim_ambush_immunity }
							has_negative_opinion_of = prev
							prowess >= 6
						}
					}
					random_living_character = {
						limit = {
							is_ai = yes
							NOT = { this = prev }
							NOT = { has_character_flag = murim_ambush_immunity }
							has_negative_opinion_of = prev
							prowess >= 6
						}
						save_scope_as = ai_ambush_victim
					}
					set_character_flag = {
						flag = murim_duel_cooldown
						months = 6
					}
					# Ambush uses intrigue for initiative, then prowess
					duel = {
						skill = intrigue
						target = scope:ai_ambush_victim
						60 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = 2.5
							}
							# Ambush succeeds - wound the victim
							scope:ai_ambush_victim = {
								add_trait = wounded_1
								add_prestige = -100
								set_character_flag = {
									flag = murim_ambush_immunity
									months = 6
								}
								# CHAIN: Victim may hire bounty hunter for revenge
								if = {
									limit = { gold >= 50 }
									random = {
										chance = 40
										trigger_event = { id = murim_jianghu.0200 days = { 14 60 } }
									}
								}
							}
							add_dread = 15
							# Notify players
							every_player = {
								limit = {
									is_alive = yes
									OR = {
										any_courtier = { this = scope:ai_ambush_victim }
										any_vassal = { this = scope:ai_ambush_victim }
									}
								}
								send_interface_message = {
									type = event_martial_bad
									title = murim_ai_ambush_attack_notification
									desc = murim_ai_ambush_attack_notification_desc
									left_icon = root
									right_icon = scope:ai_ambush_victim
								}
							}
						}
						40 = {
							compare_modifier = {
								value = scope:duel_value
								multiplier = -2.5
							}
							# Ambush fails - attacker exposed
							add_prestige = -200
							add_opinion = {
								target = scope:ai_ambush_victim
								modifier = murim_ambush_failed_shame
								years = 5
							}
							scope:ai_ambush_victim = {
								add_prestige = 100
								set_character_flag = {
									flag = murim_ambush_immunity
									months = 6
								}
							}
						}
					}
				}

				##############################
				# 5. REVENGE DUEL - triggered by grudge/enemy death
				##############################
				12 = {
					modifier = {
						add = 20
						has_trait = vengeful
					}
					modifier = {
						add = 15
						has_trait = wrathful
					}
					modifier = {
						add = 10
						has_trait = brave
					}
					modifier = {
						add = -15
						has_trait = forgiving
					}
					trigger = {
						OR = {
							has_character_flag = murim_revenge_target_flag
							any_living_character = {
								is_ai = yes
								NOT = { this = prev }
								NOT = { has_character_flag = murim_duel_cooldown }
								has_relation_rival = prev
								prowess >= 8
							}
						}
					}
					# Try to use saved revenge target first
					if = {
						limit = {
							has_character_flag = murim_revenge_target_flag
							exists = var:murim_revenge_target
							var:murim_revenge_target = {
								is_alive = yes
							}
						}
						save_scope_as = revenge_challenger
						var:murim_revenge_target = {
							save_scope_as = ai_revenge_target
						}
					}
					else = {
						random_living_character = {
							limit = {
								is_ai = yes
								NOT = { this = prev }
								NOT = { has_character_flag = murim_duel_cooldown }
								has_relation_rival = prev
								prowess >= 8
							}
							save_scope_as = ai_revenge_target
						}
					}
					set_character_flag = {
						flag = murim_duel_cooldown
						months = 6
					}
					# Revenge duels are more lethal

######################################
# ON DEATH HOOK - Duel death cascading effects
######################################

on_death = {
	effect = {
		# If killed in a duel, trigger revenge chains
		if = {
			limit = {
				exists = killer
				OR = {
					death_reason = death_duel
					has_character_flag = murim_killed_in_duel
				}
			}
			# Family revenge chain
			if = {
				limit = {
					exists = dynasty
					any_dynasty_member = {
						is_alive = yes
						prowess >= 8
						NOT = { this = root.killer }
					}
				}
				random_dynasty_member = {
					limit = {
						is_alive = yes
						prowess >= 8
						NOT = { this = root.killer }
					}
					weight = {
						base = 1
						modifier = {
							add = 5
							has_trait = vengeful
						}
						modifier = {
							add = 3
							has_trait = brave
						}
						modifier = {
							add = 3
							has_trait = wrathful
						}
					}
					set_character_flag = {
						flag = murim_revenge_target_flag
						years = 10
					}
					set_variable = {
						name = murim_revenge_target
						value = root.killer
					}
					add_opinion = {
						target = root.killer
						modifier = murim_blood_feud_hatred
					}
				}
			}

			# Sect vengeance - if victim was in a sect, the sect may hunt the killer
			if = {
				limit = {
					exists = var:murim_sect_id
					killer = {
						NOT = {
							exists = var:murim_sect_id
							var:murim_sect_id = root.var:murim_sect_id
						}
					}
				}
				every_living_character = {
					limit = {
						exists = var:murim_sect_id
						var:murim_sect_id = root.var:murim_sect_id
						prowess >= 12
						is_alive = yes
					}
					add_opinion = {
						target = root.killer
						modifier = murim_sect_enemy_opinion
					}
					# 20% chance each sect member swears revenge
					random = {
						chance = 20
						set_character_flag = {
							flag = murim_revenge_target_flag
							years = 5
						}
						set_variable = {
							name = murim_revenge_target
							value = root.killer
						}
					}
				}
			}

			# CHAIN: Qi deviation death attracts demonic sects
			if = {
				limit = {
					OR = {
						has_trait = murim_qi_deviation_crippling
						has_trait = murim_forbidden_mark_3
					}
				}
				every_living_character = {
					limit = {
						is_ai = yes
						is_alive = yes
						OR = {
							has_trait = murim_forbidden_mark_1
							has_trait = murim_forbidden_mark_2
							has_trait = murim_forbidden_mark_3
						}
						NOT = { this = root }
					}
					# Demonic cultivators investigate the death site
					random = {
						chance = 30
						trigger_event = { id = murim_jianghu.0300 days = { 30 90 } }
					}
				}
			}
		}
	}
}

######################################
# ON BATTLE END - Martial cultivators gain combat experience
######################################

on_combat_end_winner = {
	effect = {
		if = {
			limit = {
				OR = {
					has_character_flag = murim_cultivation_enabled
					AND = {
						exists = var:murim_cultivation_rank
						var:murim_cultivation_rank >= 1
					}
				}
			}
			# Combat experience boosts cultivation
			if = {
				limit = { exists = var:murim_cultivation_progress }
				change_variable = {
					name = murim_cultivation_progress
					add = 3
				}
			}
			# Small chance of combat insight triggering breakthrough readiness
			random = {
				chance = 10
				add_character_modifier = {
					modifier = murim_combat_enlightenment
					months = 6
				}
			}
		}
	}
}

on_combat_end_loser = {
	effect = {
		if = {
			limit = {
				OR = {
					has_character_flag = murim_cultivation_enabled
					AND = {
						exists = var:murim_cultivation_rank
						var:murim_cultivation_rank >= 1
					}
				}
			}
			# Defeat can also bring insight
			random = {
				chance = 15
				add_character_modifier = {
					modifier = murim_defeat_insight
					months = 3
				}
				if = {
					limit = { exists = var:murim_cultivation_progress }
					change_variable = {
						name = murim_cultivation_progress
						add = 2
					}
				}
			}
		}
	}
