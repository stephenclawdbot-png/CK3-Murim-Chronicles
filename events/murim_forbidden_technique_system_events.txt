# Technique Sealing Event
murim_forbidden_technique.0200 = {
	type = character_event
	title = murim_forbidden_technique.0200.t
	desc = murim_forbidden_technique.0200.desc
	theme = faith

	trigger = {
		murim_has_any_forbidden_technique_trigger = yes
	}

	# Seal the most dangerous technique
	option = {
		name = murim_forbidden_technique.0200.a
		# Remove the last technique learned (simplified — removes first found)
		if = {
			limit = { has_trait = murim_eternal_damnation_sutra }
			remove_trait = murim_eternal_damnation_sutra
		}
		else_if = {
			limit = { has_trait = murim_heaven_defying_sutra }
			remove_trait = murim_heaven_defying_sutra
		}
		else_if = {
			limit = { has_trait = murim_death_qi_cultivation }
			remove_trait = murim_death_qi_cultivation
		}
		else_if = {
			limit = { has_trait = murim_soul_devouring_palm }
			remove_trait = murim_soul_devouring_palm
		}
		else_if = {
			limit = { has_trait = murim_blood_reversal_art }
			remove_trait = murim_blood_reversal_art
		}
		else_if = {
			limit = { has_trait = murim_asura_body_technique }
			remove_trait = murim_asura_body_technique
		}
		else_if = {
			limit = { has_trait = murim_soul_splitting_art }
			remove_trait = murim_soul_splitting_art
		}
		else_if = {
			limit = { has_trait = murim_void_breathing_method }
			remove_trait = murim_void_breathing_method
		}
		hidden_effect = { murim_reduce_corruption_effect = yes }
		add_piety = 150
		add_prestige = -100
		ai_chance = { base = 50 }
	}

	# Keep the power — accept the corruption
	option = {
		name = murim_forbidden_technique.0200.b
		add_stress = 20
		add_prowess = 1
		ai_chance = {
			base = 30
			modifier = { add = 25 has_trait = ambitious }
		}
	}

	# Seal ALL forbidden techniques
	option = {
		name = murim_forbidden_technique.0200.c
		# Remove all forbidden technique traits
		if = { limit = { has_trait = murim_soul_devouring_palm } remove_trait = murim_soul_devouring_palm }
		if = { limit = { has_trait = murim_blood_reversal_art } remove_trait = murim_blood_reversal_art }
		if = { limit = { has_trait = murim_void_breathing_method } remove_trait = murim_void_breathing_method }
		if = { limit = { has_trait = murim_heaven_defying_sutra } remove_trait = murim_heaven_defying_sutra }
		if = { limit = { has_trait = murim_asura_body_technique } remove_trait = murim_asura_body_technique }
		if = { limit = { has_trait = murim_soul_splitting_art } remove_trait = murim_soul_splitting_art }
		if = { limit = { has_trait = murim_death_qi_cultivation } remove_trait = murim_death_qi_cultivation }
		if = { limit = { has_trait = murim_eternal_damnation_sutra } remove_trait = murim_eternal_damnation_sutra }
		hidden_effect = { murim_purge_all_corruption_effect = yes }
		add_piety = 500
		add_prestige = -300
		ai_chance = { base = 20 }
	}
}
