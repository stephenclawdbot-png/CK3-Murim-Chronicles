# === EVENT 0001: THOUSAND POISON BODY TRIAL ===
murim_tang_exp2.0001 = {
	type = character_event
	title = murim_tang_exp2.0001.t
	desc = murim_tang_exp2.0001.desc
	theme = intrigue
	
	trigger = {
		has_trait = murim_clan_tang
		has_trait_rank = { trait = murim_cultivation_rank value >= 2 }
		NOT = { has_character_flag = tang_thousand_poison_trial }
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = murim_technique_poison_master }
	}
	
	immediate = {
		set_character_flag = { flag = tang_thousand_poison_trial days = 3650 }
	}
	
	# Option A: Ingest the Thousand Poisons to build immunity
	option = {
		name = murim_tang_exp2.0001.a
		random_list = {
			50 = {
				add_trait = murim_thousand_poison_body
				add_prowess = 4
				add_intrigue = 3
				add_character_modifier = {
					modifier = murim_tang_poison_immune
					years = -1
				}
			}
			30 = {
				# Partial immunity
				add_intrigue = 2
				add_health = -1
				add_character_modifier = {
					modifier = murim_tang_partial_immunity
					years = 10
				}
			}
			20 = {
				# Poison backlash
				add_health = -4
				add_trait = murim_combat_trauma_poison_scarred
				trigger_event = { id = murim_tang_exp2.0002 days = 30 }
			}
		}
		ai_chance = { base = 60 }
	}
	
	# Option B: Master poisons through study, not ingestion
	option = {
		name = murim_tang_exp2.0001.b
		add_learning = 3
		add_intrigue = 2
		add_character_modifier = {
			modifier = murim_tang_scholarly_poisoner
			years = -1
		}
		ai_chance = { base = 30 }
	}
	
	# Option C: Reject poison arts - focus on hidden weapons only
	option = {
		name = murim_tang_exp2.0001.c
		add_martial = 2
		add_prowess = 3
		add_trait = murim_technique_hidden_weapon_master
		ai_chance = { base = 10 }
	}
}

# === EVENT 0003: ASSASSINATION CONTRACT - FIRST BLOOD ===
murim_tang_exp2.0003 = {
	type = character_event
	title = murim_tang_exp2.0003.t
	desc = murim_tang_exp2.0003.desc
	theme = intrigue
	
	trigger = {
		has_trait = murim_clan_tang
		intrigue >= 12
		NOT = { has_character_flag = tang_first_contract }
	}
	
	immediate = {
		set_character_flag = { flag = tang_first_contract days = 3650 }
		random_ruler = {
			limit = {
				murim_is_martial_artist_trigger = yes
				NOT = { has_trait = murim_clan_tang }
				NOT = { has_alliance_with = root }
			}
			save_scope_as = tang_target
		}
	}
	
	# Option A: Accept the contract - assassinate the target
	option = {
		name = murim_tang_exp2.0003.a
		add_gold = 300
		add_dread = 25
		random_list = {
			60 = {
				# Success
				scope:tang_target = {
					death = { death_reason = death_murder killer = root }
				}
				add_intrigue = 2
				add_character_modifier = {
					modifier = murim_tang_successful_assassin
					years = 5
				}
			}
			25 = {
				# Target survives but injured
				scope:tang_target = {
					add_health = -3
					add_opinion = {
						target = root
						modifier = murim_assassination_attempt_opinion
						opinion = -100
					}
				}
			}
			15 = {
				# Failed - caught
				add_prestige = -500
				imprison = { target = root type = house_arrest }
				scope:tang_target = {
					add_opinion = {
						target = root
						modifier = murim_assassination_attempt_opinion
						opinion = -100
					}
				}
			}
		}
		ai_chance = { base = 40 }
	}
	
	# Option B: Refuse - Tang Clan is not a mercenary guild
	option = {
		name = murim_tang_exp2.0003.b
		add_prestige = 200
		add_piety = 100
		ai_chance = { base = 40 }
	}
	
	# Option C: Take the contract but warn the target - double agent
	option = {
		name = murim_tang_exp2.0003.c
		add_gold = 300
		scope:tang_target = {
			add_opinion = {
				target = root
				modifier = murim_saved_life_opinion
				opinion = 50
			}
		}
		add_intrigue = 3
		add_character_modifier = {
			modifier = murim_tang_double_agent
			years = 5
		}
		ai_chance = { base = 20 }
	}
}
