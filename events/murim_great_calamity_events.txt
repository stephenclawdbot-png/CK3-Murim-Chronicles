# Murim Chronicles - The Great Calamity Event Chain
# THE APOCALYPSE. Cyclical world reset after 150-200 years.
# 4 PHASES: Gathering Storm (5yr) -> Great War (2-3yr) -> The Silence (1yr) -> Renewal
# AI AUTONOMY: All events fire autonomously via on_actions. Player participates but cannot prevent.
# INTEGRATION: murim_calamity_on_actions.txt, murim_calamity_triggers.txt, murim_calamity_traits.txt
# DESIGN: Every playthrough different. Weighted random outcomes. Clan personality drives AI behavior.

namespace = murim_calamity

# 0001 - The First Omen: Qi Disturbance
# Fires to all martial artists when omen phase begins.
murim_calamity.0001 = {
	type = character_event
	title = murim_calamity.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_trait = murim_omen_seer }
				desc = murim_calamity.0001.desc_seer
			}
			desc = murim_calamity.0001.desc
		}
	}
	theme = death
	
	trigger = {
		has_trait = martial_artist
		is_alive = yes
		NOT = { has_character_flag = received_first_omen }
	}
	
	immediate = {
		set_character_flag = received_first_omen
		create_character_memory = { type = murim_first_omen_memory }
	}
	
	# Option A: Meditate on the disturbance
	option = {
		name = murim_calamity.0001.a
		add_character_modifier = {
			modifier = murim_omen_meditation
			years = 5
		}
		# 20% chance to gain Omen Seer trait
		random = {
			chance = 20
			add_trait = murim_omen_seer
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				OR = {
					has_trait = shaolin_monk
					has_trait = wudang_disciple
				}
			}
		}
	}
	
	# Option B: Ignore it - just a tremor
	option = {
		name = murim_calamity.0001.b
		add_character_modifier = {
			modifier = murim_omen_denial
			years = 3
		}
		stress_impact = {
			paranoid = medium_stress_impact_gain
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				has_trait = iron_fist_disciple
			}
		}
	}
	
	# Option C: Warn others - something terrible approaches
	option = {
		name = murim_calamity.0001.c
		add_prestige = -100
		# Warn nearby characters
		every_courtier_or_guest = {
			limit = { has_trait = martial_artist }
			trigger_event = {
				id = murim_calamity.0002
				days = { 7 30 }
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				OR = {
					has_trait = beggar_sect_member
					has_trait = mount_hua_disciple
				}
			}
		}
	}
}

# 0010 - The Gathering Storm Begins
murim_calamity.0010 = {
	type = character_event
	title = murim_calamity.0010.t
	desc = murim_calamity.0010.desc
	theme = war
	
	trigger = {
		OR = {
			is_ruler = yes
			has_trait = martial_artist
		}
		NOT = { has_character_flag = storm_acknowledged }
	}
	
	immediate = {
		set_character_flag = storm_acknowledged
		create_character_memory = { type = murim_storm_begins_memory }
	}
	
	# Option A: Rally your clan for war
	option = {
		name = murim_calamity.0010.a
		trigger = { has_trait = martial_artist }
		add_character_modifier = {
			modifier = murim_rallying_for_war
			years = 5
		}
		add_trait = murim_calamity_accelerator
		ai_chance = {
			base = 30
			modifier = {
				add = 40
				OR = {
					has_trait = heavenly_demon_member
					has_trait = iron_fist_disciple
				}
			}
		}
	}
	
	# Option B: Seek alliances - strength in unity
	option = {
		name = murim_calamity.0010.b
		trigger = { has_trait = martial_artist }
		add_character_modifier = {
			modifier = murim_seeking_allies
			years = 5
		}
		# Find a potential ally
		random_living_character = {
			limit = {
				has_trait = martial_artist
				NOT = { this = prev }
				NOT = { has_relation_rival = prev }
			}
			add_opinion = {
				target = prev
				modifier = murim_alliance_proposal_opinion
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				OR = {
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = beggar_sect_member
				}
			}
		}
	}
	
	# Option C: Prepare to survive - hide and wait
	option = {
		name = murim_calamity.0010.c
		add_character_modifier = {
			modifier = murim_survival_preparation
			years = 5
		}
		# Slight increase to survival odds
		if = {
			limit = { NOT = { has_trait = murim_fate_blessed } }
			random = {
				chance = 15
				add_trait = murim_fate_blessed
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				OR = {
					has_trait = phantom_blade_disciple
					has_trait = poison_flower_disciple
					has_trait = tang_clan_member
				}
			}
		}
	}
	
	# Option D: Non-martial - pray for deliverance
	option = {
		name = murim_calamity.0010.d
		trigger = { NOT = { has_trait = martial_artist } }
		add_piety = 100
		add_stress = 20
		ai_chance = { base = 100 }
	}
}

# 0011 - AI Aggressive Clan Action (Heavenly Demon, Iron Fist)
murim_calamity.0011 = {
	type = character_event
	title = murim_calamity.0011.t
	desc = murim_calamity.0011.desc
	theme = war
	hidden = yes  # AI-only event
	
	trigger = {
		has_trait = martial_artist
		is_ai = yes
	}
	
	immediate = {
		# AI declares hostility toward a rival
		random_living_character = {
			limit = {
				has_trait = martial_artist
				NOT = { this = prev }
				NOT = { has_relation_ally = prev }
			}
			save_scope_as = storm_target
		}
	}
	
	option = {
		name = murim_calamity.0011.a
		trigger = { exists = scope:storm_target }
		add_opinion = {
			target = scope:storm_target
			modifier = murim_calamity_hostility_opinion
		}
		scope:storm_target = {
			add_opinion = {
				target = prev
				modifier = murim_calamity_threatened_opinion
			}
		}
		# Chance of immediate skirmish
		random = {
			chance = 30
			trigger_event = {
				id = murim_calamity.0014  # Border skirmish
				days = { 7 30 }
			}
		}
		ai_chance = { base = 100 }
	}
}

# 0020 - AI Combat Resolution (fires monthly during war)
murim_calamity.0020 = {
	type = character_event
	title = murim_calamity.0020.t
	desc = murim_calamity.0020.desc
	theme = war
	
	trigger = {
		has_trait = martial_artist
		has_global_variable = murim_calamity_war_started
		exists = scope:calamity_opponent
	}
	
	immediate = {
		create_character_memory = { type = murim_calamity_combat_memory }
	}
	
	# Duel the opponent
	option = {
		name = murim_calamity.0020.a
		duel = {
			skill = prowess
			target = scope:calamity_opponent
			50 = {
				desc = murim_calamity.0020.victory
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				add_prestige = 50
				# Opponent may die
				scope:calamity_opponent = {
					random = {
						chance = 40
						death = {
							death_reason = death_duel
							killer = prev
						}
					}
					if = {
						limit = { is_alive = yes }
						random_list = {
							40 = { add_trait = wounded_1 }
							30 = { add_trait = wounded_2 }
							20 = { add_trait = maimed }
							10 = { add_trait = one_eyed }
						}
					}
				}
			}
			50 = {
				desc = murim_calamity.0020.defeat
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				# You may die
				random = {
					chance = 30
					if = {
						limit = { has_trait = murim_fate_blessed }
						# Fate-blessed survive but barely
						add_trait = wounded_2
					}
					else = {
						death = {
							death_reason = death_duel
							killer = scope:calamity_opponent
						}
					}
				}
				if = {
					limit = { is_alive = yes }
					random_list = {
						40 = { add_trait = wounded_1 }
						30 = { add_trait = wounded_2 }
						20 = { add_trait = maimed }
						10 = { add_trait = scarred }
					}
				}
			}
		}
		ai_chance = { base = 100 }
	}
}

			20 = {
				# Become wandering teacher
				add_character_modifier = {
					modifier = murim_wandering_sage
					years = 20
				}
			}
			10 = {
				# Become a recluse - guard ruins
				add_character_modifier = {
					modifier = murim_ruin_guardian
					years = 20
				}
			}
		}
		# Generate new rivalries among survivors
		random_living_character = {
			limit = {
				has_trait = murim_calamity_survivor
				NOT = { this = prev }
				has_trait = martial_artist
			}
			# 30% chance new rivalry forms
			random = {
				chance = 30
				set_relation_rival = prev
			}
			# 40% chance friendship forms
			random = {
				chance = 40
				set_relation_friend = prev
			}
		}
		ai_chance = { base = 100 }
	}
}
