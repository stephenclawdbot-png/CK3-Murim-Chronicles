# UNDERGROUND RECRUITMENT
######################################

murim_recruit_to_underground_effect = {
	# Target is scope:recruit
	scope:recruit = {
		set_character_flag = murim_jianghu_inducted
		set_variable = {
			name = murim_jianghu_influence
			value = 10
		}
		set_variable = {
			name = murim_underground_rep
			value = 0
		}
		add_opinion = {
			target = root
			modifier = murim_recruited_me_to_jianghu
		}
		# Grant starting role based on skills
		if = {
			limit = { intrigue >= 14 }
			add_trait = sect_spy
		}
		else_if = {
			limit = { martial >= 14 }
			add_trait = jianghu_enforcer
		}
		else_if = {
			limit = { stewardship >= 14 }
			add_trait = black_market_dealer
		}
		else = {
			add_character_modifier = {
				modifier = murim_jianghu_initiate
				years = 3
			}
		}
	}
	# Recruiter gains influence
	change_variable = {
		name = murim_jianghu_influence
		add = 5
	}
	add_prestige = 50
}

murim_promote_jianghu_operative_effect = {
	# Requires scope:operative
	scope:operative = {
		change_variable = {
			name = murim_jianghu_influence
			add = 15
		}
		change_variable = {
			name = murim_underground_rep
			add = 10
		}
		add_prestige = 100
		# Promote to higher tier based on current role
		if = {
			limit = {
				has_trait = sect_spy
				intrigue >= 18
			}
			remove_trait = sect_spy
			add_trait = shadow_broker
		}
		else_if = {
			limit = {
				has_trait = jianghu_enforcer
				prowess >= 18
			}
			remove_trait = jianghu_enforcer
			add_trait = hidden_master
		}
		else_if = {
			limit = {
				has_trait = black_market_dealer
				stewardship >= 18
			}
			remove_trait = black_market_dealer
			add_trait = shadow_broker
		}
	}
}

murim_expose_spy_at_court_effect = {
	# scope:exposed_spy is the spy being revealed
	scope:exposed_spy = {
		remove_character_modifier = murim_deep_cover_operative
		remove_character_flag = murim_planted_at_court
		add_character_modifier = {
			modifier = murim_exposed_operative
			years = 10
		}
		add_prestige = -200
		# Chance of imprisonment
		if = {
			limit = {
				exists = liege
				liege = { has_trait = just }
			}
			imprison = {
				target = scope:exposed_spy
				type = dungeon
			}
		}
	}
	# Spy's handler loses influence
	if = {
		limit = {
			exists = scope:handler
		}
		scope:handler = {
			change_variable = {
				name = murim_jianghu_influence
				add = -15
			}
			add_prestige = -100
		}
	}
}

# ASSASSINATION VIA SECT CONTRACT
######################################

murim_assassinate_via_sect_effect = {
	# scope:assassination_target is the victim
	# Root is the commissioner
	add_gold = -150
	change_variable = {
		name = murim_jianghu_influence
		add = -5
	}

	random_list = {
		40 = {
			modifier = {
				factor = 1.5
				has_trait = shadow_broker
			}
			modifier = {
				factor = 1.3
				intrigue >= 18
			}
			# Assassination succeeds cleanly
			scope:assassination_target = {
				death = {
					death_reason = death_murder_unknown
				}
			}
			root = {
				add_character_modifier = {
					modifier = murim_successful_contract
					years = 3
				}
				change_variable = {
					name = murim_underground_rep
					add = 10
				}
			}
		}
		25 = {
			# Assassination succeeds but leaves evidence
			scope:assassination_target = {
				death = {
					death_reason = death_murder_unknown
				}
			}
			root = {
				add_character_modifier = {
					modifier = murim_sloppy_contract
					years = 5
				}
				random = {
					chance = 40
					trigger_event = {
						id = murim_jianghu.0070
						days = { 10 60 }
					}
				}
			}
		}
		20 = {
			# Assassination fails - target survives
			scope:assassination_target = {
				add_character_modifier = {
					modifier = murim_survived_assassination
					years = 5
				}
				add_prowess_skill = 1
			}
			root = {
				add_character_modifier = {
					modifier = murim_failed_contract
					years = 3
				}
				change_variable = {
					name = murim_underground_rep
					add = -5
				}
			}
		}
		15 = {
			# Assassination fails and traces back
			scope:assassination_target = {
				add_character_modifier = {
					modifier = murim_survived_assassination
					years = 5
				}
				add_opinion = {
					target = root
					modifier = murim_tried_to_kill_me
				}
			}
			root = {
				add_character_modifier = {
					modifier = murim_assassination_traced
					years = 10
				}
				add_prestige = -200
				change_variable = {
					name = murim_jianghu_influence
					add = -15
				}
			}
		}
	}
}

murim_control_trade_route_effect = {
	# scope:route_type determines which route
	add_gold = -300
	if = {
		limit = { scope:route_type = flag:silk_road }
		set_character_flag = murim_controls_silk_road_segment
		add_character_modifier = {
			modifier = murim_silk_road_profits
			years = -1
		}
	}
	else_if = {
		limit = { scope:route_type = flag:tea_route }
		set_character_flag = murim_controls_tea_route
		add_character_modifier = {
			modifier = murim_tea_route_profits
			years = -1
		}
	}
	else_if = {
		limit = { scope:route_type = flag:salt_route }
		set_character_flag = murim_controls_salt_route
		add_character_modifier = {
			modifier = murim_salt_route_profits
			years = -1
		}
	}
	else_if = {
		limit = { scope:route_type = flag:jade_route }
		set_character_flag = murim_controls_jade_route
		add_character_modifier = {
			modifier = murim_jade_route_profits
			years = -1
		}
	}
	add_character_modifier = {
		modifier = murim_trade_route_controller
		years = -1
	}
	change_variable = {
		name = murim_jianghu_influence
		add = 15

}

murim_lose_trade_route_effect = {
	# scope:route_type determines which is lost

		remove_character_flag = murim_controls_silk_road_segment
		remove_character_modifier = murim_silk_road_profits

		remove_character_flag = murim_controls_tea_route
		remove_character_modifier = murim_tea_route_profits

		limit = { has_character_flag = murim_controls_salt_route }
		remove_character_flag = murim_controls_salt_route
		remove_character_modifier = murim_salt_route_profits
	}

		limit = { has_character_flag = murim_controls_jade_route }
		remove_character_flag = murim_controls_jade_route
		remove_character_modifier = murim_jade_route_profits

	}

		if = {
			limit = { NOT = { murim_is_jianghu_member_trigger = yes } }

		}
	}
}
