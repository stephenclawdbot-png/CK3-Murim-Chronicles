namespace = murim_shadow_politics

# Murim Chronicles - Heavenly Demon Shadow Politics Events
# THE HEAVENLY DEMON INFILTRATES GOVERNMENTS. Rulers with aggressive/deceitful/ambitious
# traits face yearly 15% chance of HD approach. Emperor becomes puppet.
# INTEGRATION: murim_clan_triggers.txt (is_heavenly_demon_leader, faction triggers)
# INTEGRATION: murim_faction_war_events.txt (infiltration revealed during wars)
# INTEGRATION: murim_opinion_modifiers.txt (martial opinion from shadow politics)
# INTEGRATION: murim_china_center_triggers.txt (China/Korea governments targeted)
# VANILLA SEPARATION: Shadow politics uses INTRIGUE layer, not vanilla politics.
#   Puppet modifier is HIDDEN. Does not change government type or succession.
#   Effects are martial-world scope: persecute orthodox, protect unorthodox, rig tournaments.


# =================================================================
# EVENT 1: HEAVENLY DEMON APPROACHES RULER (Initial Contact)
# Fires yearly for aggressive/deceitful/ambitious rulers in China/Korea
# =================================================================
murim_shadow_politics.0001 = {
	type = character_event
	title = murim_shadow_politics.0001.t
	desc = murim_shadow_politics.0001.desc
	theme = intrigue
	
	trigger = {
		is_ruler = yes
		is_adult = yes
		NOT = { has_character_flag = flag_heavenly_demon_puppet_emperor }
		NOT = { has_character_flag = flag_rejected_heavenly_demon }
		NOT = { has_character_flag = flag_orthodox_faction_member }
		OR = {
			murim_is_china_core_trigger = yes
			murim_is_goryeo_trigger = yes
		}
		OR = {
			has_trait = ambitious
			has_trait = deceitful
			has_trait = sadistic
			has_trait = callous
			has_trait = arbitrary
			has_trait = vengeful
		}
		any_living_character = {
			has_character_flag = flag_heavenly_demon_leader
			is_alive = yes
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			factor = 1.5
			has_trait = deceitful
			has_trait = ambitious
		}
		modifier = {
			factor = 0.5
			has_trait = just
		}
		modifier = {
			factor = 2.0
			has_trait = sadistic
		}
	}
	
	immediate = {
		random_living_character = {
			limit = { has_character_flag = flag_heavenly_demon_leader }
			save_scope_as = hd_leader
		}
	}
	
	# Option A: Accept the Heavenly Demon's offer (become puppet)
	option = {
		name = murim_shadow_politics.0001.a
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = 15
				has_trait = deceitful
			}
			modifier = {
				add = -30
				has_trait = just
			}
		}
		set_character_flag = {
			flag = flag_heavenly_demon_puppet_emperor
			years = -1
		}
		add_character_modifier = {
			modifier = murim_shadow_emperor_power
			years = -1
		}
		hidden_effect = {
			set_global_variable = {
				name = murim_shadow_puppet_active
				value = yes
			}
			scope:hd_leader = {
				set_character_flag = {
					flag = flag_has_puppet_ruler
					years = -1
				}
			}
			trigger_event = { id = murim_shadow_politics.0002 days = 30 }
		}
		# Gain: military power, gold, feared
		add_dread = 20
		add_gold = 500
	}
	
	# Option B: Reject with fear (back away quietly)
	option = {
		name = murim_shadow_politics.0001.b
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				has_trait = craven
			}
		}
		set_character_flag = {
			flag = flag_rejected_heavenly_demon
			years = 5
		}
		stress_impact = {
			base = minor_stress_impact_gain
			craven = medium_stress_impact_gain
		}
	}
	
	# Option C: Reject and report to orthodox clans (heroic)
	option = {
		name = murim_shadow_politics.0001.c
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = just
			}
			modifier = {
				add = 20
				has_trait = honest
			}
		}
		set_character_flag = {
			flag = flag_rejected_heavenly_demon
			years = -1
		}
		add_prestige = 200
		hidden_effect = {
			trigger_event = { id = murim_shadow_politics.0003 days = 14 }
		}
		# Risk: HD may retaliate
		random = {
			chance = 30
			trigger_event = { id = murim_shadow_politics.0004 days = 60 }
		}
	}
	
	# Option D: Pretend to accept, secretly work against them (double agent ruler)
	option = {
		name = murim_shadow_politics.0001.d
		trigger = {
			intrigue >= 14
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				intrigue >= 18
			}
		}
		set_character_flag = {
			flag = flag_heavenly_demon_puppet_emperor
			years = -1
		}
		set_character_flag = {
			flag = flag_secretly_anti_hd
			years = -1
		}
		add_character_modifier = {
			modifier = murim_double_agent_ruler
			years = -1
		}
		hidden_effect = {
			trigger_event = { id = murim_shadow_politics.0005 days = 90 }
		}
	}
}
