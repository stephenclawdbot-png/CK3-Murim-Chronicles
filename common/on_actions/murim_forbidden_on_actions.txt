# === YEARLY PULSE: FORBIDDEN TECHNIQUE CORRUPTION ===
# Checks corruption effects annually for all corrupted characters
murim_forbidden_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_forbidden_yearly_pulse
	}

	effect = {
		every_ruler = {
			limit = {
				murim_has_any_corruption_trigger = yes
				is_alive = yes
			}
			# Corruption has ongoing health costs
			random_list = {
				70 = { }
				20 = {
					trigger_event = { id = murim_forbidden_technique.0400 }
				}
				10 = {
					# Severe corruption episode
					trigger_event = { id = murim_forbidden_technique.0410 }
				}
			}
		}
	}
}

# === YEARLY PULSE: ASCENSION MONITORING ===
# Checks for potential ascension candidates
murim_ascension_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_ascension_yearly_pulse
	}

	effect = {
		every_ruler = {
			limit = {
				has_variable = murim_martial_artist
				var:murim_cultivation_rank >= 6
				NOT = { murim_is_immortal_trigger = yes }
				NOT = { has_character_modifier = murim_heavens_attention }
			}
			random = {
				chance = 15
				add_character_modifier = { modifier = murim_heavens_attention years = 3 }
			}
		}
	}
}

# === MONTHLY PULSE: AI FORBIDDEN TECHNIQUE DECISIONS ===
# AI characters autonomously seek or avoid forbidden knowledge
murim_ai_forbidden_monthly_pulse = {
	on_actions = {
		delay = { months = 3 }
		murim_ai_forbidden_monthly_pulse
	}

	effect = {
		every_ruler = {
			limit = {
				is_ai = yes
				has_variable = murim_martial_artist
				var:murim_cultivation_rank >= 3
				NOT = { has_character_flag = murim_ai_forbidden_cooldown }
			}
			random_list = {
				80 = { }
				10 = {
					# AI discovers forbidden knowledge opportunity
					modifier = { add = 15 has_trait = ambitious }
					modifier = { add = 10 has_trait = sadistic }
					modifier = { add = -20 has_trait = just }
					trigger_event = { id = murim_forbidden_technique.0001 days = { 7 30 } }
					set_character_flag = { flag = murim_ai_forbidden_cooldown years = 3 }
				}
				10 = {
					# AI is hunted for knowing forbidden techniques
					modifier = { factor = 0 NOT = { murim_has_any_forbidden_technique_trigger = yes } }
					trigger_event = { id = murim_forbidden_technique.0500 days = { 7 30 } }
				}
			}
		}
	}
}

# === YEARLY PULSE: CONTINENTAL WAR TENSION ===
# Monitors faction tension and can trigger war
murim_continental_war_tension_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_continental_war_tension_pulse
	}

	effect = {
		# Only fire if no war active and no cooldown
		if = {
			limit = {
				NOT = { has_global_variable = murim_continental_war_active }
				NOT = { has_global_variable = murim_continental_war_cooldown }
			}
			random = {
				chance = 8
				# Tension builds — small incident
				random_ruler = {
					limit = {
						has_variable = murim_martial_artist
						var:murim_cultivation_rank >= 4
					}
					trigger_event = { id = murim_continental_war.0090 }
				}
			}
		}
		# Clear cooldown after 25 years
		if = {
			limit = {
				has_global_variable = murim_continental_war_cooldown
				# Approximate cooldown check
			}
			random = {
				chance = 4
				remove_global_variable = murim_continental_war_cooldown
			}
		}
	}
}

# === YEARLY PULSE: TRIBUNAL OPERATIONS ===
murim_tribunal_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_tribunal_yearly_pulse
	}

	effect = {
		# Tribunal summons random offenders
		random_ruler = {
			limit = {
				has_trait = murim_tribunal_judge
				is_alive = yes
			}
			random_list = {
				70 = { }
				30 = {
					trigger_event = { id = murim_tribunal_system.0020 days = { 7 60 } }
				}
			}
		}

		# Fill empty tribunal seats
		if = {
			limit = {
				any_ruler = {
					count < 7
					has_trait = murim_tribunal_judge
				}
			}
			random_ruler = {
				limit = {
					has_variable = murim_martial_artist
					var:murim_cultivation_rank >= 5
					NOT = { has_trait = murim_tribunal_judge }
					NOT = { has_trait = murim_tribunal_exile }
					NOT = { murim_has_any_corruption_trigger = yes }
					prowess >= 20
				}
				random = {
					chance = 10
					trigger_event = { id = murim_tribunal_system.0030 days = { 30 90 } }
				}
			}
		}
	}
}

# === YEARLY PULSE: ASCENSION EVENT TRIGGERS ===
# High-level cultivators get ascension-related events
murim_ascension_event_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_ascension_event_yearly_pulse
	}

	effect = {
		# Half-immortals get immortal rivalry/mentoring events
		every_ruler = {
			limit = {
				has_trait = murim_half_immortal
				is_alive = yes
				NOT = { has_character_flag = murim_ascension_event_cooldown }
			}
			random_list = {
				60 = { }
				15 = {
					trigger_event = { id = murim_ascension.0040 days = { 30 90 } }
					set_character_flag = { flag = murim_ascension_event_cooldown years = 2 }
				}
				15 = {
					trigger_event = { id = murim_ascension.0050 days = { 30 90 } }
					set_character_flag = { flag = murim_ascension_event_cooldown years = 2 }
				}
				10 = {
					trigger_event = { id = murim_ascension.0060 days = { 30 90 } }
					set_character_flag = { flag = murim_ascension_event_cooldown years = 2 }
				}
			}
		}

		# Eligible cultivators get ascension attempt prompts
		every_ruler = {
			limit = {
				has_variable = murim_martial_artist
				var:murim_cultivation_rank >= 6
				NOT = { has_trait = murim_half_immortal }
				NOT = { has_trait = murim_true_immortal }
				NOT = { has_trait = murim_ascension_failed }
				NOT = { has_character_flag = murim_ascension_prompt_cooldown }
				has_character_modifier = murim_heavens_attention
			}
			random = {
				chance = 20
				trigger_event = { id = murim_ascension.0001 days = { 30 180 } }
				set_character_flag = { flag = murim_ascension_prompt_cooldown years = 5 }
			}
		}
	}
}

# === YEARLY PULSE: IMPERIAL COURT EVENTS ===
# AI and player get imperial court events when relevant
murim_imperial_court_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_imperial_court_yearly_pulse
	}

	effect = {
		# Imperial edicts affect all martial artists periodically
		if = {
			limit = {
				NOT = { has_global_variable = murim_imperial_edict_cooldown }
			}
			random = {
				chance = 12
				random_ruler = {
					limit = {
						has_variable = murim_martial_artist
						var:murim_cultivation_rank >= 3
					}
					trigger_event = { id = murim_imperial_court.0010 days = { 30 60 } }
				}
				set_global_variable = { name = murim_imperial_edict_cooldown value = 1 }
			}
		}

		# Clear edict cooldown
		if = {
			limit = { has_global_variable = murim_imperial_edict_cooldown }
			random = {
				chance = 20
				remove_global_variable = murim_imperial_edict_cooldown
			}
		}

		# High-ranking cultivators attract imperial attention
		every_ruler = {
			limit = {
				has_variable = murim_martial_artist
				var:murim_cultivation_rank >= 5
				NOT = { has_character_flag = murim_imperial_contact_cooldown }
				NOT = { has_character_modifier = murim_imperial_enforcer }
			}
			random_list = {
				80 = { }
				10 = {
					# Imperial spy recruitment
					trigger_event = { id = murim_imperial_court.0030 days = { 30 120 } }
					set_character_flag = { flag = murim_imperial_contact_cooldown years = 5 }
				}
				5 = {
					# Imperial tournament invitation
					trigger_event = { id = murim_imperial_court.0060 days = { 30 90 } }
					set_character_flag = { flag = murim_imperial_contact_cooldown years = 5 }
				}
				5 = {
					# Marriage proposal from throne
					modifier = { factor = 0 is_married = yes }
					trigger_event = { id = murim_imperial_court.0070 days = { 30 90 } }
					set_character_flag = { flag = murim_imperial_contact_cooldown years = 5 }
				}
			}
		}

		# Imperial enforcers get conscription events during war
		every_ruler = {
			limit = {
				has_character_modifier = murim_imperial_enforcer
				is_alive = yes
				NOT = { has_character_flag = murim_conscription_cooldown }
			}
			random = {
				chance = 15
				trigger_event = { id = murim_imperial_court.0020 days = { 30 60 } }
				set_character_flag = { flag = murim_conscription_cooldown years = 3 }
			}
		}
	}
}

# === YEARLY PULSE: CALAMITY ESCALATION ===
# Auto-fires escalation events based on world tension
murim_calamity_escalation_yearly_pulse = {
	on_actions = {
		delay = { years = 1 }
		murim_calamity_escalation_yearly_pulse
	}

	effect = {
		# Only fire if no active calamity
		if = {
			limit = {
				NOT = { has_global_variable = murim_great_calamity_active }
				NOT = { has_global_variable = murim_escalation_cooldown }
			}
			random = {
				chance = 10
				random_ruler = {
					limit = {
						has_variable = murim_martial_artist
						var:murim_cultivation_rank >= 3
					}
					trigger_event = { id = murim_calamity_escalation.0001 days = { 30 90 } }
				}
				set_global_variable = { name = murim_escalation_cooldown value = 1 }
			}
		}

		# During calamity, fire refugee and demon invasion events
		if = {
			limit = { has_global_variable = murim_great_calamity_active }
			random_ruler = {
				limit = {
					has_variable = murim_martial_artist
					NOT = { has_character_flag = murim_calamity_expansion_cooldown }
				}
				random_list = {
					60 = { }
					15 = {
						trigger_event = { id = murim_calamity_refugee.0001 days = { 15 60 } }
						set_character_flag = { flag = murim_calamity_expansion_cooldown years = 2 }
					}
					15 = {
						trigger_event = { id = murim_demonic_invasion.0001 days = { 15 60 } }
						set_character_flag = { flag = murim_calamity_expansion_cooldown years = 2 }
					}
					10 = {
						trigger_event = { id = murim_calamity_refugee.0030 days = { 15 60 } }
						set_character_flag = { flag = murim_calamity_expansion_cooldown years = 2 }
					}
				}
			}
		}

		# Clear escalation cooldown
		if = {
			limit = { has_global_variable = murim_escalation_cooldown }
			random = {
				chance = 15
				remove_global_variable = murim_escalation_cooldown
			}
		}
	}
}

# === ON DEATH: LEGACY GRUDGE INHERITANCE ===
on_death = {
	effect = {
		if = {
			limit = {
				has_trait = murim_sworn_nemesis
				has_variable = murim_martial_artist
			}
			# Pass grudge to a disciple/child
			random_child = {
				limit = {
					has_variable = murim_martial_artist
					is_alive = yes
				}
				murim_inherit_grudge_effect = yes
			}
		}
	}
}
