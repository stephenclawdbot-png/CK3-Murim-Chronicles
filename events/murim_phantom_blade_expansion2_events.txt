### EVENT 0004 — Assassination Contract: Political Target

	type = character_event
	title = murim_phantom_blade_expansion2.0004.t

	left_portrait = {
		character = root
		animation = scheme
	}
	right_portrait = {
		character = scope:political_target
		animation = worry
	}
	
	trigger = {
		has_trait = murim_phantom_blade_clan
		murim_is_martial_artist_trigger = yes

		NOT = { has_character_flag = phantom_political_assassination }

	immediate = {
		set_character_flag = phantom_political_assassination
		random_living_character = {
			limit = {
				is_ruler = yes

			skill = intrigue

				desc = murim_phantom_blade_expansion2.0004.a.success

					value = scope:duel_value

					min = -49
				}
				add_dread = 30
				add_prestige = 300
				scope:political_target = {
					death = {
						death_reason = death_murder_unknown

					}
				}
			}

				desc = murim_phantom_blade_expansion2.0004.a.fail

					value = scope:duel_value
