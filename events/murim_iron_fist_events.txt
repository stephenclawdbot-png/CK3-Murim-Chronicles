# EVENT 1: THE IRON BODY TRIAL
# Every Iron Fist disciple must endure the Iron Body Trial.
# Stand under a waterfall while elders strike you with iron rods.
# If you don't break, you become a true member.
##############################
murim_iron_fist.0001 = {
	type = character_event
	title = murim_iron_fist.0001.t
	desc = murim_iron_fist.0001.desc
	theme = martial_arts
	
	trigger = {
		is_iron_fist_disciple_trigger = yes
		NOT = { has_character_flag = iron_fist_body_trial }
		age >= 15
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			prowess >= 12
		}
		modifier = {
			add = 1
			has_trait = brave
		}
	}
	
	immediate = {
		set_character_flag = iron_fist_body_trial
		create_character_memory = {
			type = iron_fist_body_trial_memory
		}
	}
	
	# Option A: Endure through sheer willpower (pure grit)
	option = {
		name = murim_iron_fist.0001.a
		random_list = {
			50 = {
				# Body holds - you pass
				add_prowess_xp = 500
				add_character_modifier = {
					modifier = iron_fist_iron_body_stage_one_modifier
					years = -1
				}
				add_trait = brave
				create_character_memory = {
					type = iron_fist_trial_passed_memory
				}
			}
			30 = {
				# Body breaks but you don't scream - partial pass
				add_prowess_xp = 200
				add_trait = wounded_1
				add_character_modifier = {
					modifier = iron_fist_trial_partial_modifier
					years = 3
				}
			}
			20 = {
				# Collapse - must retake the trial later
				add_trait = wounded_2
				add_stress = 40
				add_character_modifier = {
					modifier = iron_fist_trial_failed_modifier
					years = 1
				}
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}
	
	# Option B: Channel internal qi to reinforce your body (reformist technique)

		name = murim_iron_fist.0001.b

			has_trait = qi_cultivator

			70 = {

					modifier = iron_fist_qi_reinforced_body_modifier
					years = -1
				}
			}

				# Traditionalist elder detects qi use - you're punished
				add_prowess_xp = 100
				add_character_modifier = {
					modifier = iron_fist_cheater_stigma_modifier

				}

					limit = {
						is_iron_fist_disciple_trigger = yes
					}

						modifier = iron_fist_qi_cheater_opinion

		}

			base = 25
