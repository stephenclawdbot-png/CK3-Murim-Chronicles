# ==========================================
# ALCHEMY INITIALIZATION EFFECTS
# Called at game start for characters who
# naturally handle herbs and pills.
# Tang Clan: poison/medicine bloodline specialists
# Poison Flower Sect: toxin and herb masters
# ==========================================

murim_init_tang_alchemy_effect = {
	# Tang Clan members start with an herb stockpile
	# and a baseline alchemy skill from family training
	if = {
		limit = { NOT = { has_variable = murim_herb_stockpile } }
		set_variable = { name = murim_herb_stockpile value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = murim_alchemy_skill } }
		set_variable = { name = murim_alchemy_skill value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = murim_alchemy_xp } }
		set_variable = { name = murim_alchemy_xp value = 0 }
	}
	# Elders start with more resources
	random_list = {
		30 = {
			# Full member / elder
			modifier = {
				factor = 2.0
				OR = {
					has_trait = tang_clan_elder
					has_trait = tang_clan_inner_member
				}
			}
			change_variable = { name = murim_herb_stockpile add = 8 }
			change_variable = { name = murim_alchemy_skill add = 3 }
			change_variable = { name = murim_alchemy_xp add = 10 }
		}
		50 = {
			# Standard member
			change_variable = { name = murim_herb_stockpile add = 4 }
			change_variable = { name = murim_alchemy_skill add = 2 }
			change_variable = { name = murim_alchemy_xp add = 5 }
		}
		20 = {
			# Newer / younger member
			change_variable = { name = murim_herb_stockpile add = 2 }
			change_variable = { name = murim_alchemy_skill add = 1 }
		}
	}
}

murim_init_poison_flower_alchemy_effect = {
	# Poison Flower sect members start with herb knowledge
	# focused on toxins - higher stockpile, moderate skill
	if = {
		limit = { NOT = { has_variable = murim_herb_stockpile } }
		set_variable = { name = murim_herb_stockpile value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = murim_alchemy_skill } }
		set_variable = { name = murim_alchemy_skill value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = murim_alchemy_xp } }
		set_variable = { name = murim_alchemy_xp value = 0 }
	}
	random_list = {
		40 = {
			modifier = {
				factor = 1.5
				has_trait = poison_flower_elder
			}
			change_variable = { name = murim_herb_stockpile add = 10 }
			change_variable = { name = murim_alchemy_skill add = 2 }
			change_variable = { name = murim_alchemy_xp add = 8 }
		}
		60 = {
			change_variable = { name = murim_herb_stockpile add = 5 }
			change_variable = { name = murim_alchemy_skill add = 1 }
			change_variable = { name = murim_alchemy_xp add = 3 }
		}
	}
}
