# Check celestial alignment based on 60-year cycle
# Safe: initializes all variables with defaults
check_celestial_alignment_effect = {
	# Safety: ensure variable exists before modifying
	if = {
		limit = {
			NOT = { exists = var:current_celestial_phase }
		}
		set_variable = {
			name = current_celestial_phase
			value = 0
		}
	}

	# Calculate based on game year (60-year Sexagenary cycle)
	set_variable = {
		name = current_celestial_phase
		value = {
			add = current_year
			modulo = 60
		}
	}

	# Azure Dragon (East, Spring) - Orthodox favored, Wood element
	# Years 0-14 of the cycle
	if = {
		limit = {
			exists = var:current_celestial_phase
			var:current_celestial_phase < 15
		}
		set_variable = { name = celestial_beast value = flag:azure_dragon }
		set_variable = { name = orthodox_favor value = 20 }
		set_variable = { name = dominant_element value = flag:element_wood }
		set_variable = { name = celestial_season value = flag:spring }
	}
	# Vermilion Bird (South, Summer) - Orthodox peak, Fire element
	# Years 15-29
	else_if = {
		limit = {
			exists = var:current_celestial_phase
			var:current_celestial_phase < 30
		}
		set_variable = { name = celestial_beast value = flag:vermilion_bird }
		set_variable = { name = orthodox_favor value = 30 }
		set_variable = { name = dominant_element value = flag:element_fire }
		set_variable = { name = celestial_season value = flag:summer }
	}
	# White Tiger (West, Autumn) - Unorthodox rising, Metal element
	# Years 30-44
	else_if = {
		limit = {
			exists = var:current_celestial_phase
			var:current_celestial_phase < 45
		}
		set_variable = { name = celestial_beast value = flag:white_tiger }
		set_variable = { name = orthodox_favor value = -10 }
		set_variable = { name = dominant_element value = flag:element_metal }
		set_variable = { name = celestial_season value = flag:autumn }
	}
	# Black Tortoise (North, Winter) - Unorthodox peak, Water element
	# Years 45-59
	else = {
		set_variable = { name = celestial_beast value = flag:black_tortoise }
		set_variable = { name = orthodox_favor value = -20 }
		set_variable = { name = dominant_element value = flag:element_water }
		set_variable = { name = celestial_season value = flag:winter }
	}
}

				set_variable = { name = elemental_power value = 10 }

				modifier = {
					add = 15
					OR = {
						has_trait = stubborn

# Metal collects Water, Water nourishes Wood

		}

				modifier = elemental_generation_bonus

				years = 1

			}
			set_variable = { name = elemental_harmony value = 1 }

}

# Advance elemental mastery -- called when cultivating in harmony
advance_elemental_mastery_effect = {
	if = {
		limit = {
			exists = var:elemental_power
		}
		change_variable = {
			name = elemental_power
			add = 5
		}
		# Milestone: Minor Elemental Attunement (25+)
		if = {
			limit = {
				var:elemental_power >= 25
				NOT = { has_character_modifier = minor_elemental_attunement }
			}
			add_character_modifier = {
				modifier = minor_elemental_attunement
				duration = -1
			}
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_minor_elemental_attunement
			}
		}
		# Milestone: Major Elemental Attunement (50+)
		if = {
			limit = {
				var:elemental_power >= 50
				NOT = { has_character_modifier = major_elemental_attunement }
			}
			remove_character_modifier = minor_elemental_attunement
			add_character_modifier = {
				modifier = major_elemental_attunement
				duration = -1
			}
			add_prowess = 2
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_major_elemental_attunement
			}
		}
		# Milestone: Elemental Transcendence (100+)
		if = {
			limit = {
				var:elemental_power >= 100
				NOT = { has_character_modifier = elemental_transcendence }
			}
			remove_character_modifier = major_elemental_attunement
			add_character_modifier = {
				modifier = elemental_transcendence
				duration = -1
			}
			add_prowess = 5
			add_martial = 3
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_elemental_transcendence
				trigger_event = { id = murim_celestial.2001 }
			}
		}
	}
}

# Initialize Yin-Yang balance for a character
# Balance ranges from -100 (pure Yin) to +100 (pure Yang)
# 0 = perfect balance (ideal state)
initialize_yin_yang_balance_effect = {
	if = {
		limit = {
			NOT = { exists = var:yin_yang_balance }
		}
		# Start near neutral with slight personality bias
		set_variable = { name = yin_yang_balance value = 0 }

		# Personality shifts initial balance
		if = {
			limit = { has_trait = wrathful }
			change_variable = { name = yin_yang_balance add = 15 }
		}
		if = {
			limit = { has_trait = brave }
			change_variable = { name = yin_yang_balance add = 10 }
		}
		if = {
			limit = { has_trait = calm }
			change_variable = { name = yin_yang_balance add = -15 }
		}
		if = {
			limit = { has_trait = patient }
			change_variable = { name = yin_yang_balance add = -10 }
		}
		if = {
			limit = { has_trait = zealous }
			change_variable = { name = yin_yang_balance add = 20 }
		}
		if = {
			limit = { has_trait = cynical }
			change_variable = { name = yin_yang_balance add = -20 }
		}
	}
}


# Shift balance toward Yang (active, aggressive, hot)
shift_toward_yang_effect = {
	if = {
		limit = { exists = var:yin_yang_balance }
		change_variable = { name = yin_yang_balance add = 10 }

		# Cap at +100
		if = {
			limit = { var:yin_yang_balance > 100 }
			set_variable = { name = yin_yang_balance value = 100 }
		}

		# Extreme Yang imbalance causes qi deviation
		if = {
			limit = { var:yin_yang_balance >= 80 }
			random = {
				chance = 30
				add_character_modifier = {
					modifier = yang_qi_deviation
					years = 1
				}
				add_stress = 15
				if = {
					limit = { is_ai = no }
					custom_tooltip = tt_yang_deviation_warning
				}
			}
		}
	}
	else = {
		initialize_yin_yang_balance_effect = yes
	}
}


# Shift balance toward Yin (passive, defensive, cold)
shift_toward_yin_effect = {
	if = {
		limit = { exists = var:yin_yang_balance }
		change_variable = { name = yin_yang_balance add = -10 }

		# Cap at -100
		if = {
			limit = { var:yin_yang_balance < -100 }
			set_variable = { name = yin_yang_balance value = -100 }
		}

		# Extreme Yin imbalance causes qi stagnation
		if = {
			limit = { var:yin_yang_balance <= -80 }
			random = {
				chance = 30
				add_character_modifier = {
					modifier = yin_qi_stagnation
					years = 1
				}
				add_stress = 15
				if = {
					limit = { is_ai = no }
					custom_tooltip = tt_yin_stagnation_warning
				}
			}
		}
	}
	else = {
		initialize_yin_yang_balance_effect = yes
	}
}


# Attempt to restore Yin-Yang balance through meditation
restore_yin_yang_balance_effect = {
	if = {
		limit = { exists = var:yin_yang_balance }

		# Move 20% toward center
		if = {
			limit = { var:yin_yang_balance > 0 }
			change_variable = {
				name = yin_yang_balance
				add = {
					value = var:yin_yang_balance
					multiply = -0.2
				}
			}
		}
		else_if = {
			limit = { var:yin_yang_balance < 0 }
			change_variable = {
				name = yin_yang_balance
				add = {
					value = var:yin_yang_balance
					multiply = -0.2
				}
			}
		}

		# Perfect balance reward
		if = {
			limit = {
				var:yin_yang_balance >= -5
				var:yin_yang_balance <= 5
			}
			add_character_modifier = {
				modifier = perfect_yin_yang_harmony
				years = 1
			}
			add_stress = -10
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_perfect_harmony_achieved
			}
		}

		# Remove deviation modifiers if balance restored
		if = {
			limit = {
				var:yin_yang_balance > -80
				var:yin_yang_balance < 80
				has_character_modifier = yang_qi_deviation
			}
			remove_character_modifier = yang_qi_deviation
		}
		if = {
			limit = {
				var:yin_yang_balance > -80
				var:yin_yang_balance < 80
				has_character_modifier = yin_qi_stagnation
			}
			remove_character_modifier = yin_qi_stagnation
		}
	}
}

# Initialize meridian channels -- 12 primary, 8 extraordinary
# Each meridian can be: blocked (0), partially open (1), open (2), perfected (3)
initialize_meridian_system_effect = {
	if = {
		limit = {
			NOT = { exists = var:meridians_opened }
		}
		set_variable = { name = meridians_opened value = 0 }
		set_variable = { name = meridian_total value = 20 }
		set_variable = { name = extraordinary_meridians value = 0 }
	}
}


# Open a meridian channel -- called during cultivation breakthroughs
open_meridian_channel_effect = {
	# Safety
	if = {
		limit = {
			NOT = { exists = var:meridians_opened }
		}
		initialize_meridian_system_effect = yes
	}

	if = {
		limit = {
			exists = var:meridians_opened
			exists = var:meridian_total
			var:meridians_opened < var:meridian_total
		}
		change_variable = { name = meridians_opened add = 1 }

		# Prowess bonus every 4 meridians
		if = {
			limit = {
				trigger_if = {
					limit = { exists = var:meridians_opened }
					var:meridians_opened >= 4
				}
			}
			random = {
				chance = 50
				add_prowess = 1
			}
		}

		# Milestone: 8 Primary Meridians Open
		if = {
			limit = {
				var:meridians_opened >= 8
				NOT = { has_character_modifier = eight_meridians_open }
			}
			add_character_modifier = {
				modifier = eight_meridians_open
				duration = -1
			}
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_eight_meridians_milestone
			}
		}

		# Milestone: All 12 Primary Meridians Open
		if = {
			limit = {
				var:meridians_opened >= 12
				NOT = { has_character_modifier = twelve_meridians_open }
			}
			remove_character_modifier = eight_meridians_open
			add_character_modifier = {
				modifier = twelve_meridians_open
				duration = -1
			}
			add_prowess = 3
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_twelve_meridians_milestone
				trigger_event = { id = murim_celestial.3001 }
			}
		}

		# Milestone: Extraordinary Meridians (13-20)
		if = {
			limit = {
				var:meridians_opened >= 13
			}
			change_variable = { name = extraordinary_meridians add = 1 }
		}

		# Milestone: All 20 Meridians Perfected
		if = {
			limit = {
				var:meridians_opened >= 20
				NOT = { has_character_modifier = perfect_meridian_network }
			}
			remove_character_modifier = twelve_meridians_open
			add_character_modifier = {
				modifier = perfect_meridian_network
				duration = -1
			}
			add_prowess = 8
			add_martial = 5
			add_character_modifier = { modifier = murim_light_healing years = 1 }
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_perfect_meridians
				trigger_event = { id = murim_celestial.3002 }
			}
		}
	}
}


# Damage meridian -- from qi deviation, combat injury, or poison
damage_meridian_channel_effect = {
	if = {
		limit = {
			exists = var:meridians_opened
			var:meridians_opened > 0
		}
		change_variable = { name = meridians_opened add = -1 }

		add_character_modifier = {
			modifier = damaged_meridian_channel
			years = 2
		}
		add_stress = 20

		# Catastrophic: if too many damaged, cultivation regresses
		if = {
			limit = { var:meridians_opened <= 4 }
			add_character_modifier = {
				modifier = shattered_meridian_network
				years = 5
			}
			add_prowess = -5
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_meridian_network_shattered
			}
		}
	}
}

# Grant Orthodox Mandate to Empire-tier ruler
# Requires celestial alignment check first
grant_orthodox_mandate_effect = {
	# Safety checks
	if = {
		limit = {
			is_ruler = yes
			highest_held_title_tier >= tier_empire
		}
		add_character_modifier = {
			modifier = mandate_of_heaven_orthodox
			duration = -1
		}
		add_prestige = 500

		# All vassals acknowledge the mandate
		every_vassal = {
			limit = { is_alive = yes }
			add_opinion = {
				target = root
				modifier = acknowledges_celestial_mandate
			}
		}

		if = {
			limit = { is_ai = no }
			custom_tooltip = tt_mandate_of_heaven_granted
		}
	}
}

			limit = { is_ai = no }

		}
	}
}

# SECTION 6: HEAVENLY TRIBULATION
# ============================================================

# Begin heavenly tribulation -- ultimate cultivation test
# Called when a cultivator attempts to break through to the highest tier
begin_heavenly_tribulation_effect = {
	if = {
		limit = {
			exists = var:murim_cultivation_rank
			var:murim_cultivation_rank >= 7
		}
		# Initialize tribulation variables
		set_variable = { name = tribulation_stage value = 1 }
		set_variable = { name = tribulation_power value = 0 }

		# Calculate tribulation difficulty based on cultivation level
		set_variable = {
			name = tribulation_difficulty
			value = {
				add = var:murim_cultivation_rank
				multiply = 10
				add = prowess
			}
		}

		add_character_modifier = {
			modifier = undergoing_heavenly_tribulation
			years = 1
		}

		if = {
			limit = { is_ai = no }
			custom_tooltip = tt_tribulation_begins
			trigger_event = { id = murim_celestial.5001 }
		}
		else = {
			# AI auto-resolves tribulation
			resolve_ai_tribulation_effect = yes
		}
	}
}


# Advance tribulation stage (player event chain)
advance_tribulation_stage_effect = {
	if = {
		limit = {
			exists = var:tribulation_stage
			var:tribulation_stage < 9
		}
		change_variable = { name = tribulation_stage add = 1 }

		# Each stage tests differently
		# Stages 1-3: Physical (prowess check)
		# Stages 4-6: Mental (stress/learning check)
		# Stages 7-9: Spiritual (cultivation rank + yin-yang balance)

		# Accumulate tribulation power
		change_variable = {
			name = tribulation_power
			add = {
				add = prowess
				add = martial
				divide = 4
			}
		}

		if = {
			limit = { is_ai = no }
			custom_tooltip = tt_tribulation_stage_advanced
		}
	}
}


# Survive heavenly tribulation -- massive rewards
survive_heavenly_tribulation_effect = {
	if = {
		limit = {
			exists = var:tribulation_stage
			has_character_modifier = undergoing_heavenly_tribulation
		}
		remove_character_modifier = undergoing_heavenly_tribulation
		add_character_modifier = {
			modifier = survived_heavenly_tribulation
			duration = -1
		}

		# Massive stat boost
		add_prowess = 10
		add_martial = 5
		add_character_modifier = { modifier = murim_moderate_healing years = 1 }
		add_prestige = 1000
		add_piety = 500

		# Open remaining meridians
		if = {
			limit = {
				exists = var:meridians_opened
				var:meridians_opened < 20
			}
			set_variable = { name = meridians_opened value = 20 }
		}

		# Clean up variables
		remove_variable = tribulation_stage
		remove_variable = tribulation_power
		remove_variable = tribulation_difficulty

		if = {
			limit = { is_ai = no }
			custom_tooltip = tt_tribulation_survived
			trigger_event = { id = murim_celestial.5009 }
		}
	}
}


# Fail heavenly tribulation -- devastating consequences
fail_heavenly_tribulation_effect = {
	if = {
		limit = {
			has_character_modifier = undergoing_heavenly_tribulation
		}
		remove_character_modifier = undergoing_heavenly_tribulation

		# Catastrophic backlash
		add_character_modifier = {
			modifier = tribulation_backlash
			years = 10
		}
		add_prowess = -10
		add_martial = -5
		add_stress = 80

		# Shatter meridians
		if = {
			limit = { exists = var:meridians_opened }
			change_variable = {
				name = meridians_opened
				add = -8
			}
			if = {
				limit = { var:meridians_opened < 0 }
				set_variable = { name = meridians_opened value = 0 }
			}
		}

		# Risk of death or permanent injury
		random_list = {
			30 = {
				# Survive but crippled
				add_trait = incapable
				if = {
					limit = { is_ai = no }
					custom_tooltip = tt_tribulation_crippled
				}
			}
			40 = {
				# Survive with severe wounds
				add_trait = wounded_3
				if = {
					limit = { is_ai = no }
					custom_tooltip = tt_tribulation_wounded
				}
			}
			30 = {
				# Die from heavenly wrath
				death = { death_reason = death_tribulation }
			}
		}

		# Clean up
		remove_variable = tribulation_stage
		remove_variable = tribulation_power
		remove_variable = tribulation_difficulty

		if = {
			limit = { is_ai = no }
			trigger_event = { id = murim_celestial.5010 }
		}
	}
}


# AI auto-resolve tribulation (simplified for performance)
resolve_ai_tribulation_effect = {
	if = {
		limit = {
			is_ai = yes
			exists = var:murim_cultivation_rank
		}
		random_list = {
			# Success chance based on cultivation + prowess
			40 = {
				modifier = {
					add = 20
					prowess >= 35
				}
				modifier = {
					add = 15
					exists = var:murim_cultivation_rank
					var:murim_cultivation_rank >= 8
				}
				modifier = {
					add = 10
					exists = var:yin_yang_balance
					var:yin_yang_balance >= -20
					var:yin_yang_balance <= 20
				}
				survive_heavenly_tribulation_effect = yes
			}
			60 = {
				fail_heavenly_tribulation_effect = yes
			}
		}
	}
}

# SECTION 7: CELESTIAL QI ABSORPTION
# ============================================================

# Absorb ambient celestial qi during cultivation
# Bonus is higher when aligned with current celestial beast
absorb_celestial_qi_effect = {
	# Safety
	check_celestial_alignment_effect = yes
	if = {
		limit = {
			NOT = { exists = var:elemental_affinity }
		}
		initialize_elemental_affinity_effect = yes
	}

	if = {
		limit = {
			exists = var:murim_cultivation_rank
		}
		# Base qi absorption
		if = {
			limit = {
				NOT = { exists = var:celestial_qi_absorbed }
			}
			set_variable = { name = celestial_qi_absorbed value = 0 }
		}

		change_variable = { name = celestial_qi_absorbed add = 10 }

		# Bonus if element matches celestial season
		if = {
			limit = {
				exists = var:elemental_harmony
				var:elemental_harmony >= 1
			}
			change_variable = { name = celestial_qi_absorbed add = 5 }
		}

		# Penalty if element conflicts
		if = {
			limit = {
				exists = var:elemental_conflict
				var:elemental_conflict >= 1
			}
			change_variable = { name = celestial_qi_absorbed add = -3 }
		}

		# Milestone: 100 qi absorbed grants temporary cultivation boost
		if = {
			limit = {
				var:celestial_qi_absorbed >= 100
			}
			set_variable = { name = celestial_qi_absorbed value = 0 }
			add_character_modifier = {
				modifier = celestial_qi_saturation
				years = 1
			}
			if = {
				limit = { is_ai = no }
				custom_tooltip = tt_celestial_qi_saturation
			}
		}
	}
}

# Master yearly check -- runs all celestial subsystems
yearly_celestial_maintenance_effect = {
	# Step 1: Update celestial alignment
	check_celestial_alignment_effect = yes

	# Step 2: Initialize systems if needed
	if = {
		limit = {
			NOT = { exists = var:elemental_affinity }
		}
		initialize_elemental_affinity_effect = yes
	}
	if = {
		limit = {
			NOT = { exists = var:yin_yang_balance }
		}
		initialize_yin_yang_balance_effect = yes
	}
	if = {
		limit = {
			NOT = { exists = var:meridians_opened }
		}
		initialize_meridian_system_effect = yes
	}

	# Step 3: Run elemental cycle checks
	check_elemental_generation_cycle_effect = yes
	check_elemental_destruction_cycle_effect = yes

	# Step 4: Absorb celestial qi if cultivating
	if = {
		limit = {
			exists = var:murim_cultivation_rank
			var:murim_cultivation_rank >= 1
		}
		absorb_celestial_qi_effect = yes
	}

	# Step 5: Notify player of celestial phase changes
	if = {
		limit = { is_ai = no }
		trigger_event = { id = murim_celestial.0001 }
	}

	# Step 6: Empire-tier mandate check
	if = {
		limit = {
			is_ruler = yes
			highest_held_title_tier >= tier_empire
			has_character_modifier = mandate_of_heaven_orthodox
		}
		add_character_modifier = {
			modifier = renewed_mandate
			years = 1
		}
	}
}
