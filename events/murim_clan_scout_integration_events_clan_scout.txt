# Scout discovers a hidden prodigy
murim_clan_scout.0001 = {
	type = character_event
	title = murim_clan_scout.0001.t
	desc = murim_clan_scout.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		has_variable = murim_martial_artist
		has_character_flag = murim_active_scout
	}

	immediate = {
		random_courtier = {
			limit = {
				age >= 10
				age <= 25
				NOT = { has_variable = murim_martial_artist }
			}
			save_scope_as = scout_candidate
		}
	}

	# Recruit the prodigy immediately
	option = {
		name = murim_clan_scout.0001.a
		trigger = { exists = scope:scout_candidate }
		hidden_effect = {
			scope:scout_candidate = {
				set_variable = { name = murim_martial_artist value = 1 }
				set_variable = { name = murim_cultivation_rank value = 1 }
			}
		}
		add_prestige = 100
		ai_chance = { base = 60 }
	}

	# Test them further before committing
	option = {
		name = murim_clan_scout.0001.b
		trigger = { exists = scope:scout_candidate }
		hidden_effect = {
			trigger_event = { id = murim_clan_scout.0010 days = 30 }
		}
		ai_chance = { base = 30 }
	}

	# Pass — not worthy of our clan
	option = {
		name = murim_clan_scout.0001.c
		add_stress = -10
		ai_chance = { base = 10 }
	}
}


# Scout trials — testing the candidate

	option = {
		name = murim_clan_scout.0010.c
		add_stress = 10
		ai_chance = { base = 15 }
	}
}
