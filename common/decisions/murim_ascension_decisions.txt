# === ATTEMPT HALF-IMMORTAL ASCENSION ===
murim_attempt_half_immortal_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_destiny_quest.dds"
	title = murim_attempt_half_immortal_decision
	desc = murim_attempt_half_immortal_decision_desc
	selection_tooltip = murim_attempt_half_immortal_decision_tooltip

	is_shown = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 6
		NOT = { murim_is_immortal_trigger = yes }
		NOT = { has_trait = murim_ascension_failed }
	}

	is_valid = {
		murim_can_attempt_ascension_trigger = yes
		NOT = { has_character_flag = murim_ascension_cooldown }
	}

	effect = {
		trigger_event = { id = murim_ascension.0001 days = 30 }
		custom_tooltip = murim_attempt_half_immortal_tt
	}

	ai_check_interval = 120
	ai_will_do = {
		base = 15
		modifier = { add = 20 has_trait = ambitious }
		modifier = { add = 15 prowess >= 28 }
		modifier = { add = -20 has_trait = craven }
		modifier = { add = -10 stress >= 50 }
	}
}

# === ATTEMPT TRUE IMMORTAL ASCENSION ===

		base = 10

		modifier = { add = 20 prowess >= 35 }
		modifier = { add = -30 has_trait = craven }
	}
}

# === FACE HEAVEN'S TRIBULATION ===
murim_face_heavens_tribulation_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_destiny_quest.dds"
	title = murim_face_heavens_tribulation_decision
	desc = murim_face_heavens_tribulation_decision_desc
	selection_tooltip = murim_face_heavens_tribulation_decision_tooltip

	is_shown = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 5
		has_character_modifier = murim_heavens_attention
	}

	is_valid = {
		NOT = { has_character_flag = murim_tribulation_cooldown }
	}

	effect = {
		set_character_flag = { flag = murim_tribulation_cooldown years = 10 }
		trigger_event = { id = murim_ascension.0020 days = 3 }
	}

	ai_check_interval = 30
	ai_will_do = {
		base = 30
		modifier = { add = 30 has_trait = brave }
		modifier = { add = 20 prowess >= 25 }
	}
}
