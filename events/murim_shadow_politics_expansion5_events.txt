murim_shadow_politics_expansion5.0001 = {
	type = character_event
	title = murim_shadow_politics_expansion5.0001.t
	desc = murim_shadow_politics_expansion5.0001.desc
	theme = martial_arts
	
	# The Emperor's New Advisor — the Heavenly Demon has installed his agent as the
	# Imperial Grand Advisor. Imperial decrees now subtly favor unorthodox sects.
	# Taxes on Orthodox territories increase. Military posts go to Demon allies.
	
	trigger = {
		is_ruler = yes
		any_held_title = { tier >= county }
		NOT = { has_character_flag = noticed_imperial_corruption }
		OR = {
			has_trait = shaolin_disciple
			has_trait = mount_hua_disciple
			has_trait = wudang_disciple
			has_trait = tang_clan_member
			has_trait = beggar_sect_member
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = paranoid }
		modifier = { add = 3 intrigue >= 12 }
	}
	
	immediate = {
		set_character_flag = noticed_imperial_corruption
	}
	
	option = {
		name = murim_shadow_politics_expansion5.0001.a
		# Investigate the new Grand Advisor
		add_intrigue = 4
		add_character_modifier = {
			modifier = investigating_imperial_court
			years = 3
		}
		hidden_effect = {
			trigger_event = {
				id = murim_shadow_politics_expansion5.0002
				days = { 30 90 }
			}
		}
		ai_chance = { base = 50 }
	}
	
	option = {
		name = murim_shadow_politics_expansion5.0001.b
		# Petition the Emperor directly — bypass the advisor
		add_diplomacy = 3
