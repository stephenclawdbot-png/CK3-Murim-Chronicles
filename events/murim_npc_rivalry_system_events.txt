namespace = murim_npc_rivalry

# Rivalry sparks — two NPCs clash
murim_npc_rivalry.0001 = {
	type = character_event
	title = murim_npc_rivalry.0001.t
	desc = murim_npc_rivalry.0001.desc
	theme = martial

	trigger = {
		has_variable = murim_martial_artist
		murim_can_generate_rivalry_trigger = yes
	}

	immediate = {
		random_ruler = {
			limit = {
				has_variable = murim_martial_artist
				NOT = { this = root }
				var:murim_cultivation_rank >= 2
			}
			save_scope_as = new_rival
		}
	}

	# Accept the rivalry — this is personal
	option = {
		name = murim_npc_rivalry.0001.a
		trigger = { exists = scope:new_rival }
		hidden_effect = { murim_generate_rivalry_effect = yes }
		add_prowess = 1
		ai_chance = { base = 50 modifier = { add = 25 has_trait = wrathful } }
	}

	# Let it go — not worth the energy
	option = {
		name = murim_npc_rivalry.0001.b
		add_stress = -10
		add_piety = 25
		ai_chance = { base = 50 modifier = { add = 20 has_trait = murim_forgiving } }
	}
}
