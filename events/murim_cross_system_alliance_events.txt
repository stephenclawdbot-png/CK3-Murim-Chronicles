# Cross-System Alliance & Title Events
# Fired from on_alliance_added, on_alliance_broken, on_title_gained hooks

# === NEW ALLIANCE: INTELLIGENCE SHARING ===
murim_cross_system.0030 = {
	type = character_event
	title = murim_cross_system.0030.t
	desc = murim_cross_system.0030.desc
	theme = diplomacy
	
	trigger = {
		has_character_flag = murim_spy_network_active
		any_ally = {
			NOT = { has_character_flag = murim_spy_network_active }
		}
	}
	
	# Share intelligence with new ally
	option = {
		name = murim_cross_system.0030.a
		random_ally = {
			limit = {
				NOT = { has_character_flag = murim_spy_network_active }
			}
			add_character_modifier = {
				modifier = murim_shared_intelligence
				years = 3
			}
			add_opinion = {
				target = root
				modifier = murim_intelligence_gratitude
				opinion = 20
			}
		}
		add_gold = -30
	}
	# Keep intelligence separate
	option = {
		name = murim_cross_system.0030.b
		add_stress = 5
	}
}

# === ALLIANCE BROKEN: SPY NETWORK BETRAYAL ===
murim_cross_system.0040 = {
	type = character_event
	title = murim_cross_system.0040.t
	desc = murim_cross_system.0040.desc
	theme = intrigue
	
	trigger = {
		has_character_flag = murim_spy_network_active
	}
	
	immediate = {
		# Former ally may have compromised agents
		random_list = {
			60 = {
				set_variable = {
					name = spy_compromise_level
					value = 1
				}
			}
			30 = {
				set_variable = {
					name = spy_compromise_level
					value = 2
				}
			}
			10 = {
				set_variable = {
					name = spy_compromise_level
					value = 3
				}
			}
		}
	}
	
	# Purge compromised agents
	option = {
		name = murim_cross_system.0040.a
		add_gold = -75
		add_intrigue = 1
		add_character_modifier = {
			modifier = murim_network_purged
			years = 2
		}
	}
	# Ignore the risk
	option = {
		name = murim_cross_system.0040.b
		random_list = {
			50 = {
				# Nothing happens
				add_prestige = 50
			}
			50 = {
				# Spy network exposed!
				remove_character_flag = murim_spy_network_active
				add_prestige = -300
				add_character_modifier = {
					modifier = murim_intelligence_compromised
					years = 3
				}
			}
		}
	}
}

# === ALLIANCE BROKEN: FORBIDDEN TECHNIQUE LEAK ===

	desc = murim_cross_system.0041.desc
	theme = dread
	
	trigger = {
		has_character_flag = murim_forbidden_technique_user
	}

		name = murim_cross_system.0041.a
		# Deny everything
		add_diplomacy = 1
		random_list = {
			40 = { add_prestige = 50 }

				add_character_flag = murim_techniques_exposed
				add_character_modifier = {
					modifier = murim_tribunal_accused

				}
			}
		}
	}
	# Strike first - silence the former ally

		name = murim_cross_system.0041.b

		add_tyranny = 20
		add_dread = 25
		add_character_modifier = {
			modifier = murim_silenced_witness
			years = 5
		}
	}
	# Flee before exposure
	option = {
		name = murim_cross_system.0041.c
		add_character_flag = murim_in_hiding
		add_prestige = -200
		add_intrigue = 2
	}
}
