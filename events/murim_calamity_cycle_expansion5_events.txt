murim_calamity_cycle_expansion5.0001 = {
	type = character_event
	title = murim_calamity_cycle_expansion5.0001.t
	desc = murim_calamity_cycle_expansion5.0001.desc
	theme = martial_arts
	
	# The Sky Burns Red — the final phase of the Great Calamity begins. All nine clans
	# are at war simultaneously. The Heavenly Demon has declared himself Emperor of Murim.
	# Orthodox Alliance headquarters is burning. Unorthodox factions tear each other apart.
	
	trigger = {
		has_global_variable = great_calamity_phase_5
		is_ruler = yes
		NOT = { has_character_flag = witnessed_sky_burns }
	}
	
	weight_multiplier = {
		base = 10 # High priority — everyone should see this
	}
	
	immediate = {
		set_character_flag = witnessed_sky_burns
		add_character_modifier = {
			modifier = calamity_final_phase
			years = 5
		}
	}
	
	option = {
		name = murim_calamity_cycle_expansion5.0001.a
		# Rally what remains — make a last stand
		add_martial = 5
		add_prowess = 5
		add_character_modifier = {
			modifier = last_stand_commander
			years = -1
		}
		hidden_effect = {
			trigger_event = {
				id = murim_calamity_cycle_expansion5.0002
				days = { 30 60 }
			}
		}
		ai_chance = { base = 40 modifier = { add = 30 has_trait = brave } }
	}
	
	option = {
		name = murim_calamity_cycle_expansion5.0001.b
		# Flee to the mountains — preserve what you can
		add_stewardship = 3
		add_character_modifier = {
			modifier = mountain_refuge
			years = 10
		}
		hidden_effect = {
			set_character_flag = chose_refuge
			trigger_event = {
				id = murim_calamity_cycle_expansion5.0005
				days = { 90 180 }
			}
		}
		ai_chance = { base = 40 modifier = { add = 20 has_trait = craven } }
	}
	
	option = {
		name = murim_calamity_cycle_expansion5.0001.c
		# Seek out the Heavenly Demon directly — end this at the source
		add_prowess = 8
		hidden_effect = {
			trigger_event = {
				id = murim_calamity_cycle_expansion5.0003
				days = { 15 30 }
			}
		}
		ai_chance = { base = 20 modifier = { add = 30 has_trait = ambitious } }
	}
}
