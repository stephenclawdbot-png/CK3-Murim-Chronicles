murim_faction_war_expansion3.0001 = {
	type = character_event
	title = murim_faction_war_exp3.0001.t
	desc = murim_faction_war_exp3.0001.desc
	theme = war
	left_portrait = { character = root animation = rage }

	trigger = {
		has_trait = murim_martial_artist
		is_ruler = yes
		has_global_variable = murim_faction_war_active
		NOT = { has_character_flag = total_war_declared }
	}

	weight_multiplier = {
		base = 20
		modifier = { add = 30 has_trait = wrathful }
	}

	immediate = {
		set_character_flag = total_war_declared
		every_living_character = {
			limit = { has_trait = murim_martial_artist is_ruler = yes NOT = { this = root } }
			add_to_temporary_list = war_participants
		}
	}

	option = {
		name = murim_faction_war_exp3.0001.a
		add_character_modifier = { modifier = murim_total_war_declared years = 10 }
		add_prestige = 500
		add_martial = 3
		every_in_list = {
			list = war_participants
			trigger_event = { id = murim_faction_war_expansion3.0002 days = { 5 15 } }
		}
	}
	option = {
		name = murim_faction_war_exp3.0001.b
		add_character_modifier = { modifier = murim_war_strategist years = 10 }
		add_martial = 5
		trigger_event = { id = murim_faction_war_expansion3.0003 days = { 30 60 } }
	}
	option = {
		name = murim_faction_war_exp3.0001.c
		add_character_modifier = { modifier = murim_neutral_faction years = 15 }
		add_diplomacy = 3
		trigger_event = { id = murim_faction_war_expansion3.0004 days = { 15 30 } }
	}
}

	type = character_event

	left_portrait = { character = root animation = personality_callous }

	trigger = { has_character_modifier = murim_ambush_specialist has_global_variable = murim_faction_war_active }

	immediate = {
		random_living_character = {
			limit = { has_trait = murim_martial_artist has_character_modifier = murim_joined_total_war NOT = { this = root } }
			save_scope_as = ambush_target
		}
	}

	option = {
		name = murim_faction_war_exp3.0005.a
		trigger = { exists = scope:ambush_target }
		duel = {
			skill = intrigue
			target = scope:ambush_target
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -49 }
				desc = murim_faction_war_exp3.0005.a.win
				root = { add_prestige = 1000 add_character_modifier = { modifier = murim_ambush_master years = 20 } }
				scope:ambush_target = { add_trait = wounded_3 add_prestige = -500 }
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }
				desc = murim_faction_war_exp3.0005.a.lose
				root = { add_trait = wounded_2 add_prestige = -300 }
			}
		}
	}
	option = {
		name = murim_faction_war_exp3.0005.b
		add_character_modifier = { modifier = murim_war_crime_poison years = 15 }
		add_tyranny = 30
		every_living_character = {
