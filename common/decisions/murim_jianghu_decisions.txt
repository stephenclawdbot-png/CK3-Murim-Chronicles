# =============================================================
# 1. ENTER THE JIANGHU
# Formally declare yourself a member of the martial world.
# One-time, irreversible. Sets murim_initiated and initializes
# all cultivation / standing / combat variables.
# Major decision -- appears in the Important Decisions tab.
# =============================================================

enter_the_jianghu = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	major = yes
	ai_check_interval = 60

	desc = enter_the_jianghu_desc
	selection_tooltip = enter_the_jianghu_tooltip

	is_shown = {
		is_landed = yes
		is_adult = yes
		NOT = { has_character_flag = murim_initiated }
		NOT = { has_character_flag = murim_retired_from_jianghu }
	}

	is_valid_showing_failures_only = {
		is_available = yes
		is_imprisoned = no
		prestige >= 150
		OR = {
			prowess >= 8
			martial >= 8
			learning >= 12
		}
		custom_description = {
			text = "decision_enter_jianghu_age_req"
			age >= 16
		}
	}

	cost = {
		prestige = {
			value = 150
			if = {
				limit = { highest_held_title_tier >= tier_kingdom }
				add = 100
			}
		}
	}

	effect = {
		# Initialize all murim variables via scripted effect
		murim_init_character_variables = yes

		# Notification event: welcome to the martial world
		trigger_event = {
			id = murim_jiang.0001
			days = 1
		}

		show_as_tooltip = {
			add_character_flag = murim_initiated
			custom_tooltip = enter_the_jianghu_effect_tt
		}

		# Starting standing boost based on existing fame
		if = {
			limit = { prestige_level >= 3 }
			change_variable = {
				name = murim_jianghu_standing
				add = 5
			}
		}
		if = {
			limit = { prowess >= 15 }
			change_variable = {
				name = murim_jianghu_standing
				add = 3
			}
		}

		# Add starting cultivation trait tooltip
		show_as_tooltip = {
			add_trait = murim_outer_qi
		}

		# Stress from such a momentous life change
		stress_impact = {
			craven = minor_stress_impact_gain
			content = minor_stress_impact_gain
			lazy = minor_stress_impact_gain
		}

		# Courtiers and vassals react
		every_vassal = {
			limit = {
				has_character_flag = murim_initiated
			}
			add_opinion = {
				target = root
				modifier = impressed_opinion
				opinion = 10
			}
		}
	}

	ai_potential = {
		prowess >= 10
		OR = {
			has_trait = brave
			has_trait = ambitious
			martial >= 12
		}
	}

	ai_will_do = {
		base = 30
		modifier = {
			add = 20
			has_trait = ambitious
		}
		modifier = {
			add = 15
			has_trait = brave
		}
		modifier = {
			add = -20
			has_trait = craven
		}
		modifier = {
			add = -15
			has_trait = content
		}
		modifier = {
			add = 10
			prowess >= 15
		}
		modifier = {
			add = 10
			any_neighbor_independent_ruler = {
				has_character_flag = murim_initiated
			}
		}
	}
}

# 6. ESTABLISH SPY NETWORK
# Create an intelligence network across the jianghu.
# Costs gold. Provides intrigue bonuses and early warning of
# hostile schemes. Takes 1 year to fully establish.
# =============================================================

establish_spy_network = {
	picture = "gfx/interface/illustrations/decisions/decision_misc.dds"
	ai_check_interval = 120

	desc = establish_spy_network_desc
	selection_tooltip = establish_spy_network_tooltip

	is_shown = {
		has_character_flag = murim_initiated
		is_adult = yes
		NOT = { has_character_flag = murim_spy_network_active }
		NOT = { has_character_flag = murim_spy_network_building }
	}

		is_available = yes
		is_imprisoned = no
		NOT = { has_character_flag = murim_retired_from_jianghu }

		intrigue >= 8
		custom_description = {
			text = "decision_spy_net_standing_req"
			var:murim_jianghu_standing >= 15
		}
	}

	cost = {
		gold = {
			value = 150
			if = {
				limit = { has_character_flag = murim_sect_master }
				# Sect masters get a discount -- use sect resources
				multiply = 0.75
			}
		}
	}

	effect = {
		# Mark network as under construction
		add_character_flag = murim_spy_network_building

		# Apply building modifier (1 year = 365 days)
		add_character_modifier = {
			modifier = murim_spy_network_establishing
			days = 365
		}

		# After 1 year, the network event fires to finalize
		trigger_event = {
			id = murim_jiang.0006
			days = 365
		}

		# Immediate minor intrigue bonus while building
		add_character_modifier = {
			modifier = murim_spy_network_partial
			days = 365
		}

		custom_tooltip = establish_spy_network_effect_tt

		# Minor standing loss -- spying is not honorable
		if = {
			limit = { has_character_flag = murim_orthodox_cultivator }
			change_variable = {
				name = murim_jianghu_standing
				add = -2
			}
			change_variable = {
				name = murim_alignment
				add = -3
			}
		}

		# Demonic cultivators find this natural
		if = {
			limit = { has_character_flag = murim_demonic_cultivator }
			change_variable = {
				name = murim_jianghu_standing
				add = 1
			}
		}

		stress_impact = {
			honest = medium_stress_impact_gain
			trusting = minor_stress_impact_gain
			paranoid = minor_stress_impact_loss
		}
	}

	ai_potential = {
		intrigue >= 10
		gold >= 200
		OR = {
			has_trait = deceitful
			has_trait = paranoid
			has_trait = ambitious
			has_character_flag = murim_sect_master
		}
	}

	ai_will_do = {
		base = 20
		modifier = {
			add = 25
			has_trait = paranoid
		}
		modifier = {
			add = 20
			has_trait = deceitful
		}
		modifier = {
			add = 15
			has_character_flag = murim_sect_master
		}
		modifier = {
			add = -30
			has_trait = honest
		}
		modifier = {
			add = -20
			has_trait = trusting
		}

# =============================================================
# 9. RETIRE FROM JIANGHU
# Formally withdraw from the martial world. Removes most murim
# mechanics but grants lasting peace. Can be reversed by
# re-entering the jianghu, but at a massive standing penalty.
# For characters who want out of the endless cycle of violence.
# =============================================================

retire_from_jianghu = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	major = yes
	ai_check_interval = 180

	desc = retire_from_jianghu_desc
	selection_tooltip = retire_from_jianghu_tooltip

	is_shown = {
		has_character_flag = murim_initiated
		is_adult = yes
		NOT = { has_character_flag = murim_retired_from_jianghu }
	}

	is_valid_showing_failures_only = {
		is_available = yes
		is_imprisoned = no
		is_at_war = no
		custom_description = {
			text = "decision_retire_not_sect_master"
			NOT = { has_character_flag = murim_sect_master }
		}
		custom_description = {
			text = "decision_retire_not_alliance_lord"
			NOT = { has_character_flag = murim_wulin_alliance_lord }
		}
	}

	cost = {
		prestige = 100
	}

	effect = {
		# Mark as retired
		add_character_flag = murim_retired_from_jianghu

		# Remove active jianghu flags
		remove_character_flag = murim_wulin_alliance_member
		remove_character_flag = murim_wandering_hero
		remove_character_flag = murim_spy_network_active
		remove_character_flag = murim_spy_network_building

		# Halve standing -- the world forgets
		change_variable = {
			name = murim_jianghu_standing
			multiply = 0.5
		}

		# Remove combat modifiers
		if = {
			limit = { has_character_modifier = murim_on_jianghu_quest }
			remove_character_modifier = murim_on_jianghu_quest
		}
		if = {
			limit = { has_character_modifier = murim_spy_network_partial }
			remove_character_modifier = murim_spy_network_partial
		}

		# Grant peace of mind
		add_character_modifier = {
			modifier = murim_retired_cultivator_peace
		}

		# Stress relief -- you are free from the martial world
		add_stress = -30

		# Fire retirement event
		trigger_event = {
			id = murim_jiang.0009
			days = 1
		}

		custom_tooltip = retire_from_jianghu_effect_tt

		# Reactions
		every_living_character = {
			limit = {
				has_character_flag = murim_initiated
				has_relation_rival = root
			}
			# Rivals are disappointed
			add_opinion = {
				target = root
				modifier = murim_retired_rival_opinion
				opinion = -10
			}
		}
		every_living_character = {
			limit = {
				has_character_flag = murim_initiated
				has_relation_friend = root
			}
			# Friends understand
			add_opinion = {
				target = root
				modifier = murim_retired_friend_opinion
				opinion = 5
			}
		}
	}

	ai_potential = {
		age >= 50
		OR = {
			has_trait = content
			has_trait = temperate
			has_trait = calm
			health <= 3
		}
		NOT = { has_trait = ambitious }
	}

	ai_will_do = {
		base = 5
		modifier = {
			add = 15
			has_trait = content
		}
		modifier = {
			add = 10
			has_trait = calm
		}
		modifier = {
			add = 20
			age >= 65
		}
		modifier = {
			add = 25
			health <= 2
		}
		modifier = {
			add = -50
			has_trait = ambitious
		}
		modifier = {
			add = -30
			has_trait = brave
		}
		modifier = {
			add = 15
			stress >= 80
		}
	}
}
