# Murim Chronicles - Clan Youth Recruitment Events
# Global scouting system: clans recruit talented children worldwide.
# 10 events covering scouting, parent decisions, culture shock, rivalry, homecoming.
# TRIGGER: yearly pulse. Fires when murim_clan_needs_recruits_trigger = yes (< 3 youth).
# INTEGRATION: Uses murim_discovery_triggers.txt (murim_is_recruitable_youth_trigger)
# INTEGRATION: Uses murim_clan_triggers.txt (rank checks for scouts)
# INTEGRATION: Uses murim_clan_membership_effects.txt (join_sect effects)
# INTEGRATION: Feeds into murim_youth_tournament_events.txt (recruited kids enter tournaments)
# INTEGRATION: References murim_sect_politics_events.txt (recruited kid in succession later)
# INTEGRATION: Cross-refs murim_training_events.txt (training after recruitment)
# GENDER: ALL events fully gender-neutral. Boys AND girls equally scouted.
# AI: AI clans auto-recruit. AI parents decide based on personality. AI kids generate stories.
# GAME START NOTE: On game start, only 5-10 kids (age 8-15) have martial training.
#                  This scarcity drives the recruitment system. Clans NEED to scout.

namespace = murim_recruit

# 0001: Clan Sends Scouts Abroad
# Auto-fires when clan has < 3 youth members. AI-driven.
murim_recruit.0001 = {
	type = character_event
	title = murim_recruit.0001.t
	desc = murim_recruit.0001.desc
	theme = diplomacy
	
	trigger = {
		# This fires for the clan LEADER
		murim_is_clan_member_trigger = yes
		OR = {
			is_sect_leader_trigger = yes
			is_sect_elder_trigger = yes
		}
		murim_clan_needs_recruits_trigger = yes
		NOT = { has_character_flag = sent_scouts_this_year }
		# Cooldown: once per year per clan
	}
	
	immediate = {
		set_character_flag = {
			flag = sent_scouts_this_year
			years = 1
		}
		# Find an eligible scout from the clan
		random_courtier_or_guest = {
			limit = {
				murim_is_eligible_scout_trigger = yes
			}
			save_scope_as = clan_scout
		}
		# Determine scouting region based on clan's reach
		random_list = {
			40 = { set_variable = { name = scout_region value = flag:korea } }
			30 = { set_variable = { name = scout_region value = flag:china } }
			15 = { set_variable = { name = scout_region value = flag:japan } }
			5 = { set_variable = { name = scout_region value = flag:india } }
			3 = { set_variable = { name = scout_region value = flag:persia } }
			3 = { set_variable = { name = scout_region value = flag:steppe } }
			2 = { set_variable = { name = scout_region value = flag:europe } }
			2 = { set_variable = { name = scout_region value = flag:africa } }
		}
	}
	
	# Option A: Send the scout — standard
	option = {
		name = murim_recruit.0001.a
		# Scout departs, results in 30-90 days
		trigger_event = {
			id = murim_recruit.0002
			days = { 30 90 }
		}
		ai_chance = {
			base = 80
			# AI almost always sends scouts when they need recruits
		}
	}
	
	# Option B: Scout personally — the leader goes
	option = {
		name = murim_recruit.0001.b
		# Leader travels — risky but finds better candidates
		add_character_modifier = {
			modifier = personal_scouting_trip_modifier
			years = 1
		}
		trigger_event = {
			id = murim_recruit.0002
			days = { 14 60 }
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 15
				has_trait = diligent
			}
		}
	}
	
	# Option C: Lower standards — accept any willing youth
	option = {
		name = murim_recruit.0001.c
		# Faster but weaker recruits
		add_character_modifier = {
			modifier = lowered_recruitment_standards_modifier
			years = 3
		}
		trigger_event = {
			id = murim_recruit.0008
			days = { 7 14 }
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				# Desperation: no youth at all
				NOT = {
					any_courtier = {
						murim_is_clan_member_trigger = yes
						age >= 8
						age < 16
					}
				}
			}
		}
	}
}

# 0002: Scout Arrives at Foreign Court
# Scout evaluates children for recruitment potential.
murim_recruit.0002 = {
	type = character_event
	title = murim_recruit.0002.t
	desc = murim_recruit.0002.desc
	theme = education
	
	trigger = {
		# Fires as follow-up from 0001
		always = yes
	}
	
	immediate = {
		# Find a talented child in the scouted region
		random_living_character = {
			limit = {
				murim_is_recruitable_youth_trigger = yes
				age >= 8
				age < 16
				# Talent check: at least one strong indicator
				OR = {
					has_trait = brave
					has_trait = strong
					has_trait = rowdy
					has_trait = genius
					has_trait = quick
					has_trait = intelligent
					prowess >= 5
					martial >= 6
				}
			}
			weight = {
				base = 1
				# Korean/Chinese children more likely found
				modifier = {
					add = 3
					murim_is_martial_culture_trigger = yes
				}
				# Genius/quick children stand out
				modifier = {
					add = 5
					OR = {
						has_trait = genius
						has_trait = quick
					}
				}
				# Strong/brave children noticed in physical activities
				modifier = {
					add = 3
					OR = {
						has_trait = strong
						has_trait = brave
					}
				}
				# High prowess for age
				modifier = {
					add = 2
					prowess >= 8
				}
			}
			save_scope_as = scouted_child
		}
		# Save the child's parent for the decision event
		scope:scouted_child = {
			if = {
				limit = { exists = father }
				father = { save_scope_as = child_parent }
			}
			else_if = {
				limit = { exists = mother }
				mother = { save_scope_as = child_parent }
			}
		}
	}
	
	# Scout found a candidate — approach parent
	option = {
		name = murim_recruit.0002.a
		# If parent is player, fire decision event
		# If parent is AI, fire AI decision event
		scope:child_parent = {
			trigger_event = {
				id = murim_recruit.0003
				days = { 3 7 }
			}
		}
		ai_chance = {
			base = 90
		}
	}
	
	# Scout found nobody worthy — report back
	option = {
		name = murim_recruit.0002.b
		trigger_event = {
			id = murim_recruit.0008
			days = { 7 14 }
		}
		ai_chance = {
			base = 10
		}
	}
}

# 0003: Player's Child Scouted
# "A representative of [Clan] wishes to take your child to train..."
murim_recruit.0003 = {
	type = character_event
	title = murim_recruit.0003.t
	desc = murim_recruit.0003.desc
	theme = education
	
	trigger = {
		any_child = {
			murim_is_recruitable_youth_trigger = yes
		}
	}
	
	immediate = {
		# scope:scouted_child and scope:clan_scout carried from previous event
	}
	
	# Option A: Accept proudly
	option = {
		name = murim_recruit.0003.a
		# "Go, my child. Make us proud."
		scope:scouted_child = {
			murim_join_sect_as_outer_disciple_effect = yes
			add_trait = basic_martial_training
			add_character_modifier = {
				modifier = sent_abroad_for_training_modifier
				years = 8
			}
		}
		add_prestige = 20
		# Child moves to clan HQ region
		trigger_event = {
			id = murim_recruit.0005
			days = { 14 30 }
		}
		stress_impact = {
			compassionate = minor_stress_impact_gain
			ambitious = minor_stress_impact_loss
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = 10
				murim_is_martial_culture_trigger = yes
			}
			# More kids = more willing to let one go
			modifier = {
				add = 15
				num_of_children >= 3
			}
		}
	}
	
	# Option B: Accept reluctantly
	option = {
		name = murim_recruit.0003.b
		# "I... suppose it's for the best."
		scope:scouted_child = {
			murim_join_sect_as_outer_disciple_effect = yes
			add_trait = basic_martial_training
			add_character_modifier = {
				modifier = homesick_recruit_modifier
				years = 3
			}
		}
		add_opinion = {
			modifier = reluctant_farewell_opinion
			target = scope:scouted_child
			opinion = -10
		}
		trigger_event = {
			id = murim_recruit.0005
			days = { 14 30 }
		}
		stress_impact = {
			compassionate = medium_stress_impact_gain
		}
		ai_chance = {
			base = 25
		}
	}
	
	# Option C: Negotiate terms — demand compensation
	option = {
		name = murim_recruit.0003.c
		# "My child is valuable. What do you offer in return?"
		random_list = {
			50 = {
				# Clan agrees — gold and alliance
				add_gold = 100
				add_character_modifier = {
					modifier = clan_patronage_agreement_modifier
					years = 10
				}
				scope:scouted_child = {
					murim_join_sect_as_outer_disciple_effect = yes
					add_trait = basic_martial_training
				}
				trigger_event = {
					id = murim_recruit.0005
					days = { 14 30 }
				}
			}
			50 = {
				# Clan withdraws — too expensive
				add_prestige = -10
				# They may scout elsewhere
			}
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 15
				has_trait = greedy
			}
		}
	}
	
	# Option D: Refuse
	option = {
		name = murim_recruit.0003.d
		# "My child stays with me."
		scope:scouted_child = {
			set_character_flag = {
				flag = recruitment_rejected_recently
				years = 2
			}
			# Child may resent the decision
			random = {
				chance = 30
				add_opinion = {
					modifier = denied_my_dream_opinion
					target = root
					opinion = -20
				}
			}
		}
		# 10% chance child runs away to join anyway
		random = {
			chance = 10
			trigger_event = {
				id = murim_recruit.0003_runaway
				days = { 30 180 }
			}
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 10
				has_trait = paranoid
			}
			modifier = {
				add = 15
				num_of_children < 2
				# Only child — harder to let go
			}
		}
	}
	
	# Option E: "Over my dead body" — hostile refusal
	option = {
		name = murim_recruit.0003.e
		# "Get away from my family!"
		add_character_modifier = {
			modifier = hostile_to_sect_modifier
			years = 20
		}
		# Clan remembers the insult
		scope:scouted_child = {
			set_character_flag = {
				flag = recruitment_rejected_permanently
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				add = 15
				has_trait = wrathful
			}
		}
	}
}
