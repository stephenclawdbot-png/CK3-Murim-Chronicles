# EVENT 0001 - Discovery of the Underground Market
# Triggered: on_action yearly pulse, any martial artist with intrigue >= 8
# AI: Full weight scripting for all options
murim_black_market.0001 = {
    type = character_event
    title = murim_black_market.0001.t
    desc = murim_black_market.0001.desc
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = thinking
    }
    right_portrait = {
        character = scope:black_market_contact
        animation = scheme
    }
    
    trigger = {
        is_alive = yes
        age >= 16
        intrigue >= 8
        NOT = { has_character_flag = knows_black_market }
        NOT = { has_trait = murim_righteous_elder }
        OR = {
            has_trait = murim_martial_artist
            has_trait = murim_sect_leader
            has_trait = murim_wandering_cultivator
            has_trait = murim_shadow_operative
            has_trait = murim_unorthodox_elder
        }
        OR = {
            any_held_title = { tier >= tier_county }
            gold >= 50
            intrigue >= 14
        }
    }
    
    immediate = {
        random_courtier = {
            limit = {
                intrigue >= 10
                NOT = { has_trait = murim_righteous_elder }
            }
            save_scope_as = black_market_contact
            alternative = {
                create_character = {
                    age = { 30 45 }
                    gender_female_chance = 40
                    trait = murim_shadow_operative
                    trait = deceitful
                    trait = greedy
                    culture = root.culture
                    faith = root.faith
                    save_scope_as = black_market_contact
                }
            }
        }
        add_character_flag = {
            flag = discovered_black_market_event
            days = 30
        }
    }
    
    # Option A: Enter the underground market
    option = {
        name = murim_black_market.0001.a
        add_character_flag = knows_black_market
        add_character_modifier = {
            modifier = murim_black_market_access
            years = 5
        }
        reverse_add_opinion = {
            target = scope:black_market_contact
            modifier = murim_black_market_trust
            opinion = 15
        }
        add_intrigue_skill = 1
        stress_impact = {
            just = medium_stress_impact_gain
            honest = medium_stress_impact_gain
        }
        ai_chance = {
            base = 60
            modifier = {
                factor = 2.0
                has_trait = greedy
            }
            modifier = {
                factor = 1.5
                has_trait = deceitful
            }
            modifier = {
                factor = 0.3
                has_trait = just
            }
            modifier = {
                factor = 2.0
                has_trait = murim_unorthodox_elder
            }
            modifier = {
                factor = 1.5
                has_trait = murim_shadow_operative
            }
            modifier = {
                factor = 0.5
                has_trait = murim_orthodox_disciple
            }
        }
    }
    
    # Option B: Report to orthodox authorities
    option = {
        name = murim_black_market.0001.b
        add_piety = 50
        add_prestige = 100
        add_character_flag = reported_black_market
        if = {
            limit = { exists = scope:black_market_contact }
            scope:black_market_contact = {
                add_opinion = {
                    target = root
                    modifier = murim_betrayed_trust
                    opinion = -40
                }
                trigger_event = {
                    id = murim_black_market.0002
                    days = { 15 45 }
                }
            }
        }
        stress_impact = {
            deceitful = medium_stress_impact_gain
            greedy = minor_stress_impact_gain
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 3.0
                has_trait = just
            }
            modifier = {
                factor = 2.0
                has_trait = honest
            }
            modifier = {
                factor = 2.5
                has_trait = murim_righteous_elder
            }
            modifier = {
                factor = 0.2
                has_trait = murim_unorthodox_elder
            }
        }
    }
    
    # Option C: Extort the market for protection money
    option = {
        name = murim_black_market.0001.c
        trigger = {
            OR = {
                prowess >= 15
                has_trait = murim_sect_leader
                has_trait = intimidating
            }
        }
        add_gold = 200
        add_dread = 25
        add_character_flag = knows_black_market
        add_character_flag = black_market_extortionist
        add_character_modifier = {
            modifier = murim_black_market_extortion_income
            years = 3
        }
        stress_impact = {
            compassionate = major_stress_impact_gain
            generous = medium_stress_impact_gain
        }
        ai_chance = {
            base = 20
            modifier = {
                factor = 3.0
                has_trait = greedy
            }
            modifier = {
                factor = 2.5
                has_trait = intimidating
            }
            modifier = {
                factor = 2.0
                prowess >= 20
            }
            modifier = {
                factor = 0.1
                has_trait = compassionate
            }
        }
    }
}
