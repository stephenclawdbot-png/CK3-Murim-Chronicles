# EVENT 1: THE BEGGAR'S INITIATION
# Joining the Beggar Sect. You must prove you understand poverty and humility.
# Three paths based on your background.
##############################
murim_beggar.0001 = {
	type = character_event
	title = murim_beggar.0001.t
	desc = murim_beggar.0001.desc
	theme = intrigue
	
	trigger = {
		is_beggar_sect_member_trigger = yes
		NOT = { has_character_flag = beggar_initiation_complete }
		age >= 14
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = humble
		}
	}
	
	immediate = {
		set_character_flag = beggar_initiation_complete
		create_character_memory = {
			type = beggar_sect_initiation_memory
		}
	}
	
	# Option A: Live as a true beggar for 30 days (Dirty Faction approval)
	option = {
		name = murim_beggar.0001.a
		add_character_modifier = {
			modifier = beggar_dirty_faction_respect_modifier
			years = 10
		}
		add_prowess_xp = 200
		add_intrigue_xp = 300
		stress_impact = {
			arrogant = major_stress_impact_gain
			humble = minor_stress_impact_loss
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = humble
			}
		}
	}
	
	# Option B: Donate your wealth to the poor (Clean Faction approval)
	option = {
		name = murim_beggar.0001.b
		trigger = {
			gold >= 200
		}
		remove_short_term_gold = 200
		add_character_modifier = {
			modifier = beggar_clean_faction_patron_modifier
			years = 10
		}
		add_diplomacy_xp = 300
		add_prestige = 200
		ai_chance = {
			base = 35
			modifier = {
				add = 15
				gold >= 500
			}
		}
	}
	
	# Option C: Prove your worth through intelligence gathering (spy path)
	option = {
		name = murim_beggar.0001.c
		trigger = {
			intrigue >= 10
		}
		add_intrigue_xp = 500

# EVENT 3: CLEAN FACTION VS DIRTY FACTION SHOWDOWN
# The Beggar Sect's defining internal conflict. Clean = wealthy donors who buy rank.
# Dirty = true beggars who earned everything through suffering.
##############################
murim_beggar.0003 = {
	type = character_event
	title = murim_beggar.0003.t
	desc = murim_beggar.0003.desc
	theme = intrigue
	
	trigger = {
		is_beggar_sect_member_trigger = yes
		is_adult = yes
		NOT = { has_character_flag = beggar_faction_showdown }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_beggar_rank_elder_trigger = yes
		}
	}
	
	immediate = {
		set_character_flag = beggar_faction_showdown
		create_character_memory = {
			type = beggar_faction_showdown_memory
		}
	}
	
	# Option A: Side with Clean Faction (wealth and political power)
	option = {
		name = murim_beggar.0003.a
		add_gold = 300
		add_diplomacy_xp = 400
		add_character_modifier = {
			modifier = beggar_clean_faction_leader_modifier
			years = 10
		}
		# Dirty Faction becomes hostile
		random_courtier = {
			limit = {
				is_beggar_sect_member_trigger = yes
				has_trait = humble
			}
			add_opinion = {
				modifier = beggar_faction_betrayal_opinion
				target = root
				opinion = -35
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				gold >= 500
			}
		}
	}
	
	# Option B: Side with Dirty Faction (martial legitimacy)
	option = {
		name = murim_beggar.0003.b
		add_prowess_xp = 400
		add_prestige = 300
		add_character_modifier = {
			modifier = beggar_dirty_faction_champion_modifier
			years = 10
		}
		# Clean Faction withdraws funding
		random_courtier = {
			limit = {
				is_beggar_sect_member_trigger = yes
				stewardship >= 12
			}
			add_opinion = {
				modifier = beggar_faction_betrayal_opinion
				target = root
				opinion = -30
			}
		}
		ai_chance = {
			base = 35
			modifier = {
				add = 20
				has_trait = humble
			}
		}
	}
	
	# Option C: Unite both factions under your leadership (hardest path)
	option = {
		name = murim_beggar.0003.c
		trigger = {
			diplomacy >= 16
			OR = {
				has_trait = gregarious
				has_trait = just
			}
		}
		random_list = {
			35 = {
				# Both factions follow you
				add_diplomacy_xp = 600
				add_prestige = 500
				add_character_modifier = {
					modifier = beggar_sect_unifier_modifier
					years = -1
				}
				every_courtier = {
					limit = {
						is_beggar_sect_member_trigger = yes
					}
					add_opinion = {
						modifier = beggar_unified_leader_opinion
						target = root
						opinion = 20
					}
				}
			}
			65 = {
				# Failed mediation - civil war erupts
				add_stress = 50
				add_character_modifier = {
					modifier = beggar_sect_civil_war_modifier
					years = 3
				}
				# Chains to sect war events
				trigger_event = {
					id = murim_sect_war.0001
					days = { 30 60 }
				}
			}
		}
		ai_chance = { base = 20 }
	}
	
	# Option D: Secretly work with Heavenly Demon to split the sect
	option = {
		name = murim_beggar.0003.d
		trigger = {
			OR = {
				has_trait = deceitful
				intrigue >= 16
			}
		}
		add_intrigue_xp = 500
		add_character_modifier = {
			modifier = beggar_heavenly_demon_spy_modifier
			years = 10
		}
		add_secret = {
			type = secret_demonic_sect_spy
			target = root
		}

		trigger_event = {
			id = murim_shadow.0003
			days = { 60 120 }
		}
