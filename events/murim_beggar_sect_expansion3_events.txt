murim_beggar_sect_expansion3.0001 = {
	type = character_event
	title = murim_beggar_sect_expansion3.0001.t
	desc = murim_beggar_sect_expansion3.0001.desc
	theme = martial_arts
	
	# The Beggar Network Awakens — as Beggar Sect leader, your intelligence network
	# spanning every city, town, and village reports disturbing patterns. The Heavenly Demon
	# Sect is moving agents into position across all major trade cities simultaneously.
	
	trigger = {
		has_trait = beggar_sect_member
		is_ruler = yes
		any_held_title = { tier >= county }
		NOT = { has_character_flag = beggar_network_awakened }
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = diligent }
		modifier = { add = 2 has_trait = paranoid }
	}
	
	immediate = {
		set_character_flag = beggar_network_awakened
		set_global_variable = {
			name = beggar_intelligence_alert
			value = yes
		}
	}
	
	option = {
		name = murim_beggar_sect_expansion3.0001.a
		# Mobilize the full network — shadow every Demon agent
		add_intrigue = 5
		add_gold = -150
		add_character_modifier = {
			modifier = full_network_mobilization
			years = 3
		}
		hidden_effect = {
			trigger_event = {
				id = murim_beggar_sect_expansion3.0002
				days = { 30 60 }
			}
		}
		ai_chance = { base = 50 }
	}
	
	option = {
		name = murim_beggar_sect_expansion3.0001.b
		# Alert the Orthodox Alliance immediately
		add_diplomacy = 4
		add_prestige = 300
		hidden_effect = {
			every_known_character = {
				limit = {
					OR = {
						has_trait = shaolin_disciple
						has_trait = mount_hua_disciple
						has_trait = wudang_disciple
					}
				}
				add_opinion = {
					target = root
					modifier = intelligence_warning
					opinion = 25
				}
			}
			trigger_event = {
				id = murim_beggar_sect_expansion3.0004
				days = { 15 30 }
			}
		}
		ai_chance = { base = 30 }
	}
	
	option = {
		name = murim_beggar_sect_expansion3.0001.c
		# Sell the information to the highest bidder — profit from chaos
		add_gold = 500
		add_intrigue = 3
		add_piety = -200
		stress_impact = {
			honest = major_stress_impact_gain
			just = major_stress_impact_gain
		}
		ai_chance = { base = 20 modifier = { add = 20 has_trait = greedy } }
	}
}
