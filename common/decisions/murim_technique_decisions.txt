# MURIM CHRONICLES — TECHNIQUE UNLOCK DECISIONS
# Path: common/decisions/murim_technique_decisions.txt
#
# Player-facing decisions. Appear in the decision menu when
# the player meets the stat requirements.
# Uses the SAME scripted triggers as AI on_actions.
#
# Players can also wait for the yearly event, but decisions
# let them actively pursue techniques when ready.

murim_pursue_harmonious_palm_decision = {
	title = murim_pursue_harmonious_palm_decision_title
	desc = murim_pursue_harmonious_palm_decision_desc

	is_shown = {
		prowess >= 10
		diplomacy >= 10
		NOT = { has_trait = murim_harmonious_palm }
		age >= 16
	}

	is_valid = {
		murim_can_unlock_harmonious_palm_trigger = yes
	}

	cost = {
		prestige = 150
	}

	cooldown = { months = 6 }

	ai_check_interval = 0
	# AI uses on_actions instead, not decisions

	effect = {
		trigger_event = murim_technique.0001
	}
}
