# 10 events: deep Wudang chain with cultivation, philosophy, ultimate transcendence

murim_wudang_expansion3.0001 = {
	type = character_event
	title = murim_wudang_expansion3.0001.t
	desc = murim_wudang_expansion3.0001.desc
	theme = martial_arts
	
	# The Three Purities Meditation — on Wudang's highest peak during the winter
	# solstice, you achieve a state of complete stillness. Your internal qi circulates
	# on its own. The Taoist texts call this the "Gateway to Immortality."
	
	trigger = {
		has_trait = wudang_disciple
		age >= 25
		learning >= 12
		NOT = { has_character_flag = three_purities_attempted }
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 4 has_trait = patient }
		modifier = { add = 3 has_trait = diligent }
		modifier = { add = -5 has_trait = impatient }
	}
	
	immediate = {
		set_character_flag = three_purities_attempted
	}
	
	option = {
		name = murim_wudang_expansion3.0001.a
		# Embrace the stillness — pursue the immortality path
		add_trait = immortality_seeker
		add_learning = 5
		add_character_modifier = { modifier = murim_light_healing years = 1 }
		add_character_modifier = {
			modifier = gateway_to_immortality
			years = -1
		}
		hidden_effect = {
			trigger_event = {
				id = murim_wudang_expansion3.0002
				days = { 60 180 }
			}
		}
		ai_chance = { base = 50 }
	}
	
	option = {
		name = murim_wudang_expansion3.0001.b
		# Channel the energy into martial power instead
		add_prowess = 6
		add_martial = 3
		add_character_modifier = {
			modifier = qi_to_combat_conversion
			years = -1
		}
		ai_chance = { base = 30 }
	}
	
	option = {
		name = murim_wudang_expansion3.0001.c
		# Pull back — the void beyond feels too vast, too alien
		add_stress = 30
		add_learning = 2
		add_character_modifier = {
			modifier = touched_the_void
			years = 5
		}
		ai_chance = { base = 20 modifier = { add = 20 has_trait = craven } }
	}
}
