######################################
# EVENT 0001 — AI Clan War Escalation
# AI clans autonomously escalate from skirmish to total war
######################################
murim_calamity_ai_exp.0001 = {
	type = character_event
	title = murim_calamity_ai_exp.0001.t
	desc = murim_calamity_ai_exp.0001.desc
	theme = war_theme
	left_portrait = {
		character = root
		animation = war_over_win
	}
	right_portrait = {
		character = scope:enemy_clan_leader
		animation = personality_vengeful
	}

	trigger = {
		is_ai = yes
		has_variable = murim_calamity_active
		has_variable = murim_clan_id
		any_ruler = {
			has_variable = murim_clan_id
			NOT = { var:murim_clan_id = root.var:murim_clan_id }
			NOT = { has_character_flag = calamity_destroyed }
		}
		NOT = { has_character_flag = calamity_war_escalated }
	}

	weight_multiplier = {
		base = 5
		modifier = {
			add = 10
			has_trait = wrathful
		}
		modifier = {
			add = 5
			has_trait = ambitious
		}
	}

	immediate = {
		set_character_flag = { flag = calamity_war_escalated days = 180 }
		random_ruler = {
			limit = {
				has_variable = murim_clan_id
				NOT = { var:murim_clan_id = root.var:murim_clan_id }
				NOT = { has_character_flag = calamity_destroyed }
			}
			save_scope_as = enemy_clan_leader
		}
	}

	# Full-scale clan war declaration
	option = {
		name = murim_calamity_ai_exp.0001.a
		start_war = {
			cb = murim_clan_annihilation_cb
			target = scope:enemy_clan_leader
		}
		add_character_modifier = {
			modifier = calamity_total_war
			years = 5
		}
		add_dread = 40
		every_courtier = {
			limit = { has_variable = murim_clan_id }
			add_opinion = {
				target = root
				modifier = clan_war_rally
				opinion = 20
			}
		}
		ai_chance = { base = 70 }
	}

	# Targeted assassination of enemy leader
	option = {
		name = murim_calamity_ai_exp.0001.b
		trigger = {
			intrigue >= 12
		}
		duel = {
			skill = intrigue
			target = scope:enemy_clan_leader
			50 = {
				desc = murim_calamity_ai_exp.0001.b.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				scope:enemy_clan_leader = {
					death = {
						death_reason = death_murder
						killer = root
					}
				}
				add_prestige = 500
			}
			50 = {
				desc = murim_calamity_ai_exp.0001.b.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
				}
				health = -2
				add_prestige = -200
			}
		}
		ai_chance = { base = 30 }
	}

	# Betray own alliance and switch sides
	option = {
		name = murim_calamity_ai_exp.0001.c
		add_trait = betrayer
		scope:enemy_clan_leader = {
			add_opinion = {
				target = root
				modifier = calamity_betrayal_alliance
				opinion = 30
			}
		}
		every_courtier = {
			limit = { has_variable = murim_clan_id }
			add_opinion = {
				target = root
				modifier = betrayed_our_clan
				opinion = -50
			}
		}
		add_character_modifier = {
			modifier = calamity_turncoat
			years = 10
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = deceitful
			}
		}
	}
}

murim_calamity_ai_exp.0003 = {
	type = character_event
	title = murim_calamity_ai_exp.0003.t
	desc = murim_calamity_ai_exp.0003.desc
	theme = war_theme
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:murim_emperor
		animation = fear
	}

	trigger = {
		is_ai = yes
		has_variable = murim_calamity_active
		has_trait = heavenly_demon_member
		any_ruler = {
			has_trait = murim_emperor
			NOT = { has_character_flag = emperor_already_fallen }
		}
	}

	weight_multiplier = {
		base = 8
		modifier = {
			add = 10
			has_trait = ambitious
		}
	}

	immediate = {
		random_ruler = {
			limit = {
				has_trait = murim_emperor
				NOT = { has_character_flag = emperor_already_fallen }
			}
			save_scope_as = murim_emperor
		}
	}

	# Storm the Imperial Palace
	option = {
		name = murim_calamity_ai_exp.0003.a
		duel = {
			skill = prowess
			target = scope:murim_emperor
			45 = {
				desc = murim_calamity_ai_exp.0003.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 4.0
				}
				scope:murim_emperor = {
					set_character_flag = emperor_already_fallen
					murim_combat_death_or_injury_with_killer_effect = { REASON = death_battle KILLER = root }
				}
				add_trait = emperor_slayer
				add_prestige = 3000
				add_dread = 100
				add_character_modifier = {
					modifier = slew_the_murim_emperor
					years = 50
				}
			}
			55 = {
				desc = murim_calamity_ai_exp.0003.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -4.0
				}
				murim_combat_death_or_injury_with_killer_effect = { REASON = death_battle KILLER = scope:murim_emperor }
			}
		}
		ai_chance = { base = 80 }
	}

	# Poison the Emperor instead
	option = {
		name = murim_calamity_ai_exp.0003.b
		trigger = {
			intrigue >= 18
		}
		scope:murim_emperor = {
			set_character_flag = emperor_already_fallen
			add_character_modifier = {
				modifier = fatally_poisoned
				years = 1
			}
			health = -6
		}
		add_dread = 60
		add_character_modifier = {
			modifier = poisoned_the_emperor
			years = 20
		}
		ai_chance = { base = 20 }
	}
}
