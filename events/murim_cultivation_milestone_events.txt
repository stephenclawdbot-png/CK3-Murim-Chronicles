###############################################################################
# MURIM CULTIVATION MILESTONE EVENTS
# These fire when a cultivator crosses a significant rank threshold.
# Rank 3: Mastery of the physical body acknowledged by the heavens.
# Rank 5: The qi begins to reflect the soul — internal and external merge.
# Rank 7: A figure of genuine power in the jianghu.
# Rank 9: Approaching the boundary of the mortal and the divine.
#
# Cross-references: murim_fate, murim_rival, murim_sect_politics
###############################################################################

namespace = murim_milestone

##############################
# 0001 - RANK 3: BODY REFINEMENT COMPLETE
# The character's physical body has been fundamentally transformed.
# They are no longer ordinary.
##############################
murim_milestone.0001 = {
	type = character_event
	title = murim_milestone.0001.t
	desc = murim_milestone.0001.desc
	theme = martial_arts

	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		is_alive = yes
		is_imprisoned = no
		has_variable = murim_cultivation_level
		var:murim_cultivation_level >= 3
		NOT = { has_character_flag = murim_milestone_rank3_fired }
	}

	immediate = {
		add_character_flag = murim_milestone_rank3_fired
	}

	# Mark the crossing — train harder now that the foundation is real
	option = {
		name = murim_milestone.0001.a

		add_prestige = 150
		add_character_modifier = {
			modifier = murim_refined_body_foundation
			years = 5
		}
		change_variable = { name = murim_cultivation_xp add = 30 }

		ai_chance = {
			base = 60
			modifier = {
				add = 20
				has_trait = diligent
			}
		}
	}

	# Celebrate with the sect — share the moment
	option = {
		name = murim_milestone.0001.b

		add_prestige = 100
		add_stress = -15
		add_character_modifier = {
			modifier = murim_sect_celebrated
			months = 12
		}

		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = gregarious
			}
		}
	}
}

##############################
# 0004 - RANK 9: THE BOUNDARY
# The character stands at the edge of what the mortal cultivation path allows.
# Beyond this is either death, transcendence, or something neither.
##############################
murim_milestone.0004 = {
	type = character_event
	title = murim_milestone.0004.t
	desc = murim_milestone.0004.desc
	theme = mystic

	left_portrait = {
		character = root
		animation = worry
	}

	trigger = {
		is_alive = yes
		is_imprisoned = no
		has_variable = murim_cultivation_level
		var:murim_cultivation_level >= 9
		NOT = { has_character_flag = murim_milestone_rank9_fired }
	}

	immediate = {
		add_character_flag = murim_milestone_rank9_fired
	}

	# Push forward — cross the boundary regardless of cost
	option = {
		name = murim_milestone.0004.a

		random_list = {
			50 = {
				add_character_modifier = {
					modifier = murim_boundary_transcended
					years = 20
				}
				add_prestige = 1000
				add_health = 0.5
			}
			30 = {
				# Partial transcendence — powerful but incomplete
				add_character_modifier = {
					modifier = murim_partial_transcendence
					years = 10
				}
				add_prestige = 500
				add_stress = 20
			}
			20 = {
				# The boundary pushes back — survive but at great cost
				add_stress = 40
				add_health = -1.0
				add_character_modifier = {
					modifier = murim_boundary_backlash
					years = 5
				}
			}
		}

		ai_chance = {
			base = 45
			modifier = {
				add = 30
				has_trait = brave
			}
			modifier = {
				add = 20
				has_trait = ambitious
			}
		}
	}

	# Wait — consolidate everything before attempting the crossing
	option = {
		name = murim_milestone.0004.b

		change_variable = { name = murim_cultivation_xp add = 100 }
		add_character_modifier = {
			modifier = murim_boundary_consolidation
			years = 5
		}
		add_stress = -10

		ai_chance = {
			base = 40
			modifier = {
				add = 25
				has_trait = patient
			}
		}
	}

	# Seek guidance from the heavens — pray for a sign before crossing
	option = {
		name = murim_milestone.0004.c

		add_piety = 300
		change_variable = { name = murim_cultivation_xp add = 60 }
		trigger_event = { id = murim_fate.0001 days = { 60 120 } }

		ai_chance = {
			base = 15
			modifier = {
				add = 25
				has_trait = humble
			}
		}
	}
}
