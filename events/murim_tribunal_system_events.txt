# Tribunal Nomination Quest
murim_tribunal_system.0001 = {
	type = character_event
	title = murim_tribunal_system.0001.t
	desc = murim_tribunal_system.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = { murim_can_join_tribunal_trigger = yes }

	# Present your case to the elders
	option = {
		name = murim_tribunal_system.0001.a
		random_list = {
			50 = {
				modifier = { add = diplomacy factor = 0.3 }
				modifier = { add = prowess factor = 0.2 }
				hidden_effect = { murim_join_tribunal_effect = yes }
			}
			50 = {
				add_prestige = 100
				add_stress = 20
			}
		}
		ai_chance = { base = 60 }
	}

	# Challenge a current judge for their seat
	option = {
		name = murim_tribunal_system.0001.b
		trigger = { prowess >= 25 }
		random_list = {
			40 = {
				modifier = { add = prowess factor = 0.3 }
				hidden_effect = { murim_join_tribunal_effect = yes }
				add_dread = 15
			}
			60 = {
				add_stress = 30
				add_character_modifier = { modifier = murim_light_injury years = 1 }
			}
		}
		ai_chance = { base = 30 }
	}

	# Withdraw nomination
	option = {
		name = murim_tribunal_system.0001.c
		add_stress = 10
		ai_chance = { base = 10 }
	}
}
