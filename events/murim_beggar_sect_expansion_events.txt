# EVENT 1: CLEAN VS DIRTY FACTION SHOWDOWN
# The eternal internal struggle of the Beggar Sect.
# Clean Faction: Wealthy patrons who fund the sect but never lived as beggars.
# Dirty Faction: True street beggars who built the sect from nothing.
# When the Chief dies, this tension erupts into open conflict.
##############################
murim_beggar_exp.0001 = {
	type = character_event
	title = murim_beggar_exp.0001.t
	desc = murim_beggar_exp.0001.desc
	theme = intrigue
	
	trigger = {
		is_beggar_sect_member_trigger = yes
		OR = {
			murim_is_clan_leader_trigger = yes
			murim_is_clan_elder_trigger = yes
		}
		NOT = { has_character_flag = beggar_faction_showdown }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_variable = murim_clan_harmony
			var:murim_clan_harmony < 30
		}
	}
	
	immediate = {
		set_character_flag = beggar_faction_showdown
		# Find the opposing faction leader
		random_courtier = {
			limit = {
				is_beggar_sect_member_trigger = yes
				prowess >= 12
				NOT = { this = root }
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					has_trait = ambitious
				}
				modifier = {
					add = 2
					intrigue >= 14
				}
			}
			save_scope_as = faction_rival
		}
		create_character_memory = {
			type = beggar_faction_showdown_memory
			participants = { rival = scope:faction_rival }
		}
	}
	
	# Option A: Call for a martial contest to settle it (Dirty Faction tradition)
	option = {
		name = murim_beggar_exp.0001.a
		duel = {
			skill = prowess
			target = scope:faction_rival
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				min_compared_to = root.prowess
				# Your faction wins control
				add_prowess_xp = 600
				add_prestige = 400
				add_character_modifier = {
					modifier = beggar_faction_champion_modifier
					years = 10
				}
				scope:faction_rival = {
					add_trait = wounded_1
					add_opinion = {
						modifier = beggar_faction_defeated_opinion
						target = root
						opinion = -30
					}
				}
				change_variable = {
					name = murim_clan_harmony
					add = -10
				}
				create_character_memory = {
					type = beggar_faction_duel_victory_memory
					participants = { defeated = scope:faction_rival }
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min_compared_to = scope:faction_rival.prowess
				# Rival faction takes control
				add_trait = wounded_1
				add_stress = 35
				scope:faction_rival = {
					add_prestige = 400
				}
				change_variable = {
					name = murim_clan_harmony
					add = -15
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = wrathful
			}
		}
	}
	
	# Option B: Propose a democratic vote among all members (Clean Faction tradition)
	option = {
		name = murim_beggar_exp.0001.b
		add_diplomacy_xp = 500
		random_list = {
			50 = {
				# Vote goes your way
				add_prestige = 300
				add_character_modifier = {
					modifier = beggar_elected_leader_modifier
					years = 10
				}
				change_variable = {
					name = murim_clan_harmony
					add = 5
				}
			}
			50 = {
				# Vote goes against you
				add_stress = 30
				scope:faction_rival = {
					add_prestige = 300
				}
				change_variable = {
					name = murim_clan_harmony
					add = 5
				}
			}
		}
		ai_chance = {
			base = 35
			modifier = {
				add = 15
				has_trait = just
			}
		}
	}
	
	# Option C: Secretly eliminate the rival (assassination path)
	option = {
		name = murim_beggar_exp.0001.c
		trigger = {
			intrigue >= 14
		}
		add_intrigue_xp = 500
		add_dread = 20
		random_list = {
			60 = {
				modifier = {
					add = 15
					intrigue >= 18
				}
				# Assassination succeeds
				scope:faction_rival = {
					death = {
						death_reason = death_murder_known
						killer = root
					}
				}
				add_character_modifier = {
					modifier = beggar_shadow_chief_modifier
					years = -1
				}
				change_variable = {
					name = murim_clan_harmony
					add = -25
				}
			}
			40 = {
				# Assassination discovered!
				add_trait = murderer
				add_character_modifier = {
					modifier = beggar_traitor_exposed_modifier
					years = 10
				}
				every_courtier = {
					limit = {
						is_beggar_sect_member_trigger = yes
					}
					add_opinion = {
						modifier = beggar_traitor_opinion
						target = root
						opinion = -50
					}
				}
			}
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 10
				has_trait = deceitful
			}
		}
	}
}
