# 0001 — Shadow Contract: Imperial Target
##############################
murim_phantom_blade_exp4.0001 = {
	type = character_event
	title = murim_phantom_blade_exp4.0001.t
	desc = murim_phantom_blade_exp4.0001.desc
	theme = intrigue
	
	left_portrait = {
		character = root
		animation = scheme
	}
	right_portrait = {
		character = scope:target_official
		animation = worry
	}
	
	trigger = {
		has_trait = phantom_blade_disciple
		has_trait_rank = { trait = martial_cultivation value >= 4 }
		is_ai = no
		NOT = { has_character_flag = phantom_blade_imperial_contract }
		any_ruler = {
			has_title_rank = kingdom
			NOR = {
				has_trait = phantom_blade_disciple
				has_trait = phantom_blade_elder
			}
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = phantom_blade_elder
		}
		modifier = {
			add = 2
			intrigue >= 20
		}
	}
	
	immediate = {
		set_character_flag = { flag = phantom_blade_imperial_contract days = 1825 }
		random_ruler = {
			limit = {
				has_title_rank = kingdom
				NOR = {
					has_trait = phantom_blade_disciple
					has_trait = phantom_blade_elder
				}
			}
			save_scope_as = target_official
		}
		hidden_effect = {
			set_variable = { name = blade_contract_target value = scope:target_official }
		}
	}
	
	# Accept the imperial assassination contract
	option = {
		name = murim_phantom_blade_exp4.0001.a
		add_internal_flag = dangerous
		
		duel = {
			skill = intrigue
			target = scope:target_official
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				desc = murim_phantom_blade_exp4.0001.a.success
				send_interface_toast = {
					title = murim_phantom_blade_exp4.0001.a.success
					scope:target_official = {
						death = { death_reason = death_murder_unknown killer = root }
					}
				}
				add_prestige = 500
				add_gold = 300
				add_character_modifier = {
					modifier = phantom_blade_imperial_assassin
					years = 10
				}
				if = {
					limit = { NOT = { has_trait = phantom_blade_elder } }
					add_trait = phantom_blade_elder
				}
			}
			30 = {
				desc = murim_phantom_blade_exp4.0001.a.partial
				add_prestige = 100
				scope:target_official = {
					add_character_modifier = {
						modifier = assassination_attempt_survivor
						years = 5
					}
				}
				add_character_modifier = {
					modifier = phantom_blade_exposed
					years = 3
				}
			}
			20 = {
				desc = murim_phantom_blade_exp4.0001.a.failure
				add_prestige = -200
				add_dread = -30
				add_character_modifier = {
					modifier = phantom_blade_wounded_retreat
					years = 5
				}
				random = {
					chance = 40
					add_trait = wounded_3
				}
			}
		}
		
		stress_impact = {
			compassionate = major_stress_impact_gain
			honest = medium_stress_impact_gain
		}
		
		ai_chance = { base = 60 }
	}
	
	# Refuse — too dangerous
	option = {
		name = murim_phantom_blade_exp4.0001.b
		add_prestige = -100
		add_internal_flag = boring
		
		reverse_add_opinion = {
			target = scope:target_official
			modifier = grateful_opinion
			opinion = 10
		}
		
		if = {
			limit = { has_trait = phantom_blade_elder }
			custom_tooltip = murim_phantom_blade_exp4.0001.b.elder_shame
			add_prestige = -300
		}
		
		ai_chance = { base = 20 }
	}
	
	# Warn the target instead — betray the sect
	option = {
		name = murim_phantom_blade_exp4.0001.c
		trigger = { intrigue >= 15 }
		
		scope:target_official = {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 40
			}
		}
		
		add_character_flag = { flag = phantom_blade_traitor days = 3650 }
		add_character_modifier = {
			modifier = phantom_blade_sect_enemy
			years = 20
		}
		
		hidden_effect = {
			every_living_character = {
				limit = { has_trait = phantom_blade_disciple }
				add_opinion = {
					target = root
					modifier = angry_opinion
					opinion = -60
				}
			}
		}
		
		ai_chance = { base = 5 }
	}
}
