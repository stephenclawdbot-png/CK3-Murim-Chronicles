# Murim Chronicles - Clan/Sect Membership Management Effects
# INTEGRATION: ALL sect/clan operations go through these effects.
# Do NOT add/remove sect traits directly in event files - call these effects.
# Cross-refs: murim_triggers.txt (murim_is_clan_member_trigger),
#             murim_clan_triggers.txt (rank checks),
#             murim_sect_politics_events.txt (promotions/betrayal),
#             murim_game_start_effects.txt (initial clan seeding),
#             murim_harmony_values.txt (harmony updates on membership changes)
# GENDER: No gender restrictions on any membership operation.
# TANG EXCEPTION: Tang Clan bloodline membership handled separately
#                 in murim_tang_birthright_effects.txt
# HEARTBEAT #56 POLISH: Updated to support ALL 9 CLANS:
#   Orthodox: Mount Hua, Shaolin, Wudang, Tang Clan, Beggar Sect, Iron Fist Gate
#   Unorthodox: Heavenly Demon Sect, Poison Flower Palace, Phantom Blade Sect

########################################
# JOIN SECT EFFECTS
########################################

# Join any sect as outer disciple (standard recruitment)
# Scope: character joining
# Parameters: uses saved scope sect_type flag
murim_join_sect_as_outer_disciple_effect = {
	if = {
		limit = {
			NOR = {
				has_trait = mount_hua_disciple
				has_trait = wudang_disciple
				has_trait = shaolin_monk
				has_trait = heavenly_demon_disciple
				has_trait = beggar_sect_member
				has_trait = iron_fist_disciple
				has_trait = poison_flower_disciple
				has_trait = phantom_blade_disciple
				has_trait = tang_clan_member
			}
		}
		# Orthodox Clans
		if = {
			limit = { scope:sect_type = flag:mount_hua }
			add_trait = mount_hua_disciple
			set_variable = { name = murim_faction_alignment value = 1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:wudang }
			add_trait = wudang_disciple
			set_variable = { name = murim_faction_alignment value = 1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:shaolin }
			add_trait = shaolin_monk
			set_variable = { name = murim_faction_alignment value = 1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:beggar }
			add_trait = beggar_sect_member
			set_variable = { name = murim_faction_alignment value = 1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:iron_fist }
			add_trait = iron_fist_disciple
			set_variable = { name = murim_faction_alignment value = 1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:tang }
			# Tang Clan non-blood outsiders become associates only
			add_trait = tang_clan_associate
			set_variable = { name = murim_faction_alignment value = 1 }
			# Tang Clan: share alchemy resources with new members
			murim_init_tang_alchemy_effect = yes
		}
		# Unorthodox Clans
		else_if = {
			limit = { scope:sect_type = flag:heavenly_demon }
			add_trait = heavenly_demon_disciple
			set_variable = { name = murim_faction_alignment value = -1 }
		}
		else_if = {
			limit = { scope:sect_type = flag:poison_flower }
			add_trait = poison_flower_disciple
			set_variable = { name = murim_faction_alignment value = -1 }
			# Poison Flower: initiates receive herb knowledge immediately
			murim_init_poison_flower_alchemy_effect = yes
		}
		else_if = {
			limit = { scope:sect_type = flag:phantom_blade }
			add_trait = phantom_blade_disciple
			set_variable = { name = murim_faction_alignment value = -1 }
		}
		# Set rank to outer disciple
		set_variable = { name = murim_clan_rank value = 1 }
		add_character_modifier = { modifier = sect_outer_disciple }
		set_variable = { name = sect_join_year value = current_year }
		add_character_modifier = { modifier = new_sect_member_enthusiasm years = 1 }
		add_character_flag = { flag = recent_sect_joiner }
	}
}

# Tang Clan associate (non-blood allies only)
murim_join_tang_as_associate_effect = {
	if = {
		limit = {
			NOT = { has_trait = tang_clan_member }
			NOT = { has_trait = tang_clan_associate }
		}
		add_trait = tang_clan_associate
		add_character_modifier = { modifier = tang_associate_poison_knowledge years = 5 }
		set_variable = { name = murim_faction_alignment value = 1 }
		set_variable = { name = murim_clan_rank value = 2 }
	}
}

# Tang birthright enrollment (for Tang dynasty bloodlines)
murim_tang_birthright_enrollment_effect = {
	if = {
		limit = {
			NOT = { has_trait = tang_clan_member }
			OR = {
				has_dynasty_flag = tang_imperial_bloodline
				dynasty = { has_dynasty_modifier = tang_dynasty_legacy }
				culture = { has_cultural_pillar = heritage_chinese }
			}
		}
		add_trait = tang_clan_member
		set_variable = { name = murim_faction_alignment value = 1 }
		set_variable = { name = murim_clan_rank value = 3 }
		add_character_flag = tang_bloodline_verified
	}
}

# PROMOTION EFFECTS (all 9 clans)
########################################

# Promote from outer to inner disciple
# murim_clan_rank: 1=outer, 2=associate, 3=inner, 4=elder, 5=vice_leader, 6=leader
murim_promote_to_inner_disciple_effect = {
	if = {
		limit = {
			has_variable = murim_clan_rank
			var:murim_clan_rank <= 2
		}
		remove_character_modifier = sect_outer_disciple
		add_character_modifier = { modifier = sect_inner_disciple }
		set_variable = { name = murim_clan_rank value = 3 }
		add_prestige = 50
		add_character_flag = { flag = murim_promoted_inner }
	}
}

# Promote to elder
murim_promote_to_elder_effect = {
	if = {
		limit = {
			has_variable = murim_clan_rank
			var:murim_clan_rank >= 3
			var:murim_clan_rank < 4
			age >= 30
		}
		remove_character_modifier = sect_inner_disciple
		add_character_modifier = { modifier = sect_elder }
		set_variable = { name = murim_clan_rank value = 4 }
		add_prestige = 150
		add_character_flag = { flag = murim_promoted_elder }
	}
}

# Promote to vice leader
murim_promote_to_vice_leader_effect = {
	if = {
		limit = {
			has_variable = murim_clan_rank
			var:murim_clan_rank >= 4
			var:murim_clan_rank < 5
		}
		remove_character_modifier = sect_elder
		add_character_modifier = { modifier = sect_vice_leader }
		set_variable = { name = murim_clan_rank value = 5 }
		add_prestige = 300
		add_character_flag = { flag = murim_promoted_vice_leader }
	}
}

# Promote to leader
murim_promote_to_leader_effect = {
	if = {
		limit = {
			has_variable = murim_clan_rank
			var:murim_clan_rank >= 4
		}
		remove_character_modifier = sect_vice_leader
		remove_character_modifier = sect_elder
		remove_character_modifier = sect_inner_disciple
		add_character_modifier = { modifier = sect_leader }
		set_variable = { name = murim_clan_rank value = 6 }
		add_prestige = 500
		add_character_flag = { flag = murim_is_clan_leader }
	}
}

########################################
# LEAVE SECT EFFECTS
########################################

murim_leave_sect_effect = {
	# Remove all clan traits
	if = { limit = { has_trait = mount_hua_disciple } remove_trait = mount_hua_disciple }
	if = { limit = { has_trait = wudang_disciple } remove_trait = wudang_disciple }
	if = { limit = { has_trait = shaolin_monk } remove_trait = shaolin_monk }
	if = { limit = { has_trait = heavenly_demon_disciple } remove_trait = heavenly_demon_disciple }
	if = { limit = { has_trait = beggar_sect_member } remove_trait = beggar_sect_member }
	if = { limit = { has_trait = iron_fist_disciple } remove_trait = iron_fist_disciple }
	if = { limit = { has_trait = poison_flower_disciple } remove_trait = poison_flower_disciple }
	if = { limit = { has_trait = phantom_blade_disciple } remove_trait = phantom_blade_disciple }
	if = { limit = { has_trait = tang_clan_associate } remove_trait = tang_clan_associate }
	# Note: tang_clan_member (blood) is NEVER removed - you can't leave your bloodline
	# Remove all rank modifiers
	remove_character_modifier = sect_outer_disciple
	remove_character_modifier = sect_inner_disciple
	remove_character_modifier = sect_elder
	remove_character_modifier = sect_vice_leader
	remove_character_modifier = sect_leader
	remove_character_flag = murim_is_clan_leader
	# Add ex-member status
	add_character_flag = { flag = murim_ex_sect_member }
	set_variable = { name = murim_clan_rank value = 0 }
	# Faction alignment reset
	set_variable = { name = murim_faction_alignment value = 0 }
	# Opinion penalty with former sect members
	add_character_modifier = { modifier = sect_deserter_stigma years = 5 }
}

########################################
# EXPULSION EFFECT (forced removal)
########################################

murim_expel_from_sect_effect = {
	murim_leave_sect_effect = yes
	add_character_modifier = { modifier = sect_expelled_shame years = 10 }
	add_prestige = -300
	add_stress = 30
	add_character_flag = { flag = murim_expelled_from_sect }
}

########################################
# EMERGENCY RECRUITMENT (clan youth < 3)
########################################

murim_emergency_recruitment_effect = {
	# When a clan has fewer than 3 members under 30, trigger mass recruitment
	# Scope: should be called from on_actions with clan flag checks
	random_living_character = {
		limit = {
			OR = {
				culture = { has_cultural_pillar = heritage_chinese }
				culture = { has_cultural_pillar = heritage_korean }
			}
			age >= 14
			age <= 25
			prowess >= 8
			is_alive = yes
			NOR = {
				has_trait = mount_hua_disciple
				has_trait = wudang_disciple
				has_trait = shaolin_monk
				has_trait = heavenly_demon_disciple
				has_trait = beggar_sect_member
				has_trait = iron_fist_disciple
				has_trait = poison_flower_disciple
				has_trait = phantom_blade_disciple
				has_trait = tang_clan_member
			}
		}
		weight = {
			base = 10
			modifier = { add = 20 prowess >= 12 }
			modifier = { add = 10 has_trait = brave }
			modifier = { add = 10 has_trait = ambitious }
			modifier = { add = 15 has_trait = martial_artist }
		}
		save_scope_as = emergency_recruit
	}
	if = {
		limit = { exists = scope:emergency_recruit }
		scope:emergency_recruit = {
			murim_join_sect_as_outer_disciple_effect = yes
			add_character_flag = { flag = emergency_recruited }
			trigger_event = { id = murim_sect.0050 days = { 5 15 } }
		}
	}
}

########################################
# UTILITY: DETERMINE FACTION
########################################

murim_determine_faction_effect = {
	# Sets murim_is_orthodox or murim_is_unorthodox flag
	if = {
		limit = {
			OR = {
				has_trait = mount_hua_disciple
				has_trait = wudang_disciple
				has_trait = shaolin_monk
				has_trait = beggar_sect_member
				has_trait = iron_fist_disciple
				has_trait = tang_clan_member
				has_trait = tang_clan_associate
			}
		}
		add_character_flag = { flag = murim_is_orthodox }
		remove_character_flag = murim_is_unorthodox
	}
	else_if = {
		limit = {
			OR = {
				has_trait = heavenly_demon_disciple
				has_trait = poison_flower_disciple
				has_trait = phantom_blade_disciple
			}
		}
		add_character_flag = { flag = murim_is_unorthodox }
		remove_character_flag = murim_is_orthodox
	}
}
