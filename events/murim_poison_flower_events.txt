# Murim Chronicles - Poison Flower Palace Clan Events
# UNORTHODOX. All-female sect (with rare male exceptions). Masters of poison, seduction, and assassination.
# SIGNATURE: Thousand Poison Hand, Flower Petal Blade Dance, Heart Devouring Gu, Bewitching Arts
# PHILOSOPHY: Beauty is the deadliest weapon. Control through desire and fear.
# INTERNAL POLITICS: Palace Mistress vs Elder Council. Pure poison path vs seduction path rivalry.
# UNIQUE: Only clan with gu (poison insect) cultivation. Gender-locked recruitment (mostly).
# INTEGRATION: murim_poison_arts_events.txt, murim_romance_events.txt, murim_shadow_politics_events.txt

##############################
# EVENT 1: INITIATION - THE GARDEN OF TEN THOUSAND POISONS
# New disciples must walk through a garden filled with poisonous plants.
# Those who survive develop basic poison resistance.
##############################
murim_poison_flower.0001 = {
	type = character_event
	title = murim_poison_flower.0001.t
	desc = murim_poison_flower.0001.desc
	theme = intrigue
	
	trigger = {
		is_poison_flower_disciple_trigger = yes
		NOT = { has_character_flag = poison_flower_garden_trial }
		age >= 14
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			learning >= 12
		}
	}
	
	immediate = {
		set_character_flag = poison_flower_garden_trial
		create_character_memory = {
			type = poison_flower_garden_trial_memory
		}
	}
	
	# Option A: Walk through slowly, building resistance naturally
	option = {
		name = murim_poison_flower.0001.a
		random_list = {
			50 = {
				add_character_modifier = {
					modifier = poison_flower_basic_resistance_modifier
					years = -1
				}
				add_intrigue_xp = 300
			}
			30 = {
				# Partial resistance - some poisons still affect you
				add_character_modifier = {
					modifier = poison_flower_partial_resistance_modifier
					years = 5
				}
				add_trait = wounded_1
			}
			20 = {
				# Severe poisoning - bedridden for months
				add_trait = wounded_2
				add_character_modifier = {
					modifier = poison_flower_chronic_poisoning_modifier
					years = 3
				}
				create_character_memory = {
					type = poison_flower_garden_collapse_memory
				}
			}
		}
		ai_chance = { base = 45 }
	}
	
	# Option B: Use a secret antidote prepared beforehand (cheating)
	option = {
		name = murim_poison_flower.0001.b
		trigger = {
			OR = {
				has_trait = deceitful
				learning >= 14
			}
		}
		random_list = {
			60 = {
				add_character_modifier = {
					modifier = poison_flower_artificial_resistance_modifier
					years = 5
				}
				add_intrigue_xp = 200
			}
			40 = {
				# Elder detects the antidote - punishment
				add_character_modifier = {
					modifier = poison_flower_cheater_mark_modifier
					years = 5
				}
				every_courtier = {
					limit = {
						is_poison_flower_disciple_trigger = yes
					}
					add_opinion = {
						modifier = poison_flower_cheater_opinion
						target = root
						opinion = -20
					}
				}
			}
		}
		ai_chance = { base = 25 }
	}
	
	# Option C: Deliberately inhale the strongest poisons (extreme path)
	option = {
		name = murim_poison_flower.0001.c
		trigger = {
			OR = {
				has_trait = brave
				has_trait = ambitious
			}
		}
		random_list = {
			25 = {
				# Supreme resistance achieved immediately
				add_character_modifier = {
					modifier = poison_flower_supreme_resistance_modifier
					years = -1
				}
				add_intrigue_xp = 500
				add_prowess_xp = 300
			}
			35 = {
				# Near death but transformed
				add_trait = wounded_2
				add_character_modifier = {
					modifier = poison_flower_poison_body_modifier
					years = -1
				}
				# Your very touch becomes poisonous
			}
			40 = {
				# Nearly die - permanent damage
				add_trait = wounded_3
				add_trait = incapable
				create_character_memory = {
					type = poison_flower_near_death_poisoning_memory
				}
			}
		}
		ai_chance = { base = 15 }
	}
}
