# YOUNG DRAGON TOURNAMENT (every 5 years)
# For martial artists under 30.
# Global recruitment pipeline - clan scouts
# invite promising outsiders.
############################################

murim_tournament.0001 = {
	type = character_event
	title = murim_tournament.0001.t
	desc = murim_tournament.0001.desc
	theme = martial
	override_background = { reference = tournament_grounds }

	trigger = {
		murim_is_martial_artist_trigger = yes
		age >= 14
		age <= 30
		is_alive = yes
		NOT = { has_character_modifier = murim_young_dragon_cooldown }
	}

	immediate = {
		# Find opponent from different clan
		random_living_character = {
			limit = {
				has_trait = martial_artist
				age >= 14
				age <= 30
				is_alive = yes
				NOT = { this = root }
				NOT = { has_character_modifier = murim_young_dragon_cooldown }
			}
			weight = {
				base = 10
				modifier = { add = 20 prowess >= 14 }
				modifier = { add = 15 OR = { has_trait = mount_hua_disciple has_trait = shaolin_monk has_trait = heavenly_demon_disciple } }
			}
			save_scope_as = tournament_opponent
		}
	}

	# Fight with full power
	option = {
		name = murim_tournament.0001.a
		trigger = { exists = scope:tournament_opponent }
		duel = {
			skill = prowess
			target = scope:tournament_opponent
			50 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_tournament.0001.a.victory
				# Victory - advance to next round
				add_prestige = 100
				add_character_modifier = { modifier = murim_young_dragon_victor years = 5 }
				add_character_flag = { flag = murim_young_dragon_winner }
				# Memory of glorious victory
				add_character_flag = { flag = murim_tournament_memory_win }
				# Clan scouts notice your prowess
				if = {
					limit = {
						NOR = {
							has_trait = mount_hua_disciple
							has_trait = wudang_disciple
							has_trait = shaolin_monk
							has_trait = heavenly_demon_disciple
							has_trait = beggar_sect_member
							has_trait = iron_fist_disciple
							has_trait = poison_flower_disciple
							has_trait = phantom_blade_disciple
							has_trait = tang_clan_member
						}
					}
					# Unaffiliated winner gets scouted
					trigger_event = { id = murim_tournament.0010 days = { 15 30 } }
				}
				trigger_event = { id = murim_tournament.0002 days = { 3 7 } }
			}
			50 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:tournament_opponent }
				desc = murim_tournament.0001.a.defeat
				# Defeat - eliminated from tournament
				add_prestige = 25
				add_stress = 10
				add_character_flag = { flag = murim_tournament_memory_loss }
				# Injuries possible
				random_list = {
					60 = { }
					25 = { add_trait = wounded_1 }
					10 = { add_trait = wounded_2 }
					5 = { add_trait = murim_scarred }
				}
			}
		}
		add_character_modifier = { modifier = murim_young_dragon_cooldown years = 5 }
		stress_impact = { brave = minor_stress_impact_loss craven = major_stress_impact_gain }
		ai_chance = { base = 70 modifier = { add = 20 has_trait = brave } modifier = { add = -40 has_trait = craven } }
	}

	# Forfeit - preserve your body
	option = {
		name = murim_tournament.0001.b
		add_prestige = -50
		add_character_modifier = { modifier = murim_young_dragon_cooldown years = 5 }
		add_character_flag = { flag = murim_tournament_forfeit }
		ai_chance = { base = 30 modifier = { add = 40 has_trait = craven } }
	}
}

# GRAND TOURNAMENT (every 10 years)
# Open to all martial artists. No age limit.
# Orthodox vs Unorthodox faction tension.
############################################

murim_tournament.0004 = {
	type = character_event
	title = murim_tournament.0004.t
	desc = murim_tournament.0004.desc
	theme = martial
	override_background = { reference = tournament_grounds }

	trigger = {
		murim_is_martial_artist_trigger = yes
		is_alive = yes
		prowess >= 10
		NOT = { has_character_modifier = murim_grand_tournament_cooldown }
	}

	immediate = {
		random_living_character = {
			limit = {
				has_trait = martial_artist
				prowess >= 12
				is_alive = yes
				NOT = { this = root }
			}
			weight = {
				base = 10
				modifier = { add = 30 prowess >= 20 }
				modifier = { add = 20 OR = { has_trait = heavenly_demon_disciple has_trait = mount_hua_disciple has_trait = shaolin_monk } }
			}
			save_scope_as = grand_opponent
		}
	}

	# Enter the Grand Tournament
	option = {
		name = murim_tournament.0004.a
		trigger = { exists = scope:grand_opponent }
		duel = {
			skill = prowess
			target = scope:grand_opponent
			50 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_tournament.0004.a.victory
				add_prestige = 150
				add_character_flag = { flag = murim_grand_tournament_advancing }
				# Check for faction clash
				if = {
					limit = {
						has_character_flag = murim_is_orthodox
						scope:grand_opponent = { has_character_flag = murim_is_unorthodox }
					}
					add_character_flag = { flag = murim_orthodox_defeated_unorthodox }
					add_prestige = 50
				}
				else_if = {
					limit = {
						has_character_flag = murim_is_unorthodox
						scope:grand_opponent = { has_character_flag = murim_is_orthodox }
					}
					add_character_flag = { flag = murim_unorthodox_defeated_orthodox }
					add_prestige = 50
				}
				trigger_event = { id = murim_tournament.0005 days = { 3 7 } }
			}
			50 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:grand_opponent }
				desc = murim_tournament.0004.a.defeat
				add_prestige = 50
				add_stress = 15
				random_list = {
					50 = { }
					25 = { add_trait = wounded_1 }
					15 = { add_trait = wounded_2 }
					10 = { add_trait = murim_scarred }
				}
			}
		}
		add_character_modifier = { modifier = murim_grand_tournament_cooldown years = 10 }
		ai_chance = { base = 70 }
	}

	# Observe only
	option = {
		name = murim_tournament.0004.b
		add_character_modifier = { modifier = murim_grand_tournament_cooldown years = 10 }
		add_prestige = 10
		ai_chance = { base = 30 }
	}
}
