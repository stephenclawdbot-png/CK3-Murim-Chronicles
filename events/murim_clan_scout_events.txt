# Murim Chronicles - Clan Scout Travel Events
# Events that fire as scouts traverse the map recruiting for their clans.
# CHINA-CENTRIC: All scouts depart from and return to China (Song Dynasty region).
# VANILLA SEPARATION: Scout travel is martial-layer only. Does NOT trigger political events.
# INTEGRATION: Uses murim_clan_scout_effects.txt (scout spawning, evaluation, return),
#              murim_clan_recruitment_events.txt (dual-mode recruitment pipeline),
#              murim_discovery_triggers.txt (candidate eligibility),
#              murim_clan_triggers.txt (clan membership checks),
#              murim_clan_membership_effects.txt (join_sect_as_outer_disciple_effect)
# GENDER: All candidates equally evaluated regardless of sex.

namespace = murim_scout

# ===========================
# murim_scout.0001 — Scout Arrives at New Region
# Fires each year as scout advances their travel route.
# Determines region type and triggers appropriate sub-events.
# ===========================

murim_scout.0001 = {
    type = character_event
    title = murim_scout.0001.t
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = { var:scout_route_stage = 1 }
                desc = murim_scout.0001.desc.goryeo
                # "You cross the border into Goryeo. The martial traditions here are strong,
                #  cousins to those of the Middle Kingdom. Students practice in temple courtyards
                #  and village squares. There may be talent worth bringing home."
            }
            triggered_desc = {
                trigger = { var:scout_route_stage = 2 }
                desc = murim_scout.0001.desc.japan
                # "The island nation across the sea. Their sword arts are refined but different —
                #  shorter blades, tighter footwork. You observe from the shadows, looking for
                #  anyone whose talent transcends their local style."
            }
            triggered_desc = {
                trigger = { var:scout_route_stage = 3 }
                desc = murim_scout.0001.desc.india
                # "The subcontinent stretches before you. Warriors here fight with flowing movements,
                #  grappling holds, and internal breathing techniques unknown in the Middle Kingdom.
                #  Different, but sometimes different is exactly what the clan needs."
            }
            triggered_desc = {
                trigger = { var:scout_route_stage = 4 }
                desc = murim_scout.0001.desc.persia
                # "Persian warriors blend horsemanship with blade work. Their court martial traditions
                #  are ancient and proud. You must tread carefully — these people do not welcome
                #  foreign meddling in their youth."
            }
            triggered_desc = {
                trigger = { var:scout_route_stage >= 5 }
                desc = murim_scout.0001.desc.frontier
                # "You have traveled far from the Middle Kingdom. These lands are wild, their fighting
                #  styles raw and unrefined. But raw talent is still talent. A diamond in the rough
                #  could become the clan's greatest asset."
            }
        }
    }
    theme = martial
    
    # AI auto-handles: scout evaluates local population
    
    option = {
        name = murim_scout.0001.a
        # "Survey the local courts for talent"
        trigger = {
            # Find all eligible candidates in the region
            any_ruler = {
                any_courtier = {
                    murim_is_recruitable_candidate_trigger = yes
                    age >= 8
                    age <= 40
                }
            }
        }
        # Pick the best candidate and evaluate
        random_ruler = {
            limit = {
                any_courtier = {
                    murim_is_recruitable_candidate_trigger = yes
                }
            }
            random_courtier = {
                limit = { murim_is_recruitable_candidate_trigger = yes }
                weight = {
                    base = 1
                    modifier = { add = 5 has_trait = genius }
                    modifier = { add = 3 has_trait = brave }
                    modifier = { add = 3 has_trait = strong }
                    modifier = { add = 2 prowess >= 10 }
                    modifier = { add = 5 has_trait = martial_awareness }
                }
                save_scope_as = potential_recruit
            }
        }
        root = {
            set_variable = { name = candidate_score value = 0 }
            murim_scout_evaluate_candidate_effect = yes
        }
        # If score high enough, trigger recruitment pitch
        if = {
            limit = { var:candidate_score >= 8 }
            trigger_event = { id = murim_scout.0002 days = { 15 30 } }
        }
        # Otherwise scout moves on
        else = {
            trigger_event = { id = murim_scout.0008 days = { 30 60 } }    # No worthy candidates
        }
        ai_chance = { base = 80 }
    }
    
    option = {
        name = murim_scout.0001.b
        # "Focus on gathering intelligence about local martial arts"
        # Scout doesn't recruit but learns about foreign styles
        root = {
            add_character_modifier = {
                modifier = murim_foreign_style_knowledge
                years = 10
            }
            change_variable = { name = scout_intelligence_gathered add = 1 }
        }
        ai_chance = { base = 20 }
    }
    
    option = {
        name = murim_scout.0001.c
        # "This region is too dangerous. Move on quickly."
        trigger = {
            # Hostile region or at war
            location = {
                holder = {
                    is_at_war = yes
                }
            }
        }
        # Skip to next route stage early
        root = { murim_advance_scout_route_effect = yes }
        ai_chance = {
            base = 10
            modifier = { add = 30 has_trait = craven }
        }
    }
}

# ===========================
# murim_scout.0002 — Scout Evaluates Candidate (Detailed Assessment)
# Scout has found someone interesting. Detailed evaluation before approach.
# ===========================

murim_scout.0002 = {
    type = character_event
    title = murim_scout.0002.t
    desc = murim_scout.0002.desc
    # "You have been observing [potential_recruit.GetFirstName] for several days.
    #  [They are/She is/He is] [age] years old, [traits description].
    #  Your trained eye sees [high/moderate/extraordinary] potential.
    #  [Clan-specific assessment based on scout preferences]."
    theme = martial
    
    option = {
        name = murim_scout.0002.a
        # "Approach them directly with an offer"
        # Standard recruitment pitch — fires parent notification if child, direct offer if adult
        if = {
            limit = { scope:potential_recruit = { age < 16 } }
            # Child — must approach parent
            scope:potential_recruit.father = {
                trigger_event = { id = murim_scout.0003 days = { 3 7 } }
            }
            # Fallback to mother if no father
            if = {
                limit = { scope:potential_recruit = { NOT = { exists = father } } }
                scope:potential_recruit.mother = {
                    trigger_event = { id = murim_scout.0003 days = { 3 7 } }
                }
            }
            # Fallback to liege if orphan
            if = {
                limit = {
                    scope:potential_recruit = {
                        NOT = { exists = father }
                        NOT = { exists = mother }
                    }
                }
                scope:potential_recruit.host = {
                    trigger_event = { id = murim_scout.0003 days = { 3 7 } }
                }
            }
        }
        else = {
            # Adult — direct offer
            scope:potential_recruit = {
                trigger_event = { id = murim_scout.0004 days = { 3 7 } }
            }
        }
        ai_chance = { base = 70 }
    }
    
    option = {
        name = murim_scout.0002.b
        # "Test them first — arrange a trial"
        # Set up a combat or skill challenge before making an offer
        scope:potential_recruit = {
            trigger_event = { id = murim_scout.0005 days = { 5 15 } }
        }
        ai_chance = { base = 20 }
    }
    
    option = {
        name = murim_scout.0002.c
        # "They're not quite right. Keep searching."
        # Scout passes on this candidate
        root = {
            change_variable = { name = scout_recruits_found add = 0 }    # No change
        }
        ai_chance = {
            base = 10
            modifier = { add = 20 var:candidate_score < 10 }
        }
    }
}

# ===========================
# murim_scout.0003 — Parent Notification (Child Recruitment)
# A clan scout wants to take your child. Player event if parent is player.
# ===========================

murim_scout.0003 = {
    type = character_event
    title = murim_scout.0003.t
    desc = murim_scout.0003.desc
    # "A [mysterious stranger from the East / martial sect representative] has arrived at your court.
    #  [They] represent [Clan Name], one of the great martial clans of the Middle Kingdom.
    #  [They] wish to take your [son/daughter] [Child Name] to train in [Region].
    #  'Your child has talent,' [they] say. 'Rare talent. We would hone it into something extraordinary.
    #  In return, [Child Name] would join our clan, learn our ways, and become part of something
    #  far greater than any single court or kingdom.'"
    theme = martial
    
    # Option A: Accept proudly
    option = {
        name = murim_scout.0003.a
        # "Go, my child. Make us proud."
        scope:potential_recruit = {
            add_trait = murim_outer_disciple
            add_character_modifier = {
                modifier = murim_recruited_by_clan
                years = 100
            }
            set_variable = { name = recruitment_origin value = root.primary_title }
            set_variable = { name = recruited_by_clan value = scope:scout.var:scout_clan }
            # Move to China — clan HQ region
            # VANILLA SEPARATION: This is a martial-layer move. Does NOT affect titles or succession.
            set_variable = { name = murim_relocated_to_china value = yes }
        }
        root = {
            add_prestige = 100    # Honor that your child was chosen
            create_character_memory = {
                type = murim_child_recruited_memory
                participants = { recruit = scope:potential_recruit }
            }
        }
        scope:scout = {
            change_variable = { name = scout_recruits_found add = 1 }
        }
        ai_chance = {
            base = 30
            modifier = { add = 20 has_trait = ambitious }
            modifier = { add = 10 num_of_children >= 3 }
            modifier = { add = 15 opinion = { target = scope:scout value >= 20 } }
            modifier = { add = -20 has_trait = paranoid }
            modifier = { add = -10 num_of_children <= 1 }
        }
    }
    
    # Option B: Accept reluctantly
    option = {
        name = murim_scout.0003.b
        # "I... suppose this is for the best. Take care of [them]."
        scope:potential_recruit = {
            add_trait = murim_outer_disciple
            add_character_modifier = {
                modifier = murim_recruited_by_clan
                years = 100
            }
            add_character_modifier = {
                modifier = murim_homesick
                years = 5
            }
            set_variable = { name = murim_relocated_to_china value = yes }
            set_variable = { name = recruited_by_clan value = scope:scout.var:scout_clan }
        }
        root = {
            add_stress = 30
            add_prestige = 50
            create_character_memory = {
                type = murim_child_recruited_reluctant_memory
                participants = { recruit = scope:potential_recruit }
            }
        }
        scope:scout = {
            change_variable = { name = scout_recruits_found add = 1 }
        }
        ai_chance = {
            base = 25
            modifier = { add = 15 has_trait = compassionate }
        }
    }
    
    # Option C: Negotiate terms
    option = {
        name = murim_scout.0003.c
        # "My child's talent has a price. What does your clan offer?"
        # Demand gold/favor/alliance in exchange
        trigger_event = { id = murim_scout.0006 days = { 3 7 } }    # Negotiation chain
        ai_chance = {
            base = 20
            modifier = { add = 20 has_trait = greedy }
            modifier = { add = 10 has_trait = ambitious }
        }
    }
    
    # Option D: Refuse
    option = {
        name = murim_scout.0003.d
        # "No. My child stays with me."
        # Scout leaves. 10% chance child runs away to join anyway.
        random = {
            chance = 10
            scope:potential_recruit = {
                trigger_event = { id = murim_scout.0007 days = { 60 180 } }    # Child runs away
            }
        }
        root = {
            create_character_memory = {
                type = murim_refused_clan_recruitment_memory
                participants = { recruit = scope:potential_recruit }
            }
        }
        # Child may resent parent later
        scope:potential_recruit = {
            random = {
                chance = 30
                add_opinion = {
                    target = root
                    modifier = murim_resents_denied_opportunity
                    opinion = -15
                }
            }
        }
        ai_chance = {
            base = 15
            modifier = { add = 20 has_trait = paranoid }
            modifier = { add = 10 num_of_children <= 1 }
            modifier = { add = -10 has_trait = ambitious }
        }
    }
    
    # Option E: "Over my dead body" — hostile refusal
    option = {
        name = murim_scout.0003.e
        # "Get out. If I see your kind near my children again, I'll have you executed."
        root = {
            add_opinion = {
                target = scope:scout
                modifier = murim_hostile_to_clan
                opinion = -50
            }
        }
        scope:scout = {
            add_opinion = {
                target = root
                modifier = murim_hostile_refusal_received
                opinion = -30
            }
            # Blacklist this ruler — won't approach again
            set_variable = {
                name = murim_blacklisted_ruler
                value = root
            }
        }
        ai_chance = {
            base = 10
            modifier = { add = 20 has_trait = wrathful }
            modifier = { add = 15 has_trait = paranoid }
            modifier = { add = -20 has_trait = calm }
        }
    }
}

# murim_scout.0005 — Recruitment Trial
# Scout tests candidate before making formal offer.
# ===========================

murim_scout.0005 = {
    type = character_event
    title = murim_scout.0005.t
    desc = murim_scout.0005.desc
    # "The stranger does not introduce themselves. Instead, they throw a wooden sword at your feet.
    #  'Show me,' they say simply. [If child: Your parents watch nervously from the doorway.]
    #  [Clan-specific trial: Mount Hua = form test, Shaolin = endurance, Heavenly Demon = survival,
    #   Tang = poison identification, Wudang = puzzle, Beggar = street survival]"
    theme = martial
    
    option = {
        name = murim_scout.0005.a
        # "Accept the trial with confidence"
        # Prowess check determines result
        skill_check = {
            skill = prowess
            value = { 10 15 }    # Adjusted for age
            success_desc = murim_scout.0005.a.success
            failure_desc = murim_scout.0005.a.failure
        }
        # On success: scout is impressed, makes offer
        # On failure: scout may still offer (partial success) or leave
        ai_chance = {
            base = 60
            modifier = { add = 20 has_trait = brave }
        }
    }
    
    option = {
        name = murim_scout.0005.b
        # "Refuse the trial — 'I don't perform tricks for strangers'"
        root = {
            add_prestige = 10    # Self-respect
        }
        # Scout is either impressed by the backbone or offended
        random = {
            chance = 40
            # Impressed — makes offer anyway
            trigger_event = { id = murim_scout.0004 days = { 1 3 } }
        }
        ai_chance = {
            base = 20
            modifier = { add = 15 has_trait = arrogant }
        }
    }
    
    option = {
        name = murim_scout.0005.c
        # "Ask questions first — 'Who are you? What clan? Why me?'"
        # Information gathering — smart candidates ask questions
        root = {
            if = {
                limit = { learning >= 10 }
                # Smart candidate gets full info
                add_character_modifier = {
                    modifier = murim_informed_recruit
                    years = 5
                }
            }
        }
        # Then proceed to trial
        trigger_event = { id = murim_scout.0005 days = { 1 3 } }    # Loop back with more info
        ai_chance = {
            base = 20
            modifier = { add = 20 has_trait = patient }
            modifier = { add = 15 learning >= 12 }
        }
    }
}
