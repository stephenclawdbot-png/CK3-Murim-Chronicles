murim_shaolin_integration.0001 = {
	type = character_event
	title = murim_shaolin_integration.0001.t
	desc = murim_shaolin_integration.0001.desc
	theme = faith
	left_portrait = { character = root animation = personality_zealous }

	trigger = {
		has_variable = murim_martial_artist
		has_trait = murim_shaolin_monk
		NOT = { has_character_flag = murim_shaolin_integration_done }
	}

	immediate = {
		set_character_flag = { flag = murim_shaolin_integration_done years = 5 }
	}

	# Open the Shaolin Martial Hall to all sects for joint training
	option = {
		name = murim_shaolin_integration.0001.a
		add_piety = 200
		add_prestige = 150
		change_variable = { name = murim_cultivation_xp add = 35 }
		hidden_effect = {
			every_ruler = {
				limit = { has_variable = murim_martial_artist }
				add_opinion = { target = root modifier = grateful_opinion opinion = 10 }
			}
		}
		ai_chance = {
			base = 45
			modifier = { add = 25 has_trait = compassionate }
		}
	}

	# Attempt the Eighteen Arhat Formation alone
	option = {
		name = murim_shaolin_integration.0001.b
		add_prowess = 3
		add_martial = 1
		change_variable = { name = murim_cultivation_xp add = 75 }
		add_stress = 20
		ai_chance = {
			base = 30
			modifier = { add = 25 prowess >= 20 }
		}
	}

	# Meditate on the Diamond Sutra for inner peace
	option = {
		name = murim_shaolin_integration.0001.c
		add_learning = 2
		add_piety = 100
		add_stress = -30
		change_variable = { name = murim_cultivation_xp add = 50 }
		ai_chance = {
			base = 25
			modifier = { add = 20 learning >= 16 }
		}
	}
}
