# Romance blossoms between martial artists
murim_romance_integration.0001 = {
	type = character_event
	title = murim_romance_integration.0001.t
	desc = murim_romance_integration.0001.desc
	theme = seduction
	left_portrait = { character = root animation = personality_compassionate }

	trigger = {
		has_variable = murim_martial_artist
		is_married = no
		age >= 16
	}

	# Pursue the relationship — love transcends faction
	option = {
		name = murim_romance_integration.0001.a
		add_stress = -20
		add_prestige = 50
		change_variable = { name = murim_cultivation_xp add = 15 }
		ai_chance = {
			base = 45
			modifier = { add = 20 has_trait = lustful }
		}
	}

	# Reject romance — cultivation demands discipline
	option = {
		name = murim_romance_integration.0001.b
		change_variable = { name = murim_cultivation_xp add = 40 }
		add_stress = 15
		ai_chance = {
			base = 35
			modifier = { add = 20 has_trait = chaste }
		}
	}

	# Keep it secret
	option = {
		name = murim_romance_integration.0001.c
		add_intrigue = 1
		add_stress = -10
		ai_chance = { base = 20 }
	}
}
