### EVENT 0002 — The Palace Mistress Selection
# Competition to choose the next Poison Flower Palace Mistress
murim_poison_flower_expansion2.0002 = {
	type = character_event
	title = murim_poison_flower_expansion2.0002.t
	desc = murim_poison_flower_expansion2.0002.desc
	theme = intrigue
	
	left_portrait = {
		character = root
		animation = scheme
	}
	right_portrait = {
		character = scope:palace_rival
		animation = personality_bold
	}
	
	trigger = {
		has_trait = murim_poison_flower_clan
		has_trait = murim_martial_artist
		is_female = yes
		intrigue >= 12
		age >= 25
		NOT = { has_character_flag = palace_mistress_selection }
	}
	
	immediate = {
		set_character_flag = palace_mistress_selection
		random_living_character = {
			limit = {
				has_trait = murim_poison_flower_clan
				is_female = yes
				NOT = { this = root }
				age >= 20
			}
			weight = {
				base = 1
				modifier = { add = 5 intrigue >= 14 }
				modifier = { add = 3 has_trait = ambitious }
			}
			save_scope_as = palace_rival
		}
	}
	
	# Option A: Compete through poison mastery demonstration
	option = {
		name = murim_poison_flower_expansion2.0002.a
		duel = {
			skill = intrigue
			target = scope:palace_rival
			50 = {
				desc = murim_poison_flower_expansion2.0002.a.win
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				add_prestige = 600
				add_character_modifier = {
					modifier = murim_palace_mistress_candidate
					years = 20
				}
			}
			50 = {
				desc = murim_poison_flower_expansion2.0002.a.lose
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				add_prestige = -200
				add_stress = 25
			}
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Undermine the rival with gossip and manipulation
	option = {
		name = murim_poison_flower_expansion2.0002.b
		add_intrigue_skill = 3
		scope:palace_rival = {
			add_opinion = {
				target = root
				modifier = murim_palace_scheming
				opinion = -40
			}
		}
		random_list = {
			50 = { add_prestige = 400 custom_tooltip = murim_rival_discredited_tt }
			50 = { add_prestige = -100 custom_tooltip = murim_scheme_exposed_tt }
		}
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = deceitful }
		}
	}
	
	# Option C: Support the rival and become her advisor
	option = {
		name = murim_poison_flower_expansion2.0002.c
		scope:palace_rival = {
			add_opinion = {
				target = root
				modifier = murim_loyal_palace_sister
				opinion = 40
			}
		}
		add_character_modifier = {

			years = 20
		}
