### EVENT 1: The Blood Contract ###
murim_phantom_blade_exp5.0001 = {
	type = character_event
	title = murim_phantom_blade_exp5.0001.t
	desc = murim_phantom_blade_exp5.0001.desc
	theme = martial_arts
	
	trigger = {
		has_trait = phantom_blade_disciple
		martial >= 18
		has_variable = phantom_blade_rank
		var:phantom_blade_rank >= 3
		NOT = { has_character_flag = blood_contract_taken }
		age >= 20
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = assassin_trained
		}
		modifier = {
			add = 3
			has_trait = blade_master
		}
	}
	
	# Option A: Accept the Blood Contract - become elite assassin
	option = {
		name = murim_phantom_blade_exp5.0001.a
		add_character_flag = {
			flag = blood_contract_taken
			days = -1
		}
		add_trait = blood_oath_assassin
		add_martial_skill = 3
		add_intrigue_skill = 4
		add_dread = 30
		add_prestige = 500
		stress_impact = {
			compassionate = major_stress_impact_gain
			honest = medium_stress_impact_gain
		}
		hidden_effect = {
			set_variable = {
				name = assassination_contracts_completed
				value = 0
			}
			set_variable = {
				name = blood_contract_tier
				value = 1
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				has_trait = sadistic
			}
		}
	}
	
	# Option B: Refuse and face consequences
	option = {
		name = murim_phantom_blade_exp5.0001.b
		add_character_flag = {
			flag = refused_blood_contract
			days = 365
		}
		add_prestige = -200
		add_stress = 20
		reverse_add_opinion = {
			target = scope:phantom_blade_leader
			modifier = disobedient_blade_opinion
			opinion = -40
		}
		hidden_effect = {
			trigger_event = {
				id = murim_phantom_blade_exp5.0003
				days = { 30 90 }
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 40
				has_trait = compassionate
			}
		}
	}
	
	# Option C: Negotiate modified terms
	option = {
		name = murim_phantom_blade_exp5.0001.c
		trigger = {
			diplomacy >= 15
			OR = {
				has_trait = diplomat
				has_trait = schemer
			}
		}
		add_character_flag = {
			flag = modified_blood_contract
			days = -1
		}
		add_intrigue_skill = 2
		add_prestige = 200
		add_stress = 10
		hidden_effect = {
			set_variable = {
				name = blood_contract_tier
				value = 0
			}
		}
		ai_chance = {
			base = 20
		}
	}
	
	# Option D: Challenge the contract giver for leadership
	option = {
		name = murim_phantom_blade_exp5.0001.d
		trigger = {
			martial >= 25
			has_trait = blade_master
			prowess >= 20
		}
		duel = {
			skill = prowess
			target = scope:phantom_blade_leader
			50 = {
				desc = murim_phantom_blade_exp5.0001.d.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				add_prestige = 1000
				add_trait = phantom_blade_grandmaster
				remove_trait = phantom_blade_disciple
				add_dread = 50
				set_variable = {
					name = phantom_blade_rank
					value = 10
				}
			}
			50 = {
				desc = murim_phantom_blade_exp5.0001.d.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				add_prestige = -500
				increase_wounds_effect = { reason = duel }
				add_stress = 40
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = ambitious
			}
		}
	}
}

### EVENT 2: First Assassination Target ###
murim_phantom_blade_exp5.0002 = {
	type = character_event
	title = murim_phantom_blade_exp5.0002.t
	desc = murim_phantom_blade_exp5.0002.desc
	theme = intrigue
	
	trigger = {
		has_character_flag = blood_contract_taken
		has_variable = assassination_contracts_completed
		var:assassination_contracts_completed < 3
		NOT = { has_character_flag = on_active_contract }
	}
	
	weight_multiplier = {
		base = 2
		modifier = {
			add = 3
			var:blood_contract_tier >= 1
		}
	}
	
	immediate = {
		random_ruler = {
			limit = {
				NOT = { this = root }
				NOT = { has_trait = phantom_blade_disciple }
				martial >= 10
				is_alive = yes
			}
			save_scope_as = assassination_target
		}
	}
	
	# Option A: Strike from the shadows - stealth kill
	option = {
		name = murim_phantom_blade_exp5.0002.a
		add_character_flag = {
			flag = on_active_contract
			days = 60
		}
		duel = {
			skill = intrigue
			target = scope:assassination_target
			60 = {
				desc = murim_phantom_blade_exp5.0002.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
					min = -49
				}
				scope:assassination_target = {
					death = {
						death_reason = death_murder_known
						killer = root
					}
				}
				add_prestige = 800
				add_gold = 200
				change_variable = {
					name = assassination_contracts_completed
					add = 1
				}
				add_trait = shadow_killer
			}
			40 = {
				desc = murim_phantom_blade_exp5.0002.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
					min = -49
				}
				add_prestige = -300
				increase_wounds_effect = { reason = assassination_failed }
				add_character_flag = {
					flag = known_assassin
					days = 730
				}
			}
		}
		ai_chance = {
			base = 50
		}
	}
	
	# Option B: Poison approach - subtle kill
	option = {
		name = murim_phantom_blade_exp5.0002.b
		trigger = {
			learning >= 12
		}
		add_character_flag = {
			flag = on_active_contract
			days = 90
		}
		hidden_effect = {
			random_list = {
				70 = {
					modifier = {
						add = 20
						intrigue >= 20
					}
					scope:assassination_target = {
						death = {
							death_reason = death_murder_unknown
							killer = root
						}
					}
					root = {
						add_prestige = 600
						add_gold = 300
						change_variable = {
							name = assassination_contracts_completed
							add = 1
						}
					}
				}
				30 = {
					root = {
						add_prestige = -200
						add_stress = 30
					}
				}
			}
		}
		ai_chance = {
			base = 30
		}
	}
	
	# Option C: Challenge openly - honorable kill
	option = {
		name = murim_phantom_blade_exp5.0002.c
		trigger = {
			has_trait = honorable
		}
		duel = {
			skill = martial
			target = scope:assassination_target
			50 = {
				desc = murim_phantom_blade_exp5.0002.c.success
				scope:assassination_target = {
					death = {
						death_reason = death_duel
						killer = root
					}
				}
				add_prestige = 500
				add_gold = 100
				change_variable = {
					name = assassination_contracts_completed
					add = 1
				}
				add_piety = 100
			}
			50 = {
				desc = murim_phantom_blade_exp5.0002.c.failure
				increase_wounds_effect = { reason = duel }
				add_prestige = -200
			}
		}
		ai_chance = {
			base = 20
		}
	}
	
	# Option D: Refuse the contract
	option = {
		name = murim_phantom_blade_exp5.0002.d
		add_stress = 30
		add_prestige = -500
		reverse_add_opinion = {
			target = scope:phantom_blade_leader
			modifier = contract_breaker_opinion
			opinion = -60
		}
		ai_chance = {
			base = 10
		}
	}
}
