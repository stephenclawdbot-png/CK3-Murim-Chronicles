# EVENT 1: THE IRON BODY FORGING FURNACE
# The legendary volcanic furnace beneath Iron Fist Gate's mountain.
# Only those who have completed Stage 3 Iron Body may attempt this.
# Success = near-invulnerable body. Failure = crippling burns.
##############################
murim_iron_fist_exp.0001 = {
	type = character_event
	title = murim_iron_fist_exp.0001.t
	desc = murim_iron_fist_exp.0001.desc
	theme = martial_arts
	
	trigger = {
		is_iron_fist_disciple_trigger = yes
		has_character_modifier = iron_fist_iron_body_stage_one_modifier
		prowess >= 20
		age >= 25
		NOT = { has_character_flag = iron_fist_furnace_attempted }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = brave
		}
		modifier = {
			add = 2
			prowess >= 25
		}
	}
	
	immediate = {
		set_character_flag = iron_fist_furnace_attempted
		create_character_memory = {
			type = iron_fist_furnace_trial_memory
		}
	}
	
	# Option A: Enter the furnace with pure body conditioning (traditionalist)
	option = {
		name = murim_iron_fist_exp.0001.a
		random_list = {
			35 = {
				modifier = {
					add = 15
					prowess >= 28
				}
				# Perfect forging - body becomes near-invulnerable
				add_prowess_xp = 1000
				add_character_modifier = {
					modifier = iron_fist_forged_body_modifier
					years = -1
				}
				add_prestige = 500
				create_character_memory = {
					type = iron_fist_furnace_mastery_memory
				}
				# All Iron Fist members hear of your achievement
				every_ruler = {
					limit = {
						is_iron_fist_disciple_trigger = yes
						NOT = { this = root }
					}
					add_opinion = {
						modifier = iron_fist_furnace_respect_opinion
						target = root
						opinion = 30
					}
				}
			}
			40 = {
				# Partial success - scarred but stronger
				add_prowess_xp = 500
				add_trait = scarred
				add_character_modifier = {
					modifier = iron_fist_partial_forging_modifier
					years = 10
				}
			}
			25 = {
				# Failure - severe burns, may lose martial ability
				add_trait = wounded_3
				add_stress = 60
				add_character_modifier = {
					modifier = iron_fist_furnace_burns_modifier
					years = 5
				}
				create_character_memory = {
					type = iron_fist_furnace_failure_memory
				}
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}
	
	# Option B: Use qi reinforcement during the forging (reformist approach)
	option = {
		name = murim_iron_fist_exp.0001.b
		trigger = {
			has_trait = qi_cultivator
		}
		random_list = {
			55 = {
				# Qi-body fusion - creates hybrid defense
				add_prowess_xp = 800
				add_character_modifier = {
					modifier = iron_fist_qi_body_fusion_modifier
					years = -1
				}
				add_prestige = 300
			}
			25 = {
				# Qi disrupted by volcanic heat - internal injury
				add_trait = wounded_2
				add_character_modifier = {
					modifier = iron_fist_qi_disruption_modifier
					years = 3
				}
			}
			20 = {
				# Elder discovers your qi use mid-trial
				add_character_modifier = {
					modifier = iron_fist_cheater_stigma_modifier
					years = 10
				}
				add_stress = 30
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 10
				has_trait = deceitful
			}
		}
	}
	
	# Option C: Decline - the furnace is not worth the risk
	option = {
		name = murim_iron_fist_exp.0001.c
		add_stress = -15
		add_prestige = -100
		ai_chance = {
			base = 15
			modifier = {
				add = 15
				has_trait = craven
			}
		}
	}
}

# EVENT 2: MOUNTAIN SPLITTING COMPETITION
# Annual strength contest held at Iron Fist Gate.
# All 9 clans may send their strongest to compete.
# Pure physical power contest - no weapons, no qi tricks.
##############################
murim_iron_fist_exp.0002 = {
	type = character_event
	title = murim_iron_fist_exp.0002.t
	desc = murim_iron_fist_exp.0002.desc
	theme = martial_arts
	
	trigger = {
		murim_is_martial_artist_trigger = yes
		prowess >= 14
		NOT = { has_character_flag = mountain_split_competed_recently }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			is_iron_fist_disciple_trigger = yes
		}
		modifier = {
			add = 1
			prowess >= 20
		}
	}
	
	immediate = {
		set_character_flag = {
			flag = mountain_split_competed_recently
			years = 3
		}
		# Generate opponent from another clan
		random_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
				prowess >= 14
				NOT = { this = root }
				NOT = {
					OR = {
						AND = {
							is_iron_fist_disciple_trigger = yes
							root = { is_iron_fist_disciple_trigger = yes }
						}
					}
				}
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					prowess >= 18
				}
				modifier = {
					add = 2
					is_iron_fist_disciple_trigger = yes
				}
				modifier = {
					add = 1
					is_shaolin_monk_trigger = yes
				}
			}
			save_scope_as = competition_rival
		}
		create_character_memory = {
			type = mountain_splitting_competition_memory
			participants = { rival = scope:competition_rival }
		}
	}
	
	# Option A: Compete with raw strength (standard approach)
	option = {
		name = murim_iron_fist_exp.0002.a
		duel = {
			skill = prowess
			target = scope:competition_rival
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				min_compared_to = root.prowess
				opinion_modifier = {
					who = scope:competition_rival
					modifier = mountain_split_defeated_opinion
					opinion = -15
				}
				# Victory!
				add_prowess_xp = 600
				add_prestige = 300
				add_character_modifier = {
					modifier = mountain_split_champion_modifier
					years = 3
				}
				create_character_memory = {
					type = mountain_split_victory_memory
					participants = { defeated = scope:competition_rival }
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min_compared_to = scope:competition_rival.prowess
				# Defeat
				add_prowess_xp = 200
				add_stress = 20
				add_character_modifier = {
					modifier = mountain_split_humbled_modifier
					years = 1
				}
			}
		}
		ai_chance = {
			base = 60
		}
	}
	
	# Option B: Use a special technique during the contest
	option = {
		name = murim_iron_fist_exp.0002.b
		trigger = {
			OR = {
				has_trait = vajra_fist_user
				has_trait = mountain_splitting_palm_user
				has_trait = iron_shirt_master
			}
		}
		add_prowess_xp = 800
		add_prestige = 400
		add_character_modifier = {
			modifier = mountain_split_technique_champion_modifier
			years = 5
		}
		scope:competition_rival = {
			add_opinion = {
				modifier = mountain_split_technique_awe_opinion
				target = root
				opinion = 10
			}
		}
		ai_chance = {
			base = 30
		}
	}
	
	# Option C: Forfeit to show respect (diplomatic path)
	option = {
		name = murim_iron_fist_exp.0002.c
		add_diplomacy_xp = 300
		scope:competition_rival = {
			add_opinion = {
				modifier = mountain_split_gracious_opinion
				target = root
				opinion = 20
			}
		}
		stress_impact = {
			ambitious = major_stress_impact_gain
		}
		ai_chance = {
			base = 10
		}
	}
}

# EVENT 3: REFORMIST REBELLION
# The Reformist faction within Iron Fist Gate rises up.
# They want to incorporate qi cultivation alongside body training.
# Traditionalists see this as betrayal of the Iron Fist Way.
##############################
murim_iron_fist_exp.0003 = {
	type = character_event
	title = murim_iron_fist_exp.0003.t
	desc = murim_iron_fist_exp.0003.desc
	theme = intrigue
	
	trigger = {
		is_iron_fist_disciple_trigger = yes
		OR = {
			murim_is_clan_leader_trigger = yes
			murim_is_clan_elder_trigger = yes
		}
		NOT = { has_character_flag = iron_fist_reformist_crisis }
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			any_courtier = {
				count >= 3
				is_iron_fist_disciple_trigger = yes
				has_trait = qi_cultivator
			}
		}
	}
	
	immediate = {
		set_character_flag = iron_fist_reformist_crisis
		# Find the reformist leader
		random_courtier = {
			limit = {
				is_iron_fist_disciple_trigger = yes
				has_trait = qi_cultivator
				prowess >= 12
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					has_trait = ambitious
				}
			}
			save_scope_as = reformist_leader
		}
		create_character_memory = {
			type = iron_fist_reformist_rebellion_memory
			participants = { rebel = scope:reformist_leader }
		}
	}
	
	# Option A: Crush the rebellion with force (traditionalist)
	option = {
		name = murim_iron_fist_exp.0003.a
		duel = {
			skill = prowess
			target = scope:reformist_leader
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.0
				}
				min_compared_to = root.prowess
				# Crush the reformists
				scope:reformist_leader = {
					add_trait = wounded_2
					add_opinion = {
						modifier = iron_fist_crushed_rebel_opinion
						target = root
						opinion = -40
					}
				}
				add_prestige = 300
				add_character_modifier = {
					modifier = iron_fist_tradition_enforcer_modifier
					years = 10
				}
				# Clan harmony drops from internal violence
				change_variable = {
					name = murim_clan_harmony
					add = -15
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.0
				}
				min_compared_to = scope:reformist_leader.prowess
				# Reformist proves qi-body hybrid works
				add_trait = wounded_1
				add_stress = 40
				scope:reformist_leader = {
					add_prestige = 400
				}
				# Gate splits into two factions
				change_variable = {
					name = murim_clan_harmony
					add = -25
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = wrathful
			}
		}
	}
	
	# Option B: Negotiate a compromise (allow limited qi training)
	option = {
		name = murim_iron_fist_exp.0003.b
		add_diplomacy_xp = 500
		scope:reformist_leader = {
			add_opinion = {
				modifier = iron_fist_compromise_opinion
				target = root
				opinion = 15
			}
		}
		add_character_modifier = {
			modifier = iron_fist_moderate_leader_modifier
			years = 10
		}
		# Harmony preserved
		change_variable = {
			name = murim_clan_harmony
			add = 5
		}
		ai_chance = {
			base = 35
			modifier = {
				add = 15
				has_trait = just
			}
		}
	}
	
	# Option C: Side with the reformists (betray tradition)
	option = {
		name = murim_iron_fist_exp.0003.c
		trigger = {
			has_trait = qi_cultivator
		}
		add_character_modifier = {
			modifier = iron_fist_reformed_leader_modifier
			years = -1
		}
		scope:reformist_leader = {
			add_opinion = {
				modifier = iron_fist_ally_opinion
				target = root
				opinion = 30
			}
		}
		# Traditionalist elders furious
		every_courtier = {
			limit = {
				is_iron_fist_disciple_trigger = yes
				NOT = { has_trait = qi_cultivator }
				age >= 40
			}
			add_opinion = {
				modifier = iron_fist_tradition_betrayer_opinion
				target = root
				opinion = -35
			}
		}
		change_variable = {
			name = murim_clan_harmony
			add = -10
		}
		ai_chance = {
			base = 15
		}
	}
}
