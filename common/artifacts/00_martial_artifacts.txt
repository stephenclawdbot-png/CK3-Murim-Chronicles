# Artifact Decisions
study_orthodox_manual = {
	descision = {
		name = "Study Orthodox Manual"
		desc = "Study the righteous techniques recorded within."
		
		is_shown = {
			has_trait = orthodox_compendium
		}
		
		effect = {
			add_learning = 1
			random_list = {
				30 = { add_trait = sword_aptitude_1 }
				20 = { add_trait = alchemist_1 }
				50 = { add_prestige = 50 }
			}
		}
		
		cooldown = { months = 6 }
	}
}
