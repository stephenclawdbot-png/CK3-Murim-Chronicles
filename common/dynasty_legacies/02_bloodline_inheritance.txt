# BLOODLINE 1: FIRST SWORDSMAN LEGACY
# ==========================================
bloodline_legacy = {
	name = "Blood of the First Swordsman"
	desc = "The [dynasty] carries the legacy of the first martial cultivator."
	family = martial
	
	stage_age = child
	
	bloodline_sword_seed = {
		name = "Inheriting the Sword Seed"
		desc = "The bloodline's martial potential awakens early. The child grasps wooden swords with uncanny precision."
		
		inherit_chance = {
			base = 15
			modifier = {
				add = 20
				has_trait = sword_aptitude_3
			}
			modifier = {
				add = 30
				has_trait = sword_aptitude_4
			}
			modifier = {
				add = 10
				any_parent = {
					has_trait = sword_aptitude_3
				}
			}
		}
		
		effect = {
			if = {
				limit = { NOT = { has_trait = sword_aptitude_1 } }
				add_trait = sword_aptitude_1
			}
			add_prestige = 50
		}
	}
	
	stage_age = young_adult
	
	bloodline_sword_awakening = {
		name = "Bloodline Awakening"
		desc = "The ancestral sword dao resonates within. Each swing carries the weight of generations."
		
		inherit_chance = {
			base = 10
			modifier = {
				add = 25
				age >= 16
				martial >= 10
			}
			modifier = {
				add = 15
				dynasty_prestige >= 500
			}
			modifier = {
				add = 10
				any_parent = {
					has_trait = sword_aptitude_4
				}
			}
		}
		
		effect = {
			if = {
				limit = { has_trait = sword_aptitude_1 }
				remove_trait = sword_aptitude_1
			}
			if = {
				limit = { NOT = { has_trait = sword_aptitude_2 } }
				add_trait = sword_aptitude_2
			}
			add_martial = 2
			add_prowess = 3
		}
	}
	
	stage_age = adult
	
	bloodline_inheritor = {
		name = "True Bloodline Inheritor"
		desc = "The full potential of the First Swordsman manifests. Sword intent radiates from every movement."
		
		inherit_chance = {
			base = 5
			modifier = {
				add = 20
				has_trait = sword_aptitude_3
			}
			modifier = {
				add = 30
				has_trait = sword_aptitude_4
			}
			modifier = {
				add = 10
				dynasty_prestige >= 1000
			}
			modifier = {
				add = 15
				any_parent = {
					OR = {
						has_trait = sword_aptitude_4
						has_trait = sword_aptitude_5
					}
				}
			}
		}
		
		effect = {
			if = {
				limit = {
					OR = {
						has_trait = sword_aptitude_1
						has_trait = sword_aptitude_2
					}
				}
				if = {
					limit = { has_trait = sword_aptitude_1 }
					remove_trait = sword_aptitude_1
				}
				if = {
					limit = { has_trait = sword_aptitude_2 }
					remove_trait = sword_aptitude_2
				}
			}
			if = {
				limit = { NOT = { has_trait = sword_aptitude_3 } }
				add_trait = sword_aptitude_3
			}
			add_martial = 5
			add_prowess = 5
