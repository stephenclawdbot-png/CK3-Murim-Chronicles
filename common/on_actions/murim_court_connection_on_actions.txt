# Murim Chronicles - Court Connection On Actions
# Links martial world to vanilla rulers via lightweight yearly/quarterly events
# Martial stays BEHIND vanilla — shadow layer only

on_yearly_pulse = {
	# Enforce ruler rank minimum for martial artist rulers
	# Runs yearly to catch characters who gained titles outside on_title_gained
	effect = {
		every_ruler = {
			limit = {
				has_variable = murim_martial_artist
				has_variable = murim_clan_rank
				highest_held_title_tier >= tier_county
				OR = {
					AND = {
						highest_held_title_tier >= tier_empire
						var:murim_clan_rank < 5
					}
					AND = {
						highest_held_title_tier >= tier_duchy
						var:murim_clan_rank < 4
					}
					AND = {
						highest_held_title_tier >= tier_county
						var:murim_clan_rank < 3
					}
				}
			}
			murim_enforce_ruler_rank_effect = yes
		}
	}

	random_events = {
		# Martial advisor arrives at court (player rulers with martial courtiers)
		200 = murim_court.0001

		# Sect leader seeks audience with ruler (kingdoms+)
		150 = murim_court.0003

		# Emperor's martial tournament (kingdoms+, 10yr cooldown)
		100 = murim_court.0004

		# Martial world crisis reaches the throne
		120 = murim_court.0005

		# Shadow behind the throne — martial influence on ruler
		100 = murim_court.0006

		# AI: martial leaders seek court influence
		150 = murim_court.0007

		# Non-martial rulers learn about the martial world
		180 = murim_court.0008

		# Foreign master hears of rankings — starts migration
		120 = murim_global.0001

		# Tournament attracts foreign challengers
		80 = murim_global.0006

		# Wandering master arrives at foreign court
		100 = murim_global.0007

		# AI foreign masters auto-migrate east
		150 = murim_global.0008
	}
}

on_quarterly_character_pulse = {
	random_events = {
		# Martial advisor gives counsel (rulers with advisors)
		50 = murim_court.0002
	}
}
