######################################
# EVENT 0001 — Thousand Petal Poison Trial
# Palace disciple must survive exposure to 1000 different poisons
######################################
murim_poison_flower_exp3.0001 = {
	type = character_event
	title = murim_poison_flower_exp3.0001.t
	desc = murim_poison_flower_exp3.0001.desc
	theme = intrigue_scheme_theme
	left_portrait = {
		character = root
		animation = pain
	}
	right_portrait = {
		character = scope:poison_master
		animation = personality_cynical
	}

	trigger = {
		has_trait = poison_flower_palace_member
		has_variable = murim_cultivation_level
		var:murim_cultivation_level >= 3
		NOT = { has_character_flag = completed_thousand_petal_trial }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = poison_resistance
		}
		modifier = {
			add = 1.5
			has_trait = poison_flower_inner_disciple
		}
	}

	immediate = {
		set_character_flag = { flag = attempting_thousand_petal_trial days = 365 }
		if = {
			limit = {
				any_courtier = {
					has_trait = poison_flower_elder
				}
			}
			random_courtier = {
				limit = { has_trait = poison_flower_elder }
				save_scope_as = poison_master
			}
		}
		else = {
			create_character = {
				culture = root.culture
				location = root.capital_province
				age = { 45 65 }

				random_traits = yes

				dynasty = none

				faith = root.faith

				gender_female_chance = 70

				trait = murim_martial_artist
				save_scope_as = poison_master
			}
		}
	}

	# Endure the full trial
	option = {
		name = murim_poison_flower_exp3.0001.a
		duel = {
			skill = learning
			target = scope:poison_master
			50 = {
				desc = murim_poison_flower_exp3.0001.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				send_interface_toast = {
					title = murim_poison_flower_exp3.0001.a.success.toast
					left_icon = root
					add_trait = poison_immunity
					add_character_modifier = {
						modifier = thousand_petal_survivor
						years = 10
					}
					set_character_flag = completed_thousand_petal_trial
					add_prestige = 500
					if = {
						limit = { has_variable = murim_cultivation_level }
						change_variable = { name = murim_cultivation_level add = 1 }
					}
				}
			}
			50 = {
				desc = murim_poison_flower_exp3.0001.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
				}
				send_interface_toast = {
					title = murim_poison_flower_exp3.0001.a.failure.toast
					left_icon = root
					add_trait = poison_scarred
					add_character_modifier = {
						modifier = failed_petal_trial_recovery
						years = 3
					}
					health = -2
				}
			}
		}
		ai_chance = {
			base = 70
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}

	# Use antidote secretly
	option = {
		name = murim_poison_flower_exp3.0001.b
		trigger = {
			has_trait = deceitful
		}
		add_character_modifier = {
			modifier = cheated_petal_trial
			years = 5
		}
		set_character_flag = completed_thousand_petal_trial
		add_secret = {
			type = secret_cheated_trial
			target = root
		}
		add_prestige = 200
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = craven
			}
		}
	}

	# Refuse the trial
	option = {
		name = murim_poison_flower_exp3.0001.c
		add_prestige = -200
		add_character_modifier = {
			modifier = refused_palace_trial
			years = 2
		}
		ai_chance = {
			base = 10
		}
	}
}

	}

		animation = personality_vengeful
	}

		has_trait = poison_flower_palace_member
		has_character_flag = completed_thousand_petal_trial

			add = 3

				any_courtier = {

				}
			}

				limit = { has_trait = poison_flower_palace_leader }

		name = murim_poison_flower_exp3.0002.a

			modifier = forbidden_garden_knowledge
			years = 15
		}
		add_trait = forbidden_poison_arts
		change_variable = { name = murim_cultivation_level add = 1 }
		add_secret = {
			type = secret_forbidden_garden

			honest = major_stress_impact_gain
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 30
				has_trait = ambitious
			}
		}
	}

	# Report the garden to the Palace Mistress
	option = {
		name = murim_poison_flower_exp3.0002.b
		reverse_add_opinion = {
			target = scope:garden_keeper
			modifier = reported_secret_garden

		}

		add_piety = 100
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = honest
			}
		}
	}

	# Destroy the garden
	option = {
		name = murim_poison_flower_exp3.0002.c

			has_trait = zealous
		}

				opinion = -50
			}
		}
