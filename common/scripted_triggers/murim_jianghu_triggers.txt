####################################
# SECTION 1: CORE JIANGHU STATUS TRIGGERS
####################################

# Check if character is part of the jianghu underground
murim_is_jianghu_member_trigger = {
	OR = {
		has_trait = shadow_broker
		has_trait = sect_enforcer
		has_trait = jianghu_informant
		has_trait = underground_merchant
		has_trait = hidden_cultivator
		has_trait = wulin_outcast
		has_trait = jianghu_assassin
		has_trait = puppet_master
		has_character_flag = joined_jianghu_underground
		has_variable = jianghu_standing
	}
}

# Check if character has significant jianghu influence
murim_has_jianghu_influence_trigger = {
	murim_is_jianghu_member_trigger = yes
	OR = {
		has_variable_list = jianghu_contacts
		AND = {
			has_variable = jianghu_standing
			var:jianghu_standing >= 25
		}
		has_trait = shadow_broker
		has_trait = puppet_master
	}
}

# Check if character is a jianghu leader (sect master, alliance head, etc.)
murim_is_jianghu_leader_trigger = {
	OR = {
		has_trait = sect_master
		has_trait = wulin_alliance_leader
		has_trait = shadow_broker
		AND = {
			has_variable = jianghu_standing
			var:jianghu_standing >= 75
		}
		has_character_flag = recognized_jianghu_authority
	}
}

# Check if character is a hidden cultivator (concealed martial power)
murim_is_hidden_cultivator_trigger = {
	OR = {
		has_trait = hidden_cultivator
		has_character_flag = concealed_cultivation_base
		AND = {
			has_variable = cultivation_realm
			var:cultivation_realm >= 3
			has_character_flag = disguised_as_mortal
		}
	}
}

# Check if character is operating covertly in the mundane world
murim_is_operating_covertly_trigger = {
	murim_is_jianghu_member_trigger = yes
	NOR = {
		has_trait = known_cultivator
		has_character_flag = exposed_jianghu_identity
		has_character_flag = publicly_declared_sect_allegiance
	}
	OR = {
		has_trait = hidden_cultivator
		has_trait = jianghu_informant
		has_character_flag = covert_operations_active
	}
}

# SECTION 2: SHADOW BROKER TRIGGERS
####################################

# Check if character can act as shadow broker
murim_can_be_shadow_broker_trigger = {
	is_adult = yes
	is_imprisoned = no
	NOR = {
		has_trait = honest
		has_trait = just
		has_trait = zealous
	}
	OR = {
		intrigue >= 12
		has_trait = schemer
		has_trait = deceitful
		has_trait = shadow_broker
	}
	OR = {
		has_variable = jianghu_standing
		has_trait = jianghu_informant
		has_character_flag = shadow_network_established
	}
}

# Check if character has an active shadow broker network
murim_has_shadow_network_trigger = {
	OR = {
		has_trait = shadow_broker
		has_character_flag = shadow_network_established
	}
	has_variable = shadow_network_reach
	var:shadow_network_reach >= 1
	NOT = { has_character_flag = shadow_network_compromised }
}

# Check if character can install puppet governors via shadow broker network
murim_can_install_puppet_governor_trigger = {
	murim_has_shadow_network_trigger = yes
	OR = {
		has_trait = shadow_broker
		has_trait = puppet_master
	}
	gold >= 150
	has_variable = shadow_network_reach
	var:shadow_network_reach >= 3
	any_vassal = {
		is_ai = yes
		NOT = { has_character_flag = puppet_governor_installed }
	}
}

# Check if a character is a puppet governor
murim_is_puppet_governor_trigger = {
	has_character_flag = puppet_governor_installed
	has_variable = puppet_master_id
	is_ai = yes
}

# Check if shadow broker can intercept communications
murim_can_intercept_messages_trigger = {
	murim_has_shadow_network_trigger = yes
	intrigue >= 15
	OR = {
		has_trait = shadow_broker
		has_perk = truth_is_relative_perk
		has_character_flag = interceptor_network_active
	}
	NOT = { has_character_modifier = network_exposed_cooldown }
}
