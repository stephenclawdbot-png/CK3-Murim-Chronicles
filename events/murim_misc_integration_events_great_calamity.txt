# Great Calamity: First Omen
murim_great_calamity.0001 = {
	type = character_event
	title = murim_great_calamity.0001.t
	desc = murim_great_calamity.0001.desc
	theme = skull
	left_portrait = { character = root animation = personality_zealous }

	trigger = {
		has_variable = murim_martial_artist
		NOT = { has_global_variable = murim_great_calamity_active }
		NOT = { has_character_flag = murim_witnessed_first_omen }
	}

	immediate = {
		set_character_flag = murim_witnessed_first_omen
	}

	# Prepare for the storm — gather resources and allies
	option = {
		name = murim_great_calamity.0001.a
		add_martial = 1
		add_diplomacy = 1
		add_stress = 20
		ai_chance = {
			base = 45
			modifier = { add = 20 has_trait = diligent }
		}
	}

	# Investigate the omen's source
	option = {
		name = murim_great_calamity.0001.b
		add_learning = 2
		add_intrigue = 1
		ai_chance = {
			base = 35
			modifier = { add = 20 learning >= 14 }
		}
	}

	# Ignore it — omens are superstition
	option = {
		name = murim_great_calamity.0001.c
		add_stress = -10
		add_prestige = -25
		ai_chance = { base = 20 }
	}
}
