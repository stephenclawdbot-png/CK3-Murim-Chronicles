# Continental War Declaration
murim_continental_war.0001 = {
	type = character_event
	title = murim_continental_war.0001.t
	desc = murim_continental_war.0001.desc
	theme = martial
	left_portrait = { character = root animation = personality_bold }

	trigger = {
		murim_is_faction_leader_trigger = yes
		NOT = { murim_continental_war_active_trigger = yes }
	}

	immediate = {
		hidden_effect = { murim_declare_continental_war_effect = yes }
	}

	# Rally the orthodox sects — purge the unorthodox menace
	option = {
		name = murim_continental_war.0001.a
		trigger = { murim_is_orthodox_faction_trigger = yes }
		add_prestige = 500
		add_dread = 20
		add_character_modifier = { modifier = murim_faction_leader_wartime years = 10 }
		hidden_effect = {
			every_ruler = {
				limit = { has_variable = murim_martial_artist }
				trigger_event = { id = murim_continental_war.0005 days = 7 }
			}
		}
		ai_chance = { base = 60 }
	}

	# Unleash the unorthodox — tear down the old order
	option = {
		name = murim_continental_war.0001.b
		trigger = { murim_is_unorthodox_faction_trigger = yes }
		add_prestige = 500
		add_dread = 30
		add_character_modifier = { modifier = murim_faction_leader_wartime years = 10 }
		hidden_effect = {
			every_ruler = {
				limit = { has_variable = murim_martial_artist }
				trigger_event = { id = murim_continental_war.0005 days = 7 }
			}
		}
		ai_chance = { base = 60 }
	}

	# Imperial decree — the Emperor demands martial submission
	option = {

		trigger = { has_character_flag = murim_imperial_faction_leader }

# War turning point — assassination of faction leader
murim_continental_war.0035 = {
	type = character_event
	title = murim_continental_war.0035.t
	desc = murim_continental_war.0035.desc
	theme = intrigue
	left_portrait = { character = root animation = personality_irrational }

	trigger = {
		has_variable = murim_martial_artist
		murim_continental_war_active_trigger = yes
		intrigue >= 14
	}

	# Assassinate the enemy faction leader
	option = {
		name = murim_continental_war.0035.a
		random_list = {
			40 = {
				modifier = { add = intrigue factor = 0.3 }
				add_prestige = 500
				add_dread = 30
				change_variable = { name = murim_war_contribution add = 5 }
			}
			60 = {
				add_stress = 40
				add_health = -1
				add_prestige = -100
			}
		}
		ai_chance = {
			base = 30
			modifier = { add = 25 intrigue >= 18 }
		}
	}

	# Betray your faction — switch sides
	option = {
		name = murim_continental_war.0035.b
		add_intrigue = 2
		add_prestige = -200
		add_dread = 15
		# Flip faction flags
		if = {
			limit = { has_character_flag = murim_war_orthodox_side }
			remove_character_flag = murim_war_orthodox_side
			set_character_flag = { flag = murim_war_unorthodox_side years = 20 }
		}
		else_if = {
			limit = { has_character_flag = murim_war_unorthodox_side }
			remove_character_flag = murim_war_unorthodox_side
			set_character_flag = { flag = murim_war_orthodox_side years = 20 }
		}
		ai_chance = {
			base = 10
			modifier = { add = 20 has_trait = deceitful }
		}
	}

	# Report the plot to your leader
	option = {
		name = murim_continental_war.0035.c
		add_prestige = 150
		add_piety = 50
		ai_chance = { base = 40 }
	}
}
