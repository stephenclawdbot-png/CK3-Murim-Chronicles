######################################
# 0001 - EMPEROR DISCOVERS CULTIVATORS
######################################

murim_imperial_cultivator.0001 = {
	type = character_event
	title = murim_imperial_cultivator.0001.t
	desc = murim_imperial_cultivator.0001.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = shock
	}
	right_portrait = {
		character = scope:exposed_cultivator
		animation = shame
	}

	trigger = {
		is_ruler = yes
		highest_held_title_tier >= tier_duchy
		NOT = { murim_is_cultivator_trigger = yes }
		exists = scope:exposed_cultivator
	}

	immediate = {
		set_character_flag = murim_discovered_cultivators
	}

	# Option A: Demand they serve the empire
	option = {
		name = murim_imperial_cultivator.0001.a
		scope:exposed_cultivator = {
			set_character_flag = murim_cultivation_public
			add_opinion = {
				target = root
				modifier = murim_forced_into_service
			}
			add_character_modifier = {
				modifier = murim_imperial_cultivator_conscript
				years = -1
			}
		}
		root = {
			add_prestige = 100
			add_character_modifier = {
				modifier = murim_has_cultivator_servant
				years = 10
			}
		}
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = ambitious }
			modifier = { add = 10 has_trait = pragmatic }
		}
	}

	# Option B: Imprison them as a threat
	option = {
		name = murim_imperial_cultivator.0001.b
		scope:exposed_cultivator = {
			imprison = {
				target = scope:exposed_cultivator
				type = dungeon
			}
			add_opinion = {
				target = root
				modifier = murim_imprisoned_for_cultivation
			}
		}
		root = {
			add_piety = 50
			add_tyranny = 10
		}
		ai_chance = {
			base = 25
			modifier = { add = 20 has_trait = zealous }
			modifier = { add = 15 has_trait = paranoid }
		}
	}

	# Option C: Secretly learn from them
	option = {
		name = murim_imperial_cultivator.0001.c
		scope:exposed_cultivator = {
			add_opinion = {
				target = root
				modifier = murim_interested_in_cultivation
			}
		}
		root = {
			set_character_flag = murim_secretly_studying_cultivation
			add_character_modifier = {
				modifier = murim_hidden_cultivation_student
				years = 5
			}
			add_learning = 1
		}
		ai_chance = {
			base = 20
			modifier = { add = 20 has_trait = curious }
			modifier = { add = 15 learning >= 14 }
		}
	}

	# Option D: Ignore and pretend nothing happened
	option = {
		name = murim_imperial_cultivator.0001.d
		root = {
			add_stress = 10
		}
		ai_chance = {
			base = 15
			modifier = { add = 15 has_trait = lazy }
			modifier = { add = 10 has_trait = content }
		}
	}
}
