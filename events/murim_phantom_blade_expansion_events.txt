# PHANTOM CHASE - PURSUED BY ORTHODOX HUNTERS
# Orthodox faction sends hunters after a
# Phantom Blade assassin.
############################################
murim_phantom_exp.0002 = {
	type = character_event
	title = murim_phantom_exp.0002.t
	desc = murim_phantom_exp.0002.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = fear
	}
	right_portrait = {
		character = scope:orthodox_hunter
		animation = personality_bold
	}

	trigger = {
		has_trait = phantom_blade_disciple
		murim_is_martial_artist_trigger = yes
		has_variable = murim_faction
		var:murim_faction = flag:unorthodox
		prowess >= 12
		NOT = { has_character_modifier = murim_phantom_chase_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 3 dread >= 30 }
		modifier = { add = 2 has_character_flag = assassination_contract_cooldown }
	}

	immediate = {
		add_character_modifier = { modifier = murim_phantom_chase_cooldown years = 3 }
		# Orthodox hunter
		random_living_character = {
			limit = {
				has_trait = martial_artist
				is_alive = yes
				prowess >= 14
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = iron_fist_disciple
					has_trait = beggar_sect_member
					has_trait = tang_clan_member
				}
				NOT = { this = root }
			}
			weight = {
				base = 10
				modifier = { add = 15 prowess >= 18 }
				modifier = { add = 10 has_trait = just }
			}
			save_scope_as = orthodox_hunter
		}
	}

	# Use Phantom Step to vanish - escape through stealth
	option = {
		name = murim_phantom_exp.0002.a
		trigger = { exists = scope:orthodox_hunter }
		random_list = {
			55 = {
				modifier = { add = 15 intrigue >= 16 }
				modifier = { add = 10 has_character_flag = blade_spirit_communion_done }
				desc = murim_phantom_exp.0002.a.escape
				add_prestige = 100
				add_intrigue = 1
				add_character_modifier = { modifier = murim_phantom_escape_artist years = 3 }
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:phantom_chase_escaped }
				}
			}
			45 = {
				modifier = { add = 15 scope:orthodox_hunter = { prowess >= 20 } }
				desc = murim_phantom_exp.0002.a.caught
				# Caught - forced to fight
				trigger_event = { id = murim_phantom_exp.0003 days = 1 }
			}
		}
		stress_impact = { brave = medium_stress_impact_gain craven = minor_stress_impact_loss }
		ai_chance = { base = 50 modifier = { add = 20 intrigue >= 16 } }
	}

	# Stand and fight - Phantom Blade never runs
	option = {
		name = murim_phantom_exp.0002.b
		trigger = { exists = scope:orthodox_hunter }
		duel = {
			skill = prowess
			target = scope:orthodox_hunter
			50 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_phantom_exp.0002.b.victory
				add_prestige = 200
				add_dread = 15
				scope:orthodox_hunter = {
					random_list = {
						40 = { add_trait = wounded_1 }
						35 = { add_trait = wounded_2 }
						15 = { add_trait = maimed }
						10 = { death = { death_reason = death_duel killer = root } }
					}
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:phantom_hunter_slain }
				}
			}
			50 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:orthodox_hunter }
				desc = murim_phantom_exp.0002.b.defeat
				add_prestige = -100
				random_list = {
					30 = { add_trait = wounded_1 }
					30 = { add_trait = wounded_2 }
					25 = { add_trait = maimed }
					15 = {
						# Captured - faction war implications
						add_character_modifier = { modifier = murim_captured_by_orthodox years = 2 }
						trigger_event = { id = murim_faction_war.0005 days = { 5 15 } }
					}
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:phantom_hunter_defeated_me }
				}
			}
		}
		stress_impact = { brave = minor_stress_impact_loss craven = major_stress_impact_gain }
		ai_chance = { base = 40 modifier = { add = 30 has_trait = brave } }
	}

	# Negotiate - offer information for freedom
	option = {
		name = murim_phantom_exp.0002.c
		trigger = { exists = scope:orthodox_hunter diplomacy >= 10 }
		add_prestige = -50
		add_dread = -10
		scope:orthodox_hunter = {
			add_opinion = { target = root modifier = grudging_respect_opinion opinion = 10 }
		}
		add_character_modifier = { modifier = murim_informant_shame years = 5 }
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:phantom_chase_negotiated }
		}
		stress_impact = { deceitful = minor_stress_impact_loss honest = major_stress_impact_gain }
		ai_chance = { base = 10 modifier = { add = 20 has_trait = deceitful } }
	}
}
