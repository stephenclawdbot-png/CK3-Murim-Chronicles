# Murim Chronicles - General Scripted Effects
# Core effects for cultivation, duels, memories, injuries, and cross-system integration
# Cross-references: murim_modifiers, murim_triggers, murim_ranking_effects, all event files
# FIXED: All trait references now match actual trait definitions in 01_martial_traits.txt

# ==========================================
# CULTIVATION EFFECTS
# ==========================================

murim_gain_qi_effect = {
	# Gain qi cultivation progress based on learning and piety
	if = {
		limit = {
			OR = {
				has_trait = murim_nascent_soul
				has_trait = cultivation_nascent_soul
			}
		}
		# Already at high cultivation, minor benefit
		add_piety = 50
		create_character_memory = {
			type = murim_qi_cultivation_memory
		}
	}
	else_if = {
		limit = {
			OR = {
				has_trait = murim_golden_core
				has_trait = cultivation_core_formation
			}
		}
		change_variable = {
			name = murim_cultivation_progress
			add = 10
		}
		add_piety = 25
	}
	else_if = {
		limit = {
			OR = {
				has_trait = murim_foundation
				has_trait = cultivation_foundation
			}
		}
		change_variable = {
			name = murim_cultivation_progress
			add = 15
		}
		add_piety = 15
	}
	else = {
		change_variable = {
			name = murim_cultivation_progress
			add = 20
		}
		add_piety = 10
	}
}

murim_attempt_breakthrough_effect = {
	# Attempt cultivation breakthrough with risk of qi deviation
	# Uses martial power and cultivation progress to determine success
	random_list = {
		40 = {
			modifier = {
				add = learning
			}
			modifier = {
				factor = 1.5
				OR = {
					has_trait = cultivation_prodigy
					has_trait = genius
				}
			}
			# Breakthrough success
			murim_cultivation_breakthrough_success_effect = yes
			trigger_event = {
				id = murim_cultivation.0100
				days = 1
			}
		}
		30 = {
			# Partial success - progress but no breakthrough
			change_variable = {
				name = murim_cultivation_progress
				add = 25
			}
			add_character_modifier = {
				modifier = qi_stagnation
				years = 1
			}
			create_character_memory = {
				type = murim_failed_breakthrough_memory
			}
		}
		30 = {
			modifier = {
				factor = 1.5
				has_trait = reckless
			}
			modifier = {
				factor = 0.5
				has_trait = patient
			}
			# Qi deviation - dangerous failure
			murim_qi_deviation_effect = yes
			trigger_event = {
				id = murim_cultivation.0200
				days = 1
			}
		}
	}
}

murim_cultivation_breakthrough_success_effect = {
	# Promote to next cultivation tier
	# Uses correct trait names from 01_martial_traits.txt
	if = {
		limit = {
			NOR = {
				has_trait = murim_outer_qi
				has_trait = murim_qi_gathering
				has_trait = murim_foundation
				has_trait = murim_golden_core
				has_trait = murim_nascent_soul
				has_trait = murim_spirit_severing
				has_trait = murim_heavenly_tribulation
			}
		}
		add_trait = murim_outer_qi
	}
	else_if = {
		limit = { has_trait = murim_outer_qi }
		remove_trait = murim_outer_qi
		add_trait = murim_qi_gathering
	}
	else_if = {
		limit = { has_trait = murim_qi_gathering }
		remove_trait = murim_qi_gathering
		add_trait = murim_foundation
	}
	else_if = {
		limit = { has_trait = murim_foundation }
		remove_trait = murim_foundation
		add_trait = murim_golden_core
		add_character_modifier = {
			modifier = core_formation_radiance
			years = 3
		}
	}
	else_if = {
		limit = { has_trait = murim_golden_core }
		remove_trait = murim_golden_core
		add_trait = murim_nascent_soul
		add_character_modifier = {
			modifier = nascent_soul_presence
			years = 5
		}
	}
	else_if = {
		limit = { has_trait = murim_nascent_soul }
		remove_trait = murim_nascent_soul
		add_trait = murim_spirit_severing
	}
	else_if = {
		limit = { has_trait = murim_spirit_severing }
		remove_trait = murim_spirit_severing
		add_trait = murim_heavenly_tribulation
	}
	add_character_modifier = {
		modifier = breakthrough_euphoria
		years = 1
	}
	set_variable = {
		name = murim_cultivation_progress
		value = 0
	}
	add_prestige = 200
	add_piety = 100
	create_character_memory = {
		type = murim_breakthrough_success_memory
	}
}

murim_qi_deviation_effect = {
	# Qi deviation - cultivation gone wrong
	add_character_modifier = {
		modifier = qi_deviation_recovery
		years = 2
	}
	random_list = {
		40 = {
			# Minor deviation - just pain
			add_stress = 30
			add_character_modifier = {
				modifier = meridian_opening_strain
				years = 1
			}
		}
		35 = {
			# Moderate deviation - wounded
			add_trait = wounded_1
			add_stress = 50
			create_character_memory = {
				type = murim_qi_deviation_wounded_memory
			}
		}
		20 = {
			# Severe deviation - seriously wounded
			add_trait = wounded_2
			add_stress = 80
			create_character_memory = {
				type = murim_qi_deviation_severe_memory
			}
		}
		5 = {
			# Critical deviation - incapable or death
			random_list = {
				60 = {
					add_trait = incapable
					create_character_memory = {
						type = murim_qi_deviation_crippled_memory
					}
				}
				40 = {
					death = {
						death_reason = death_qi_deviation
					}
				}
			}
		}
	}
}

murim_duel_victory_effect = {
	# Apply effects of winning a duel
	add_prestige = 100
	add_stress = -10
	# Track duel wins
	if = {
		limit = {
			NOT = { has_variable = murim_duel_wins }
		}
		set_variable = {
			name = murim_duel_wins
			value = 1
		}
	}
	else = {
		change_variable = {
			name = murim_duel_wins
			add = 1
		}
	}
	create_character_memory = {
		type = murim_duel_victory_memory
	}
	# Check for ranking impact
	murim_recalculate_martial_rank_effect = yes
}

murim_duel_defeat_effect = {
	# Apply effects of losing a duel
	add_prestige = -50
	add_stress = 25
	# Track duel losses
	if = {
		limit = {
			NOT = { has_variable = murim_duel_losses }
		}
		set_variable = {
			name = murim_duel_losses
			value = 1
		}
	}
	else = {
		change_variable = {
			name = murim_duel_losses
			add = 1
		}
	}
	# Random injury from combat
	random_list = {
		50 = {
			# No lasting injury
		}
		25 = {
			add_trait = wounded_1
		}
		15 = {
			add_trait = wounded_2
		}
		7 = {
			# Serious injury - scarred
			add_trait = scarred_1
		}
		3 = {
			# Crippling defeat
			add_trait = incapable
		}
	}
	add_character_modifier = {
		modifier = defeat_shame
		years = 2
	}
	create_character_memory = {
		type = murim_duel_defeat_memory
	}
	# Check for ranking impact
	murim_recalculate_martial_rank_effect = yes
}

murim_duel_death_effect = {
	# Character died in a duel
	# Remove from rankings first
	murim_remove_from_rankings_on_death_effect = yes
	# Create memories for witnesses
	every_close_family_member = {
		limit = { is_alive = yes }
		create_character_memory = {
			type = murim_family_member_killed_in_duel_memory
			participants = {
				dead_fighter = root
			}
		}
		add_stress = 40
	}
}

murim_apply_craven_check_effect = {
	# After a devastating defeat, check for becoming craven
	random = {
		chance = 25
		modifier = {
			factor = 2
			has_trait = shy
		}
		modifier = {
			factor = 0.5
			has_trait = brave
		}
		modifier = {
			factor = 1.5
			stress >= 80
		}
		add_trait = craven
		create_character_memory = {
			type = murim_spirit_broken_memory
		}
	}
}

# MEMORY CREATION EFFECTS
# ==========================================

murim_create_fight_memory_effect = {
	# Create a memory of fighting someone
	# Scope: root = self, scope:target = opponent
	create_character_memory = {
		type = murim_fight_memory
		participants = {
			opponent = scope:target
		}
	}
	scope:target = {
		create_character_memory = {
			type = murim_fight_memory
			participants = {
				opponent = root
			}
		}
	}
}

murim_join_sect_effect = {
	# Join a sect - set variables and modifiers
	set_variable = {
		name = murim_sect_member
		value = yes
	}
	add_prestige = 50
	create_character_memory = {
		type = murim_joined_sect_memory
	}
}

murim_leave_sect_effect = {
	# Leave or be expelled from sect
	if = {
		limit = { has_variable = murim_sect_member }
		remove_variable = murim_sect_member
	}
	add_character_modifier = {
		modifier = murim_sect_exile_shame
		years = 3
	}
	add_stress = 30
	create_character_memory = {
		type = murim_left_sect_memory
	}
}

murim_gain_ancient_weapon_effect = {
	# Gain a powerful ancient weapon
	add_character_modifier = {
		modifier = ancient_weapon_wielder
		years = -1
	}
	add_prestige = 150
	create_character_memory = {
		type = murim_gained_ancient_weapon_memory
	}
	# Trigger jealousy from rivals
	every_known_character = {
		limit = {
			murim_is_martial_artist_trigger = yes
			NOT = { this = root }
			is_alive = yes
		}
		random = {
			chance = 15
			trigger_event = {
				id = murim_artifact.0010
				days = { 15 60 }
			}
		}
	}
}
