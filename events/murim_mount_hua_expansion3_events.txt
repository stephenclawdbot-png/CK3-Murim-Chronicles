murim_mount_hua_expansion3.0001 = {
	type = character_event
	title = murim_mount_hua_expansion3.0001.t
	desc = murim_mount_hua_expansion3.0001.desc
	theme = martial_arts
	
	# The Sword Resonance Phenomenon — during midnight meditation on the peak, your sword
	# begins vibrating without being touched. It resonates with the mountain itself.
	# The ancient texts call this "Sword Heart Awakening" — it happens once per generation.
	
	trigger = {
		has_trait = mount_hua_disciple
		prowess >= 15
		age >= 20
		NOT = { has_character_flag = sword_resonance_experienced }
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 5 has_trait = diligent }
		modifier = { add = 3 has_trait = patient }
	}
	
	immediate = {
		set_character_flag = sword_resonance_experienced
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.a
		# Follow the resonance — let it guide your sword
		add_prowess = 6
		add_martial = 3
		add_trait = sword_heart_awakened
		add_character_modifier = {
			modifier = mountain_sword_resonance
			years = -1
		}
		hidden_effect = {
			trigger_event = {
				id = murim_mount_hua_expansion3.0002
				days = { 30 90 }
			}
		}
		ai_chance = { base = 60 }
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.b
		# Resist it — this power feels too foreign, too dangerous
		add_learning = 3
		add_stress = 20
		add_character_modifier = {
			modifier = rejected_sword_resonance
			years = 5
		}
		ai_chance = { base = 25 modifier = { add = 20 has_trait = paranoid } }
	}
	
	option = {
		name = murim_mount_hua_expansion3.0001.c
		# Study the phenomenon scientifically — understand before embracing
		add_learning = 5
		add_character_modifier = {
			modifier = analytical_swordsman
			years = 5
		}
		hidden_effect = {
			trigger_event = {
				id = murim_mount_hua_expansion3.0003
				days = { 60 120 }
			}
		}
		ai_chance = { base = 15 }
	}
}
