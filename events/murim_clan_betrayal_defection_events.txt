# EVENT 0001 - Seeds of Discontent: Growing Dissatisfaction
# Triggered: on_action yearly pulse, courtiers with low opinion of liege
# AI: Full betrayal motivation system
murim_betrayal.0001 = {
    type = character_event
    title = murim_betrayal.0001.t
    desc = murim_betrayal.0001.desc
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = thinking
    }
    right_portrait = {
        character = scope:rival_recruiter
        animation = scheme
    }
    
    trigger = {
        is_alive = yes
        age >= 18
        NOT = { has_character_flag = betrayal_considered_recently }
        NOT = { has_trait = murim_traitor }
        OR = {
            murim_is_martial_artist_trigger = yes
            has_trait = murim_shadow_operative
        }
        OR = {
            # Low opinion of current leader
            any_liege_or_above = {
                reverse_opinion = {
                    target = root
                    value <= -20
                }
            }
            # Ambitious and passed over
            AND = {
                has_trait = ambitious
                age >= 30
                NOT = { has_trait = murim_sect_leader }
            }
            # Bitter about something
            stress >= 60
            has_character_modifier = murim_passed_over_for_weapon
            has_character_modifier = murim_expelled_from_sect
            has_character_modifier = murim_clan_disapproval
        }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 3.0
            has_character_modifier = murim_expelled_from_sect
        }
        modifier = {
            factor = 2.0
            has_trait = ambitious
        }
        modifier = {
            factor = 2.0
            stress >= 70
        }
        modifier = {
            factor = 0.3
            has_trait = content
        }
        modifier = {
            factor = 0.2
            has_trait = honest
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = betrayal_considered_recently
            days = 730
        }
        # Find a rival clan recruiter
        random_known_character = {
            limit = {
                is_alive = yes
                NOT = { this = root }
                OR = {
                    has_trait = murim_sect_leader
                    has_trait = murim_shadow_operative
                }
                NOT = { is_allied_to = root }
            }
            save_scope_as = rival_recruiter
        }
    }
    
    # Option A: Begin plotting defection
    option = {
        name = murim_betrayal.0001.a
        trigger = {
            exists = scope:rival_recruiter
        }
        add_character_flag = {
            flag = murim_plotting_defection
            days = 1825
        }
        add_character_modifier = {
            modifier = murim_secret_defector
            years = 3
        }
        reverse_add_opinion = {
            target = scope:rival_recruiter
            modifier = murim_potential_recruit
            opinion = 15
        }
        trigger_event = {
            id = murim_betrayal.0002
            days = { 30 120 }
        }
        stress_impact = {
            honest = major_stress_impact_gain
            loyal = major_stress_impact_gain
        }
        ai_chance = {
            base = 35
            modifier = {
                factor = 3.0
                has_trait = ambitious
            }
            modifier = {
                factor = 2.0
                has_trait = deceitful
            }
            modifier = {
                factor = 2.0
                has_character_modifier = murim_expelled_from_sect
            }
            modifier = {
                factor = 0.1
                has_trait = honest
            }
            modifier = {
                factor = 0.1
                has_trait = content
            }
        }
    }
    
    # Option B: Voice complaints openly
    option = {
        name = murim_betrayal.0001.b
        add_stress = -10
        add_prestige = -25
        if = {
            limit = {
                any_liege_or_above = { is_alive = yes }
            }
            random_liege_or_above = {
                add_opinion = {
                    target = root
                    modifier = murim_vocal_complainer
                    opinion = -15
                }
            }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                has_trait = honest
            }
            modifier = {
                factor = 2.0
                has_trait = brave
            }
        }
    }
    
    # Option C: Suppress feelings - remain loyal
    option = {
        name = murim_betrayal.0001.c
        add_stress = 20
        add_piety = 25
        add_character_modifier = {
            modifier = murim_suppressed_dissent
            years = 3
        }
        ai_chance = {
            base = 35
            modifier = {
                factor = 2.0
                has_trait = content
            }
            modifier = {
                factor = 2.0
                has_trait = patient
            }
        }
    }
}

# EVENT 0002 - The Offer: Rival Clan's Proposal
# Triggered: from 0001 plotting defection
# AI: Full negotiation with rival clan
murim_betrayal.0002 = {
    type = character_event
    title = murim_betrayal.0002.t
    desc = murim_betrayal.0002.desc
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = scheme
    }
    right_portrait = {
        character = scope:rival_recruiter
        animation = personality_diplomacy
    }
    
    trigger = {
        is_alive = yes
        has_character_flag = murim_plotting_defection
    }
    
    immediate = {
        random_known_character = {
            limit = {
                is_alive = yes
                NOT = { this = root }
                OR = {
                    has_trait = murim_sect_leader
                    has_trait = murim_shadow_operative
                }
                NOT = { is_allied_to = root }
            }
            save_scope_as = rival_recruiter
        }
        # What do they offer?
        random_list = {
            25 = { set_variable = { name = defection_offer value = 1 } } # Power and position
            25 = { set_variable = { name = defection_offer value = 2 } } # Gold and resources
            25 = { set_variable = { name = defection_offer value = 3 } } # Secret techniques
            25 = { set_variable = { name = defection_offer value = 4 } } # Protection from enemies
        }
    }
    
    # Option A: Accept and defect with clan secrets
    option = {
        name = murim_betrayal.0002.a
        trigger = {
            exists = scope:rival_recruiter
        }
        add_trait = murim_traitor
        remove_character_flag = murim_plotting_defection
        remove_character_modifier = murim_secret_defector
        # Gain from defection
        random_list = {
            25 = {
                add_gold = 500
                add_prestige = 200
            }
            25 = {
                add_prowess_skill = 3
                add_character_modifier = {
                    modifier = murim_stolen_clan_techniques
                    years = 10
                }
            }
            25 = {
                add_character_modifier = {
                    modifier = murim_high_position_in_new_clan
                    years = 10
                }
                add_prestige = 300
            }
            25 = {
                add_character_modifier = {
                    modifier = murim_protection_from_enemies
                    years = 5
                }
                add_gold = 200
            }
        }
        # Former clan reacts
        if = {
            limit = {
                any_liege_or_above = { is_alive = yes }
            }
            random_liege_or_above = {
                add_opinion = {
                    target = root
                    modifier = murim_traitor_defected
                    opinion = -100
                }
                trigger_event = {
                    id = murim_betrayal.0003
                    days = { 5 30 }
                }
            }
        }
        # Every clan member hostile
        every_known_character = {
            limit = {
                is_alive = yes
                OR = {
                    is_vassal_of = root.liege
                    is_courtier_of = root.liege
                }
            }
            add_opinion = {
                target = root
                modifier = murim_traitor_to_clan
                opinion = -40
            }
        }
        reverse_add_opinion = {
            target = scope:rival_recruiter
            modifier = murim_defector_welcomed
            opinion = 20
        }
        trigger_event = {
            id = murim_betrayal.0004
            days = { 30 90 }
        }
        ai_chance = {
            base = 40
            modifier = {
                factor = 2.0
                has_trait = ambitious
            }
            modifier = {
                factor = 2.0
                has_trait = deceitful
            }
            modifier = {
                factor = 0.2
                has_trait = honest
            }
        }
    }
    
    # Option B: Accept but plan to be a double agent
    option = {
        name = murim_betrayal.0002.b
        trigger = {
            exists = scope:rival_recruiter
            intrigue >= 16
        }
        add_character_flag = {
            flag = murim_double_agent_defector
            days = 3650
        }
        add_character_modifier = {
            modifier = murim_double_agent_infiltrator
            years = 5
        }
        add_intrigue_skill = 2
        remove_character_flag = murim_plotting_defection
        # Pretend to defect
        reverse_add_opinion = {
            target = scope:rival_recruiter
            modifier = murim_defector_welcomed
            opinion = 20
        }
        trigger_event = {
            id = murim_betrayal.0006
            days = { 90 365 }
        }
        ai_chance = {
            base = 25
            modifier = {
                factor = 3.0
                intrigue >= 20
            }
            modifier = {
                factor = 2.0
                has_trait = deceitful
            }
        }
    }
    
    # Option C: Change your mind - stay loyal
    option = {
        name = murim_betrayal.0002.c
        remove_character_flag = murim_plotting_defection
        remove_character_modifier = murim_secret_defector
        add_stress = 15
        add_piety = 50
        add_character_modifier = {
            modifier = murim_loyalty_reaffirmed
            years = 3
        }
        ai_chance = {
            base = 35
            modifier = {
                factor = 3.0
                has_trait = honest
            }
            modifier = {
                factor = 2.0
                has_trait = content
            }
        }
    }
}

            limit = {
                is_alive = yes

            exists = scope:traitor

                modifier = {
                    factor = 1.5

                    death = {
                        death_reason = death_murder_unknown

                    }
                }
                add_dread = 30
                add_prestige = 100

                desc = murim_betrayal.0003.a.failed
                add_prestige = -50
                scope:traitor = {

            }

            base = 40

                factor = 2.0
                has_trait = wrathful
            }
            modifier = {
                factor = 2.0
                has_trait = sadistic
            }
        }
    }

        name = murim_betrayal.0003.b
        trigger = {
            exists = scope:traitor
        }
        add_prestige = 100
        add_dread = 10

            add_prestige = -300

                modifier = murim_publicly_denounced_traitor

            limit = { is_alive = yes }

                target = scope:traitor

            base = 35
            modifier = {
                factor = 2.0
                has_trait = just
            }
            modifier = {
                factor = 1.5
                diplomacy >= 14
            }
        }
    }
    
    # Option C: Initiate loyalty purge of all members
    option = {
        name = murim_betrayal.0003.c
        add_dread = 20
        add_tyranny = 10
        add_character_modifier = {
            modifier = murim_loyalty_purge_active
            years = 2
        }

            limit = { is_alive = yes }

                80 = {
                    add_opinion = {
                        target = root
                        modifier = murim_loyalty_tested
                        opinion = -10
                    }

# EVENT 0005 - Loyalty Purge: Finding Suspects

            has_character_flag = suspected_traitor

    }
    
    immediate = {
        random_courtier = {
            limit = {
                has_character_flag = suspected_traitor
            }
            save_scope_as = purge_suspect
        }
        # Is the suspect actually guilty?
        random_list = {
            30 = { set_variable = { name = suspect_guilty value = 1 } }

    }

            exists = scope:purge_suspect
        }
        scope:purge_suspect = {
            death = {
                death_reason = death_execution
                killer = root
            }
        }
        add_dread = 20
        add_tyranny = 15
        every_courtier = {
            limit = { is_alive = yes }
            add_opinion = {
                target = root
                modifier = murim_executed_during_purge
                opinion = -20
            }
        }
        ai_chance = {
            base = 25
            modifier = {
                factor = 3.0
                has_trait = sadistic
            }
            modifier = {
                factor = 2.0
                has_trait = paranoid
            }
        }
    }
    
    # Option B: Interrogate thoroughly
    option = {
        name = murim_betrayal.0005.b
        trigger = {
            exists = scope:purge_suspect
        }
        random_list = {
            50 = {
                modifier = {
                    factor = 1.5
                    intrigue >= 16
                }
                desc = murim_betrayal.0005.b.truth_found
                if = {
                    limit = { has_variable = suspect_guilty }
                    # Correctly identified
                    scope:purge_suspect = {
                        imprison = {
                            target = prev
                            type = dungeon
                        }
                    }
                    add_prestige = 100
                }
                if = {
                    limit = { NOT = { has_variable = suspect_guilty } }
                    # Innocent - cleared
                    scope:purge_suspect = {
                        remove_character_flag = suspected_traitor
                        add_opinion = {
                            target = root
                            modifier = murim_cleared_of_suspicion
                            opinion = 10
                        }
                    }
                    add_prestige = 50
                }
            }
            50 = {
                desc = murim_betrayal.0005.b.inconclusive
                add_stress = 10
            }
        }
        ai_chance = {
            base = 45
            modifier = {
                factor = 2.0
                intrigue >= 16
            }
            modifier = {
                factor = 1.5
                has_trait = patient
            }
        }
    }
    
    # Option C: Release suspect with warning
    option = {
        name = murim_betrayal.0005.c
        trigger = {
            exists = scope:purge_suspect
        }
        scope:purge_suspect = {

# EVENT 0007 - Betrayal in Battle: Switching Sides Mid-Fight
# Triggered: during wars or major conflicts
# AI: Full battlefield betrayal
murim_betrayal.0007 = {
    type = character_event
    title = murim_betrayal.0007.t
    desc = murim_betrayal.0007.desc
    theme = martial
    
    left_portrait = {
        character = root
        animation = scheme
    }
    
    trigger = {
        is_alive = yes
        is_at_war = yes
        OR = {
            has_character_flag = murim_plotting_defection
            has_character_modifier = murim_secret_defector
            has_trait = ambitious
        }
        NOT = { has_trait = murim_traitor }
        NOT = { has_character_flag = battlefield_betrayal_considered }
    }
    
    weight_multiplier = {
        base = 0.3
        modifier = {
            factor = 3.0
            has_character_flag = murim_plotting_defection
        }
        modifier = {
            factor = 2.0
            has_character_modifier = murim_secret_defector
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = battlefield_betrayal_considered
            days = 730
        }
    }
    
    # Option A: Switch sides in the heat of battle
    option = {
        name = murim_betrayal.0007.a
        add_trait = murim_traitor
        add_character_modifier = {
            modifier = murim_battlefield_turncoat
            years = 10
        }
        add_dread = 20
        add_prestige = -300
        # Devastating blow to former side
        if = {
            limit = {
                any_liege_or_above = { is_alive = yes }
            }
            random_liege_or_above = {
                add_opinion = {
                    target = root
                    modifier = murim_battlefield_betrayal
                    opinion = -100
                }
                add_character_modifier = {
                    modifier = murim_betrayed_in_battle
                    years = 3
                }
            }
        }
        # New side rewards you
        add_gold = 300
        add_character_modifier = {
            modifier = murim_turncoat_reward
            years = 3
        }
        ai_chance = {
            base = 15
            modifier = {
                factor = 3.0
                has_character_flag = murim_plotting_defection
            }
            modifier = {
                factor = 2.0
                has_trait = deceitful
            }
            modifier = {
                factor = 0.05
                has_trait = honest
            }
        }
    }
    
    # Option B: Stay loyal despite temptation
    option = {
        name = murim_betrayal.0007.b
        add_piety = 50
        add_prestige = 50
        add_stress = 15
        add_character_modifier = {
            modifier = murim_loyalty_under_fire
            years = 5
        }
        ai_chance = {
            base = 60
            modifier = {
                factor = 2.0
                has_trait = brave
            }
            modifier = {
                factor = 2.0
                has_trait = honest
            }
        }
    }
    
    # Option C: Flee the battlefield entirely
    option = {
        name = murim_betrayal.0007.c
        add_prestige = -200
        add_character_modifier = {
            modifier = murim_battlefield_deserter
            years = 5
        }
        add_stress = -10
        ai_chance = {
            base = 20
            modifier = {
                factor = 3.0
                has_trait = craven
            }
        }
    }
}

# EVENT 0008 - Traitor's Return: Seeking Forgiveness
# Triggered: defectors who regret their choice
# AI: Full redemption/rejection system
murim_betrayal.0008 = {
    type = character_event
    title = murim_betrayal.0008.t
    desc = murim_betrayal.0008.desc
    theme = diplomacy
    
    left_portrait = {
        character = root
        animation = grief
    }
    right_portrait = {
        character = scope:former_leader
        animation = anger
    }
    
    trigger = {
        is_alive = yes
        has_trait = murim_traitor
        has_character_modifier = murim_defector_regret
    }
    
    immediate = {
        random_known_character = {
            limit = {
                is_alive = yes
                has_trait = murim_sect_leader
                opinion = {
                    target = root
                    value <= -30
                }
            }
            save_scope_as = former_leader
        }
    }
    
    # Option A: Return and beg forgiveness
    option = {
        name = murim_betrayal.0008.a
        trigger = {
            exists = scope:former_leader
        }
        random_list = {
            30 = {
                modifier = {
                    factor = 1.5
                    diplomacy >= 16
                }
                desc = murim_betrayal.0008.a.forgiven
                remove_trait = murim_traitor
                add_trait = murim_redeemed_cultivator
                add_piety = 200
                add_prestige = -200
                scope:former_leader = {
                    add_opinion = {
                        target = root
                        modifier = murim_forgave_traitor
                        opinion = -10
                    }
                }
                add_character_modifier = {
                    modifier = murim_on_probation
                    years = 5
                }
            }
            40 = {
                desc = murim_betrayal.0008.a.punished
                # Severe punishment but allowed back
                add_prowess_skill = -2
                add_prestige = -300
                imprison = {
                    target = root
                    type = dungeon
                }
                add_character_modifier = {
                    modifier = murim_punished_traitor
                    years = 3
                }
            }
            30 = {
                desc = murim_betrayal.0008.a.rejected
                add_prestige = -200
                add_stress = 30
                # Permanently exiled
                add_character_modifier = {
                    modifier = murim_permanently_exiled
                    years = 20
                }
            }
        }
        ai_chance = {
            base = 40
            modifier = {
                factor = 2.0
                has_trait = honest
            }
            modifier = {
                factor = 2.0
                has_trait = compassionate
            }
        }
    }
    
    # Option B: Offer valuable intelligence as a peace offering
    option = {
        name = murim_betrayal.0008.b
        trigger = {
            exists = scope:former_leader
            has_character_modifier = murim_kept_insurance_secrets
        }
        remove_character_modifier = murim_kept_insurance_secrets
        random_list = {
            50 = {
                desc = murim_betrayal.0008.b.accepted
                remove_trait = murim_traitor
                add_character_modifier = {
                    modifier = murim_intelligence_bargain_accepted
                    years = 5
                }
                scope:former_leader = {
                    add_opinion = {
                        target = root
                        modifier = murim_useful_intelligence
                        opinion = 15
                    }
                }
            }
            50 = {
                desc = murim_betrayal.0008.b.not_enough
                add_stress = 20
            }
        }
        ai_chance = {
            base = 35
            modifier = {
                factor = 2.0
                has_character_modifier = murim_kept_insurance_secrets
            }
        }
    }
    
    # Option C: Accept exile and move on
    option = {
        name = murim_betrayal.0008.c
        remove_character_modifier = murim_defector_regret
        add_trait = murim_wandering_cultivator
        add_character_modifier = {
            modifier = murim_exiled_wanderer
            years = 10
        }
        add_stress = -15
        ai_chance = {
            base = 25
            modifier = {
                factor = 2.0
                has_trait = content
            }
        }
    }
}
