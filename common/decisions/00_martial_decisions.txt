# Abandon Mortal Coil - Ascend to Void Realm
abandon_mortal_coil = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = decision_abandon_mortal_desc
	selection_tooltip = decision_abandon_mortal_tooltip
	major = yes

	is_shown = {
		is_ruler = yes
		age >= 40
		has_trait = nascent_soul
		NOT = { has_trait = void_realm }
	}

	is_valid = {
		prestige >= 1000
		piety >= 500
		stress < 50
		custom_description = {
			text = "decision_abandon_mortal_valid_tt"
			NOT = { has_character_modifier = qi_deviation_recovery }
		}
	}

	cost = {
		prestige = 500
		piety = 250
	}

	effect = {
		if = {
			limit = { has_trait = nascent_soul }
			remove_trait = nascent_soul
		}
		add_trait = void_realm
		add_prestige = 200
		add_piety = 500
		if = {
			limit = {
				has_variable = cultivation_rank
			}
			change_variable = { name = cultivation_rank add = 1 }
		}
		else = {
			set_variable = { name = cultivation_rank value = 7 }
		}
		# Lifespan extension
		add_character_modifier = {
			modifier = void_realm_longevity
			years = 200
		}
		custom_tooltip = tt_ascending_void
		# Notify sect members
		every_vassal = {
			limit = { is_alive = yes }
			custom_tooltip = tt_master_ascended_void
		}
	}

	cooldown = { years = 20 }

	ai_check_interval = 60
	ai_potential = {
		age >= 45
		has_trait = nascent_soul
	}
	ai_will_do = {
		base = 80
		modifier = {
			add = 20
			age >= 60
		}
		modifier = {
			add = -30
			is_at_war = yes
		}
	}
}

# Recruit Close Disciple
close_sect_disciple = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = decision_close_disciple_desc
	selection_tooltip = decision_close_disciple_tooltip

	is_shown = {
		is_ruler = yes
		age >= 30
		martial >= 15
	}

	is_valid = {
		gold >= 100
		prestige >= 100
		custom_description = {
			text = "decision_close_disciple_limit"
			NOT = { has_character_modifier = recently_recruited_disciple }
		}
	}

	cost = {
		gold = 100
		prestige = 50
	}

	effect = {
		add_character_modifier = {
			modifier = recently_recruited_disciple
			years = 5
		}
		create_character = {
			location = root.capital_province
			age = { 16 20 }
			trait = qi_refinement
			faith = root.faith
			culture = root.culture
			dynasty = root.dynasty
			employer = root
			save_scope_as = new_disciple
		}
		hidden_effect = {
			scope:new_disciple = {
				set_variable = { name = murim_master value = root }
				add_opinion = {
					target = root
					modifier = master_disciple_bond
				}
				# Initialize cultivation
				set_variable = { name = cultivation_rank value = 1 }
				set_variable = { name = cultivation_progress value = 0 }
			}
		}
		# Track disciple count
		if = {
			limit = { has_variable = murim_disciple_count }
			change_variable = { name = murim_disciple_count add = 1 }
		}
		else = {
			set_variable = { name = murim_disciple_count value = 1 }
		}
		if = {
			limit = { NOT = { has_trait = respected_master } }
			add_trait = respected_master
		}
		custom_tooltip = tt_new_disciple_recruited
	}

	cooldown = { years = 5 }

	ai_check_interval = 60
	ai_potential = {
		age >= 35
		martial >= 18
	}
	ai_will_do = {
		base = 40
		modifier = {
			add = 20
			has_trait = generous
		}
		modifier = {
			add = -20
			gold < 200
		}
	}
}

# Comprehend Advanced Technique
comprehend_technique = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	desc = decision_comprehend_desc
	selection_tooltip = decision_comprehend_tooltip

	is_shown = {
		is_ruler = yes
		OR = {
			has_trait = sword_aptitude_2
			has_trait = fist_aptitude_2
			has_trait = palm_aptitude_2
			has_trait = movement_aptitude_2
			AND = {
				has_variable = technique_mastery
				var:technique_mastery >= 3
			}
		}
		NOT = { has_character_modifier = recently_comprehended }
	}

	is_valid = {
		learning >= 10
		stress < 60
		custom_description = {
			text = "decision_comprehend_requires_study"
			OR = {
				has_trait = secret_manual_holder
				prestige >= 500
				AND = {
					has_variable = technique_mastery
					var:technique_mastery >= 5
				}
			}
		}
	}

	cost = {
		prestige = 200
	}

	effect = {
		add_character_modifier = {
			modifier = recently_comprehended
			years = 10
		}
		# Determine which technique to advance
		if = {
			limit = { has_trait = sword_aptitude_2 }
			remove_trait = sword_aptitude_2
			add_trait = sword_aptitude_3
			add_prowess = 5
			custom_tooltip = tt_sword_technique_advanced
		}
		else_if = {
			limit = { has_trait = fist_aptitude_2 }
			remove_trait = fist_aptitude_2
			add_trait = fist_aptitude_3
			add_prowess = 5
			custom_tooltip = tt_fist_technique_advanced
		}
		else_if = {
			limit = { has_trait = palm_aptitude_2 }
			remove_trait = palm_aptitude_2
			add_trait = palm_aptitude_3
			add_prowess = 4
			custom_tooltip = tt_palm_technique_advanced
		}
		else_if = {
			limit = { has_trait = movement_aptitude_2 }
			remove_trait = movement_aptitude_2
			add_trait = movement_aptitude_3
			add_prowess = 3
			custom_tooltip = tt_movement_technique_advanced
		}
		# Track comprehension
		if = {
			limit = { has_variable = technique_mastery }
			change_variable = { name = technique_mastery add = 1 }
		}
		else = {
			set_variable = { name = technique_mastery value = 1 }
		}
		# Remove manual if used
		if = {
			limit = { has_trait = secret_manual_holder }
			remove_trait = secret_manual_holder
		}
		add_dynasty_prestige = 100
	}

	cooldown = { years = 10 }

	ai_check_interval = 60
	ai_will_do = {
		base = 60
		modifier = {
			add = 20
			has_trait = diligent
		}
	}
}
