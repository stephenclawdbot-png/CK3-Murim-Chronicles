### EVENT 1: Declaration of Holy War ###
murim_faction_war_exp2.0001 = {
	type = character_event
	title = murim_faction_war_exp2.0001.t
	desc = murim_faction_war_exp2.0001.desc
	theme = war
	
	trigger = {
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_disciple
			has_trait = wudang_disciple
		}
		martial >= 18
		NOT = { has_character_flag = holy_war_declared }
		any_ruler = {
			OR = {
				has_trait = phantom_blade_disciple
				has_trait = poison_flower_disciple
				has_trait = heavenly_demon_heir
			}
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = zealous }
		modifier = { add = 2 martial >= 22 }
	}
	
	immediate = {
		random_ruler = {
			limit = {
				OR = {
					has_trait = phantom_blade_disciple
					has_trait = poison_flower_disciple
					has_trait = heavenly_demon_heir
				}
				martial >= 12
			}
			save_scope_as = unorthodox_target
		}
	}
	
	# Option A: Declare holy purge on all unorthodox
	option = {
		name = murim_faction_war_exp2.0001.a
		add_character_flag = { flag = holy_war_declared days = -1 }
		add_character_flag = { flag = orthodox_crusader days = -1 }
		add_prestige = 800
		add_piety = 500
		add_dread = 30
		reverse_add_opinion = {
			target = scope:unorthodox_target
			modifier = holy_war_enemy_opinion
			opinion = -100
		}
		hidden_effect = {
			every_ruler = {
				limit = {
					OR = {
						has_trait = mount_hua_disciple
						has_trait = shaolin_disciple
						has_trait = wudang_disciple
					}
				}
				add_opinion = { target = root modifier = holy_war_ally_opinion opinion = 30 }
			}
			trigger_event = { id = murim_faction_war_exp2.0004 days = { 30 90 } }
		}
		ai_chance = { base = 40 modifier = { add = 30 has_trait = zealous } }
	}
	
	# Option B: Targeted strike on one unorthodox clan only
	option = {
		name = murim_faction_war_exp2.0001.b
		add_character_flag = { flag = holy_war_declared days = -1 }
		add_character_flag = { flag = targeted_purge days = -1 }
		add_prestige = 400
		add_piety = 200
		reverse_add_opinion = {
			target = scope:unorthodox_target
			modifier = targeted_enemy_opinion
			opinion = -60
		}
		ai_chance = { base = 35 }
	}
	
	# Option C: Propose peace summit instead
	option = {
		name = murim_faction_war_exp2.0001.c
		trigger = { diplomacy >= 16 }
		add_prestige = 300
		add_piety = 300
		add_diplomacy_skill = 1
		hidden_effect = {
			trigger_event = { id = murim_faction_war_exp2.0003 days = { 15 45 } }
		}
		ai_chance = { base = 20 modifier = { add = 20 has_trait = compassionate } }
	}
	
	# Option D: Stay neutral - let others fight
	option = {
		name = murim_faction_war_exp2.0001.d
		add_prestige = -200
		add_stress = 15
		stress_impact = { zealous = major_stress_impact_gain }
		ai_chance = { base = 10 }
	}
}
