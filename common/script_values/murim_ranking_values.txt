# Murim Chronicles - Top 50 Martial Artist Ranking System
# Hidden rank score calculated yearly for all martial artists
# Formula: (prowess x 2) + (cultivation_rank_weight x 10) + (technique_count x 3) + (duel_wins x 5) - (duel_losses x 2) - (wounds x 3)
#          + age_experience_bonus + element_mastery_bonus + bloodline_bonus + sect_prestige_bonus
#          + weapon_mastery_bonus + fame_bonus + streak_bonus + title_weight + special_condition_bonus
# Load-safe: all values have fallback defaults, exists checks on all variables
# POLISH FIX: All trait references now match actual definitions in trait files
# Cross-references: 01_martial_traits.txt, murim_cultivation_traits.txt, murim_technique_traits.txt, murim_combat_trauma_traits.txt

######################################
# CULTIVATION RANK WEIGHT
# Converts cultivation tier to numeric weight for ranking formula
# Uses OR blocks to match both 01_martial_traits.txt AND murim_cultivation_traits.txt names
######################################

murim_cultivation_rank_weight_value = {
	value = 0
	# Tier 1: Outer Qi / Mortal Realm (lowest cultivators)
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_outer_qi
				has_trait = mortal_realm
			}
		}
		add = 1
	}
	# Tier 2: Qi Gathering / Qi Condensation Realm
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_qi_gathering
				has_trait = qi_condensation_realm
			}
		}
		add = 2
	}
	# Tier 3: Foundation / Foundation Realm
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_foundation
				has_trait = foundation_realm
			}
		}
		add = 3
	}
	# Tier 4: Golden Core / Core Formation Realm
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_golden_core
				has_trait = core_formation_realm
			}
		}
		add = 5
	}
	# Tier 5: Nascent Soul / Nascent Soul Realm
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_nascent_soul
				has_trait = nascent_soul_realm
			}
		}
		add = 8
	}
	# Tier 6: Spirit Severing / Ascension Realm
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_spirit_severing
				has_trait = ascension_realm
			}
		}
		add = 12
	}
	# Tier 7: Heavenly Tribulation (highest - no equivalent in cultivation_traits)
	if = {
		limit = {
			exists = this
			has_trait = murim_heavenly_tribulation
		}
		add = 25
	}
}

######################################

murim_age_experience_bonus_value = {
	value = 0
	if = {
		limit = {
			exists = this
			age >= 20
			age < 30
		}
		add = 2
	}
	if = {
		limit = {
			exists = this

		}
		add = 5

		limit = {
			exists = this

			age < 70
		}
		add = 8
	}
	if = {
		limit = {
			exists = this
			age >= 70
		}
		add = 10
	}

######################################
# FAME AND PRESTIGE BONUS
######################################

murim_fame_bonus_value = {
	value = 0
	if = {
		limit = {
			exists = this
			prestige_level >= 3
		}
		add = 3
	}
	if = {
		limit = {
			exists = this
			prestige_level >= 5
		}
		add = 3
	}
	if = {
		limit = {
			exists = this
			prestige_level >= 7
		}
		add = 5
	}
	# Kill streak bonus
	if = {
		limit = {
			exists = this
			exists = var:murim_duel_kills
			var:murim_duel_kills >= 5
		}
		add = 3
	}
	if = {
		limit = {
			exists = this
			exists = var:murim_duel_kills
			var:murim_duel_kills >= 10
		}
		add = 5
	}
}

######################################
# MASTER COMPOSITE RANKING SCORE
# The final formula that determines Top 50 position
######################################

murim_total_ranking_score_value = {
	value = 0
	# Base: prowess x 2
	add = {
		value = prowess
		multiply = 2
	}
	# Cultivation weight x 10
	add = {
		value = murim_cultivation_rank_weight_value
		multiply = 10
	}
	# Technique count x 3
	add = {
		value = murim_technique_count_value
		multiply = 3
	}
	# Duel wins x 5
	add = {
		value = murim_duel_wins_value
		multiply = 5
	}
	# Duel losses x -2
	subtract = {
		value = murim_duel_losses_value
		multiply = 2
	}
	# Wound penalties x -3
	subtract = {
		value = murim_wound_penalty_value
		multiply = 3
	}
	# Age experience bonus
	add = murim_age_experience_bonus_value
	# Special condition bonuses
	add = murim_special_condition_bonus_value
	# Fame bonus
	add = murim_fame_bonus_value
	# Title weight
	add = murim_title_weight_value
	# Floor at 0
	min = 0
}
