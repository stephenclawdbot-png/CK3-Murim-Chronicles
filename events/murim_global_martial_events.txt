# Murim Chronicles - Global Martial Arts Events
# Foreign masters are RARE: 1 appears every 10-20 years in the murim heartland
# Danger comes from UNKNOWN style, not raw power — once fought, you learn their patterns
# Use has_character_flag = fought_X_style to track familiarity (first fight disadvantage, subsequent normal)
# Balance: every foreign style has a counter. No style dominates.
# Cross-refs: murim_duel (foreign duels), murim_ranking (foreign challenger to Top 50),
#             murim_training (learn foreign technique), murim_politics (foreign alliance),
#             murim_tang_clan_events (Tang meets Greek Fire), murim_mount_hua_events (sword exchange),
#             murim_heavenly_demon_events (Berserker vs Demon), murim_artifact (foreign weapons),
#             murim_clan_war (foreign faction intervention), murim_daily_life (wandering master arrival),
#             murim_injury_chance_values (foreign style modifier), murim_global_martial_traits (all traits)
# All combat uses murim_injury_chance. Every event chains to 3+ other files.
# WANDERING MASTER system: landless adventurers with martial traits trigger court arrival events.
# AI vs AI: foreign masters duel AI martial artists automatically.

namespace = murim_global

# ============================================================
# EVENT 0001: Foreign Martial Master Arrives in the Murim Heartland
# ENTRY POINT for all global martial content
# Chains to: duel, training, politics, daily_life, ranking
# Fires from on_action yearly pulse with very low MTTH (~10-20 year cycle)
# Also triggered by: wandering_martial_master arriving at court
# ============================================================
murim_global.0001 = {
	type = character_event
	title = murim_global.0001.t
	desc = murim_global.0001.desc
	theme = martial
	
	left_portrait = {
		character = root
		animation = shock
	}
	
	trigger = {
		is_alive = yes
		is_landed = yes
		# Must have some martial awareness
		OR = {
			prowess >= 10
			any_held_title = { tier >= tier_county }
		}
		NOT = { has_character_flag = foreign_master_arrival_cooldown }
	}
	
	weight_multiplier = {
		base = 1
		# More likely in murim heartland (Korea/China)
		modifier = {
			add = 2.0
			culture = { has_cultural_pillar = heritage_korean }
		}
		modifier = {
			add = 2.0
			culture = { has_cultural_pillar = heritage_chinese }
		}
		# Less likely elsewhere
		modifier = {
			add = -0.5
			NOR = {
				culture = { has_cultural_pillar = heritage_korean }
				culture = { has_cultural_pillar = heritage_chinese }
			}
		}
	}
	
	immediate = {
		set_character_flag = {
			flag = foreign_master_arrival_cooldown
			years = 10  # Rare: once per decade
		}
		# Determine what kind of foreign master arrives (random)
		random_list = {
			15 = { set_variable = { name = foreign_master_type value = 1 } }  # Japanese
			12 = { set_variable = { name = foreign_master_type value = 2 } }  # Indian
			10 = { set_variable = { name = foreign_master_type value = 3 } }  # Siam/Khmer
			10 = { set_variable = { name = foreign_master_type value = 4 } }  # Persian
			12 = { set_variable = { name = foreign_master_type value = 5 } }  # Mongol
			10 = { set_variable = { name = foreign_master_type value = 6 } }  # Norse
			8 = { set_variable = { name = foreign_master_type value = 7 } }   # Byzantine
			8 = { set_variable = { name = foreign_master_type value = 8 } }   # African
			15 = { set_variable = { name = foreign_master_type value = 9 } }  # Goryeo/Chinese wanderer
		}
	}
	
	# Option A: Challenge the foreign master — test their strength
	option = {
		name = murim_global.0001.a
		
		# First encounter penalty if you haven't fought this style before
		# Chain to: DUEL — foreign style combat
		duel = {
			skill = prowess
			value = high
			
			# Unknown style disadvantage: -10% success if first encounter
			# Tracked by: fought_japanese_style, fought_indian_style, etc.
			40 = {
				desc = murim_global.0001.a.win
				add_prestige = 400
				# Learn the style — gain foreign_style_veteran trait
				if = {
					limit = {
						NOT = { has_trait = foreign_style_veteran }
					}
					add_trait = foreign_style_veteran
				}
				# Set style familiarity flag based on type
				if = {
					limit = { var:foreign_master_type = 1 }
					set_character_flag = { flag = fought_japanese_style }
				}
				if = {
					limit = { var:foreign_master_type = 2 }
					set_character_flag = { flag = fought_indian_style }
				}
				if = {
					limit = { var:foreign_master_type = 3 }
					set_character_flag = { flag = fought_southeast_asian_style }
				}
				if = {
					limit = { var:foreign_master_type = 4 }
					set_character_flag = { flag = fought_persian_style }
				}
				if = {
					limit = { var:foreign_master_type = 5 }
					set_character_flag = { flag = fought_mongol_style }
				}
				if = {
					limit = { var:foreign_master_type = 6 }
					set_character_flag = { flag = fought_norse_style }
				}
				if = {
					limit = { var:foreign_master_type = 7 }
					set_character_flag = { flag = fought_byzantine_style }
				}
				if = {
					limit = { var:foreign_master_type = 8 }
					set_character_flag = { flag = fought_african_style }
				}
				# Chain to: RANKING — defeating foreign master may boost rank
				change_variable = {
					name = murim_rank_prestige
					add = 10
				}
				# Injury roll — foreign fights are dangerous due to unfamiliarity
				random = {
					chance = murim_injury_chance_honor_duel_winner
					random_list = {
						60 = { add_trait = wounded }
						25 = { add_character_modifier = { modifier = foreign_combat_scar days = 365 } }
						15 = { add_trait = one_eyed }
					}
				}
				create_character_memory = {
					type = defeated_foreign_master_memory
				}
			}
			
			60 = {
				desc = murim_global.0001.a.lose
				# Unknown style = higher loss chance first time
				add_prestige = -100
				stress_impact = {
					base = 20
					arrogant = 15
				}
				# Still learn from the loss
				if = {
					limit = {
						NOT = { has_trait = foreign_style_veteran }
					}
					add_trait = foreign_style_veteran
				}
				# Injury — loser against unknown style, higher severity
				random = {
					chance = murim_injury_chance_honor_duel_loser
					random_list = {
						40 = { add_trait = wounded }
						25 = { add_character_modifier = { modifier = mysterious_wound days = 365 } }
						20 = { add_trait = one_eyed }
						10 = { add_trait = disfigured }
						5 = { add_character_modifier = { modifier = shattered_meridians days = 1825 } }
					}
				}
				# Chain to: TRAINING — motivate to learn the foreign style
				trigger_event = {
					id = murim_training.0002  # Intense training after loss
					days = { 14 30 }
				}
				create_character_memory = {
					type = defeated_by_foreign_master_memory
				}
			}
		}
		
		ai_chance = {
			base = 35
			modifier = {
				add = 20
				has_trait = brave
			}
			modifier = {
				add = 15
				has_trait = ambitious
			}
			modifier = {
				add = -20
				has_trait = craven
			}
		}
	}
	
	# Option B: Welcome and recruit — offer a position at court
	option = {
		name = murim_global.0001.b
		
		add_gold = -100  # hosting costs
		add_prestige = 100
		# Recruit foreign master as courtier (wandering master joins court)
		add_character_modifier = {
			modifier = foreign_master_at_court
			years = 5
		}
		# Chain to: TRAINING — foreign master can teach you
		random = {
			chance = 40
			trigger_event = {
				id = murim_global.0003  # Cross-cultural technique exchange
				days = { 30 60 }
			}
		}
		# Chain to: POLITICS — having a foreign master raises your prestige
		add_character_modifier = {
			modifier = cosmopolitan_court
			years = 3
		}
		# Other sects notice
		random = {
			chance = 20
			trigger_event = {
				id = murim_politics.0008  # Neighbors concerned about foreign fighter
				days = { 30 90 }
			}
		}
		
		ai_chance = {
			base = 30
			modifier = {
				add = 15
				diplomacy >= 12
			}
		}
	}
	
	# Option C: Spy on their techniques — observe from the shadows
	option = {
		name = murim_global.0001.c
		
		intrigue >= 8
		
		duel = {
			skill = intrigue
			value = average
			
			60 = {
				desc = murim_global.0001.c.success
				# Learn their weaknesses without fighting
				add_intrigue = 1
				add_character_flag = {
					flag = studied_foreign_techniques
					years = 5
				}
				# Chain to: JIANGHU — intelligence network gains
				change_variable = {
					name = jianghu_intelligence
					add = 5
				}
				# Next encounter with this style: no disadvantage
				if = {
					limit = { var:foreign_master_type = 1 }
					set_character_flag = { flag = fought_japanese_style }
				}
				if = {
					limit = { var:foreign_master_type = 2 }
					set_character_flag = { flag = fought_indian_style }
				}
				# (repeat for all types)
			}
			
			40 = {
				desc = murim_global.0001.c.failure
				# Caught spying — foreign master is offended
				add_prestige = -50
				stress_impact = {
					base = 15
				}
				# Chain to: DUEL — forced confrontation
				random = {
					chance = 50
					trigger_event = {
						id = murim_duel.0005  # Hostile encounter
						days = { 1 3 }
					}
				}
			}
		}
		
		ai_chance = {
			base = 20
			modifier = {
				add = 15
				has_trait = deceitful
			}
		}
	}
	
	# Option D: Ignore — the murim world has enough problems
	option = {
		name = murim_global.0001.d
		
		stress_impact = {
			base = -5
		}
		# Foreign master moves on — may appear at another court
		# Chain to: DAILY LIFE — rumors about the foreign master spread
		random = {
			chance = 30
			add_character_modifier = {
				modifier = missed_opportunity_regret
				years = 1
			}
		}
		
		ai_chance = {
			base = 15
			modifier = {
				add = 10
				has_trait = content
			}
		}
	}
}

		is_alive = yes

		set_character_flag = {
			flag = unknown_style_cooldown
			years = 3
		}
	}
	
	# Option A: Engage and try to adapt mid-fight

		name = murim_global.0002.a
		
		# DISADVANTAGE first round — can't read their moves
		duel = {

				}

						days = { 7 14 }
					}
				}
				# Injury from unknown style — unpredictable
				random = {
					chance = murim_injury_chance_honor_duel_winner

						15 = { add_trait = one_eyed }
						10 = { add_trait = disfigured }
						5 = { add_character_modifier = { modifier = shattered_meridians days = 1825 } }
					}
				}
				# Chain to: TRAINING — defeat motivates study
				trigger_event = {
					id = murim_training.0001  # Training montage
					days = { 14 30 }
				}
				create_character_memory = {
					type = defeated_by_unknown_style_memory
				}
			}
		}

			base = 40
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}
	
	# Option B: Observe without engaging — learn their patterns first
	option = {
		name = murim_global.0002.b
		
		add_learning = 1
		# Chain to: TRAINING — study the style before fighting
		add_character_modifier = {
			modifier = studying_unknown_style
			years = 1
		}

			flag = observed_unknown_style
			years = 5
		}
		
		ai_chance = {
			base = 30
			modifier = {
				add = 15
				has_trait = patient
			}
		}
	}
	
	# Option C: Flee — this is too dangerous
	option = {
		name = murim_global.0002.c
		
		stress_impact = {
			base = 10
			brave = 20
			craven = -10
		}
		add_prestige = -50
		# Chain to: DAILY LIFE — rumor spreads that you ran
		random = {
			chance = 30
			add_character_modifier = {
				modifier = reputation_for_cowardice
				years = 2
			}
		}
		
		ai_chance = {
			base = 15
			modifier = {
				add = 20
				has_trait = craven
			}
		}
	}
	
	# Option D: Try to communicate — maybe they're friendly
	option = {
		name = murim_global.0002.d
		
		diplomacy >= 10
		
		duel = {
			skill = diplomacy
			value = average

					years = 10
				}
			}

				# Misunderstanding — they attack
				stress_impact = {

# EVENT 0003: Cross-Cultural Technique Exchange

# ============================================================
# EVENT 0004: Foreign Assassination Attempt
# Hashashin or Shinobi sent after a ranked fighter
# Unique stealth duel mechanics — intrigue vs awareness
# Chains to: duel, ranking, jianghu, politics
# ============================================================
murim_global.0004 = {
	type = character_event
	title = murim_global.0004.t
	desc = murim_global.0004.desc
	theme = skull
	
	left_portrait = {
		character = root
		animation = fear
	}
	
	trigger = {
		is_alive = yes
		is_landed = yes
		# Only targets ranked or important fighters
		OR = {
			prowess >= 15
			has_variable = murim_rank
		}
		NOT = { has_character_flag = assassination_attempt_cooldown }
	}
	
	immediate = {
		set_character_flag = {
			flag = assassination_attempt_cooldown
			years = 5
		}
	}
	
	# Option A: Your instincts warn you — fight back (prowess check)
	option = {
		name = murim_global.0004.a
		
		duel = {
			skill = prowess
			value = very_hard  # assassin has surprise advantage
			
			35 = {
				desc = murim_global.0004.a.win
				add_prestige = 500
				add_dread = 20
				# Chain to: RANKING — surviving assassination boosts reputation
				change_variable = {
					name = murim_rank_prestige
					add = 10
				}
				# Capture or kill assassin
				random_list = {
					50 = {
						# Killed the assassin
						add_prestige = 100
						create_character_memory = {
							type = killed_foreign_assassin_memory
						}
					}
					50 = {
						# Captured alive — can interrogate
						# Chain to: JIANGHU — discover who sent them
						trigger_event = {
							id = murim_jianghu.0005  # Interrogation chain
							days = { 3 7 }
						}
					}
				}
				# Injury from ambush — even if you win
				random = {
					chance = murim_injury_chance_ambush_defender_success
					random_list = {
						50 = { add_trait = wounded }
						30 = { add_character_modifier = { modifier = assassin_poison_wound days = 180 } }
						20 = { add_trait = one_eyed }
					}
				}
			}
			
			65 = {
				desc = murim_global.0004.a.lose
				# Assassination partially succeeds
				random_list = {
					30 = {
						# Near death — survived but badly hurt
						add_trait = wounded
						add_character_modifier = { modifier = near_death_experience days = 365 }
						# Chain to: FATE — near death changes perspective
						trigger_event = {
							id = murim_fate.0001  # Brush with death
							days = { 7 14 }
						}
					}
					40 = {
						# Poisoned wound — slow death unless treated
