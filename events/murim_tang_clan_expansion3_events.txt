murim_tang_clan_expansion3.0001 = {
	type = character_event
	title = murim_tang_clan_expansion3.0001.t
	desc = murim_tang_clan_expansion3.0001.desc
	theme = martial_arts
	
	# The Thousand Poison Cauldron — Tang Clan's ultimate test. A disciple must survive
	# three days sealed inside a room with 1000 different poisons. Those who emerge alive
	# gain immunity to all known toxins. 30% death rate historically.
	
	trigger = {
		has_trait = tang_clan_member
		age >= 18
		age <= 35
		NOT = { has_character_flag = attempted_poison_cauldron }
		prowess >= 8
	}
	
	weight_multiplier = {
		base = 1
		modifier = { add = 3 has_trait = ambitious }
		modifier = { add = 2 has_trait = brave }
		modifier = { add = -5 has_trait = craven }
	}
	
	immediate = {
		set_character_flag = attempted_poison_cauldron
	}
	
	option = {
		name = murim_tang_clan_expansion3.0001.a
		# Enter the Cauldron Room
		random_list = {
			35 = {
				# Perfect immunity achieved
				add_trait = poison_immune
				add_prowess = 6
				add_character_modifier = {
					modifier = thousand_poison_body
					years = -1
				}
				add_prestige = 500
				hidden_effect = {
					trigger_event = {
						id = murim_tang_clan_expansion3.0003
						days = { 30 60 }
					}
				}
			}
			35 = {
				# Partial immunity — scarred but stronger
				add_trait = poison_resistant
				add_prowess = 3
				add_health = -1
				add_character_modifier = {
					modifier = poison_scarred_body
					years = -1
				}
			}
			30 = {
				# Death in the cauldron
				death = { death_reason = death_poison_trial }
			}
		}
		ai_chance = {
			base = 50
			modifier = { add = 30 has_trait = brave }
		}
	}
	
	option = {
		name = murim_tang_clan_expansion3.0001.b
		# Refuse — too risky
		add_stress = 20
		add_character_modifier = {
			modifier = cauldron_refusal_shame
			years = 5
		}
		ai_chance = {
			base = 30
			modifier = { add = 30 has_trait = craven }
		}
	}
	
	option = {
		name = murim_tang_clan_expansion3.0001.c
		# Secretly take an antidote beforehand (cheating)
		add_intrigue = 4
		random_list = {
			60 = {
				add_trait = poison_resistant
				add_prowess = 2
				add_character_modifier = {
					modifier = false_immunity
					years = 10
				}
			}
			40 = {
				# Caught cheating — expelled from inner circle
				add_trait = tang_clan_disgraced
				add_prestige = -300
				add_opinion = {
					target = liege
					modifier = cheater_exposed
					opinion = -50
				}
			}
		}
		ai_chance = {
			base = 20
			modifier = { add = 20 has_trait = deceitful }
		}
	}
}
