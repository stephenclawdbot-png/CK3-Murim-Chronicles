# A character discovers the remnants of a legendary battle from centuries past
######################################
murim_world_history.0001 = {
	type = character_event
	title = murim_world_history.0001.t
	desc = murim_world_history.0001.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		is_landed = yes
		has_variable = murim_cultivation_level
		NOT = { has_character_flag = found_ancient_battlefield }

	weight_multiplier = {
		base = 1
		modifier = {
			factor = 2
			has_trait = curious
		}
		modifier = {
			factor = 1.5
			var:murim_cultivation_level >= 5
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = found_ancient_battlefield
			days = 3650
		}
		create_character_memory = {
			type = murim_discovered_ancient_battlefield
		}
	}
	
	# Option A: Search the battlefield for relics
	option = {
		name = murim_world_history.0001.a
		
		random_list = {
			40 = {
				desc = murim_world_history.0001.a.success

					modifier = ancient_battlefield_insights

					id = murim_artifact.0001
					days = { 15 45 }
				}
			}
			30 = {
				desc = murim_world_history.0001.a.partial
				add_prestige = 200
				add_character_modifier = {
					modifier = ancient_qi_residue_exposure
					years = 5
				}
			}
			30 = {
				desc = murim_world_history.0001.a.trap
				add_trait = wounded
				add_prestige = -100
				create_character_memory = {
					type = murim_wounded_by_ancient_formation
				}
				trigger_event = {
					id = murim_fate.0001
					days = { 30 90 }
				}
			}
		}
		
		ai_chance = {
			base = 60
			modifier = {
				factor = 1.5
				has_trait = brave
			}

					type = murim_vision_of_ancient_masters
