###################################
# SECTION 1: CULTIVATION ELIGIBILITY
###################################

murim_can_attempt_body_tempering_trigger = {
	is_alive = yes
	age >= 16
	NOT = { has_trait = murim_qi_deviation_crippling }
	NOT = { has_trait = murim_breakthrough_failure }
	prowess >= 8
	NOT = { has_character_flag = murim_tempering_cooldown }
}

murim_can_attempt_breakthrough_trigger = {
	is_alive = yes
	age >= 16
	exists = var:murim_cultivation_rank
	NOT = {
		custom_description = {
			text = murim_max_cultivation_rank_reached
			var:murim_cultivation_rank >= 6
		}
	}
	murim_has_qi_deviation_trigger = no
	NOT = { has_trait = murim_breakthrough_failure }
	NOT = { has_character_flag = murim_breakthrough_cooldown }
}

murim_can_attempt_tribulation_trigger = {
	is_alive = yes
	age >= 16
	exists = var:murim_cultivation_rank
	prowess >= 20
	murim_has_qi_deviation_trigger = no
}

murim_can_purify_qi_trigger = {
	is_alive = yes
	OR = {
		has_character_modifier = murim_corrupted_qi_poisoning
		has_character_flag = murim_corrupted_qi
	}

murim_is_cultivator_trigger = {
	OR = {
		exists = var:murim_cultivation_rank
		murim_has_any_technique_trait_trigger = yes
		murim_has_body_tempering_trigger = yes
		has_character_flag = murim_is_cultivator
	}
}

murim_is_high_realm_cultivator_trigger = {
	exists = var:murim_cultivation_rank
	trigger_if = {
		limit = { exists = var:murim_cultivation_rank }
		var:murim_cultivation_rank >= 4
	}
}

murim_is_peak_realm_cultivator_trigger = {
	exists = var:murim_cultivation_rank
	trigger_if = {
		limit = { exists = var:murim_cultivation_rank }
		var:murim_cultivation_rank >= 6
	}
}
