# Murim Chronicles - Tang Poison Clan Events
# The Tang Family: masters of poison arts, hidden weapons, assassination
# Fighting style: indirect, deceptive, lethal from distance
# High lethality but low direct prowess — fights through attrition and poison stacking
# Cross-refs: murim_duel (poison duels), murim_jianghu (black market poison trade),
#             murim_artifact (alchemy cauldrons), murim_rank (infamy), murim_clan_war (righteous vs demonic)
# All combat uses murim_injury_chance + murim_poison_injury_chance values
# Tang Clan is NEUTRAL — sells poison to both righteous and demonic sides

namespace = murim_tang

########################################
# 0001 - TANG CLAN POISON REFINEMENT
# Core alchemy chain: refine poisons for use/trade
# Entry point to Tang Clan identity
########################################

murim_tang.0001 = {
	type = character_event
	title = murim_tang.0001.t
	desc = murim_tang.0001.desc
	theme = intrigue_activity
	
	left_portrait = {
		character = root
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
		is_imprisoned = no
		has_trait = tang_poison_master
		NOT = { has_character_flag = tang_refining_cooldown }
		OR = {
			intrigue >= 12
			learning >= 10
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_trait = tang_clan_heir
		}
		modifier = {
			add = 0.3
			intrigue >= 16
		}
		modifier = {
			add = 0.2
			learning >= 14
		}
	}

	immediate = {
		add_character_flag = {
			flag = tang_refining_cooldown
			years = 1
		}
	}

	# Option A: Refine lethal combat poison (high risk, high reward)
	option = {
		name = murim_tang.0001.a
		trigger = {
			intrigue >= 14
		}
		random_list = {
			# Success: deadly poison created
			60 = {
				modifier = {
					add = 15
					learning >= 16
				}
				modifier = {
					add = 10
					has_trait = alchemy_master
				}
				send_interface_toast = {
					title = murim_tang_refine_success
					left_icon = root
				}
				add_character_modifier = {
					modifier = tang_lethal_poison_prepared
					years = 3
				}
				# Poison gives huge intrigue boost in next combat
				add_intrigue = 1
				add_prestige = 100
				if = {
					limit = { has_variable = tang_poisons_refined }
					change_variable = { name = tang_poisons_refined add = 1 }
				}
				if = {
					limit = { NOT = { has_variable = tang_poisons_refined } }
					set_variable = { name = tang_poisons_refined value = 1 }
				}
			}
			# Partial success: weaker poison
			25 = {
				send_interface_toast = {
					title = murim_tang_refine_partial
					left_icon = root
				}
				add_character_modifier = {
					modifier = tang_mild_poison_prepared
					years = 2
				}
				add_prestige = 30
			}
			# Failure: poisoned yourself
			15 = {
				modifier = {
					add = -10
					has_trait = poison_resistance_trained
				}
				send_interface_toast = {
					title = murim_tang_refine_fail
					left_icon = root
				}
				add_character_modifier = {
					modifier = tang_self_poisoned
					years = 1
				}
				# Health damage from accidental exposure
				add_trait = wounded_1
				add_intrigue = -1
				stress_impact = {
					base = 15
					paranoid = medium_stress_impact_gain
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = 15
				intrigue >= 18
			}
		}
	}

	# Option B: Brew healing antidotes (safe, diplomatic value)
	option = {
		name = murim_tang.0001.b
		# Antidotes are always safe to make
		add_character_modifier = {
			modifier = tang_antidote_stock
			years = 3
		}
		add_prestige = 50
		add_piety = 25
		# Can be traded for favors later
		if = {
			limit = { NOT = { has_variable = tang_antidotes_stocked } }
			set_variable = { name = tang_antidotes_stocked value = 1 }
		}
		if = {
			limit = { has_variable = tang_antidotes_stocked }
			change_variable = { name = tang_antidotes_stocked add = 1 }
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				has_trait = compassionate
			}
			modifier = {
				add = 15
				diplomacy >= 14
			}
		}
	}

	# Option C: Craft hidden weapons (throwing needles, smoke bombs)
	option = {
		name = murim_tang.0001.c
		trigger = {
			OR = {
				has_trait = tang_hidden_weapon_expert
				martial >= 10
			}
		}
		random_list = {
			70 = {
				modifier = {
					add = 10
					has_trait = tang_hidden_weapon_expert
				}
				add_character_modifier = {
					modifier = tang_hidden_weapons_ready
					years = 3
				}
				add_prowess = 2
				add_prestige = 75
			}
			30 = {
				# Accident during crafting
				send_interface_toast = {
					title = murim_tang_weapon_accident
					left_icon = root
				}
				add_trait = wounded_1
				stress_impact = {
					base = 10
				}
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 25
				has_trait = tang_hidden_weapon_expert
			}
		}
	}

	# Option D: Study poison theory (learning path, no risk)
	option = {
		name = murim_tang.0001.d
		add_learning = 1
		add_prestige = 25
		stress_impact = {
			base = -5
			diligent = minor_stress_loss
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 15
				has_trait = diligent
			}
		}
	}
}

########################################
# 0002 - ASSASSINATION CONTRACT
# Tang Clan receives contract to eliminate a target
# Morality choice + execution chain
########################################

murim_tang.0002 = {
	type = character_event
	title = murim_tang.0002.t
	desc = murim_tang.0002.desc
	theme = intrigue_activity

	left_portrait = {
		character = root
		animation = personality_vengeful
	}
	right_portrait = {
		character = scope:tang_target
		animation = worry
	}

	trigger = {
		is_alive = yes
		is_imprisoned = no
		OR = {
			has_trait = tang_poison_master
			has_trait = tang_hidden_weapon_expert
		}
		intrigue >= 10
		any_relation = {
			type = rival
			is_alive = yes
		}
		NOT = { has_character_flag = tang_contract_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			has_trait = callous
		}
		modifier = {
			add = 0.3
			has_trait = sadistic
		}
		modifier = {
			subtract = 0.5
			has_trait = compassionate
		}
	}

	immediate = {
		add_character_flag = {
			flag = tang_contract_cooldown
			years = 2
		}
		# Pick a rival as the target
		random_relation = {
			type = rival
			limit = { is_alive = yes }
			save_scope_as = tang_target
		}
	}

	# Option A: Accept the contract — poison assassination
	option = {
		name = murim_tang.0002.a
		trigger = {
			exists = scope:tang_target
			has_character_modifier = tang_lethal_poison_prepared
		}
		# Create fight memory
		create_character_memory = {
			type = tang_assassination_attempt_memory
			participants = { target = scope:tang_target }
		}
		random_list = {
			# Clean kill — target dies, no evidence
			40 = {
				modifier = {
					add = 15
					intrigue >= 18
				}
				modifier = {
					add = 10
					has_trait = tang_poison_master
				}
				modifier = {
					add = 5
					has_character_modifier = tang_hidden_weapons_ready
				}
				scope:tang_target = {
					death = {
						death_reason = death_murder_unknown
						killer = root
					}
				}
				add_prestige = 200
				add_gold = 150
				remove_character_modifier = tang_lethal_poison_prepared
				# Infamy if anyone suspects
				random_list = {
					70 = {
						# Clean — no suspicion
						custom_tooltip = murim_tang_clean_kill_tt
					}
					30 = {
						# Rumored
						add_character_modifier = {
							modifier = suspected_poisoner
							years = 5
						}
					}
				}
			}
			# Target survives but poisoned
			35 = {
				scope:tang_target = {
					add_character_modifier = {
						modifier = tang_slow_poison_afflicted
						years = 3
					}
					add_trait = wounded_2
					# Target creates memory of assassination attempt
					create_character_memory = {
						type = tang_survived_poison_memory
						participants = { attacker = root }
					}
				}
				add_prestige = 50
				remove_character_modifier = tang_lethal_poison_prepared
				# Target now knows and wants revenge
				scope:tang_target = {
					add_opinion = {
						target = root
						modifier = attempted_murder_opinion
						opinion = -80
					}
				}
			}
			# Complete failure — caught
			25 = {
				modifier = {
					add = -10
					intrigue >= 16
				}
				# Exposed as assassin
				add_character_modifier = {
					modifier = exposed_tang_assassin
					years = 5
				}
				add_prestige = -200
				# All righteous faction members hate you
				every_ruler = {
					limit = {
						OR = {
							has_trait = mount_hua_disciple
							has_trait = wudang_taoist
							has_trait = shaolin_monk
						}
						NOT = { this = root }
					}
					add_opinion = {
						target = root
						modifier = exposed_assassin_opinion
						opinion = -40
					}
				}
				# Target gets +opinion from sympathizers
				scope:tang_target = {
					add_prestige = 100
				}
				stress_impact = {
					base = 30
					paranoid = major_stress_impact_gain
				}
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 25
				has_trait = sadistic
			}
			modifier = {
				add = 15
				has_trait = murim_vengeful
			}
			modifier = {
				add = -30
				has_trait = just
			}
		}
	}

	# Option B: Decline — too dishonorable
	option = {
		name = murim_tang.0002.b
		add_piety = 50
		stress_impact = {
			base = -10
			just = minor_stress_loss
			compassionate = minor_stress_loss
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = compassionate
			}
			modifier = {
				add = 20
				has_trait = just
			}
		}
	}

	# Option C: Warn the target instead (flip sides)
	option = {
		name = murim_tang.0002.c
		trigger = {
			exists = scope:tang_target
		}
		scope:tang_target = {
			add_opinion = {
				target = root
				modifier = warned_of_danger_opinion
				opinion = 40
			}
		}
		add_piety = 75
		add_prestige = -50
		# Whoever hired you is now your enemy
		stress_impact = {
			base = 5
			honest = minor_stress_loss
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 25
				has_trait = honest
			}
			modifier = {
				add = 15
				has_trait = compassionate
			}
		}
	}

	# Option D: Blackmail the contractor instead
	option = {
		name = murim_tang.0002.d
		trigger = {
			intrigue >= 16
		}
		add_gold = 200
		add_character_modifier = {
			modifier = tang_blackmail_leverage
			years = 3
		}
		stress_impact = {
			base = 10
			greedy = minor_stress_loss
		}
		ai_chance = {
			base = 15
			modifier = {
				add = 20
				has_trait = greedy
			}
			modifier = {
				add = 10
				intrigue >= 20
			}
		}
	}
}

	title = murim_tang.0010.t
	desc = murim_tang.0010.desc
	theme = diplomacy_majesty_focus

	left_portrait = {
		character = root
		animation = personality_rational
	}

	trigger = {
		is_alive = yes
		has_trait = tang_poison_master
		is_ruler = yes
		NOT = { has_character_flag = tang_defense_pact_cooldown }
	}

	weight_multiplier = {
		base = 0.5
		modifier = {
			add = 0.5
			any_war_enemy = {
				is_alive = yes
			}
		}
	}

	immediate = {
		add_character_flag = {
			flag = tang_defense_pact_cooldown
			years = 5
		}
	}

	# Option A: Offer poison traps to defend territory
	option = {
		name = murim_tang.0010.a
		add_character_modifier = {
			modifier = tang_poison_defenses_active
			years = 3
		}
		add_prestige = 200
		# Enemies approaching territory risk poisoning
		add_martial = 1
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				martial >= 12
			}
		}
	}

	# Option B: Offer antidotes to allied forces
	option = {
		name = murim_tang.0010.b
		add_prestige = 150
		add_piety = 50
		# Allied forces gain poison resistance in wars
		add_character_modifier = {
			modifier = tang_allied_antidote_supply
			years = 3
		}
		ai_chance = {
			base = 35
			modifier = {
				add = 20
				has_trait = compassionate
			}
		}
	}

	# Option C: Stay neutral (Tang tradition)
	option = {
		name = murim_tang.0010.c
		add_prestige = 50
		stress_impact = {
			base = -5
		}
		ai_chance = {
			base = 25
		}
	}
}
