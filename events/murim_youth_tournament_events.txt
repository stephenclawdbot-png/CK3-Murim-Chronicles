# Murim Chronicles - Youth Sparring Tournament Events
# The Young Dragons Tournament: YEARLY, ages 8-15 ONLY, sparring only, NO death
# INTEGRATION: Uses shared triggers from murim_triggers.txt:
#   is_youth_eligible (age >= 8, age < 16, has any martial trait)
#   is_martial_artist (has any martial trait)
# INTEGRATION: Uses shared effects from murim_effects.txt:
#   create_fight_memory_effect (every sparring match creates memory)
# INTEGRATION: Calls on_yearly_pulse via murim_on_actions.txt
# Cross-refs: murim_mentor_events.txt (murim_mentor.0001 - master scouts youth champion)
#             murim_training_events.txt (training bonuses from tournament experience)
#             murim_ranking_events.txt (prodigy enters ranking system at 16)
#             murim_grand_tournament_events.txt (youth champions fast-track to adult qualifiers)
#             murim_seeding_values.txt (power score for youth tracking)
#
# BALANCE: All youth rewards are TEMPORARY. They give a head start, not permanent dominance.
# - Champion: +3 prowess, +1 martial for 5 years (expires)
# - Finalist: +1 prowess, +1 martial for 3 years (expires)
# - Participant: +5 opinion with masters for 1 year
# - No stacking with adult tournament traits
# AGE GATE: Characters turn 16 -> youth modifiers continue their duration but character
#           becomes eligible for adult tournaments (murim_triggers.txt: is_eligible_for_tournament)

namespace = murim_youth

##############################
# murim_youth.0001 - THE YOUNG DRAGONS TOURNAMENT ANNOUNCED
# Fires yearly via on_yearly_pulse for characters with martial youth in court
# Trigger: Any ruler/head of family with martial children aged 8-15
##############################

murim_youth.0001 = {
	type = character_event
	title = murim_youth.0001.t
	desc = murim_youth.0001.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		is_landed = yes
		any_child = {
			is_alive = yes
			age >= 8
			age < 16
			has_trait = martial_artist
		}
	}
	
	# Cooldown: once per year (on_yearly_pulse fires this)
	cooldown = { years = 1 }
	
	immediate = {
		# Find eligible children
		every_child = {
			limit = {
				is_alive = yes
				age >= 8
				age < 16
				has_trait = martial_artist
			}
			save_scope_as = eligible_youth
			add_character_flag = {
				flag = youth_tournament_eligible_this_year
				years = 1
			}
		}
		# Pick the most talented child as focus
		ordered_child = {
			limit = {
				is_alive = yes
				age >= 8
				age < 16
				has_trait = martial_artist
			}
			order_by = prowess
			max = 1
			save_scope_as = best_youth
		}
	}
	
	# Option A: Encourage your child to enter
	option = {
		name = murim_youth.0001.a
		# "Go forth, young dragon. Show them what our family can do."
		scope:best_youth = {
			add_character_flag = {
				flag = entering_youth_tournament
				years = 1
			}
		}
		add_prestige = 25
		trigger_event = {
			id = murim_youth.0002
			days = { 7 14 }
		}
		ai_chance = {
			base = 60
			modifier = {
				factor = 1.5
				has_trait = ambitious
			}
		}
	}
	
	# Option B: Coach them personally (+prowess bonus in tournament)
	option = {
		name = murim_youth.0001.b
		# "I will train you myself. No child of mine enters unprepared."
		trigger = {
			has_trait = martial_artist
			prowess >= 12
		}
		scope:best_youth = {
			add_character_flag = {
				flag = entering_youth_tournament
				years = 1
			}
			add_character_flag = {
				flag = parent_coached_for_tournament
				years = 1
			}
			# Temporary training bonus
			add_character_modifier = {
				modifier = parent_coached_modifier
				years = 1
			}
		}
		add_stress = 15 # Personal investment
		trigger_event = {
			id = murim_youth.0002
			days = { 7 14 }
		}
		ai_chance = {
			base = 30
			modifier = {
				factor = 2.0
				has_trait = diligent
			}
		}
	}
	
	# Option C: Hire a trainer for them
	option = {
		name = murim_youth.0001.c
		# "I'll find the best instructor money can buy."
		add_gold = -30
		scope:best_youth = {
			add_character_flag = {
				flag = entering_youth_tournament
				years = 1
			}
			add_character_flag = {
				flag = hired_trainer_for_tournament
				years = 1
			}
		}
		trigger_event = {
			id = murim_youth.0002
			days = { 7 14 }
		}
		ai_chance = {
			base = 20
			modifier = {
				factor = 2.0
				gold >= 100
			}
		}
	}
	
	# Option D: Forbid them from entering (protective parent)
	option = {
		name = murim_youth.0001.d
		# "You're not ready. Maybe next year."
		scope:best_youth = {
			add_opinion = {
				target = root
				modifier = disappointed_in_parent_opinion
				opinion = -15
			}
		}
		add_stress = -10 # Relief from not risking your child
		ai_chance = {
			base = 10
			modifier = {
				factor = 3.0
				has_trait = craven
			}
			modifier = {
				factor = 2.0
				has_trait = paranoid
			}
		}
	}
}

##############################
# murim_youth.0002 - PLAYER'S CHILD ENTERS THE TOURNAMENT
# Pre-tournament preparation and scouting
##############################

murim_youth.0002 = {
	type = character_event
	title = murim_youth.0002.t
	desc = murim_youth.0002.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		any_child = {
			has_character_flag = entering_youth_tournament
		}
	}
	
	immediate = {
		ordered_child = {
			limit = {
				has_character_flag = entering_youth_tournament
				is_alive = yes
			}
			order_by = prowess
			max = 1
			save_scope_as = tournament_youth
		}
		# Generate a rival youth from the realm
		random_living_character = {
			limit = {
				age >= 8
				age < 16
				has_trait = martial_artist
				is_alive = yes
				NOT = { is_child_of = root }
			}
			weight = {
				base = 1
				modifier = { factor = 3 prowess >= 8 }
				modifier = {
					factor = 2
					any_parent = {
						OR = {
							has_trait = mount_hua_disciple
							has_trait = wudang_disciple
							has_trait = shaolin_disciple
							has_trait = tang_clan_member
							has_trait = heavenly_demon_disciple
							has_trait = beggar_sect_member
						}
					}
				}
			}
			save_scope_as = youth_rival
		}
	}
	
	# Option A: "They'll do great. I believe in them."
	option = {
		name = murim_youth.0002.a
		scope:tournament_youth = {
			add_opinion = {
				target = root
				modifier = parental_support_opinion
				opinion = 10
			}
		}
		trigger_event = {
			id = murim_youth.0003
			days = { 3 7 }
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Spy on the competition
	option = {
		name = murim_youth.0002.b
		trigger = {
			OR = {
				has_trait = schemer
				intrigue >= 10
			}
		}
		# Learn about the rival's fighting style
		scope:tournament_youth = {
			add_character_flag = {
				flag = scouted_youth_rival
				years = 1
			}
		}
		add_prestige = -10 # Slightly dishonorable
		trigger_event = {
			id = murim_youth.0003
			days = { 3 7 }
		}
		ai_chance = {
			base = 20
			modifier = { factor = 2 has_trait = deceitful }
		}
	}
	
	# Option C: Pray for their safety
	option = {
		name = murim_youth.0002.c
		add_piety = 25
		trigger_event = {
			id = murim_youth.0003
			days = { 3 7 }
		}
		ai_chance = {
			base = 30
			modifier = { factor = 2 has_trait = zealous }
		}
	}
}

##############################
# murim_youth.0003 - YOUTH SPARRING ROUND 1
# Light prowess check with age weighting. NO injury. Win/lose only.
##############################

murim_youth.0003 = {
	type = character_event
	title = murim_youth.0003.t
	desc = murim_youth.0003.desc
	theme = martial_arts
	
	trigger = {
		is_alive = yes
		any_child = {
			has_character_flag = entering_youth_tournament
			is_alive = yes
		}
	}
	
	immediate = {
		ordered_child = {
			limit = {
				has_character_flag = entering_youth_tournament
				is_alive = yes
			}
			order_by = prowess
			max = 1
			save_scope_as = tournament_youth
		}
		# Generate round 1 opponent (easier)
		random_living_character = {
			limit = {
				age >= 8
				age < 16
				has_trait = martial_artist
				is_alive = yes
				NOT = { is_child_of = root }
			}
			weight = {
				base = 10
				modifier = { factor = 0.5 prowess >= 10 } # Easier opponent for round 1
			}
			save_scope_as = round1_opponent
		}
	}
	
	# The fight happens automatically. Outcome based on prowess + bonuses.
	option = {
		name = murim_youth.0003.a
		# "Watch closely..."
		
		# Prowess duel calculation (youth version -- no injuries)
		duel = {
			skill = prowess
			target = scope:round1_opponent
			
			# Bonuses from preparation
			modifier = {
				factor = 1.2
				scope:tournament_youth = { has_character_flag = parent_coached_for_tournament }
			}
			modifier = {
				factor = 1.1
				scope:tournament_youth = { has_character_flag = hired_trainer_for_tournament }
			}
			modifier = {
				factor = 1.1
				scope:tournament_youth = { has_character_flag = scouted_youth_rival }
			}
			# Age advantage: older youth have slight edge
			modifier = {
				factor = 1.15
				scope:tournament_youth = { age >= 13 }
				scope:round1_opponent = { age < 11 }
			}
			
			# WIN
			10 = {
				min_prestige_level_diff = -3
				desc = murim_youth.0003.win
				scope:tournament_youth = {
					add_character_flag = {
						flag = youth_tournament_round1_win
						years = 1
					}
					# Create fight memory
					create_character_memory = {
						type = youth_tournament_victory_memory
					}
				}
				trigger_event = {
					id = murim_youth.0004
					days = { 3 5 }
				}
			}
			# LOSE
			10 = {
				min_prestige_level_diff = 3
				desc = murim_youth.0003.lose
				scope:tournament_youth = {
					add_character_flag = {
						flag = youth_tournament_eliminated
						years = 1
					}
					create_character_memory = {
						type = youth_tournament_loss_memory
					}
				}
				trigger_event = {
					id = murim_youth.0008
					days = { 1 3 }
				}
			}
		}
		ai_chance = { base = 100 }
	}
}
