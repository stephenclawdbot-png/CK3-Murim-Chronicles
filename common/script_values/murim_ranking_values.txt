# Murim Chronicles - Top 50 Martial Artist Ranking System
# Hidden rank score calculated yearly for all martial artists
# Formula: (prowess x 2) + (cultivation_rank_weight x 10) + (technique_count x 3) + (duel_wins x 5) - (duel_losses x 2) - (wounds x 3)
#          + age_experience_bonus + element_mastery_bonus + bloodline_bonus + sect_prestige_bonus
#          + weapon_mastery_bonus + fame_bonus + streak_bonus + title_weight + special_condition_bonus
# Load-safe: all values have fallback defaults, exists checks on all variables

######################################

# Converts cultivation tier to numeric weight for ranking formula
######################################

######################################
# AGE & EXPERIENCE BONUS
# Older warriors with decades of practice get bonus
# But very old age causes decline
######################################

murim_age_experience_bonus_value = {
	value = 0
	# Young prodigy bonus (under 20 and already cultivating)
	if = {
		limit = {
			exists = this
			age < 20
			OR = {
				has_trait = qi_sensing
				has_trait = murim_qi_sensing
				has_trait = qi_condensation
				has_trait = murim_qi_condensation
				has_trait = foundation_establishment
				has_trait = murim_foundation_establishment
			}
		}
		add = 3
	}
	# Prime combat years (20-35)
	if = {
		limit = {
			exists = this
			age >= 20
			age < 36
		}
		add = 2
	}
	# Peak experience (36-55)
	if = {
		limit = {
			exists = this
			age >= 36
			age < 56
		}
		add = 4
	}
	# Veteran master (56-70)
	if = {
		limit = {
			exists = this
			age >= 56
			age < 71
		}
		add = 3
	}
	# Aging decline (71-85) unless high cultivation offsets
	if = {
		limit = {
			exists = this
			age >= 71
			age < 86
			NOR = {
				has_trait = nascent_soul
				has_trait = murim_nascent_soul
				has_trait = spirit_severing
				has_trait = murim_spirit_severing
				has_trait = dao_seeking
				has_trait = murim_dao_seeking
				has_trait = immortal_ascension
				has_trait = murim_immortal_ascension
			}
		}
		add = -2
	}
	# Severe decline (86+) unless Dao-level cultivation
	if = {
		limit = {
			exists = this
			age >= 86
			NOR = {
				has_trait = dao_seeking
				has_trait = murim_dao_seeking
				has_trait = immortal_ascension
				has_trait = murim_immortal_ascension
			}
		}
		add = -5
	}
	# Immortal cultivators transcend age
	if = {
		limit = {
			exists = this
			age >= 71
			OR = {
				has_trait = immortal_ascension
				has_trait = murim_immortal_ascension
			}
		}
		add = 5
	}
	# Years of active cultivation tracked via variable
	if = {
		limit = {
			exists = this
			exists = var:murim_cultivation_years
			var:murim_cultivation_years >= 10
		}
		add = 1
	}
	if = {
		limit = {
			exists = this
			exists = var:murim_cultivation_years
			var:murim_cultivation_years >= 25
		}
		add = 2
	}
	if = {
		limit = {
			exists = this
			exists = var:murim_cultivation_years
			var:murim_cultivation_years >= 50
		}
		add = 3
	}
}

######################################
# SECT PRESTIGE BONUS
# Being sect leader or elder adds ranking weight
######################################

murim_sect_prestige_bonus_value = {
	value = 0
	# Sect member baseline
	if = {
		limit = {
			exists = this
			exists = var:murim_sect_member
			var:murim_sect_member = yes
		}
		add = 1
	}
	# Inner disciple
	if = {
		limit = {
			exists = this
			has_trait = inner_disciple
		}
		add = 2
	}
	# Core disciple
	if = {
		limit = {
			exists = this
			has_trait = core_disciple
		}
		add = 3
	}
	# Sect elder
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = sect_elder
				has_trait = murim_sect_elder
			}
		}
		add = 5
	}
	# Sect leader / patriarch
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = sect_master
				has_trait = murim_sect_leader
				has_trait = sect_patriarch
			}
		}
		add = 8
	}
	# Alliance leader (highest sect prestige)
	if = {
		limit = {
			exists = this
			OR = {
				has_trait = murim_alliance_leader
				has_trait = martial_alliance_chief
			}
		}
		add = 12
	}
	# Rogue cultivator penalty (no sect backing)
	if = {
		limit = {
			exists = this
			NOR = {
				exists = var:murim_sect_member
				has_trait = inner_disciple
				has_trait = core_disciple
				has_trait = sect_elder
				has_trait = murim_sect_elder
				has_trait = sect_master
				has_trait = murim_sect_leader
			}
			OR = {
				has_trait = rogue_cultivator
				has_trait = wandering_swordsman
			}
		}
		add = -1
	}
}

# Difference calculator for duel matchups

	value = murim_composite_rank_score
	subtract = scope:target.murim_composite_rank_score
}

# Challenge difficulty rating
murim_challenge_difficulty_value = {
	value = 0
	# Easy (10+ points above target)
	if = {
		limit = { murim_rank_score_difference >= 10 }
		value = 1
	}
	# Even match (within 10 points)
	if = {
		limit = {
			murim_rank_score_difference > -10
			murim_rank_score_difference < 10
		}
		value = 2
	}
	# Hard (10-30 points below target)
	if = {
		limit = {
			murim_rank_score_difference <= -10
			murim_rank_score_difference > -30
		}
		value = 3
	}
	# Extreme (30+ points below target)
	if = {
		limit = {
			murim_rank_score_difference <= -30
		}
		value = 4
	}
}
