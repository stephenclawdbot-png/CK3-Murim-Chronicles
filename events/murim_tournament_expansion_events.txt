############################################
# INTER-CLAN EXHIBITION MATCH
# Two clans arrange a friendly exhibition
# during tournament season. High stakes for
# clan prestige but no killing.
############################################
murim_tourney_exp.0001 = {
	type = character_event
	title = murim_tourney_exp.0001.t
	desc = murim_tourney_exp.0001.desc
	theme = martial
	override_background = { reference = tournament_grounds }
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:exhibition_opponent
		animation = personality_bold
	}

	trigger = {
		murim_is_martial_artist_trigger = yes
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_monk
			has_trait = wudang_disciple
			has_trait = tang_clan_member
			has_trait = beggar_sect_member
			has_trait = iron_fist_disciple
			has_trait = heavenly_demon_disciple
			has_trait = poison_flower_disciple
			has_trait = phantom_blade_disciple
		}
		has_variable = murim_clan_rank
		var:murim_clan_rank >= 3
		prowess >= 14
		NOT = { has_character_modifier = murim_exhibition_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 2 prowess >= 18 }
		modifier = { add = 2 has_trait = brave }
		modifier = { add = 1 diplomacy >= 12 }
	}

	immediate = {
		# Find opponent from a DIFFERENT clan
		random_living_character = {
			limit = {
				has_trait = martial_artist
				is_alive = yes
				prowess >= 12
				NOT = { this = root }
				# Different clan check
				trigger_if = {
					limit = { root = { has_trait = mount_hua_disciple } }
					NOT = { has_trait = mount_hua_disciple }
				}
				trigger_if = {
					limit = { root = { has_trait = shaolin_monk } }
					NOT = { has_trait = shaolin_monk }
				}
				trigger_if = {
					limit = { root = { has_trait = wudang_disciple } }
					NOT = { has_trait = wudang_disciple }
				}
				trigger_if = {
					limit = { root = { has_trait = heavenly_demon_disciple } }
					NOT = { has_trait = heavenly_demon_disciple }
				}
				trigger_if = {
					limit = { root = { has_trait = tang_clan_member } }
					NOT = { has_trait = tang_clan_member }
				}
				trigger_if = {
					limit = { root = { has_trait = beggar_sect_member } }
					NOT = { has_trait = beggar_sect_member }
				}
				trigger_if = {
					limit = { root = { has_trait = iron_fist_disciple } }
					NOT = { has_trait = iron_fist_disciple }
				}
				trigger_if = {
					limit = { root = { has_trait = poison_flower_disciple } }
					NOT = { has_trait = poison_flower_disciple }
				}
				trigger_if = {
					limit = { root = { has_trait = phantom_blade_disciple } }
					NOT = { has_trait = phantom_blade_disciple }
				}
				OR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = wudang_disciple
					has_trait = tang_clan_member
					has_trait = beggar_sect_member
					has_trait = iron_fist_disciple
					has_trait = heavenly_demon_disciple
					has_trait = poison_flower_disciple
					has_trait = phantom_blade_disciple
				}
			}
			weight = {
				base = 10
				modifier = { add = 15 prowess >= 16 }
				modifier = { add = 10 has_variable = murim_clan_rank var:murim_clan_rank >= 3 }
			}
			save_scope_as = exhibition_opponent
		}
	}

	# Fight with honor - hold back killing moves
	option = {
		name = murim_tourney_exp.0001.a
		trigger = { exists = scope:exhibition_opponent }
		duel = {
			skill = prowess
			target = scope:exhibition_opponent
			50 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_tourney_exp.0001.a.win
				add_prestige = 150
				add_character_modifier = { modifier = murim_exhibition_victor years = 3 }
				scope:exhibition_opponent = {
					add_opinion = { target = root modifier = respect_martial_prowess opinion = 20 }
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:exhibition_victory }
				}
			}
			50 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:exhibition_opponent }
				desc = murim_tourney_exp.0001.a.lose
				add_prestige = 50
				add_stress = 10
				scope:exhibition_opponent = {
					add_prestige = 150
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:exhibition_defeat_graceful }
				}
			}
		}
		add_character_modifier = { modifier = murim_exhibition_cooldown years = 2 }
		stress_impact = { brave = minor_stress_impact_loss craven = medium_stress_impact_gain }
		ai_chance = { base = 60 }
	}

	# Go all out - disregard exhibition rules
	option = {
		name = murim_tourney_exp.0001.b
		trigger = { exists = scope:exhibition_opponent }
		duel = {
			skill = prowess
			target = scope:exhibition_opponent
			60 = {
				compare_modifier = { value = prowess multiplier = 3 }
				desc = murim_tourney_exp.0001.b.win
				add_prestige = 200
				add_dread = 15
				scope:exhibition_opponent = {
					random_list = {
						50 = { add_trait = wounded_1 }
						30 = { add_trait = wounded_2 }
						20 = { add_trait = scarred }
					}
					add_opinion = { target = root modifier = insult_opinion opinion = -30 }
				}
				# Cross-ref: This causes faction tension
				trigger_event = { id = murim_faction_war.0003 days = { 5 15 } }
			}
			40 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:exhibition_opponent }
				desc = murim_tourney_exp.0001.b.lose
				add_trait = wounded_1
				add_prestige = -100
				add_dread = -10
			}
		}
		add_character_modifier = { modifier = murim_exhibition_cooldown years = 2 }
		stress_impact = { compassionate = major_stress_impact_gain wrathful = minor_stress_impact_loss }
		ai_chance = { base = 25 modifier = { add = 25 has_trait = wrathful } }
	}

	# Decline the exhibition politely
	option = {
		name = murim_tourney_exp.0001.c
		add_prestige = -25
		add_character_modifier = { modifier = murim_exhibition_cooldown years = 2 }
		ai_chance = { base = 15 }
	}
}

############################################
# DARK HORSE UPSET
# Unknown martial artist defeats a ranked
# fighter, shaking up the Murim rankings.
############################################
murim_tourney_exp.0002 = {
	type = character_event
	title = murim_tourney_exp.0002.t
	desc = murim_tourney_exp.0002.desc
	theme = martial
	override_background = { reference = tournament_grounds }
	left_portrait = {
		character = root
		animation = personality_bold
	}

	trigger = {
		murim_is_martial_artist_trigger = yes
		prowess >= 16
		NOT = { has_variable = murim_ranking_position }
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_monk
			has_trait = wudang_disciple
			has_trait = tang_clan_member
			has_trait = beggar_sect_member
			has_trait = iron_fist_disciple
			has_trait = heavenly_demon_disciple
			has_trait = poison_flower_disciple
			has_trait = phantom_blade_disciple
		}
		NOT = { has_character_flag = dark_horse_upset_done }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 3 prowess >= 20 }
		modifier = { add = 2 age <= 25 }
		modifier = { add = 1 has_trait = brave }
	}

	immediate = {
		add_character_flag = { flag = dark_horse_upset_done }
		# Find a ranked fighter
		random_living_character = {
			limit = {
				has_trait = martial_artist
				has_variable = murim_ranking_position
				var:murim_ranking_position <= 50
				is_alive = yes
				NOT = { this = root }
			}
			weight = {
				base = 10
				modifier = { add = 5 var:murim_ranking_position <= 20 }
			}
			save_scope_as = ranked_victim
		}
	}

	# Challenge them openly on the tournament stage
	option = {
		name = murim_tourney_exp.0002.a
		trigger = { exists = scope:ranked_victim }
		duel = {
			skill = prowess
			target = scope:ranked_victim
			45 = {
				compare_modifier = { value = prowess multiplier = 2 }
				desc = murim_tourney_exp.0002.a.upset
				add_prestige = 500
				add_character_modifier = { modifier = murim_dark_horse_fame years = 10 }
				# Take their approximate ranking
				set_variable = { name = murim_ranking_position value = 45 }
				# Every clan wants you now
				trigger_event = { id = murim_tourney_exp.0006 days = { 10 30 } }
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:dark_horse_upset_legendary }
				}
			}
			55 = {
				compare_modifier = { value = prowess multiplier = 2 target = scope:ranked_victim }
				desc = murim_tourney_exp.0002.a.crushed
				add_prestige = 50
				add_stress = 25
				random_list = {
					40 = { add_trait = wounded_1 }
					35 = { add_trait = wounded_2 }
					15 = { add_trait = maimed }
					10 = { add_trait = scarred }
				}
				hidden_effect = {
					add_to_variable_list = { name = murim_combat_memories target = flag:dark_horse_humiliation }
				}
			}
		}
		stress_impact = { brave = minor_stress_impact_loss craven = major_stress_impact_gain }
		ai_chance = { base = 50 modifier = { add = 30 has_trait = ambitious } }
	}

	# Wait for a better opportunity
	option = {
		name = murim_tourney_exp.0002.b
		add_prestige = 25
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:dark_horse_patience }
		}
		ai_chance = { base = 50 modifier = { add = 30 has_trait = patient } }
	}
}

# TOURNAMENT RIGGING DISCOVERED
# Someone has been poisoning opponents or
# bribing judges. Orthodox/Unorthodox tension.
############################################
murim_tourney_exp.0003 = {
	type = character_event
	title = murim_tourney_exp.0003.t
	desc = murim_tourney_exp.0003.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = shock
	}

	trigger = {
		murim_is_martial_artist_trigger = yes
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_monk
			has_trait = wudang_disciple
			has_trait = tang_clan_member
			has_trait = beggar_sect_member
			has_trait = iron_fist_disciple
			has_trait = heavenly_demon_disciple
			has_trait = poison_flower_disciple
			has_trait = phantom_blade_disciple
		}
		intrigue >= 10
		NOT = { has_character_flag = tournament_rigging_discovered }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 3 intrigue >= 16 }
		modifier = { add = 2 OR = { has_trait = poison_flower_disciple has_trait = phantom_blade_disciple } }
	}

	immediate = {
		add_character_flag = { flag = tournament_rigging_discovered days = 1825 }
		# The rigger - usually unorthodox
		random_living_character = {
			limit = {
				has_trait = martial_artist
				intrigue >= 14
				is_alive = yes
				NOT = { this = root }
			}
			weight = {
				base = 10
				modifier = { add = 15 OR = { has_trait = heavenly_demon_disciple has_trait = poison_flower_disciple has_trait = phantom_blade_disciple } }
				modifier = { add = 5 intrigue >= 18 }
			}
			save_scope_as = tournament_rigger
		}
	}

	# Expose the rigging publicly - honor demands it
	option = {
		name = murim_tourney_exp.0003.a
		trigger = { exists = scope:tournament_rigger }
		add_prestige = 200
		add_piety = 100
		scope:tournament_rigger = {
			add_prestige = -500
			add_dread = -20
			add_character_modifier = { modifier = murim_exposed_cheater years = 10 }
			add_opinion = { target = root modifier = insult_opinion opinion = -40 }
		}
		# Cross-ref: faction tension rises
		trigger_event = { id = murim_faction_war.0001 days = { 5 15 } }
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:tournament_rigger_exposed }
		}
		stress_impact = { just = minor_stress_impact_loss deceitful = medium_stress_impact_gain }
		ai_chance = { base = 50 modifier = { add = 30 has_trait = just } modifier = { add = -20 has_trait = deceitful } }
	}

	# Blackmail the rigger - profit from silence
	option = {
		name = murim_tourney_exp.0003.b
		trigger = { exists = scope:tournament_rigger }
		add_gold = 150
		add_intrigue_skill = 1
		scope:tournament_rigger = {
			add_opinion = { target = root modifier = blackmail_opinion opinion = -20 }
		}
		add_character_modifier = { modifier = murim_knows_dark_secrets years = 5 }
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:tournament_rigging_blackmail }
		}
		stress_impact = { just = major_stress_impact_gain deceitful = minor_stress_impact_loss }
		ai_chance = { base = 30 modifier = { add = 30 has_trait = greedy } }
	}

	# Ignore it - not your problem
	option = {
		name = murim_tourney_exp.0003.c
		add_stress = 10
		hidden_effect = {
			add_to_variable_list = { name = murim_combat_memories target = flag:tournament_rigging_ignored }
		}
		ai_chance = { base = 20 }
	}
}

# CHAMPION RETIREMENT CEREMONY
# An aging champion decides to retire,
# passing their title and wisdom to a successor.

murim_tourney_exp.0005 = {
	type = character_event
	title = murim_tourney_exp.0005.t
	desc = murim_tourney_exp.0005.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_compassionate
	}

	trigger = {
		murim_is_martial_artist_trigger = yes
		age >= 50
		has_variable = murim_ranking_position
		var:murim_ranking_position <= 10
		OR = {
			has_trait = mount_hua_disciple
			has_trait = shaolin_monk
			has_trait = wudang_disciple
			has_trait = tang_clan_member
			has_trait = beggar_sect_member
			has_trait = iron_fist_disciple
			has_trait = heavenly_demon_disciple
			has_trait = poison_flower_disciple
			has_trait = phantom_blade_disciple
		}
		NOT = { has_character_flag = champion_retirement_done }
	}

	weight_multiplier = {
		base = 1
		modifier = { add = 3 age >= 60 }
		modifier = { add = 2 health <= 3 }
		modifier = { add = 1 OR = { has_trait = wounded_2 has_trait = wounded_3 } }
	}

	immediate = {
		add_character_flag = { flag = champion_retirement_done }
		# Find a worthy successor from same clan
		random_living_character = {
			limit = {
				has_trait = martial_artist
				is_alive = yes
				age >= 18
				age <= 40
				prowess >= 14
				NOT = { this = root }
				# Same clan
				trigger_if = {
					limit = { root = { has_trait = mount_hua_disciple } }
					has_trait = mount_hua_disciple
				}
				trigger_if = {
					limit = { root = { has_trait = shaolin_monk } }
					has_trait = shaolin_monk
				}
				trigger_if = {
					limit = { root = { has_trait = wudang_disciple } }
					has_trait = wudang_disciple
				}
				trigger_if = {
					limit = { root = { has_trait = tang_clan_member } }
					has_trait = tang_clan_member
				}
				trigger_if = {
					limit = { root = { has_trait = beggar_sect_member } }
					has_trait = beggar_sect_member
				}
				trigger_if = {
					limit = { root = { has_trait = iron_fist_disciple } }
					has_trait = iron_fist_disciple
				}
				trigger_if = {
					limit = { root = { has_trait = heavenly_demon_disciple } }
					has_trait = heavenly_demon_disciple
				}
				trigger_if = {
					limit = { root = { has_trait = poison_flower_disciple } }
					has_trait = poison_flower_disciple
				}
				trigger_if = {
					limit = { root = { has_trait = phantom_blade_disciple } }
					has_trait = phantom_blade_disciple
				}
			}
			weight = {
				base = 10
				modifier = { add = 15 prowess >= 18 }
				modifier = { add = 10 has_variable = murim_clan_rank var:murim_clan_rank >= 3 }
			}
			save_scope_as = retirement_successor
		}
	}

	# Formal retirement - pass the mantle with ceremony
	option = {
		name = murim_tourney_exp.0005.a
		trigger = { exists = scope:retirement_successor }
		add_prestige = 500
		add_piety = 200
		add_character_modifier = { modifier = murim_retired_champion years = -1 }
		# Pass ranking to successor
		scope:retirement_successor = {
			set_variable = { name = murim_ranking_position value = root.var:murim_ranking_position }
