######################################
# INITIALIZE RANKING VARIABLES
# Sets up default variables for new martial artists
# Called from murim_ranking_on_actions.txt yearly pulse
######################################

murim_initialize_ranking_variables_effect = {
	if = {
		limit = {
			exists = this
			is_alive = yes
			NOT = { has_variable = murim_martial_rank_score }
		}
		set_variable = {
			name = murim_martial_rank_score
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_cultivation_xp }
		}
		set_variable = {
			name = murim_cultivation_xp
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_duel_wins }
		}
		set_variable = {
			name = murim_duel_wins
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = murim_duel_losses }
		}
		set_variable = {
			name = murim_duel_losses
			value = 0
		}
	}
}

######################################
# RECALCULATE MARTIAL RANK
# Yearly pulse - recalculates score for a single character
######################################

murim_recalculate_martial_rank_effect = {
	# Safety check
	if = {
		limit = {
			exists = this
			is_alive = yes
			murim_is_martial_artist_trigger = yes
		}
		# Calculate new score using the composite formula from murim_ranking_values.txt
		set_variable = {
			name = murim_martial_rank_score
			value = murim_total_ranking_score_value
		}
		# Check if score crosses top 50 threshold
		if = {
			limit = {
				var:murim_martial_rank_score >= murim_is_top_50_threshold_value
				NOT = { has_character_modifier = murim_ranked_top_50_modifier }
			}
			# New entry into Top 50
			murim_enter_top_50_effect = yes
		}
		# Check if dropped below threshold
		if = {
			limit = {
				var:murim_martial_rank_score < murim_is_top_50_threshold_value
				has_character_modifier = murim_ranked_top_50_modifier
			}
			# Dropped out of Top 50
			murim_leave_top_50_effect = yes
		}
		# Update tier modifiers
		murim_update_rank_tier_modifiers_effect = yes
	}
}

######################################
# ENTER TOP 50
# Character crosses threshold, gains modifier, notification fires
######################################

murim_enter_top_50_effect = {
	# Add ranked fighter modifier
	add_character_modifier = {
		modifier = murim_ranked_top_50_modifier
		years = 1
	}
	# Notification to the character - namespace is murim_rank
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_rank.0001
			days = 1
		}
	}
	# Prestige boost for entering rankings
	add_prestige = 200
	# Set initial rank position variable if not set
	if = {
		limit = {
			NOT = { has_variable = murim_martial_rank_position }
		}
		set_variable = {
			name = murim_martial_rank_position
			value = 50
		}
	}
	# Memory of entering rankings
	create_character_memory = {
		type = murim_entered_rankings_memory
	}
	# Court notification for liege
	if = {
		limit = {
			exists = liege
			liege = { is_ai = no }
		}
		liege = {
			send_interface_message = {
				type = event_generic_neutral
				title = murim_rank_new_entry_notification
				left_icon = root
				desc = murim_rank_new_entry_notification_desc
			}
		}
	}
	# Stress relief from glory
	add_stress = -15
}

######################################
# LEAVE TOP 50
# Dropped out, loses modifier, shame
######################################

murim_leave_top_50_effect = {
	# Remove ranked modifier
	if = {
		limit = { has_character_modifier = murim_ranked_top_50_modifier }
		remove_character_modifier = murim_ranked_top_50_modifier
	}
	# Notification to the character - namespace is murim_rank
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_rank.0002
			days = 1
		}
	}
	# Shame modifier
	add_character_modifier = {
		modifier = murim_fallen_from_rankings_modifier
		years = 2
	}
	# Memory of falling from rankings
	create_character_memory = {
		type = murim_fell_from_rankings_memory
	}
	# Prestige loss
	add_prestige = -100
	# Stress from public failure
	add_stress = 20
	# Remove position variable
	if = {
		limit = { has_variable = murim_martial_rank_position }
		remove_variable = murim_martial_rank_position
	}
}

######################################
# REMOVE FROM RANKINGS ON DEATH
# Dead fighters removed, next in line promoted
######################################

murim_remove_from_rankings_on_death_effect = {
	# Only process if was a ranked fighter
	if = {
		limit = {
			has_variable = murim_martial_rank_position
			has_character_modifier = murim_ranked_top_50_modifier
		}
		# Remove modifier (cosmetic cleanup for death screen)
		remove_character_modifier = murim_ranked_top_50_modifier
		# Notify living players about ranked fighter death
		every_player = {
			limit = {
				is_alive = yes
				murim_is_martial_artist_trigger = yes
			}
			trigger_event = {
				id = murim_rank.0003
				days = { 1 5 }
			}
		}
		# Remove variables
		remove_variable = murim_martial_rank_score
		remove_variable = murim_martial_rank_position
	}
}

######################################
# UPDATE RANK TIER MODIFIERS
# Applies/removes tier-specific modifiers based on position
######################################

murim_update_rank_tier_modifiers_effect = {
	# Top 10 elite modifier
	if = {
		limit = {
			has_variable = murim_martial_rank_position
			var:murim_martial_rank_position <= 10
			NOT = { has_character_modifier = murim_top_10_elite_modifier }
		}
		add_character_modifier = {
			modifier = murim_top_10_elite_modifier
			years = 1
		}
	}
	# Remove top 10 if dropped
	if = {
		limit = {
			has_character_modifier = murim_top_10_elite_modifier
			OR = {
				NOT = { has_variable = murim_martial_rank_position }
				AND = {
					has_variable = murim_martial_rank_position
					var:murim_martial_rank_position > 10
				}
			}
		}
		remove_character_modifier = murim_top_10_elite_modifier
	}
	# Number 1 - Strongest Under Heaven
	if = {
		limit = {
			has_variable = murim_martial_rank_position
			var:murim_martial_rank_position = 1
			NOT = { has_character_modifier = strongest_under_heaven }
		}
		add_character_modifier = {
			modifier = strongest_under_heaven
			years = 1
		}
		# Special event for reaching #1
		if = {
			limit = { is_ai = no }
			trigger_event = {
				id = murim_rank.0004
				days = 3
			}
		}
	}
	# Remove #1 modifier if not #1 anymore
	if = {
		limit = {
			has_character_modifier = strongest_under_heaven
			OR = {
				NOT = { has_variable = murim_martial_rank_position }
				AND = {
					has_variable = murim_martial_rank_position
					var:murim_martial_rank_position > 1
				}
			}
		}
		remove_character_modifier = strongest_under_heaven
	}
}

######################################
# RANKING POSITION SWAP
# Swaps positions between two fighters after a ranking duel
# Expects scope:rank_challenger and scope:rank_challenge_target (or scope:target)
# Called from murim_ranking_on_actions.txt AI duel resolution
######################################

murim_ranking_position_swap_effect = {
	# Store challenger's current position
	if = {
		limit = {
			has_variable = murim_martial_rank_position
			OR = {
				exists = scope:rank_challenge_target
				exists = scope:target
			}
		}
		save_temporary_scope_as = swap_winner
		# Get the target (support both scope names)
		if = {
			limit = { exists = scope:rank_challenge_target }
			scope:rank_challenge_target = {
				save_temporary_scope_as = swap_loser
			}
		}
		else_if = {
			limit = { exists = scope:target }
			scope:target = {
				save_temporary_scope_as = swap_loser
			}
		}
		# Only swap if both have positions and winner is ranked lower (higher number)
		if = {
			limit = {
				scope:swap_loser = {
					has_variable = murim_martial_rank_position
				}
			}
			# Store loser's position temporarily
			set_variable = {
				name = murim_temp_swap_position
				value = scope:swap_loser.var:murim_martial_rank_position
			}
			# Give loser the winner's old position
			scope:swap_loser = {
				set_variable = {
					name = murim_martial_rank_position
					value = scope:swap_winner.var:murim_martial_rank_position
				}
			}
			# Give winner the loser's old (better) position
			set_variable = {
				name = murim_martial_rank_position
				value = var:murim_temp_swap_position
			}
			# Cleanup temp
			remove_variable = murim_temp_swap_position
		}
	}
}

######################################
# RECORD DUEL WIN
# Increments win counter, adds prestige, stress relief
# Called from on_actions duel resolution blocks
######################################

murim_record_duel_win_effect = {
	if = {
		limit = {
			NOT = { has_variable = murim_duel_wins }
		}
		set_variable = {
			name = murim_duel_wins
			value = 0
		}
	}
	change_variable = {
		name = murim_duel_wins
		add = 1
	}
	# Prowess XP from combat experience
	if = {
		limit = { has_variable = murim_cultivation_xp }
		change_variable = {
			name = murim_cultivation_xp
			add = 25
		}
	}
	# Stress relief from victory
	add_stress = -10
	# Memory of victory
	create_character_memory = {
		type = murim_duel_victory_memory
	}
}

######################################
# RECORD DUEL LOSS
# Increments loss counter, shame, stress
# Called from on_actions duel resolution blocks
######################################

murim_record_duel_loss_effect = {
	if = {
		limit = {
			NOT = { has_variable = murim_duel_losses }
		}
		set_variable = {
			name = murim_duel_losses
			value = 0
		}
	}
	change_variable = {
		name = murim_duel_losses
		add = 1
	}
	# Shame modifier
	add_character_modifier = {
		modifier = defeat_shame
		years = 1
	}
	# Stress from defeat
	add_stress = 15
	# Memory of defeat
	create_character_memory = {
		type = murim_duel_defeat_memory
	}
}

######################################
# ROLL COMBAT INJURIES
# Random injury outcomes after losing a duel
# Wounds, incapacitated, craven, one-eyed, etc.
# Called from on_actions duel resolution blocks
######################################

murim_roll_combat_injuries_effect = {
	random_list = {
		40 = {
			# Minor bruises - no lasting effect
			add_character_modifier = {
				modifier = battle_scarred_veteran
				years = 1
			}
		}
		25 = {
			# Wounded
			add_trait = wounded_1
			create_character_memory = {
				type = murim_combat_wounded_memory
			}
		}
		15 = {
			# Severely wounded
			add_trait = wounded_2
			create_character_memory = {
				type = murim_combat_severely_wounded_memory
			}
		}
		8 = {
			# Disfigured - lost an eye or scarred
			modifier = {
				factor = 1.5
				prowess < 8
			}
			random_list = {
				50 = {
					add_trait = one_eyed
				}
				50 = {
					add_trait = scarred_1
				}
			}
			create_character_memory = {
				type = murim_combat_disfigured_memory
			}
		}
		7 = {
			# Incapacitated - barely survived
			modifier = {
				factor = 2.0
				prowess < 5
			}
			add_trait = incapable
			create_character_memory = {
				type = murim_combat_incapacitated_memory
			}
		}
		5 = {
			# Developed combat fear
			modifier = {
				factor = 2.0
				has_trait = shy
			}
			modifier = {
				factor = 0.5
				has_trait = brave
			}
			if = {
				limit = {
					NOT = { has_trait = craven }
					NOT = { has_trait = brave }
				}
				add_trait = craven
				create_character_memory = {
					type = murim_combat_fear_memory
				}
			}
		}
	}
}
