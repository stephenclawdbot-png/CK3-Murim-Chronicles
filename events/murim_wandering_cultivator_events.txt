# EVENT 0001 - The Road Calls: Becoming a Wanderer
# Triggered: on_action yearly pulse, martial artists without strong ties
# AI: Full wandering lifestyle initiation
murim_wanderer.0001 = {
    type = character_event
    title = murim_wanderer.0001.t
    desc = murim_wanderer.0001.desc
    theme = martial
    
    left_portrait = {
        character = root
        animation = personality_bold
    }
    
    trigger = {
        is_alive = yes
        age >= 18
        NOT = { has_trait = murim_wandering_cultivator }
        NOT = { has_character_flag = considered_wandering }
        OR = {
            has_trait = murim_martial_artist
            has_trait = murim_shadow_operative
        }
        OR = {
            is_landed = no
            has_character_modifier = murim_expelled_from_sect
            has_character_modifier = murim_heartbreak
            has_character_modifier = murim_devastating_heartbreak
            stress >= 60
            has_trait = adventurer
            AND = {
                has_trait = content
                NOT = { has_trait = ambitious }
            }
        }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 3.0
            has_character_modifier = murim_expelled_from_sect
        }
        modifier = {
            factor = 2.0
            has_character_modifier = murim_heartbreak
        }
        modifier = {
            factor = 2.0
            stress >= 70
        }
        modifier = {
            factor = 1.5
            has_trait = adventurer
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = considered_wandering
            days = 730
        }
    }
    
    # Option A: Embrace the wandering path
    option = {
        name = murim_wanderer.0001.a
        add_trait = murim_wandering_cultivator
        add_stress = -30
        add_character_modifier = {
            modifier = murim_freedom_of_the_road
            years = 10
        }
        add_prowess_skill = 1
        add_martial_skill = 1
        # Start wandering events chain
        trigger_event = {
            id = murim_wanderer.0002
            days = { 30 120 }
        }
        ai_chance = {
            base = 45
            modifier = {
                factor = 2.0
                has_character_modifier = murim_expelled_from_sect
            }
            modifier = {
                factor = 2.0
                stress >= 70
            }
            modifier = {
                factor = 2.0
                has_trait = adventurer
            }
            modifier = {
                factor = 0.3
                has_trait = ambitious
            }
        }
    }
    
    # Option B: Temporary wandering - will return
    option = {
        name = murim_wanderer.0001.b
        add_character_modifier = {
            modifier = murim_temporary_wanderer
            years = 3
        }
        add_stress = -15
        trigger_event = {
            id = murim_wanderer.0002
            days = { 30 120 }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                has_trait = patient
            }
        }
    }
    
    # Option C: Stay and face your problems
    option = {
        name = murim_wanderer.0001.c
        add_stress = 15
        add_prestige = 25
        add_character_modifier = {
            modifier = murim_steadfast_resolve
            years = 3
        }
        ai_chance = {
            base = 25
            modifier = {
                factor = 2.0
                has_trait = ambitious
            }
            modifier = {
                factor = 2.0
                is_landed = yes
            }
        }
    }
}
