# Murim Chronicles Phase 2 - Grind Loop Scripted Triggers

murim_can_attempt_body_tempering_trigger = {
	is_alive = yes
	age >= 16
	NOT = { has_trait = murim_qi_deviation_crippling }
	NOT = { has_trait = murim_breakthrough_failure }
	prowess >= 8
	NOT = { has_character_flag = murim_tempering_cooldown }
}

murim_can_attempt_breakthrough_trigger = {
	is_alive = yes
	age >= 16
	exists = var:murim_cultivation_rank
	NOT = {
		custom_description = {
			text = murim_max_cultivation_rank_reached
			var:murim_cultivation_rank >= 6
		}
	}
	NOT = {
		OR = {
			has_trait = murim_qi_deviation_minor
			has_trait = murim_qi_deviation_major
			has_trait = murim_qi_deviation_crippling
		}
	}
	NOT = { has_trait = murim_breakthrough_failure }
	NOT = { has_character_flag = murim_breakthrough_cooldown }
}

murim_can_study_forbidden_trigger = {
	is_alive = yes
	age >= 16
	NOT = { has_trait = murim_forbidden_mark_3 }
	intrigue >= 10
	NOT = { has_character_flag = murim_forbidden_study_cooldown }
}

murim_can_enter_closed_door_trigger = {
	is_alive = yes
	OR = {
		is_ruler = yes
		is_knight = yes
	}
	age >= 16
	NOT = { has_trait = murim_qi_deviation_crippling }
	NOT = { has_character_flag = murim_closed_door_active }
	learning >= 8
}

murim_can_accept_disciple_trigger = {
	is_alive = yes
	age >= 25
	murim_has_any_technique_trait_trigger = yes
	NOT = { has_character_flag = murim_training_disciple }
	prowess >= 15
}
