namespace = murim_iron_fist_gate

# Iron Fist Gate - Full Clan Event Chain
# One of the 9 Great Clans - Orthodox body cultivation sect
# Specialty: External body hardening, iron body techniques, brute force martial arts
# Philosophy: The body is the ultimate weapon. No blade, no poison, no trick surpasses forged flesh.

# ============================================
# EVENT 0001 - Iron Body Initiation Trial
# New disciples must survive the Iron Crucible - heated stones pressed against skin
# ============================================
murim_iron_fist_gate.0001 = {
	type = character_event
	title = murim_iron_fist_gate.0001.t
	desc = murim_iron_fist_gate.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = pain
	}
	right_portrait = {
		character = scope:iron_fist_elder
		animation = personality_bold
	}
	
	trigger = {
		is_alive = yes
		has_trait = murim_martial_artist
		NOT = { has_trait = iron_body_initiate }
		NOT = { has_character_flag = iron_fist_trial_done }
		any_liege_or_above = {
			has_trait = iron_fist_gate_member
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = brave
		}
		modifier = {
			add = -1
			has_trait = craven
		}
	}
	
	immediate = {
		set_character_flag = { flag = iron_fist_trial_done days = 3650 }
		random_courtier = {
			limit = {
				has_trait = iron_fist_gate_elder
			}
			save_scope_as = iron_fist_elder
		}
		if = {
			limit = {
				NOT = { exists = scope:iron_fist_elder }
			}
			random_courtier = {
				limit = {
					has_trait = iron_fist_gate_member
					age >= 40
				}
				save_scope_as = iron_fist_elder
			}
		}
	}
	
	# Endure the Iron Crucible - full trial
	option = {
		name = murim_iron_fist_gate.0001.a
		
		duel = {
			skill = prowess
			target = scope:iron_fist_elder
			50 = {
				desc = murim_iron_fist_gate.0001.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				send_interface_toast = {
					title = murim_iron_fist_gate.0001.a.success.toast
					left_icon = root
					root = {
						add_trait = iron_body_initiate
						add_trait = iron_fist_gate_member
						add_prowess_skill = 2
						add_character_modifier = {
							modifier = iron_crucible_survivor
							years = 10
						}
					}
				}
			}
			50 = {
				desc = murim_iron_fist_gate.0001.a.fail
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
				}
				send_interface_toast = {
					title = murim_iron_fist_gate.0001.a.fail.toast
					left_icon = root
					root = {
						add_trait = wounded
						add_character_modifier = {
							modifier = iron_crucible_scarred
							years = 5
						}
						add_stress = 30
					}
				}
			}
		}
		
		ai_chance = {
			base = 70
			modifier = {
				add = 20
				has_trait = brave
			}
		}
	}
	
	# Refuse - leave Iron Fist Gate
	option = {
		name = murim_iron_fist_gate.0001.b
		
		add_trait = craven
		add_stress = 20
		add_opinion = {
			target = scope:iron_fist_elder
			modifier = disappointed_opinion
			opinion = -30
		}
		
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = craven
			}
		}
	}
	
	# Demand a different trial - challenge the elder directly
	option = {
		name = murim_iron_fist_gate.0001.c
		trigger = {
			has_trait = arrogant
			prowess >= 12
		}
		
		duel = {
			skill = prowess
			target = scope:iron_fist_elder
			30 = {
				desc = murim_iron_fist_gate.0001.c.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2.0
				}
				send_interface_toast = {
					title = murim_iron_fist_gate.0001.c.success.toast
					left_icon = root
					root = {
						add_trait = iron_body_initiate
						add_trait = iron_fist_gate_member
						add_trait = iron_fist_prodigy
						add_prowess_skill = 4
						add_prestige = 200
					}
				}
			}
			70 = {
				desc = murim_iron_fist_gate.0001.c.fail
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2.0
				}
				send_interface_toast = {
					title = murim_iron_fist_gate.0001.c.fail.toast
					left_icon = root
					root = {
						add_trait = severely_injured
						add_prestige = -100
						add_stress = 50
					}
				}
			}
		}
		
		ai_chance = {
			base = 5
		}
	}
}
