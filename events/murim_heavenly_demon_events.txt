########################################
murim_demon.0002 = {
	type = character_event
	title = murim_demon.0002.t
	desc = murim_demon.0002.desc
	theme = war
	left_portrait = {
		character = root
		animation = personality_vengeful
	}
	right_portrait = {
		character = scope:demon_challenger
		animation = personality_bold
	}

	trigger = {
		has_trait = heavenly_demon_disciple
		has_variable = murim_clan_rank
		var:murim_clan_rank >= 4
		NOT = { has_character_flag = demon_succession_cooldown }
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_trait = ambitious
		}
		modifier = {
			add = 2
			prowess >= 20
		}
	}

	immediate = {
		add_character_flag = {
			flag = demon_succession_cooldown
			days = 1825
		}
		random_any_court_province_or_below = {
			limit = {
				any_pool_character = {
					has_trait = heavenly_demon_disciple
					prowess >= 15
					NOT = { this = root }
					age >= 20
				}
			}
			random_pool_character = {
				limit = {
					has_trait = heavenly_demon_disciple
					prowess >= 15
					NOT = { this = root }
				}
				save_scope_as = demon_challenger
			}
		}
	}

	# Accept the challenge - duel for leadership
	option = {
		name = murim_demon.0002.a
		trigger = {
			exists = scope:demon_challenger
		}
		duel = {
			skill = prowess
			target = scope:demon_challenger
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				desc = murim_demon.0002.a.win
				add_prestige = 500
				add_prowess = 2
				scope:demon_challenger = {
					add_prestige = -200
					random_list = {
						60 = {
							add_trait = wounded_1
						}
						30 = {
							add_trait = wounded_2
						}
						10 = {
							add_trait = wounded_3
						}
					}
				}
				hidden_effect = {
					add_to_variable_list = {
						name = murim_combat_memories
						target = flag:demon_succession_victory
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
				}
				desc = murim_demon.0002.a.lose
				add_prestige = -300
				random_list = {
					50 = {
						add_trait = wounded_2
					}
					30 = {
						add_trait = wounded_3
					}
					20 = {
						add_trait = incapable
					}
				}
				scope:demon_challenger = {
					add_prestige = 500
				}
				hidden_effect = {
					add_to_variable_list = {
						name = murim_combat_memories
						target = flag:demon_succession_defeat
					}
				}
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				prowess >= 22
			}
		}
	}

	# Refuse - political maneuvering instead
