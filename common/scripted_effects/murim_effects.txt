# CULTIVATION EFFECTS
# ==========================================

murim_gain_qi_effect = {
	# Gain qi cultivation progress based on learning and piety
	if = {
		limit = {
			has_trait = cultivation_nascent_soul
		}
		# Already at max cultivation, minor benefit
		add_piety = 50
		create_character_memory = {
			type = murim_qi_cultivation_memory
		}
	}
	else_if = {
		limit = {
			has_trait = cultivation_core_formation
		}
		change_variable = {
			name = murim_cultivation_progress
			add = 10
		}
		add_piety = 25
	}
	else_if = {
		limit = {
			has_trait = cultivation_foundation
		}
		change_variable = {
			name = murim_cultivation_progress
			add = 15
		}
		add_piety = 15
	}
	else = {
		change_variable = {
			name = murim_cultivation_progress
			add = 20
		}
		add_piety = 10
	}
}

murim_attempt_breakthrough_effect = {
	# Attempt cultivation breakthrough with risk of qi deviation
	# Uses martial power and cultivation progress to determine success
	random_list = {
		40 = {
			modifier = {
				add = learning
			}
			modifier = {
				factor = 1.5
				has_trait = cultivation_prodigy
			}
			# Breakthrough success
			run_interaction = murim_cultivation_breakthrough_success_effect
			trigger_event = {
				id = murim_cultivation.0100
				days = 1
			}
		}
		30 = {
			# Partial success - progress but no breakthrough
			change_variable = {
				name = murim_cultivation_progress
				add = 25
			}
			add_character_modifier = {
				modifier = qi_stagnation
				years = 1
			}
			create_character_memory = {
				type = murim_failed_breakthrough_memory
			}
		}
		30 = {
			modifier = {
				factor = 1.5
				has_trait = reckless
			}
			modifier = {
				factor = 0.5
				has_trait = patient
			}
			# Qi deviation - dangerous failure
			run_interaction = murim_qi_deviation_effect
			trigger_event = {
				id = murim_cultivation.0200
				days = 1
			}
		}
	}
}

murim_cultivation_breakthrough_success_effect = {
	# Promote to next cultivation tier
	if = {
		limit = {
			NOR = {
				has_trait = cultivation_body_tempering
				has_trait = cultivation_meridian_opening
				has_trait = cultivation_foundation
				has_trait = cultivation_core_formation
				has_trait = cultivation_nascent_soul
			}
		}
		add_trait = cultivation_body_tempering
	}
	else_if = {
		limit = { has_trait = cultivation_body_tempering }
		remove_trait = cultivation_body_tempering
		add_trait = cultivation_meridian_opening
	}
	else_if = {
		limit = { has_trait = cultivation_meridian_opening }
		remove_trait = cultivation_meridian_opening
		add_trait = cultivation_foundation
	}
	else_if = {
		limit = { has_trait = cultivation_foundation }
		remove_trait = cultivation_foundation
		add_trait = cultivation_core_formation
		add_character_modifier = {
			modifier = core_formation_radiance
			years = 3
		}
	}
	else_if = {
		limit = { has_trait = cultivation_core_formation }
		remove_trait = cultivation_core_formation
		add_trait = cultivation_nascent_soul
		add_character_modifier = {
			modifier = nascent_soul_presence
			years = 5
		}
	}
	add_character_modifier = {
		modifier = breakthrough_euphoria
		years = 1
	}
	set_variable = {
		name = murim_cultivation_progress
		value = 0
	}
	add_prestige = 200
	add_piety = 100
	create_character_memory = {
		type = murim_breakthrough_success_memory
	}
}

murim_qi_deviation_effect = {
	# Qi deviation - cultivation gone wrong
	add_character_modifier = {
		modifier = qi_deviation_recovery
		years = 2
	}
	random_list = {
		40 = {
			# Minor deviation - just pain
			add_stress = 30
			add_prowess_skill = -2
		}
		35 = {
			# Moderate deviation - wounded
			add_trait = wounded_1
			add_stress = 50
			create_character_memory = {
				type = murim_qi_deviation_wounded_memory
			}
		}
		20 = {
			# Severe deviation - seriously wounded
			add_trait = wounded_2
			add_stress = 75
			change_variable = {
				name = murim_cultivation_progress
				add = -50
			}
		}
		5 = {
			# Critical deviation - incapacitated
			add_trait = incapable
			create_character_memory = {
				type = murim_qi_deviation_critical_memory
			}
		}
	}
}

# DUEL EFFECTS
# ==========================================

murim_initiate_duel_effect = {
	# Set up duel variables and trigger duel event chain
	save_scope_as = murim_duel_challenger
	set_variable = {
		name = murim_in_duel
		value = yes
	}
	scope:murim_duel_target = {
		set_variable = {
			name = murim_in_duel
			value = yes
		}
	}
	create_character_memory = {
		type = murim_duel_initiated_memory
		participants = {
			opponent = scope:murim_duel_target
		}
	}
}

murim_resolve_duel_victory_effect = {
	# Duel winner consequences
	add_prestige = 100
	add_character_modifier = {
		modifier = duel_champion_fame
		years = 2
	}
	change_variable = {
		name = murim_duel_wins
		add = 1
	}
	create_character_memory = {
		type = murim_duel_victory_memory
		participants = {
			defeated = scope:murim_duel_loser
		}
	}
	# Check if this victory should trigger ranking update
	if = {
		limit = {
			has_variable = murim_martial_rank
		}
		trigger_event = {
			id = murim_ranking.0100
			days = 3
		}
	}
}

murim_resolve_duel_defeat_effect = {
	# Duel loser consequences
	add_character_modifier = {
		modifier = defeat_shame
		years = 1
	}
	change_variable = {
		name = murim_duel_losses
		add = 1
	}
	add_stress = 25
	# Combat injury roll
	random_list = {
		50 = {
			# Minor injuries, just bruised
			add_character_modifier = {
				modifier = sparring_bruises
				days = 30
			}
		}
		30 = {
			# Wounded from the duel
			add_trait = wounded_1
			create_character_memory = {
				type = murim_duel_wounded_memory
				participants = {
					victor = scope:murim_duel_winner
				}
			}
		}
		15 = {
			# Seriously wounded
			add_trait = wounded_2
			add_character_modifier = {
				modifier = crippling_wound_lingering
				years = 3
			}
		}
		5 = {
			modifier = {
				factor = 2
				has_variable = murim_death_match
			}
			# Death match risk - potentially fatal
			add_trait = incapable
			create_character_memory = {
				type = murim_duel_crippled_memory
				participants = {
					victor = scope:murim_duel_winner
				}
			}
		}
	}
	# Potential craven gain from traumatic defeat
	random = {
		chance = 15
		modifier = {
			factor = 2
			has_trait = wounded_2
		}
		add_trait = craven
		create_character_memory = {
			type = murim_combat_trauma_craven_memory
		}
	}
}

murim_death_match_kill_effect = {
	# Death match killing consequences
	add_character_modifier = {
		modifier = killing_intent_aura
		years = 3
	}
	add_prestige = 200
	add_stress = 20
	create_character_memory = {
		type = murim_killed_in_death_match_memory
		participants = {
			victim = scope:murim_duel_loser
		}
	}
	# May trigger ranker death events if victim was ranked
	scope:murim_duel_loser = {
		if = {
			limit = {
				has_variable = murim_martial_rank
				var:murim_martial_rank <= 50
			}
			trigger_event = {
				id = murim_ranker_death.0001
				days = 1
			}
		}
		death = {
			death_reason = death_duel
			killer = scope:murim_duel_winner
		}
	}
}

murim_apply_combat_injuries_effect = {
	# General combat injury roll - used by many event files
	# Severity scales with prowess difference
	random_list = {
		40 = {
			modifier = {
				factor = 1.5
				prowess >= 15
			}
			# No injury - skilled enough to avoid
			add_stress = 5
		}
		30 = {
			# Light wound
			add_trait = wounded_1
			add_stress = 15
			create_character_memory = {
				type = murim_combat_wounded_memory
			}
		}
		20 = {
			modifier = {
				factor = 0.5
				prowess >= 20
			}
			# Serious wound
			add_trait = wounded_2
			add_stress = 30
			add_character_modifier = {
				modifier = crippling_wound_lingering
				years = 2
			}
		}
		10 = {
			modifier = {
				factor = 0.25
				prowess >= 20
			}
			# Devastating injury - incapacitated
			add_trait = incapable
			add_stress = 60
			create_character_memory = {
				type = murim_combat_incapacitated_memory
			}
		}
	}
}

murim_apply_craven_check_effect = {
	# Check if traumatic experience causes craven trait
	if = {
		limit = {
			NOT = { has_trait = brave }
			NOT = { has_trait = craven }
		}
		random = {
			chance = 10
			modifier = {
				add = 10
				has_trait = wounded_2
			}
			modifier = {
				add = 20
				has_trait = incapable
			}
			modifier = {
				add = 5
				stress >= 75
			}
			add_trait = craven
			create_character_memory = {
				type = murim_developed_craven_memory
			}
		}
	}
}

murim_check_ranking_eligibility_effect = {
	# Check if character qualifies for murim ranking
	if = {
		limit = {
			prowess >= 15
			OR = {
				has_trait = cultivation_foundation
				has_trait = cultivation_core_formation
				has_trait = cultivation_nascent_soul
			}
			NOT = { has_variable = murim_martial_rank }
		}
		set_variable = {
			name = murim_martial_rank
			value = 50
		}
		add_character_modifier = {
			modifier = newly_ranked_confidence
			years = 1
		}
		create_character_memory = {
			type = murim_entered_rankings_memory
		}
		trigger_event = {
			id = murim_ranking.0001
			days = 3
		}
	}
}

murim_update_rank_after_victory_effect = {
	# Update ranking after defeating a ranked opponent
	if = {
		limit = {
			scope:murim_duel_loser = {
				has_variable = murim_martial_rank
			}
			OR = {
				NOT = { has_variable = murim_martial_rank }
				var:murim_martial_rank > scope:murim_duel_loser.var:murim_martial_rank
			}
		}
		# Take the defeated's rank
		set_variable = {
			name = murim_martial_rank
			value = scope:murim_duel_loser.var:murim_martial_rank
		}
		scope:murim_duel_loser = {
			change_variable = {
				name = murim_martial_rank
				add = 1
			}
			add_character_modifier = {
				modifier = fallen_ranker_despair
				years = 1
			}
		}
		create_character_memory = {
			type = murim_rank_climbed_memory
		}
	}
}
