######################################
# 0100 - CLOSED-DOOR CULTIVATION
# Extended isolation training - high reward, high cost
# Locks character for 180+ days, massive XP if survived
######################################

murim_grind.0100 = {
	type = character_event
	title = murim_grind.0100.t
	desc = murim_grind.0100.desc
	theme = mental_health
	left_portrait = {
		character = root
		animation = thinking
	}

	trigger = {
		is_alive = yes
		age >= 18
		has_character_flag = murim_initiated
		NOT = { has_character_flag = murim_closed_door_active }
		NOT = { is_at_war = yes }
		NOT = { is_imprisoned = yes }
		prowess >= 8
		OR = {
			has_trait = qi_condensation
			has_trait = foundation_establishment
			has_trait = core_formation
			has_trait = nascent_soul
			has_trait = spirit_severing
			has_trait = dao_seeking
		}
	}

	immediate = {
		set_character_flag = {
			flag = murim_closed_door_active
			days = 365
		}
	}

	# Standard closed-door - 180 days isolation
	option = {
		name = murim_grind.0100.a
		add_character_modifier = { modifier = murim_closed_door_cultivation days = 180 }
		change_variable = { name = murim_cultivation_xp add = 15 }
		add_character_flag = {
			flag = murim_in_seclusion
			days = 180
		}
		stress_impact = {
			base = 20
			gregarious = 30
			shy = -10
		}
		# Chance of breakthrough insight during seclusion
		random = {
			chance = 25
			trigger_event = { id = murim_grind.0101 days = { 90 150 } }
		}
		# Chance of qi deviation during extended meditation
		random = {
			chance = 15
			trigger_event = { id = murim_grind.0102 days = { 60 120 } }
		}
		ai_chance = {
			base = 20
			modifier = { add = 30 has_trait = diligent }
			modifier = { add = 20 has_trait = patient }
			modifier = { add = 15 has_trait = ambitious }
			modifier = { add = -25 has_trait = gregarious }
			modifier = { add = -20 has_trait = lazy }
			modifier = { add = -15 has_trait = impatient }
		}
	}

	# Extreme closed-door - 360 days, double XP but dangerous
	option = {
		name = murim_grind.0100.b
		add_character_modifier = { modifier = murim_closed_door_extreme days = 360 }
		change_variable = { name = murim_cultivation_xp add = 30 }
		add_character_flag = {
			flag = murim_in_seclusion
			days = 360
		}
		add_stress = 40
		# Higher breakthrough chance
		random = {
			chance = 40
			trigger_event = { id = murim_grind.0101 days = { 120 300 } }
		}
		# Higher deviation chance
		random = {
			chance = 30
			trigger_event = { id = murim_grind.0102 days = { 90 240 } }
		}
		# Chance of permanent health damage
		random = {
			chance = 10
			add_trait = ill
		}
		ai_chance = {
			base = 5
			modifier = { add = 25 has_trait = ambitious }
			modifier = { add = 15 has_trait = brave }
			modifier = { add = -30 has_trait = craven }
			modifier = { add = -20 has_trait = content }
		}
	}

	# Refuse - not the right time
	option = {
		name = murim_grind.0100.c
		remove_character_flag = murim_closed_door_active
		stress_impact = {
			base = 5
			ambitious = 15
		}
		ai_chance = {
			base = 10
			modifier = { add = 30 has_trait = lazy }
			modifier = { add = 20 is_at_war = yes }
		}
	}
}

######################################
# 0200 - SPARRING PARTNER TRAINING
# Find a sparring partner for mutual growth
######################################

murim_grind.0200 = {
	type = character_event
	title = murim_grind.0200.t
	desc = murim_grind.0200.desc
	theme = martial
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:sparring_partner
		animation = personality_bold
	}

	trigger = {
		is_alive = yes
		has_character_flag = murim_initiated
		age >= 16
		NOT = { has_character_flag = murim_in_seclusion }
		any_courtier = {
			is_alive = yes
			has_character_flag = murim_initiated
			prowess >= 6
			NOT = { this = root }
			age >= 16
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				is_alive = yes
				has_character_flag = murim_initiated
				prowess >= 6
				NOT = { this = root }
				age >= 16
			}
			weight = {
				base = 10
				modifier = { add = 20 prowess >= root.prowess }
				modifier = { add = 10 has_relation_friend = root }
				modifier = { add = 5 has_relation_rival = root }
			}
			save_scope_as = sparring_partner
		}
	}

	# Friendly sparring - safe, moderate XP for both
	option = {
		name = murim_grind.0200.a
		change_variable = { name = murim_cultivation_xp add = 2 }
		scope:sparring_partner = {
			change_variable = { name = murim_cultivation_xp add = 2 }
		}
		add_opinion = { target = scope:sparring_partner modifier = murim_sparring_partner opinion = 10 }
		scope:sparring_partner = {
			add_opinion = { target = root modifier = murim_sparring_partner opinion = 10 }
		}
		stress_impact = { base = -5 }
		ai_chance = {
			base = 40
			modifier = { add = 20 has_trait = gregarious }
			modifier = { add = 15 has_relation_friend = scope:sparring_partner }
		}
	}

	# Intense sparring - risk injury, better XP
	option = {
		name = murim_grind.0200.b
		change_variable = { name = murim_cultivation_xp add = 5 }
		scope:sparring_partner = {
			change_variable = { name = murim_cultivation_xp add = 5 }
		}
		# Risk of sparring injury for both
		random = {
			chance = 20
			add_character_modifier = { modifier = murim_sparring_bruises days = 30 }
		}
		random = {
			chance = 20
			scope:sparring_partner = {
				add_character_modifier = { modifier = murim_sparring_bruises days = 30 }
			}
		}
		# Chance sparring turns hostile if rivals
		if = {
			limit = { has_relation_rival = scope:sparring_partner }
			random = {
				chance = 30
				trigger_event = { id = murim_grind.0201 days = 1 }
			}
		}
		ai_chance = {
			base = 25
			modifier = { add = 20 has_trait = brave }
			modifier = { add = 15 has_trait = wrathful }
		}
	}

	# Decline - train alone
	option = {
		name = murim_grind.0200.c
		stress_impact = { base = 5 }
		ai_chance = {
			base = 10
			modifier = { add = 25 has_trait = shy }
			modifier = { add = 15 has_trait = paranoid }
		}
	}
}
