	title = murim_imperial_court.0010.t

# Emperor's forbidden technique vault — heist opportunity
murim_imperial_court.0040 = {
	type = character_event
	title = murim_imperial_court.0040.t
	desc = murim_imperial_court.0040.desc
	theme = intrigue
	left_portrait = { character = root animation = personality_irrational }

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 4
		intrigue >= 14
	}

	# Attempt the heist
	option = {
		name = murim_imperial_court.0040.a
		random_list = {
			40 = {
				modifier = { add = intrigue factor = 0.3 }
				change_variable = { name = murim_cultivation_xp add = 100 }
				add_prestige = 200
				trigger_event = { id = murim_forbidden_technique.0001 days = 7 }
			}
			60 = {
				add_stress = 40
				add_prestige = -200
				add_health = -0.5
			}
		}
		ai_chance = { base = 30 }
	}

		ai_chance = { base = 20 }
	}
}
