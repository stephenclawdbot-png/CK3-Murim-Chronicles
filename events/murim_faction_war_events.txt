# Murim Chronicles - Orthodox vs Unorthodox Faction War Events
# THE GREAT FACTION WAR: Erupts every 20-30 years. The defining conflict of the martial world.
# ORTHODOX (Righteous Path): Mount Hua, Shaolin, Wudang, Tang, Beggar Sect, Iron Fist Gate
# UNORTHODOX (Demonic Path): Heavenly Demon Sect (always leads), Poison Flower Palace, Phantom Blade Sect
# INTEGRATION: murim_clan_triggers.txt (is_orthodox_faction, is_unorthodox_faction, all 9 clan checks)
# INTEGRATION: murim_opinion_modifiers.txt (martial opinion changes from war events)
# INTEGRATION: murim_ranking_events.txt (war heroes gain ranking points)
# INTEGRATION: murim_china_center_triggers.txt (wars centered in China, spread outward)
# INTEGRATION: murim_shadow_politics_events.txt (HD infiltration revealed during wars)
# VANILLA SEPARATION: Faction wars are MARTIAL ONLY. No vanilla armies, no county occupation.
#   Wars are fought through duels, raids, assassinations, and martial confrontations.
#   Political rulers may be AFFECTED but wars don't use CK3 war system.

namespace = murim_faction_war

# =================================================================
# EVENT 1: WAR BREWING - TENSION RISING (Prelude, fires 1-3 years before war)
# =================================================================
murim_faction_war.0001 = {
	type = character_event
	title = murim_faction_war.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_character_flag = flag_orthodox_faction_member }
				desc = murim_faction_war.0001.desc.orthodox
			}
			triggered_desc = {
				trigger = { has_character_flag = flag_unorthodox_faction_member }
				desc = murim_faction_war.0001.desc.unorthodox
			}
			desc = murim_faction_war.0001.desc.neutral
		}
	}
	theme = martial
	
	trigger = {
		murim_is_martial_artist_trigger = yes
		NOT = { has_global_variable = murim_faction_war_active }
		NOT = { has_global_variable = murim_faction_war_cooldown }
		any_living_character = {
			count >= 3
			murim_is_clan_leader_trigger = yes
			has_character_flag = flag_orthodox_faction_member
		}
		any_living_character = {
			count >= 2
			murim_is_clan_leader_trigger = yes
			has_character_flag = flag_unorthodox_faction_member
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.5
			any_living_character = {
				has_character_flag = flag_heavenly_demon_puppet_emperor
			}
		}
		modifier = {
			add = 0.3
			has_global_variable = murim_recent_cross_faction_murder
		}
	}
	
	immediate = {
		set_global_variable = {
			name = murim_faction_war_brewing
			value = yes
		}
		set_variable = {
			name = war_tension_level
			value = 1
		}
	}
	
	# Option A: Rally your faction (faction member)
	option = {
		name = murim_faction_war.0001.a
		trigger = {
			OR = {
				has_character_flag = flag_orthodox_faction_member
				has_character_flag = flag_unorthodox_faction_member
			}
		}
		add_character_modifier = {
			modifier = murim_war_rally_leader
			years = 3
		}
		hidden_effect = {
			set_variable = {
				name = war_tension_level
				value = 2
			}
			# Cross-system: war brewing ripples across all systems
			if = {
				limit = { has_character_flag = flag_orthodox_faction_member }
				trigger_event = { id = murim_orthodox.0001 days = { 7 14 } }
			}
			if = {
				limit = { has_character_flag = flag_unorthodox_faction_member }
				trigger_event = { id = murim_unorthodox.0001 days = { 7 14 } }
			}
			trigger_event = { id = murim_underworld.0003 days = { 14 30 } }
			trigger_event = { id = murim_elder.0001 days = { 7 21 } }
		}
		stress_impact = {
			brave = minor_stress_impact_loss
			craven = medium_stress_impact_gain
		}
	}

	# Option B: Counsel peace (try to prevent war)
	option = {
		name = murim_faction_war.0001.b
		trigger = {
			OR = {
				has_character_flag = flag_orthodox_faction_member
				has_character_flag = flag_unorthodox_faction_member
			}
		}
		add_character_modifier = {
			modifier = murim_peacemaker_reputation
			years = 5
		}
		random = {
			chance = 25
			modifier = {
				add = 15
				diplomacy >= 15
			}
			modifier = {
				add = 10
				has_trait = diplomat
			}
			# Peace succeeds - war delayed
			set_global_variable = {
				name = murim_faction_war_cooldown
				value = yes
			}
			trigger_event = { id = murim_faction_war.0002 days = 30 }
		}
	}
	
	# Option C: Prepare secretly (neutral/independent)
	option = {
		name = murim_faction_war.0001.c
		add_character_modifier = {
			modifier = murim_war_profiteer_preparing
			years = 2
		}
		hidden_effect = {
			set_character_flag = {
				flag = flag_faction_war_opportunist
				years = 5
			}
			# Cross-system: war profiteers connect to underworld and black market
			trigger_event = { id = murim_underworld.0005 days = { 30 60 } }
			trigger_event = { id = murim_season.0001 days = { 14 30 } }
		}
	}
	
	# Option D: Flee the martial world temporarily
	option = {
		name = murim_faction_war.0001.d
		trigger = {
			has_trait = craven
		}
		add_character_modifier = {
			modifier = murim_fled_faction_war
			years = 5
		}
		add_prestige = -200
		stress_impact = {
			craven = medium_stress_impact_loss
			brave = major_stress_impact_gain
		}
	}
}

	}

		trigger = {
			any_living_character = {
				murim_is_clan_leader_trigger = yes
				has_character_flag = flag_shaolin_member

		}
		hidden_effect = {
			random = {
				chance = 40

					add = 20
					diplomacy >= 18

				chance = 60

				trigger_event = { id = murim_faction_war.0010 days = 60 }

	}
	
	# Option B: Propose marriage alliance (cross-faction)
	option = {
		name = murim_faction_war.0002.b
		trigger = {
			any_close_family_member = {
				is_adult = yes
				is_married = no
			}
		}
		hidden_effect = {
			trigger_event = { id = murim_faction_war.0004 days = 30 }
		}
	}
	
	# Option C: Give up on peace, prepare for war
	option = {
		name = murim_faction_war.0002.c
		remove_global_variable = murim_faction_war_cooldown
		hidden_effect = {
