# Murim Chronicles - Top 50 Martial Ranking On Actions
# Yearly recalculation pulse, death removal, duel completion hooks
# AI ranked challenges, tournament scheduling
# Load-safe: all on_actions check exists before variable access
# POLISH FIX: All event IDs use murim_rank.XXXX namespace (matching murim_ranking_events.txt)
# Cross-references: murim_ranking_effects.txt, murim_ranking_events.txt, murim_effects.txt

######################################
# YEARLY RANKING RECALCULATION
# Every January, recalculate all martial artist scores
######################################

on_yearly_pulse = {
	effect = {
		# Recalculate rankings for all martial artist rulers
		every_ruler = {
			limit = {
				is_alive = yes
				has_variable = murim_martial_artist
			}
			# Initialize variables if first time
			murim_initialize_ranking_variables_effect = yes
			# Recalculate score
			murim_recalculate_martial_rank_effect = yes
		}

		# Player yearly ranking summary
		every_player = {
			limit = {
				exists = this
				is_alive = yes
				murim_is_martial_artist_trigger = yes
			}
			trigger_event = {
				id = murim_rank.0006
				days = { 5 15 }
			}
		}

		# AI ranked fighters challenge each other (3-5 per year)
		random_living_character = {
			limit = {
				is_ai = yes
				is_alive = yes
				murim_is_top_50_martial_artist_trigger = yes
				murim_ai_should_challenge_for_rank_trigger = yes
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					has_trait = ambitious
				}
				modifier = {
					add = 3
					has_trait = brave
				}
				modifier = {
					add = -10
					has_trait = craven
				}
			}
			# Save challenger scope for reference in nested block
			save_scope_as = rank_challenger
			# Find a target ranked HIGHER (lower position number) to challenge
			random_living_character = {
				limit = {
					is_alive = yes
					murim_is_top_50_martial_artist_trigger = yes
					NOT = { this = scope:rank_challenger }
					has_variable = murim_martial_rank_position
					scope:rank_challenger = {
						has_variable = murim_martial_rank_position
					}
				}
				weight = {
					base = 1
					modifier = {
						add = 5
						has_variable = murim_martial_rank_position
						var:murim_martial_rank_position >= 1
					}
				}
				save_scope_as = rank_challenge_target
			}
			# Execute the background duel if a target was found
			if = {
				limit = {
					exists = scope:rank_challenge_target
				}
				# Prowess duel resolution
				duel = {
					skill = prowess
					target = scope:rank_challenge_target
					50 = {
						# Challenger wins - swap positions
						compare_modifier = {
							value = scope:duel_value
							multiplier = 2
						}
						desc = murim_rank_challenge_winner_desc
						add_prestige = 150
						murim_record_duel_win_effect = yes
						scope:rank_challenge_target = {
							add_prestige = -100
							murim_record_duel_loss_effect = yes
							murim_roll_combat_injuries_effect = yes
						}
						# Swap ranking positions
						scope:rank_challenge_target = {
							save_scope_as = target
						}
						murim_ranking_position_swap_effect = yes
						# Notify player if in same realm
						every_player = {
							limit = {
								is_alive = yes
								OR = {
									any_vassal = { this = scope:rank_challenger }
									any_vassal = { this = scope:rank_challenge_target }
									any_courtier = { this = scope:rank_challenger }
									any_courtier = { this = scope:rank_challenge_target }
								}
							}
							send_interface_message = {
								type = event_generic_neutral
								title = murim_rank_ai_duel_winner_notification
								left_icon = scope:rank_challenger
								right_icon = scope:rank_challenge_target
								desc = murim_rank_ai_duel_winner_desc
							}
						}
						create_character_memory = {
							type = murim_ranking_duel_victory_memory
						}
					}
					50 = {
						# Challenger loses
						compare_modifier = {
							value = scope:duel_value
							multiplier = -2
						}
						desc = murim_rank_challenge_loser_desc
						add_prestige = -100
						murim_record_duel_loss_effect = yes
						murim_roll_combat_injuries_effect = yes
						scope:rank_challenge_target = {
							add_prestige = 100
							murim_record_duel_win_effect = yes
						}
						every_player = {
							limit = {
								is_alive = yes
								OR = {
									any_vassal = { this = scope:rank_challenger }
									any_vassal = { this = scope:rank_challenge_target }
									any_courtier = { this = scope:rank_challenger }
									any_courtier = { this = scope:rank_challenge_target }
								}
							}
							send_interface_message = {
								type = event_generic_neutral
								title = murim_rank_ai_duel_loser_notification
								left_icon = scope:rank_challenge_target
								right_icon = scope:rank_challenger
								desc = murim_rank_ai_duel_loser_desc
							}
						}
						create_character_memory = {
							type = murim_ranking_duel_defeat_memory
						}
					}
				}
			}
		}
	}
}

######################################
# ON CHARACTER DEATH
# Remove dead fighter from rankings, trigger memorial events
######################################

on_death = {
	effect = {
		if = {
			limit = {
				has_variable = murim_martial_rank_position
			}
			murim_remove_from_rankings_on_death_effect = yes
			if = {
				limit = {
					has_character_modifier = murim_ranked_top_50_modifier
				}
				every_player = {
					limit = {
						is_alive = yes
					}
					trigger_event = {
						id = murim_ranker_death.0001
						days = { 1 10 }
					}
				}
			}
		}
	}
}

######################################
# ON GAME START
# Initialize ranking system variables
######################################

on_game_start_after_lobby = {
	effect = {
		every_living_character = {
			limit = {
				murim_is_martial_artist_trigger = yes
			}
			murim_initialize_ranking_variables_effect = yes
			murim_recalculate_martial_rank_effect = yes
		}
	}
}

######################################
# FIVE-YEAR PULSE: GRAND TOURNAMENT
# Every 5 years, a tournament event fires for players
######################################

on_five_year_pulse = {
	effect = {
		every_player = {
			limit = {
				is_alive = yes
				murim_is_martial_artist_trigger = yes
			}
			random_list = {
				60 = {
					trigger_event = {
						id = murim_rank.0010
						days = { 10 30 }
					}
				}
				40 = {
					# No tournament this cycle
				}
			}
		}
	}
}
