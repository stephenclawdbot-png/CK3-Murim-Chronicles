# murim_scout.0001 — Scout Arrives at Foreign Court (Western)
# Triggers when a scout enters a non-Korean/non-Chinese realm
murim_scout.0001 = {
	type = character_event
	title = murim_scout.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:mount_hua } }
				desc = murim_scout.0001.desc_mount_hua
				# "A composed traveler from the distant East requests an audience with your court.
				#  They move with an unusual grace, and their eyes seem to evaluate everyone 
				#  in the room with a single glance. They claim to represent a 'martial sect' 
				#  — a concept your courtiers struggle to understand."
			}
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:shaolin } }
				desc = murim_scout.0001.desc_shaolin
				# "A bald monk in simple robes arrives at your court, having walked from 
				#  the far East. Despite appearing unarmed and humble, your guards report 
				#  an unsettling feeling — as though this gentle pilgrim could defeat 
				#  every soldier in your retinue without breaking a sweat."
			}
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:heavenly_demon } }
				desc = murim_scout.0001.desc_heavenly_demon
				# "A stranger has been seen lurking in the less reputable districts of your 
				#  capital. Your spies report they've been speaking with prisoners, exiles, 
				#  and the desperate — offering them 'a new life of power in the East.' 
				#  Something about them makes even your hardened spymaster uneasy."
			}
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:wudang } }
				desc = murim_scout.0001.desc_wudang
				# "A soft-spoken scholar from the East has arrived, requesting access to 
				#  your royal library. They claim to seek knowledge of 'the body's hidden 
				#  meridians' — whatever that means. Your court physician is fascinated. 
				#  Your court marshal noticed the scholar wears a sword with practiced ease."
			}
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:beggar_sect } }
				desc = murim_scout.0001.desc_beggar
				# "Your steward reports something strange: a new beggar has appeared in the 
				#  city streets, but this one is... different. The other beggars defer to them. 
				#  They've been seen talking to homeless children, evaluating them with an 
				#  intensity that belies their ragged appearance. And once, a thief tried to 
				#  rob them — and was found unconscious three streets away."
			}
			triggered_desc = {
				trigger = { scope:scout = { var:murim_clan_scout = flag:tang_clan } }
				desc = murim_scout.0001.desc_tang
				# "A quiet traveler from the East has been asking unusual questions in your 
				#  court — about bloodlines, family trees, marriages to Eastern merchants. 
				#  They seem to be searching for someone specific. When asked their purpose, 
				#  they smile and say only: 'Family matters.' Your herbalist noticed they 
				#  carry vials of substances that are definitely not medicine."
			}
		}
	}
	
	theme = diplomacy
	
	# Court ruler gets options:
	option = {
		name = murim_scout.0001.a
		# "Grant them audience — I'm curious about the martial world"
		# Triggers scout evaluation of your court members
		# Grants martial_awareness to the ruler if they don't have it
		trigger_event = {
			id = murim_scout.0002
			days = 7
		}
		if = {
			limit = { NOT = { has_trait = martial_awareness } }
			add_trait = martial_awareness
		}
		# VANILLA SEPARATION: No opinion change with liege, no political consequences
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_trait = curious
			}
			modifier = {
				add = 15
				has_trait = brave
			}
			modifier = {
				add = -20
				has_trait = craven
			}
		}
	}
	
	option = {
		name = murim_scout.0001.b
		# "Allow them to stay but keep them watched"
		# Scout operates covertly, evaluates characters without formal permission
		# Lower chance of finding recruits, but no risk to ruler
		trigger_event = {
			id = murim_scout.0003
			days = 14
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 15
				has_trait = paranoid
			}
		}
	}
	
	option = {
		name = murim_scout.0001.c
		# "Expel this foreign agent immediately"
		# Scout is expelled, realm blacklisted for 10 years
		# VANILLA SEPARATION: This is a martial decision, not a diplomatic incident
		scope:scout = {
			murim_scout_expelled_effect = yes
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = paranoid
			}
			modifier = {
				add = 20
				has_trait = zealous  # Religious rulers suspicious of Eastern practices
			}
		}
	}
	
	option = {
		name = murim_scout.0001.d
		# "Capture them — I want to know more about these 'martial sects'"
		# Triggers capture event. Risky — scout may fight back.
		# If scout is powerful (prowess > ruler's martial), they escape
		# If captured successfully, intelligence event + clan hostility
		trigger_event = {
			id = murim_scout.0004
			days = 3
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 20
				has_trait = sadistic
			}
			modifier = {
				add = 15
				has_trait = ambitious
			}
		}
	}
}

# murim_scout.0002 — Scout Evaluates Court (Formal Audience Granted)
murim_scout.0002 = {
	type = character_event
	title = murim_scout.0002.t
	desc = murim_scout.0002.desc
	# "The martial sect representative spends several days at your court, 
	#  observing everyone from kitchen servants to your finest knights.
	#  They watch sparring matches with an unreadable expression, occasionally
	#  nodding at unexpected moments — not when the strongest swing lands,
	#  but when a squire dodges with natural instinct."
	
	theme = martial
	
	# Scout evaluates all characters in court age 8-40
	# Checks: prowess relative to age, brave/strong/genius traits, martial_awareness
	# If finds candidate(s), triggers recruitment pitch
	
	immediate = {
		# Scout evaluation logic
		# Weighted scoring: genius +5, quick +4, strong +3, brave +3, 
		#   robust +2, ambitious +2, rowdy +2, martial_awareness +2
		# Prowess threshold: age 8-15 needs prowess >= 4, age 16-25 needs >= 8, age 26+ needs >= 12
		# Clan-specific preferences applied:
		#   Mount Hua: +3 for just, +3 for temperate, -3 for deceitful
		#   Shaolin: +3 for calm, +3 for compassionate, +4 for learning >= 10, -3 for wrathful
		#   Heavenly Demon: +3 for ambitious, +3 for deceitful, +2 for sadistic, -3 for just
		#   Tang: +5 for Tang bloodline (var:tang_clan_member), +3 for learning >= 8 (alchemy)
		#   Wudang: +4 for learning >= 12, +3 for patient, +2 for scholar, -3 for wrathful
		#   Beggar Sect: +3 for gregarious, +2 for brave, +5 for no_liege (homeless/wanderers)
	}
	
	option = {
		name = murim_scout.0002.a
		# "They've identified [NUMBER] potential candidates in my court"
		# Scout approaches candidates. For each: recruitment pitch event.
		# If player's character or player's child is a candidate, special event.
		trigger_event = {
			id = murim_recruit.0003  # From murim_clan_recruitment_events.txt
			days = 3
		}
		ai_chance = { base = 100 }
	}
	
	option = {
		name = murim_scout.0002.b
		# "Nobody caught their eye. The scout bows and departs."
		# Scout moves to next realm. Slight disappointment.
		# May trigger murim_scout.0008 (failed recruitment trip) if pattern continues
		ai_chance = { base = 0 }  # Only fires if no candidates found
	}
}

# murim_scout.0004 — Attempted Capture of Scout
murim_scout.0004 = {
	type = character_event
	title = murim_scout.0004.t
	desc = murim_scout.0004.desc
	# "Your guards move to seize the Eastern visitor. What happens next 
	#  depends entirely on how skilled this scout truly is..."
	
	theme = martial
	
	# Outcome based on prowess comparison: scout vs. ruler's best fighter
	# If scout prowess > best guard prowess: scout escapes (fights through)
	# If scout prowess <= best guard prowess: scout captured
	
	option = {
		name = murim_scout.0004.a
		# SCOUT ESCAPES — "They moved through your guards like water through 
		#  fingers. Three soldiers are unconscious. The visitor is gone."
		trigger = {
			# Scout is stronger than any guard
			scope:scout = { prowess >= 16 }
		}
		# Scout escapes, realm blacklisted for 10 years
		# 1-2 guards wounded. Ruler gains martial_awareness from witnessing.
		scope:scout = { murim_scout_expelled_effect = yes }
		if = {
			limit = { NOT = { has_trait = martial_awareness } }
			add_trait = martial_awareness
		}
		# VANILLA SEPARATION: No war, no diplomatic incident. Just embarrassment.
		add_prestige = -50
		ai_chance = { base = 0 }  # Outcome, not choice
	}
	
	option = {
		name = murim_scout.0004.b
		# SCOUT CAPTURED — "After a brief but terrifying display of skill,
		#  your guards overwhelm the visitor through sheer numbers."
		trigger = {
			scope:scout = { prowess < 16 }
		}
		scope:scout = { murim_scout_captured_effect = yes }
		# Triggers ransom/rescue chain. Clan will respond.
		ai_chance = { base = 0 }  # Outcome, not choice
	}
}
