############################################
# MURIM FORBIDDEN ARTS ON_ACTIONS
# Fires dark cultivation path events via yearly
# and monthly pulses. Events 0001-0005 existed
# but had no on_actions hook — wired here.
# Events 0006-0007 are new additions.
#
# All events carry their own trigger blocks
# to gate eligibility (demonic_cultivator trait,
# stress, wounds, forbidden_knowledge variable).
############################################

on_yearly_pulse = {

    # 0001: Demonic qi awakening temptation
    # Fires for near-demonic characters under stress or injury
    random_events = {
        chance_to_happen = 8
        100 = murim_forbidden_arts.0001
    }

    # 0002: Blood ritual opportunity
    # Fires for established demonic cultivators
    random_events = {
        chance_to_happen = 10
        100 = murim_forbidden_arts.0002
    }

    # 0003: Dark martial technique refinement
    random_events = {
        chance_to_happen = 12
        100 = murim_forbidden_arts.0003
    }

    # 0004: Demonic qi crisis / corruption spiral
    random_events = {
        chance_to_happen = 10
        100 = murim_forbidden_arts.0004
    }

    # 0005: Demonic path culmination — redemption or damnation
    # Low chance, only fires at high corruption
    random_events = {
        chance_to_happen = 5
        100 = murim_forbidden_arts.0005
    }

    # 0006: Demonic qi hunger — dark power demands to be fed
    random_events = {
        chance_to_happen = 18
        100 = murim_forbidden_arts.0006
    }

    # 0007: Ancient demonic manual discovery
    random_events = {
        chance_to_happen = 4
        100 = murim_forbidden_arts.0007
    }
}

on_monthly_character_pulse = {

    # Demonic cultivators passively accumulate dark qi pressure
    # Fires murim_forbidden_arts.0006 at elevated rate for active demonic path
    random_events = {
        chance_to_happen = 3
        100 = murim_forbidden_arts.0006
    }
}
