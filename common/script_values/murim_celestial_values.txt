# ==============================================================
# ORTHODOX MANDATE SUPPORT
# Orthodox Sect Masters passively sustain the Mandate.
# Calculated per-character, summed by scripted_effect.
# ==============================================================

# Per-orthodox-sect-master monthly mandate contribution
murim_orthodox_mandate_aura_value = {
	value = 0

	# Must be an Orthodox Sect Master at Foundation+
	if = {
		limit = {
			has_character_flag = murim_orthodox_cultivator
			has_character_flag = murim_sect_master
			murim_cultivation_rank_value >= 3
		}
		# Base contribution
		value = 0.05

		# Higher cultivation = stronger aura
		add = {
			value = murim_cultivation_rank_value
			subtract = 3
			multiply = 0.02
			min = 0
		}

		# Sanctioned Sect bonus (Court recognizes them)
		if = {
			limit = { has_character_flag = murim_sect_sanctioned }
			add = 0.03
		}

		# Jianghu standing amplifies influence
		if = {
			limit = {
				exists = var:murim_jianghu_standing
				var:murim_jianghu_standing >= 65
			}
			add = 0.02
		}
	}

	max = 0.15
}

# Righteous Pillar decision: immediate Mandate burst
murim_righteous_pillar_mandate_gain = {
	value = 5

	# Stronger during Expansion/Advancement (Court listens)
	if = {
		limit = {
			exists = global_var:murim_current_era
			global_var:murim_current_era <= 2
		}
		add = 2
	}

	# Weaker during Division (who cares?)
	if = {
		limit = {
			exists = global_var:murim_current_era
			global_var:murim_current_era = 5
		}
		subtract = 3
	}

	# Higher rank = more impressive declaration
	add = {
		value = murim_cultivation_rank_value
		subtract = 3
		min = 0
		max = 3
	}

	min = 2
	max = 10
	round = yes
}

# Merit gained by Sect Master from Righteous Pillar
murim_righteous_pillar_merit_gain = 20

# Opinion bonus Hegemon gets toward Sect Master
murim_righteous_pillar_opinion_gain = 30


# ==============================================================
# DEMONIC MANDATE DRAIN
# Demonic Sect Masters erode the Mandate passively.
# Smaller per-character than orthodox gain, but they
# also trigger big sabotage events.
# ==============================================================

murim_demonic_mandate_drain_value = {
	value = 0

	if = {
		limit = {
			has_character_flag = murim_demonic_cultivator
			has_character_flag = murim_sect_master
			murim_cultivation_rank_value >= 2
		}
		# Base drain (intentionally smaller than orthodox gain)
		value = -0.03

		# Higher cultivation = deeper rot
		subtract = {
			value = murim_cultivation_rank_value
			subtract = 2
			multiply = 0.01
			min = 0
		}

		# Active Ministry infiltration amplifies drain
		if = {
			limit = { has_character_flag = murim_ministry_infiltrator_active }
			subtract = 0.02
		}

		# Multiple infiltrated ministries
		if = {
			limit = {
				exists = var:murim_infiltrated_ministry_count
				var:murim_infiltrated_ministry_count >= 3
			}
			subtract = 0.03
		}
	}

	min = -0.12
}
