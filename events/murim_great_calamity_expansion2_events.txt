# Event 0001 — Ashes of the Old World
murim_great_calamity_expansion2.0001 = {
	type = character_event
	title = murim_great_calamity_exp2.0001.t
	desc = murim_great_calamity_exp2.0001.desc
	theme = death
	left_portrait = {
		character = root
		animation = grief
	}

	trigger = {
		has_global_variable = murim_calamity_aftermath
		has_variable = murim_calamity_survivor
		NOT = { has_character_flag = calamity_ashes_witnessed }
	}

	weight_multiplier = {
		base = 100
		modifier = {
			add = 50
			has_trait = murim_calamity_witness
		}
	}

	immediate = {
		set_character_flag = calamity_ashes_witnessed
		# Count surviving martial artists in realm
		every_living_character = {
			limit = {
				has_trait = murim_martial_artist
				is_alive = yes
			}
			add_to_temporary_list = calamity_survivors_list
		}
	}

	# Option A — Survey the ruins and count survivors
	option = {
		name = murim_great_calamity_exp2.0001.a
		add_character_modifier = {
			modifier = murim_calamity_surveyor
			years = 10
		}
		# Set survivor count variable
		set_variable = {
			name = murim_survivors_counted
			value = 1
		}
		every_in_list = {
			list = calamity_survivors_list
			limit = { is_alive = yes }
			root = {
				change_variable = {
					name = murim_survivors_counted
					add = 1
				}
			}
		}
		stress_impact = {
			compassionate = medium_stress_impact_loss
			callous = medium_stress_impact_gain
		}
		trigger_event = {
			id = murim_great_calamity_expansion2.0002
			days = { 30 60 }
		}
	}

	# Option B — Mourn the dead, honor the fallen
	option = {
		name = murim_great_calamity_exp2.0001.b
		add_piety = 500
		add_character_modifier = {
			modifier = murim_mourning_the_fallen
			years = 5
		}
		every_in_list = {
			list = calamity_survivors_list
			add_opinion = {
				modifier = murim_shared_grief
				target = root
				opinion = 30
			}
		}
		trigger_event = {
			id = murim_great_calamity_expansion2.0003
			days = { 60 120 }
		}
	}

	# Option C — The old world is gone. We start anew.
	option = {
		name = murim_great_calamity_exp2.0001.c
		add_prestige = 1000
		add_character_modifier = {
			modifier = murim_new_era_founder
			years = 20
		}
		set_global_variable = {
			name = murim_new_era_declared
			value = yes
		}
		trigger_event = {
			id = murim_great_calamity_expansion2.0004
			days = { 15 30 }
		}
	}
}

# Event 0002 — The Survivor Census
murim_great_calamity_expansion2.0002 = {
	type = character_event
	title = murim_great_calamity_exp2.0002.t
	desc = murim_great_calamity_exp2.0002.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = thinking
	}

	trigger = {
		has_character_flag = calamity_ashes_witnessed
		has_variable = murim_survivors_counted
	}

	immediate = {
		# Categorize survivors by clan
		every_living_character = {
			limit = {
				has_trait = murim_martial_artist
				has_trait = murim_clan_member
				is_alive = yes
			}
			add_to_temporary_list = clan_survivors
		}
		every_living_character = {
			limit = {
				has_trait = murim_martial_artist
				NOT = { has_trait = murim_clan_member }
				is_alive = yes
			}
			add_to_temporary_list = clanless_survivors
		}
	}

	# Option A — Recruit clanless survivors into our ranks
	option = {
		name = murim_great_calamity_exp2.0002.a
		every_in_list = {
			list = clanless_survivors
			limit = {
				NOT = { has_relation_rival = root }
				opinion = { target = root value >= -20 }
			}
			add_opinion = {
				modifier = murim_recruited_after_calamity
				target = root
				opinion = 40
			}
			set_variable = {
				name = murim_adopted_clan
				value = root
			}
		}
		add_character_modifier = {
			modifier = murim_rebuilder_recruiter
			years = 15
		}
		trigger_event = {
			id = murim_great_calamity_expansion2.0005
			days = { 90 180 }
		}
	}

	# Option B — Focus on clan survivors only

		name = murim_great_calamity_exp2.0002.b
		every_in_list = {
			list = clan_survivors
			add_opinion = {
				modifier = murim_clan_solidarity
				target = root
				opinion = 25
			}
		}
		add_character_modifier = {
			modifier = murim_clan_purist_rebuilder
			years = 15
		}
	}

	# Option C — Create a new unified order from all survivors
	option = {
		name = murim_great_calamity_exp2.0002.c
		add_prestige = 500
		add_character_modifier = {
			modifier = murim_unifier_of_survivors
			years = 20
		}
		set_global_variable = {
			name = murim_unified_survivor_order

# Event 0004 — Declaration of the New Era

	type = character_event

	desc = murim_great_calamity_exp2.0004.desc

		animation = personality_bold
	}

	trigger = {
		has_global_variable = murim_new_era_declared
	}

	immediate = {
		# Increment era counter
		if = {
			limit = { NOT = { has_global_variable = murim_era_count } }

			}

		}

			}

	desc = murim_great_calamity_exp2.0010.desc

	left_portrait = {
		character = root
		animation = war_over_win
	}

	trigger = {
		OR = {
			has_character_modifier = murim_old_era_loyalist
			has_character_modifier = murim_supreme_survivor_leader
		}
	}

	immediate = {
		# Identify the resistance leader
		random_living_character = {
			limit = {
				has_trait = murim_martial_artist
				has_character_modifier = murim_old_era_loyalist
				NOT = { this = root }
				prowess >= 15
			}
			save_scope_as = resistance_leader
			weight = {
				base = 1
				modifier = {
					add = 5
					prowess >= 25
				}
				modifier = {
					add = 3
					has_trait = ambitious
				}
			}
		}
	}

	# Option A — Crush the resistance with force
	option = {
		name = murim_great_calamity_exp2.0010.a
		trigger = {
			prowess >= 20
		}
		add_prestige = 500
		add_tyranny = 20
		if = {
			limit = { exists = scope:resistance_leader }
			scope:resistance_leader = {
				add_character_modifier = {
					modifier = murim_crushed_rebel
					years = 10
				}
				add_opinion = {
					modifier = murim_defeated_by_tyrant
					target = root
					opinion = -60
				}
			}
		}
		add_character_modifier = {
			modifier = murim_iron_ruler
			years = 20
		}
	}

	# Option B — Negotiate with the dissidents
	option = {
		name = murim_great_calamity_exp2.0010.b
		add_diplomacy = 2
		if = {
			limit = { exists = scope:resistance_leader }
			scope:resistance_leader = {
				add_opinion = {
					modifier = murim_willing_to_talk
					target = root
					opinion = 20
				}
			}
		}
		add_character_modifier = {
			modifier = murim_peaceful_resolver
			years = 15
		}
	}

	# Option C — Step down and let them choose a new leader
	option = {
		name = murim_great_calamity_exp2.0010.c
		remove_character_modifier = murim_supreme_survivor_leader
		add_prestige = -500
		add_piety = 500
		add_character_modifier = {
			modifier = murim_graceful_abdication
			years = 30
		}
		every_living_character = {
			limit = {
				has_trait = murim_martial_artist
				is_alive = yes
			}
			add_opinion = {
				modifier = murim_respected_humility
				target = root
				opinion = 30
			}
		}
	}
}
