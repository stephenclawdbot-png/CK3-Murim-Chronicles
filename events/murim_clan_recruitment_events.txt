# Murim Chronicles - Clan Recruitment Events (Dual-Mode System)
# TWO MODES: Emergency (global scouting when youth < 3) and Normal (local opportunistic, always-on)
# INTEGRATION: Uses murim_clan_triggers.txt (clan_needs_recruits, is_recruitable_youth)
# INTEGRATION: Uses murim_discovery_triggers.txt (is_in_martial_region, martial_awareness checks)
# INTEGRATION: Uses murim_clan_membership_effects.txt (join_sect_as_outer_disciple_effect)
# INTEGRATION: Feeds murim_youth_tournament_events.txt (recruited kids enter Young Dragon Tournament)
# INTEGRATION: Feeds murim_sect_politics_events.txt (recruited members rise through ranks)
# Cross-refs: murim_training_events.txt (new recruits train), murim_mentor_events.txt (assigned masters),
#             murim_clan_war_events.txt (recruits fight), murim_wandering_events.txt (wanderers recruited)
# GENDER: Fully gender-neutral. Boys AND girls equally scouted. No restrictions.
# AI: Both modes fire automatically for AI clans. AI parents decide based on personality.

namespace = murim_recruit

# MODE 1: EMERGENCY RECRUITMENT — Global Scouting (youth < 3)
######################################################################

# murim_recruit.0001 — Clan Dispatches Scouts Abroad
# Trigger: on_yearly_pulse, clan has < 3 members aged 8-15
# The clan is desperate. Senior members are sent worldwide to find talent.
murim_recruit.0001 = {
	type = character_event
	title = murim_recruit.0001.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_trait = mount_hua_disciple }
				desc = murim_recruit.0001.desc.mount_hua
			}
			triggered_desc = {
				trigger = { has_trait = shaolin_monk }
				desc = murim_recruit.0001.desc.shaolin
			}
			triggered_desc = {
				trigger = { has_trait = heavenly_demon_member }
				desc = murim_recruit.0001.desc.demon
			}
			triggered_desc = {
				trigger = { has_trait = tang_clan_member }
				desc = murim_recruit.0001.desc.tang
			}
			triggered_desc = {
				trigger = { has_trait = wudang_disciple }
				desc = murim_recruit.0001.desc.wudang
			}
			triggered_desc = {
				trigger = { has_trait = beggar_sect_member }
				desc = murim_recruit.0001.desc.beggar
			}
			desc = murim_recruit.0001.desc.generic
		}
	}
	theme = martial
	
	trigger = {
		# Must be a sect elder or leader to dispatch scouts
		murim_is_sect_leadership_trigger = yes
		# Clan needs recruits (fewer than 3 youth aged 8-15)
		murim_clan_needs_recruits_trigger = yes
		# Cooldown: don't spam scouting
		NOT = { has_character_flag = murim_scouts_dispatched_recently }
	}
	
	weight_multiplier = {
		base = 1
		# More urgent if zero youth
		modifier = {
			add = 2
			murim_clan_youth_count_value = 0
		}
		# Slightly more urgent if only 1
		modifier = {
			add = 1
			murim_clan_youth_count_value = 1
		}
	}
	
	immediate = {
		set_character_flag = {
			flag = murim_scouts_dispatched_recently
			years = 2
		}
	}
	
	# Option A: Send scouts to Korea and China (close range)
	option = {
		name = murim_recruit.0001.a
		# Search for talented youth in the core martial regions
		# Finds random eligible youth in Goryeo/China courts
		random_list = {
			70 = {
				# Success: found a promising candidate
				trigger_event = {
					id = murim_recruit.0002
					days = { 30 90 }
				}
			}
			30 = {
				# Failure: no suitable candidates found nearby
				trigger_event = {
					id = murim_recruit.0008
					days = { 60 120 }
				}
			}
		}
		stress_impact = {
			base = 5
		}
		ai_chance = {
			base = 60
		}
	}
	
	# Option B: Send scouts GLOBALLY — expensive but wider net
	option = {
		name = murim_recruit.0001.b
		# Global scouting: can find talent anywhere in the world
		add_gold = -50
		random_list = {
			50 = {
				# Found someone far away
				trigger_event = {
					id = murim_recruit.0002
					days = { 90 180 }
				}
				# Set flag so the recruit event knows this is a foreign candidate
				set_character_flag = {
					flag = murim_global_scout_active
					years = 1
				}
			}
			30 = {
				# Found multiple candidates — bidding war possible
				trigger_event = {
					id = murim_recruit.0007
					days = { 90 180 }
				}
			}
			20 = {
				# Failed — nobody worthy found abroad
				trigger_event = {
					id = murim_recruit.0008
					days = { 120 240 }
				}
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 20
				murim_clan_youth_count_value = 0
			}
		}
	}
	
	# Option C: Lower standards — accept weaker candidates locally
	option = {
		name = murim_recruit.0001.c
		# Desperate: accept anyone with even marginal potential
		custom_tooltip = murim_recruit.0001.c.tt
		# Guaranteed to find SOMEONE, but they're mediocre
		trigger_event = {
			id = murim_recruit.0002
			days = { 15 30 }
		}
		set_character_flag = {
			flag = murim_lowered_standards_recruit
			years = 3
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 30
				murim_clan_youth_count_value = 0
				# Only pick this if truly desperate AND already tried normal scouting
				has_character_flag = murim_previous_scout_failed
			}
		}
	}
}

# murim_recruit.0002 — Scout Arrives, Evaluates Candidate
# Fires after 0001 succeeds. Scout has found a promising youth.
murim_recruit.0002 = {
	type = character_event
	title = murim_recruit.0002.t
	desc = murim_recruit.0002.desc
	theme = martial
	
	trigger = {
		murim_is_sect_leadership_trigger = yes
	}
	
	immediate = {
		# Find the best candidate in the region
		# Priority: high prowess, brave/strong/genius traits, martial_awareness
		random_character = {
			limit = {
				murim_is_recruitable_youth_trigger = yes
				age >= 8
				age < 16
				NOR = {
					has_trait = mount_hua_disciple
					has_trait = shaolin_monk
					has_trait = heavenly_demon_member
					has_trait = tang_clan_member
					has_trait = tang_clan_associate
					has_trait = wudang_disciple
					has_trait = beggar_sect_member
				}
				# If global scout, can be anywhere; otherwise must be in martial region
				trigger_if = {
					limit = {
						prev = { NOT = { has_character_flag = murim_global_scout_active } }
					}
					murim_is_in_martial_region_trigger = yes
				}
			}
			weight = {
				base = 1
				modifier = { add = 3 prowess >= 8 }
				modifier = { add = 5 prowess >= 12 }
				modifier = { add = 2 has_trait = brave }
				modifier = { add = 2 has_trait = strong }
				modifier = { add = 3 has_trait = genius }
				modifier = { add = 2 has_trait = quick }
				modifier = { add = 1 has_trait = ambitious }
				modifier = { add = 2 has_trait = martial_awareness }
				modifier = { add = 3 has_character_flag = murim_tournament_participant }
				# If lowered standards, accept weaker candidates
				modifier = {
					add = 5
					prev = { has_character_flag = murim_lowered_standards_recruit }
				}
			}
			save_scope_as = recruit_candidate
		}
	}
	
	# Option A: Approach the candidate's parent/guardian
	option = {
		name = murim_recruit.0002.a
		# Fires the parent notification event
		scope:recruit_candidate = {
			# If they have a parent alive, notify the parent
			trigger_if = {
				limit = {
					any_parent = { is_alive = yes }
				}
				random_parent = {
					limit = { is_alive = yes }
					trigger_event = {
						id = murim_recruit.0003
						days = { 7 14 }
					}
				}
			}
			# If orphan, approach them directly
			trigger_else = {
				trigger_event = {
					id = murim_recruit.0005
					days = { 7 14 }
				}
			}
		}
		ai_chance = { base = 80 }
	}
	
	# Option B: Test them first before offering
	option = {
		name = murim_recruit.0002.b
		custom_tooltip = murim_recruit.0002.b.tt
		# Trial event — candidate must prove themselves
		scope:recruit_candidate = {
			trigger_event = {
				id = murim_recruit.0006
				days = { 7 14 }
			}
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				# Wudang and Mount Hua prefer testing first
				OR = {
					has_trait = wudang_disciple
					has_trait = mount_hua_disciple
				}
			}
		}
	}
}

# murim_recruit.0003 — Player/AI Parent: Your Child Has Been Scouted
# The key decision point for parents.
murim_recruit.0003 = {
	type = character_event
	title = murim_recruit.0003.t
	desc = murim_recruit.0003.desc
	theme = martial
	
	trigger = {
		any_child = {
			is_alive = yes
			age >= 8
			age < 16
		}
	}
	
	# Option A: Accept proudly
	option = {
		name = murim_recruit.0003.a
		# Child moves to clan, gains outer_disciple
		custom_tooltip = murim_recruit.0003.a.tt
		random_child = {
			limit = {
				murim_is_recruitable_youth_trigger = yes
				age >= 8
				age < 16
			}
			# Join the clan
			murim_join_sect_as_outer_disciple_effect = yes
			# Move to clan's region if needed
			trigger_event = {
				id = murim_recruit.0005
				days = { 14 30 }
			}
		}
		# Parent gains prestige — your child is special
		add_prestige = 100
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = ambitious }
			modifier = { add = 10 num_of_children >= 3 }
			modifier = { add = 15 culture = { has_cultural_pillar = heritage_korean } }
			modifier = { add = 15 culture = { has_cultural_pillar = heritage_chinese } }
			modifier = { add = -20 has_trait = paranoid }
		}
	}
	
	# Option B: Accept reluctantly
	option = {
		name = murim_recruit.0003.b
		custom_tooltip = murim_recruit.0003.b.tt
		random_child = {
			limit = {
				murim_is_recruitable_youth_trigger = yes
				age >= 8
				age < 16
			}
			murim_join_sect_as_outer_disciple_effect = yes
			add_character_modifier = {
				modifier = murim_homesick_modifier
				years = 3
			}
			trigger_event = {
				id = murim_recruit.0005
				days = { 14 30 }
			}
		}
		add_prestige = 50
		stress_impact = {
			base = 20
		}
		ai_chance = {
			base = 20
			modifier = { add = 15 has_trait = compassionate }
		}
	}
	
	# Option C: Negotiate terms
	option = {
		name = murim_recruit.0003.c
		custom_tooltip = murim_recruit.0003.c.tt
		# Demand gold/alliance in exchange
		random_list = {
			60 = {
				# Clan agrees to terms
				add_gold = 100
				add_prestige = 50
				random_child = {
					limit = {
						murim_is_recruitable_youth_trigger = yes
						age >= 8
						age < 16
					}
					murim_join_sect_as_outer_disciple_effect = yes
					trigger_event = {
						id = murim_recruit.0005
						days = { 14 30 }
					}
				}
			}
			40 = {
				# Clan withdraws — too demanding
				custom_tooltip = murim_recruit.0003.c.fail.tt
				add_prestige = -25
			}
		}
		ai_chance = {
			base = 15
			modifier = { add = 20 has_trait = greedy }
			modifier = { add = 10 has_trait = shrewd }
		}
	}
	
	# Option D: Refuse politely
	option = {
		name = murim_recruit.0003.d
		custom_tooltip = murim_recruit.0003.d.tt
		# Child stays. Slight opinion penalty with clan.
		# 10% chance child resents you / runs away
		random_list = {
			90 = {
				# Child accepts your decision
				custom_tooltip = murim_recruit.0003.d.stay.tt
			}
			10 = {
				# Child runs away to join on their own!
				custom_tooltip = murim_recruit.0003.d.runaway.tt
				random_child = {
					limit = {
						murim_is_recruitable_youth_trigger = yes
						age >= 8
						age < 16
						has_trait = ambitious
					}
					murim_join_sect_as_outer_disciple_effect = yes
					add_opinion = {
						target = prev
						modifier = murim_prevented_my_dream
						opinion = -30
					}
				}
			}
		}
		ai_chance = {
			base = 25
			modifier = { add = 15 has_trait = paranoid }
			modifier = { add = 10 num_of_children <= 1 }
		}
	}
	
	# Option E: "Over my dead body"
	option = {
		name = murim_recruit.0003.e
		custom_tooltip = murim_recruit.0003.e.tt
		# Hostile refusal. Clan won't ask again. Strong opinion penalty.
		add_character_flag = {
			flag = murim_hostile_to_recruitment
			years = 20
		}
		# The recruiting clan remembers this insult
		stress_impact = {
			base = 10
		}
		ai_chance = {
			base = 10
			modifier = { add = 20 has_trait = wrathful }
			modifier = { add = 10 has_trait = stubborn }
		}
	}
}

# murim_recruit.0004 — AI Parent Decision Logic
# Hidden event: determines AI parent behavior for recruitment
murim_recruit.0004 = {
	type = character_event
	title = murim_recruit.0004.t
	desc = murim_recruit.0004.desc
	theme = martial
	hidden = yes
	
	trigger = {
		is_ai = yes
		any_child = {
			murim_is_recruitable_youth_trigger = yes
			age >= 8
			age < 16
		}
	}
	
	immediate = {
		# AI decision weights based on personality
		random_list = {
			# Accept proudly
			30 = {
				modifier = { add = 20 has_trait = ambitious }
				modifier = { add = 10 num_of_children >= 3 }
				modifier = { add = 15 culture = { has_cultural_pillar = heritage_korean } }
				modifier = { add = 15 culture = { has_cultural_pillar = heritage_chinese } }
				random_child = {
					limit = {
						murim_is_recruitable_youth_trigger = yes
						age >= 8
						age < 16
					}
					murim_join_sect_as_outer_disciple_effect = yes
				}
			}
			# Accept reluctantly
			20 = {
				modifier = { add = 10 has_trait = compassionate }
				random_child = {
					limit = {
						murim_is_recruitable_youth_trigger = yes
						age >= 8
						age < 16
					}
					murim_join_sect_as_outer_disciple_effect = yes
					add_character_modifier = {
						modifier = murim_homesick_modifier
						years = 3
					}
				}
			}
			# Refuse
			40 = {
				modifier = { add = 15 has_trait = paranoid }
				modifier = { add = 20 num_of_children <= 1 }
				modifier = { add = -15 has_trait = ambitious }
			}
			# Hostile refusal
			10 = {
				modifier = { add = 15 has_trait = wrathful }
				add_character_flag = {
					flag = murim_hostile_to_recruitment
					years = 20
				}
			}
		}
	}
}

# murim_recruit.0006 — Foreign Child at Young Dragon Tournament

	type = character_event
	title = murim_recruit.0006.t

		age >= 8

					days = { 7 14 }

			}
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 20
				OR = {
					has_trait = mount_hua_disciple
					has_trait = wudang_disciple

				}
			}

# A recruit who was actually a spy from another clan.
murim_recruit.0050 = {
	type = character_event
	title = murim_recruit.0050.t
	desc = murim_recruit.0050.desc
	theme = intrigue
	
	trigger = {
		murim_is_clan_member_trigger = yes
		has_character_flag = murim_planted_spy
		# Discovery chance increases over time
		has_character_flag = { flag = murim_spy_plant_date days >= 365 }
	}
	
	weight_multiplier = {
		base = 0.05  # 5% yearly discovery chance
		modifier = {
			add = 0.05
			any_character = {
				murim_is_same_sect_trigger = yes
				intrigue >= 15
			}
		}
	}
	
	# Option A: Execute the spy
	option = {
		name = murim_recruit.0050.a
		custom_tooltip = murim_recruit.0050.a.tt
		# Harsh but sends a message
		ai_chance = {
			base = 30
			modifier = { add = 20 has_trait = murim_vengeful }
		}
	}
	
	# Option B: Turn them into a double agent
	option = {
		name = murim_recruit.0050.b
		custom_tooltip = murim_recruit.0050.b.tt
		# Feed false information back to their original clan
		ai_chance = {
			base = 25
			modifier = { add = 20 has_trait = deceitful }
		}
	}
	
	# Option C: Expel them publicly
	option = {
		name = murim_recruit.0050.c
		custom_tooltip = murim_recruit.0050.c.tt
		murim_expel_from_sect_effect = yes

	}
	
	# Option D: Forgive them — they've become loyal
	option = {
