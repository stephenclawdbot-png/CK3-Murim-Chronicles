namespace = murim_cross_hooks

# Tournament winner affects rankings
murim_cross_hooks.0001 = {
	type = character_event
	title = murim_cross_hooks.0001.t
	desc = murim_cross_hooks.0001.desc
	theme = martial

	trigger = {
		has_variable = murim_martial_artist
		has_character_flag = murim_tournament_winner
	}

	immediate = {
		remove_character_flag = murim_tournament_winner
	}

	# Ranking rises — fame spreads
	option = {
		name = murim_cross_hooks.0001.a
		change_variable = { name = murim_cultivation_rank add = 1 }
		add_prestige = 300
		trigger_event = { id = murim_ranking.0001 days = 30 }
		ai_chance = { base = 100 }
	}
}

# High ranking attracts tribunal attention

	type = character_event
	title = murim_cross_hooks.0010.t
	desc = murim_cross_hooks.0010.desc
	theme = martial

	trigger = {
		has_variable = murim_martial_artist
		var:murim_cultivation_rank >= 5
		NOT = { has_trait = murim_tribunal_judge }
		NOT = { has_character_flag = murim_tribunal_noticed }
	}

	immediate = {
		set_character_flag = { flag = murim_tribunal_noticed years = 10 }
	}

	# Accept the Tribunal's interest
	option = {
		name = murim_cross_hooks.0010.a
		add_prestige = 100
		trigger_event = { id = murim_tribunal_system.0030 days = 60 }
		ai_chance = { base = 55 }
	}

	# Decline — stay independent
	option = {
		name = murim_cross_hooks.0010.b
		add_stress = -10
		ai_chance = { base = 45 }
	}
}
