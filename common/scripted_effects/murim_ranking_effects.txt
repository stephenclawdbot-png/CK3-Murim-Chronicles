######################################
# RECALCULATE MARTIAL RANK
# Yearly pulse - recalculates score for a single character
######################################

murim_recalculate_martial_rank_effect = {
	# Safety check
	if = {
		limit = {
			exists = this
			is_alive = yes
			murim_is_martial_artist_trigger = yes
		}
		# Calculate new score using the composite formula from murim_ranking_values.txt
		set_variable = {
			name = murim_martial_rank_score
			value = murim_total_ranking_score_value
		}
		# Check if score crosses top 50 threshold
		if = {
			limit = {
				var:murim_martial_rank_score >= murim_is_top_50_threshold_value
				NOT = { has_character_modifier = murim_ranked_top_50_modifier }
			}
			# New entry into Top 50
			murim_enter_top_50_effect = yes
		}
		# Check if dropped below threshold
		if = {
			limit = {
				var:murim_martial_rank_score < murim_is_top_50_threshold_value
				has_character_modifier = murim_ranked_top_50_modifier
			}
			# Dropped out of Top 50
			murim_leave_top_50_effect = yes
		}
		# Update tier modifiers
		murim_update_rank_tier_modifiers_effect = yes
	}
}

######################################
# ENTER TOP 50
# Character crosses threshold, gains modifier, notification fires
######################################

murim_enter_top_50_effect = {
	# Add ranked fighter modifier
	add_character_modifier = {
		modifier = murim_ranked_top_50_modifier
		years = 1
	}
	# Notification to the character - FIXED: namespace is murim_rank, not murim_ranking
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_rank.0001
			days = 1
		}
	}
	# Prestige boost for entering rankings
	add_prestige = 200
	# Set initial rank position variable if not set
	if = {
		limit = {
			NOT = { has_variable = murim_martial_rank_position }
		}
		set_variable = {
			name = murim_martial_rank_position
			value = 50
		}
	}
	# Memory of entering rankings
	create_character_memory = {
		type = murim_entered_rankings_memory
	}
	# Court notification for liege
	if = {
		limit = {
			exists = liege
			liege = { is_ai = no }
		}
		liege = {
			send_interface_message = {
				type = event_generic_neutral
				title = murim_rank_new_entry_notification
				left_icon = root
				desc = murim_rank_new_entry_notification_desc
			}
		}
	}
	# Stress relief from glory
	add_stress = -15
}

######################################
# LEAVE TOP 50
# Dropped out, loses modifier, shame
######################################

murim_leave_top_50_effect = {
	# Remove ranked modifier
	if = {
		limit = { has_character_modifier = murim_ranked_top_50_modifier }
		remove_character_modifier = murim_ranked_top_50_modifier
	}
	# Notification to the character - FIXED: namespace is murim_rank
	if = {
		limit = { is_ai = no }
		trigger_event = {
			id = murim_rank.0002
			days = 1
		}
	}
	# Shame modifier
	add_character_modifier = {
		modifier = murim_fallen_from_rankings_modifier
		years = 2
	}
	# Memory of falling from rankings
	create_character_memory = {
		type = murim_fell_from_rankings_memory
	}
	# Prestige loss
	add_prestige = -100
	# Stress from public failure
	add_stress = 20
	# Remove position variable
	if = {
		limit = { has_variable = murim_martial_rank_position }
		remove_variable = murim_martial_rank_position
	}
}

######################################
# REMOVE FROM RANKINGS ON DEATH
# Dead fighters removed, next in line promoted
######################################

murim_remove_from_rankings_on_death_effect = {
	# Only process if was a ranked fighter
	if = {
		limit = {
			has_variable = murim_martial_rank_position
		}
		# Store the dead fighter's rank for promotion logic
		save_temporary_scope_as = dead_ranked_fighter
		# Was this the #1?
		if = {
			limit = {
				var:murim_martial_rank_position = 1
			}
			# World event for #1 dying - FIXED: namespace murim_rank
			every_living_character = {
				limit = {
					is_ai = no
				}
				trigger_event = {
					id = murim_rank.0003
					days = 3
				}
			}
		}
		# General ranker death notification - FIXED: namespace murim_rank
		if = {
			limit = {
				any_living_character = {
					is_ai = no
					murim_is_martial_artist_trigger = yes
				}
			}
			every_living_character = {
				limit = {
					is_ai = no
					murim_is_martial_artist_trigger = yes
				}
				send_interface_message = {
					type = event_generic_neutral
					title = murim_rank_death_notification
					left_icon = scope:dead_ranked_fighter
					desc = murim_rank_death_desc
				}
			}
		}
	}
}

######################################
# RANKING DUEL EFFECT
# Resolve a formal ranking challenge between two ranked fighters
######################################

murim_ranking_duel_effect = {
	# Both combatants must exist and be alive
	if = {
		limit = {
			exists = scope:rank_challenger
			exists = scope:rank_challenge_target
			scope:rank_challenger = { is_alive = yes }
			scope:rank_challenge_target = { is_alive = yes }
		}
		# Duel resolution
		duel = {
			skill = prowess
			target = scope:rank_challenge_target
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
				}
				modifier = {
					add = 10
					scope:rank_challenger = { has_trait = brave }
				}
				desc = murim_rank_challenge_winner_desc
				# Challenger wins - swap positions
				scope:rank_challenger = {
					create_character_memory = {
						type = murim_ranking_duel_victory_memory
					}
					add_prestige = 150
					add_character_modifier = {
						modifier = murim_recent_duel_cooldown
						years = 1
					}
					# Swap rank positions
					if = {
						limit = {
							scope:rank_challenge_target = { has_variable = murim_martial_rank_position }
						}
						set_variable = {
							name = murim_martial_rank_position
							value = scope:rank_challenge_target.var:murim_martial_rank_position
						}
					}
				}
				scope:rank_challenge_target = {
					create_character_memory = {
						type = murim_ranking_duel_defeat_memory
					}
					add_character_modifier = {
						modifier = defeat_shame
						years = 2
					}
					add_character_modifier = {
						modifier = murim_recent_duel_cooldown
						years = 1
					}
					# Wound chance
					random = {
						chance = 25
						add_trait = wounded_1
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
				}
				modifier = {
					add = 10
					scope:rank_challenge_target = { has_trait = brave }
				}
				desc = murim_rank_challenge_loser_desc
				# Challenger loses
				scope:rank_challenger = {
					create_character_memory = {
						type = murim_ranking_duel_defeat_memory
					}
					add_character_modifier = {
						modifier = defeat_shame
						years = 2
					}
					add_character_modifier = {
						modifier = murim_recent_duel_cooldown
						years = 1
					}
					# Wound chance
					random = {
						chance = 30
						add_trait = wounded_1
					}
				}
				scope:rank_challenge_target = {
					create_character_memory = {
						type = murim_ranking_duel_victory_memory
					}
					add_prestige = 75
					add_character_modifier = {
						modifier = murim_recent_duel_cooldown
						years = 1
					}
				}
			}
		}
	}
}

######################################
# UPDATE RANK TIER MODIFIERS
# Adjusts modifiers to reflect current ranking tier
######################################

murim_update_rank_tier_modifiers_effect = {
	if = {
		limit = {
			has_variable = murim_martial_rank_position
			var:murim_martial_rank_position = 1
		}
		# The #1 - strongest under heaven
		if = {
			limit = { NOT = { has_character_modifier = strongest_under_heaven } }
			add_character_modifier = {
				modifier = strongest_under_heaven
				years = 1
			}
		}
	}
	else = {
		# Remove #1 modifier if no longer #1
		if = {
			limit = { has_character_modifier = strongest_under_heaven }
			remove_character_modifier = strongest_under_heaven
		}
	}
}
