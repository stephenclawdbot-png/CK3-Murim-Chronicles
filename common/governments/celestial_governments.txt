################################################################
# MURIM CHRONICLES -- GOVERNMENT TYPES
# Path: common/governments/celestial_governments.txt
# Defines three distinct governance structures for the
# martial world, each reflecting a different relationship
# between power and authority.
#
# 1. orthodox_sect_government  -- Righteous sects
# 2. unorthodox_sect_government -- Demonic/heterodox sects
# 3. free_cultivator_government -- Independent cultivators
#
# CK3 government_type syntax reference:
#   create_cadet_branches, valid_holdings, can_get_government
#   are top-level blocks; character_modifier goes inside
#   the government definition.
################################################################

orthodox_sect_government = {
	create_cadet_branches = yes

	valid_holdings = {
		castle
		city
		church
	}

	can_get_government = {
		OR = {
			has_character_flag = murim_orthodox_cultivator
			has_character_flag = murim_sect_master
			AND = {
				has_character_flag = murim_initiated
				NOT = { has_character_flag = murim_demonic_cultivator }
			}
		}
	}

	character_modifier = {
		# Sect master bonuses -- structured hierarchy
		monthly_prestige = 0.5
		monthly_piety = 0.3
		domain_limit = 2
		vassal_limit = 10
		learning = 2
		diplomacy = 1
		monthly_dynasty_prestige = 0.2
		# Righteous reputation
		dread_baseline_add = -5
	}

	ai_priority = {
		base = 50
		modifier = {
			add = 30
			has_character_flag = murim_orthodox_cultivator
		}
		modifier = {
			add = 20
			has_character_flag = murim_sect_master
		}
		modifier = {
			add = 10
			learning >= 12
		}
		modifier = {
			add = -50
			has_character_flag = murim_demonic_cultivator
		}
	}

	flag = government_is_sect
	flag = government_is_orthodox
}

unorthodox_sect_government = {
	create_cadet_branches = yes

	valid_holdings = {
		castle
		city
	}

	can_get_government = {
		OR = {
			has_character_flag = murim_demonic_cultivator
			AND = {
				has_character_flag = murim_initiated
				has_character_flag = murim_demonic_cultivator
			}
		}
	}

	character_modifier = {
		# Demonic sect bonuses -- fear-based rule
		monthly_prestige = 0.3
		monthly_piety = -0.2
		domain_limit = 3
		vassal_limit = 5
		intrigue = 3
		martial = 2
		# Rule through fear
		dread_baseline_add = 20
		dread_decay_mult = -0.25
		monthly_tyranny = -0.1
		# Demonic cultivation bonuses
		prowess = 3
	}

	ai_priority = {
		base = 30
		modifier = {
			add = 50
			has_character_flag = murim_demonic_cultivator
		}
		modifier = {
			add = 20
			intrigue >= 15
		}
		modifier = {
			add = 10
			has_trait = sadistic
		}
		modifier = {
			add = -60
			has_character_flag = murim_orthodox_cultivator
		}
	}

	flag = government_is_sect
	flag = government_is_unorthodox
}

free_cultivator_government = {
	create_cadet_branches = no

	valid_holdings = {
		castle
		city
		church
	}

	can_get_government = {
		has_character_flag = murim_initiated
		NOT = { has_character_flag = murim_sect_master }
	}

	character_modifier = {
		# Wandering cultivator -- personal freedom
		monthly_prestige = 0.2
		domain_limit = 1
		vassal_limit = 3
		# Freedom of movement and independence
		prowess = 2
		martial = 1
		intrigue = 1
		# Small bonuses for personal cultivation focus
		learning = 1
		health = 0.25
		stress_loss_mult = 0.10
	}

	ai_priority = {
		base = 20
		modifier = {
			add = 30
			has_character_flag = murim_initiated
			NOT = { has_character_flag = murim_sect_master }
		}
		modifier = {
			add = 15
			has_character_flag = murim_wandering_hero
		}
		modifier = {
			add = 10
			has_trait = content
		}
		modifier = {
			add = -20
			has_character_flag = murim_sect_master
		}
	}

	flag = government_is_cultivator
	flag = government_is_free
}
