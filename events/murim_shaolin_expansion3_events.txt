murim_shaolin_expansion3.0001 = {
	type = character_event
	title = murim_shaolin_expansion3.0001.t
	desc = murim_shaolin_expansion3.0001.desc
	theme = martial_arts
	
	# Forbidden Scripture Chamber — a young monk discovers the sealed vault beneath the pagoda
	# containing techniques banned by the First Abbot 300 years ago
	
	trigger = {
		has_trait = shaolin_disciple
		age >= 16
		age <= 30
		NOT = { has_character_flag = found_forbidden_chamber }
		is_ai = yes
		any_held_title = {
			tier = county
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = curious
		}
		modifier = {
			add = 3
			has_trait = ambitious
		}
		modifier = {
			add = -5
			has_trait = content
		}
	}
	
	immediate = {
		set_character_flag = found_forbidden_chamber
		add_character_modifier = {
			modifier = forbidden_knowledge_temptation
			years = 5
		}
	}
	
	# Option A: Enter the forbidden chamber and study the banned techniques
	option = {
		name = murim_shaolin_expansion3.0001.a
		add_trait = forbidden_arts_student
		add_prowess = 4
		add_piety = -200
		stress_impact = {
			zealous = major_stress_impact_gain
			compassionate = minor_stress_impact_gain
		}
		hidden_effect = {
			set_character_flag = chose_forbidden_path
			random_list = {
				60 = {
					trigger_event = {
						id = murim_shaolin_expansion3.0003
						days = { 30 90 }
					}
				}
				40 = {
					trigger_event = {
						id = murim_shaolin_expansion3.0004
						days = { 60 180 }
					}
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = ambitious
			}
		}
	}
	
	# Option B: Seal the chamber and report to the Abbot
	option = {
		name = murim_shaolin_expansion3.0001.b
		add_piety = 300
		add_prestige = 100
		add_opinion = {
			target = liege
			modifier = loyal_disciple_opinion
			opinion = 30
		}
		hidden_effect = {
			set_character_flag = reported_chamber
			trigger_event = {
				id = murim_shaolin_expansion3.0002
				days = { 10 30 }
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 30
				has_trait = zealous
			}
		}
	}
	
	# Option C: Tell no one, but seal it yourself — keep the secret
	option = {
		name = murim_shaolin_expansion3.0001.c
		add_intrigue = 3
		add_character_modifier = {
			modifier = keeper_of_secrets
			years = 10
		}
		stress_impact = {
			honest = major_stress_impact_gain
		}
		hidden_effect = {
			set_character_flag = secret_keeper_chamber
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 20
				has_trait = deceitful
			}
		}
	}
}
