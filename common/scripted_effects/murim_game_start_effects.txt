# CLAN SEEDING SUB-EFFECT
# Called by master seeding for each clan
##############################

murim_seed_clan_effect = {
	# Parameters: CLAN_TRAIT, ELDER_TRAIT, INNER_TRAIT, TECHNIQUE_1, TECHNIQUE_2, CULTURE_BIAS, ALIGNMENT, TARGET_COUNT
	# This is a parameterized effect -- each clan calls it with different values
	
	# Find eligible characters for this clan
	# In practice, each clan block above passes its own traits
	# The master seeding effect handles this through the random_list in Phase 1
	# This sub-effect ensures clan structure (elder/inner/outer hierarchy)
	
	# After Phase 1 random assignment, enforce hierarchy:
	# 1 elder (highest prowess in clan)
	# 3-4 inner disciples
	# Rest are outer disciples
	
	# Find the strongest clan member, make them elder
	random_living_character = {
		limit = {
			has_trait = $CLAN_TRAIT$
			is_alive = yes
			NOT = { has_trait = $ELDER_TRAIT$ }
		}
		weight = {
			base = 1
			modifier = { factor = 10 prowess >= 20 }
			modifier = { factor = 5 prowess >= 16 }
			modifier = { factor = 3 age >= 40 }
		}
		add_trait = $ELDER_TRAIT$
		add_trait = $TECHNIQUE_1$
		add_trait = $TECHNIQUE_2$
		add_prowess_skill = 8
	}
	
	# Find 3-4 strong members, make them inner disciples
	random_living_character = {
		limit = {
			has_trait = $CLAN_TRAIT$
			is_alive = yes
			NOT = { has_trait = $ELDER_TRAIT$ }
			NOT = { has_trait = $INNER_TRAIT$ }
		}
		count = 3
		weight = {
			base = 1
			modifier = { factor = 5 prowess >= 14 }
			modifier = { factor = 2 age >= 25 }
		}
		add_trait = $INNER_TRAIT$
		add_trait = $TECHNIQUE_1$
		add_prowess_skill = 4
	}
}
