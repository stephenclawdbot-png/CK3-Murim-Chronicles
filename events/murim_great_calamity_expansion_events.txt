### EVENT 0001 — Ashes of the Old World
# Fires after calamity kills 70-90% of martial world. Survivors assess devastation.
murim_great_calamity_expansion.0001 = {
	type = character_event
	title = murim_great_calamity_expansion.0001.t
	desc = murim_great_calamity_expansion.0001.desc
	theme = death
	
	left_portrait = {
		character = root
		animation = grief
	}
	
	trigger = {
		has_global_variable = murim_calamity_active
		is_alive = yes
		has_trait = murim_calamity_survivor
		NOT = { has_character_flag = witnessed_ashes_of_old_world }
	}
	
	immediate = {
		set_character_flag = witnessed_ashes_of_old_world
		add_stress = 50
		# Count surviving martial artists in the world
		every_ruler = {
			limit = {
				has_trait = murim_martial_artist
				is_alive = yes
			}
			add_to_temporary_list = surviving_martial_artists
		}
	}
	
	# Option A: Vow to rebuild the martial world
	option = {
		name = murim_great_calamity_expansion.0001.a
		add_trait = murim_rebuilder
		add_prestige = 500
		add_character_modifier = {
			modifier = murim_rebuilder_resolve
			years = 20
		}
		# Start rebuilding process
		set_character_flag = murim_rebuilding_vow
		trigger_event = {
			id = murim_great_calamity_expansion.0003
			days = { 60 180 }
		}
		ai_chance = {
			base = 60
			modifier = {
				add = 20
				has_trait = ambitious
			}
			modifier = {
				add = 20
				has_trait = compassionate
			}
		}
	}
	
	# Option B: Scavenge the ruins for lost techniques
	option = {
		name = murim_great_calamity_expansion.0001.b
		add_character_modifier = {
			modifier = murim_scavenger_of_ruins
			years = 10
		}
		random_list = {
			40 = {
				add_trait = murim_forbidden_knowledge
				custom_tooltip = murim_found_forbidden_scrolls_tt
			}
			30 = {
				add_martial_skill = 4
				custom_tooltip = murim_found_weapon_cache_tt
			}
			30 = {
				add_stress = 30
				custom_tooltip = murim_only_found_corpses_tt
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				add = 30
				has_trait = greedy
			}
		}
	}
	
	# Option C: Mourn in silence — the era is truly over
	option = {
		name = murim_great_calamity_expansion.0001.c
		add_piety = 300
		add_trait = murim_witness_of_ages
		add_character_modifier = {
			modifier = murim_deep_mourning
			years = 5
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 30
				has_trait = content
			}
		}
	}
}
