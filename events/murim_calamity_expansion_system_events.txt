# Demonic Breach — the barrier between realms tears
murim_demonic_invasion.0001 = {
	type = character_event
	title = murim_demonic_invasion.0001.t
	desc = murim_demonic_invasion.0001.desc
	theme = skull
	left_portrait = { character = root animation = personality_irrational }

	trigger = {
		has_variable = murim_martial_artist
		has_global_variable = murim_great_calamity_active
		NOT = { has_character_flag = murim_witnessed_demonic_breach }
	}

	immediate = {
		set_character_flag = { flag = murim_witnessed_demonic_breach years = 50 }
	}

	# Rally all sects — orthodox and unorthodox must unite
	option = {
		name = murim_demonic_invasion.0001.a
		add_diplomacy = 2
		add_prestige = 200
		add_character_modifier = { modifier = murim_demon_invasion_rally years = 5 }
		hidden_effect = {
			trigger_event = { id = murim_demonic_invasion.0010 days = 30 }
		}
		ai_chance = {
			base = 45
			modifier = { add = 25 has_trait = compassionate }
			modifier = { add = 15 diplomacy >= 14 }
		}
	}

	# Exploit the chaos — demonic energy can fuel forbidden techniques
	option = {
		name = murim_demonic_invasion.0001.b
		trigger = { murim_has_any_corruption_trigger = yes }
		add_prowess = 3
		change_variable = { name = murim_cultivation_xp add = 80 }
		hidden_effect = { murim_advance_corruption_effect = yes }
		ai_chance = {
			base = 25
			modifier = { add = 30 has_trait = ambitious }
			modifier = { add = 20 murim_has_any_forbidden_technique_trigger = yes }
		}
	}

	# Flee to a sealed sanctuary
	option = {
		name = murim_demonic_invasion.0001.c
		add_stress = -20
		add_prestige = -100
		ai_chance = {
			base = 15
			modifier = { add = 25 has_trait = craven }
		}
	}

	# Stand alone — face the demons head-on
	option = {
		name = murim_demonic_invasion.0001.d
		add_prowess = 2
		add_dread = 20
		add_stress = 25
		hidden_effect = {
			trigger_event = { id = murim_demonic_invasion.0020 days = 14 }
		}
		ai_chance = {
			base = 15
			modifier = { add = 30 has_trait = brave }
			modifier = { add = 20 prowess >= 22 }
		}
	}
}
