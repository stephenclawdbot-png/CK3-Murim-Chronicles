# POISON FLOWER PALACE — Full Clan Event Chain
# An all-female sect renowned for poison arts, seduction techniques, and flower-based martial arts.
# Unorthodox faction. Located in a hidden valley filled with lethal exotic flowers.
# Known for their beauty, ruthlessness, and mastery of the Hundred Poison Body.

# ============================================
# EVENT 0001 — The Flower Maiden's Trial
# New disciples must survive a garden of lethal flowers to prove poison resistance
# ============================================
murim_poison_flower_palace.0001 = {
	type = character_event
	title = murim_poison_flower_palace.0001.t
	desc = murim_poison_flower_palace.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = scope:palace_mistress
		animation = personality_compassionate
	}
	
	trigger = {
		is_alive = yes
		is_female = yes
		has_trait = murim_martial_artist
		NOT = { has_trait = poison_flower_initiate }
		NOT = { has_character_flag = flower_trial_done }
	}
	
	immediate = {
		random_relation = {
			type = mentor
			limit = { has_trait = poison_flower_palace_member }
			save_scope_as = palace_mistress
		}
		set_character_flag = {
			flag = flower_trial_done
			days = 3650
		}
	}
	
	# Option A: Walk through the Garden of a Thousand Poisons
	option = {
		name = murim_poison_flower_palace.0001.a
		random_list = {
			70 = {
				add_trait = poison_flower_initiate
				add_character_modifier = {
					modifier = poison_resistance_awakened_modifier
					days = 730
				}
				add_prowess = 2
				add_intrigue = 3
				trigger_event = {
					id = murim_poison_flower_palace.0002
					days = { 60 120 }
				}
			}
			30 = {
				add_trait = poison_scarred
				add_character_modifier = {
					modifier = flower_poison_aftermath_modifier
					days = 365
				}
				add_health = -1
			}
		}
	}
	
	# Option B: Build resistance gradually with microdosing
	option = {
		name = murim_poison_flower_palace.0001.b
		add_trait = poison_flower_initiate
		add_character_modifier = {
			modifier = slow_poison_resistance_modifier
			days = 1095
		}
		add_intrigue = 2
		trigger_event = {
			id = murim_poison_flower_palace.0002
			days = { 180 365 }
		}
	}
	
	# Option C: Refuse the trial — poisons are beneath you
	option = {
		name = murim_poison_flower_palace.0001.c
		add_prestige = -200
		add_opinion = {
			target = scope:palace_mistress
			modifier = disappointed_mistress_opinion
			opinion = -30
		}
		add_character_flag = rejected_poison_path
	}
}
