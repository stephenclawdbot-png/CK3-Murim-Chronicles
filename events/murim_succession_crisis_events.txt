###############################################
# murim_succession.0001 - The Sect Master Falls
# Current sect master dies or becomes incapable, triggering succession crisis
###############################################
murim_succession.0001 = {
	type = character_event
	title = murim_succession.0001.t
	desc = murim_succession.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = grief
	}
	
	trigger = {
		is_alive = yes
		has_variable = murim_sect_member
		NOT = { has_character_flag = succession_crisis_active }
		OR = {
			has_variable = murim_sect_elder
			has_variable = murim_sect_inner_disciple
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 3
			has_variable = murim_sect_elder
		}
		modifier = {
			add = 2
			prowess >= 20
		}
	}
	
	immediate = {
		add_character_flag = {
			flag = succession_crisis_active
			years = 3
		}
		random_living_character = {
			limit = {
				has_variable = murim_sect_member
				NOT = { this = root }
				prowess >= 15
			}
			weight = {
				base = 1
				modifier = {
					add = 5
					has_variable = murim_sect_elder
				}
			}
			save_scope_as = succession_rival
		}
	}
	
	# Option A: Claim leadership through martial might
	option = {
		name = murim_succession.0001.a
		duel = {
			skill = prowess
			target = scope:succession_rival
			50 = {
				desc = murim_succession.0001.a.victory
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_succession.0001.a.victory.toast
					left_icon = root
					root = {
						add_prestige = 500
						set_variable = {
							name = murim_sect_leader
							value = yes
						}
						add_character_modifier = {
							modifier = new_sect_master
							years = 5
						}
						create_character_memory = {
							type = claimed_sect_leadership_by_combat_memory
							participants = {
								defeated_rival = scope:succession_rival
							}
						}
						random_list = {
							30 = { add_trait = wounded }
							70 = { }
						}
						# Chain: sect politics after new leader
						trigger_event = {
							id = murim_sect.0001
							days = { 14 30 }
						}
						# Chain: ranking acknowledges new sect master
						trigger_event = {
							id = murim_ranking.0001
							days = { 3 7 }
						}
					}
					scope:succession_rival = {
						add_prestige = -300
						create_character_memory = {
							type = lost_succession_duel_memory
							participants = {
								new_sect_master = root
							}
						}
						random_list = {
							40 = { add_trait = wounded }
							20 = { add_trait = incapable }
							40 = { }
						}
						# Rival may plot revenge
						random_list = {
							40 = {
								trigger_event = {
									id = murim_jianghu.0005
									days = { 90 180 }
								}
							}
							60 = { }
						}
					}
				}
			}
			50 = {
				desc = murim_succession.0001.a.defeat
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_succession.0001.a.defeat.toast
					left_icon = root
					root = {
						add_prestige = -300
						create_character_memory = {
							type = lost_succession_duel_memory
							participants = {
								new_sect_master = scope:succession_rival
							}
						}
						random_list = {
							40 = { add_trait = wounded }
							25 = { add_trait = incapable }
							35 = { }
						}
						# Chain: expelled or leave sect
						trigger_event = {
							id = murim_sect.0003
							days = { 7 14 }
						}
					}
					scope:succession_rival = {
						set_variable = {
							name = murim_sect_leader
							value = yes
						}
					}
				}
			}
		}
		ai_chance = {
			base = 40
			modifier = {
				add = 30
				has_trait = ambitious
			}
		}
	}
	
	# Option B: Rally the elders through diplomacy
	option = {
		name = murim_succession.0001.b
		custom_tooltip = murim_succession.0001.b.tt
		create_character_memory = {
			type = diplomatic_succession_bid_memory
			participants = {
				succession_opponent = scope:succession_rival
			}
		}
		random_list = {
			40 = {
				# Elders support you
				set_variable = {
					name = murim_sect_leader

				}

					modifier = elder_backed_sect_master
					years = 5
				}

				trigger_event = {

			}

				trigger_event = {
					id = murim_sect.0003
					days = { 14 30 }
				}
				# Chain: sect war
				trigger_event = {
					id = murim_faction_war.0001
					days = { 30 60 }
				}
			}
			30 = {
				# Elders choose the rival
				add_prestige = -200
				scope:succession_rival = {
					set_variable = {
						name = murim_sect_leader
						value = yes
					}

				trigger_event = {
					id = murim_daily_life.0001
					days = { 7 14 }
				}
			}
		}
		ai_chance = {
			base = 35
			modifier = {
				add = 25
				diplomacy >= 15
			}
		}
	}
	
	# Option C: Leave the sect and forge your own path
	option = {
		name = murim_succession.0001.c
		remove_variable = murim_sect_member
		remove_variable = murim_sect_elder
		add_prestige = 100
		create_character_memory = {
			type = left_sect_during_succession_memory
		}

		trigger_event = {
			id = murim_sect.0001
			days = { 30 60 }
		}

		trigger_event = {
			id = murim_daily_life.0001

			gregarious = medium_stress_impact_gain

			base = 25
		}
