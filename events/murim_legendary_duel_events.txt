###############################################
# murim_legendary_duel.0001 - Challenge of the Heavenly Sword
# A top-10 ranked martial artist challenges the player for their position
###############################################
murim_legendary_duel.0001 = {
	type = character_event
	title = murim_legendary_duel.0001.t
	desc = murim_legendary_duel.0001.desc
	theme = martial_arts
	
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:legendary_challenger
		animation = personality_vengeful
	}
	
	trigger = {
		is_alive = yes
		has_variable = murim_ranking_position
		var:murim_ranking_position <= 10
		NOT = { has_character_flag = legendary_duel_cooldown }
		any_living_character = {
			has_variable = murim_ranking_position
			var:murim_ranking_position <= 20
			NOT = { this = root }
			prowess >= 20
		}
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 2
			has_trait = martial_arts_master
		}
		modifier = {
			add = 1
			var:murim_ranking_position <= 5
		}
	}
	
	immediate = {
		random_living_character = {
			limit = {
				has_variable = murim_ranking_position
				var:murim_ranking_position <= 20
				NOT = { this = root }
				prowess >= 20
			}
			weight = {
				base = 1
				modifier = {
					add = 3
					prowess >= root.prowess
				}
			}
			save_scope_as = legendary_challenger
		}
		add_character_flag = {
			flag = legendary_duel_cooldown
			years = 3
		}
	}
	
	# Option A: Accept the challenge with full power
	option = {
		name = murim_legendary_duel.0001.a
		duel = {
			skill = prowess
			target = scope:legendary_challenger
			50 = {
				desc = murim_legendary_duel.0001.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_legendary_duel.0001.a.success.toast
					left_icon = root
					root = {
						add_prestige = 500
						create_character_memory = {
							type = legendary_duel_victory_memory
							participants = {
								defeated_legend = scope:legendary_challenger
							}
						}
						if = {
							limit = { has_trait = wounded }
							remove_trait = wounded
						}
						add_trait_xp = {
							trait = martial_arts_master
							tracks = martial_arts_track
							value = 30
						}
						# Chain to ranking update
						trigger_event = {
							id = murim_ranking.0001
							days = { 3 7 }
						}
					}
					scope:legendary_challenger = {
						add_prestige = -300
						create_character_memory = {
							type = legendary_duel_defeat_memory
							participants = {
								victorious_legend = root
							}
						}
						random_list = {
							30 = { add_trait = wounded }
							20 = { add_trait = incapable }
							50 = { }
						}
					}
				}
			}
			50 = {
				desc = murim_legendary_duel.0001.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				send_interface_toast = {
					title = murim_legendary_duel.0001.a.failure.toast
					left_icon = root
					root = {
						add_prestige = -300
						create_character_memory = {
							type = legendary_duel_defeat_memory
							participants = {
								victorious_legend = scope:legendary_challenger
							}
						}
						random_list = {
							40 = { add_trait = wounded }
							20 = { add_trait = incapable }
							10 = { add_trait = one_eyed }
							30 = { }
						}
						# Chain to ranker death if incapable
						if = {
							limit = { has_trait = incapable }
							trigger_event = {
								id = murim_ranker_death.0001
								days = { 10 30 }
							}
						}
					}
					scope:legendary_challenger = {
						add_prestige = 500
						trigger_event = {
							id = murim_ranking.0001
							days = { 3 7 }
						}
					}
				}
			}
		}
		ai_chance = {
			base = 70
			modifier = {
				add = 30
				has_trait = brave
			}
		}
	}
	
	# Option B: Propose a conditional duel with artifact stakes
	option = {
		name = murim_legendary_duel.0001.b
		custom_tooltip = murim_legendary_duel.0001.b.tt
		create_character_memory = {
			type = artifact_stake_duel_memory
			participants = {
				duel_opponent = scope:legendary_challenger
			}
		}
		trigger_event = {
			id = murim_artifact.0001
			days = { 5 15 }
		}
		trigger_event = {
			id = murim_legendary_duel.0002
			days = { 30 60 }
		}
		stress_impact = {
			ambitious = medium_stress_loss
			greedy = medium_stress_loss
		}
		ai_chance = {
			base = 20
			modifier = {
				add = 20
				has_trait = greedy
			}
		}
	}
	
	# Option C: Refuse and lose face
	option = {
		name = murim_legendary_duel.0001.c
		add_prestige = -500
		add_character_modifier = {
			modifier = coward_of_the_martial_world
			years = 5
		}
		create_character_memory = {
			type = refused_legendary_duel_memory
			participants = {
				challenger = scope:legendary_challenger
			}
		}
		random_list = {
			40 = { add_trait = craven }
			60 = { }
		}
		stress_impact = {
			brave = major_stress_impact_gain
			ambitious = major_stress_impact_gain
		}
		ai_chance = {
			base = 10
			modifier = {
				add = 30
				has_trait = craven
			}
		}
	}
}
