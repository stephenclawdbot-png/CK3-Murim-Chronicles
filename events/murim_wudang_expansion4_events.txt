murim_wudang_expansion4.0001 = {
	type = character_event
	title = murim_wudang_exp4.0001.t
	desc = murim_wudang_exp4.0001.desc
	theme = faith
	left_portrait = { character = root animation = thinking }

	trigger = {
		has_trait = murim_clan_wudang
		has_trait = murim_cultivation_realm_5
		age >= 60
		NOT = { has_character_flag = wudang_immortal_path_offered }
	}

	weight_multiplier = {
		base = 10
		modifier = { add = 20 has_trait = patient }
		modifier = { add = 15 learning >= 20 }
	}

	immediate = {
		set_character_flag = wudang_immortal_path_offered
		random_vassal = {
			limit = { has_trait = murim_clan_wudang age >= 70 learning >= 15 }
			save_scope_as = wudang_immortal_elder
		}
	}

	option = {
		name = murim_wudang_exp4.0001.a
		add_character_modifier = { modifier = murim_seeking_immortality years = 30 }
		add_piety = 1000
		add_health = 2
		trigger_event = { id = murim_wudang_expansion4.0002 days = { 180 365 } }
	}
	option = {
		name = murim_wudang_exp4.0001.b
		add_character_modifier = { modifier = murim_sword_god_path years = 50 }
		add_prowess = 10
		add_martial = 5
		trigger_event = { id = murim_wudang_expansion4.0003 days = { 90 180 } }
	}
	option = {
		name = murim_wudang_exp4.0001.c
		add_character_modifier = { modifier = murim_wudang_worldly_sage years = 20 }
		add_diplomacy = 5
		add_prestige = 500
	}
}

murim_wudang_expansion4.0003 = {
	type = character_event
	title = murim_wudang_exp4.0003.t
	desc = murim_wudang_exp4.0003.desc
	theme = martial
	left_portrait = { character = root animation = war_over_win }

	trigger = {
		has_character_modifier = murim_sword_god_path
		has_trait = murim_clan_wudang
	}

	immediate = {
		random_living_character = {
			limit = { murim_is_martial_artist_trigger = yes prowess >= 30 NOT = { this = root } }

		}
	}

	option = {
		name = murim_wudang_exp4.0003.a
		trigger = { exists = scope:sword_rival }
		duel = {
			skill = prowess
			target = scope:sword_rival
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -49 }
				desc = murim_wudang_exp4.0003.a.win
				root = {
					add_trait = murim_sword_god
					add_prowess = 15
					add_prestige = 3000
					add_character_modifier = { modifier = murim_sword_god_aura years = 100 }
				}
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -49 }
				desc = murim_wudang_exp4.0003.a.lose
				root = { add_trait = wounded_3 add_prestige = -500 }
			}
		}
	}
	option = {
		name = murim_wudang_exp4.0003.b
		add_prowess = 8
		add_character_modifier = { modifier = murim_sword_hermit years = 30 }
		trigger_event = { id = murim_wudang_expansion4.0005 days = { 180 365 } }
	}
	option = {
		name = murim_wudang_exp4.0003.c
		add_prowess = 5
		add_martial = 3
		add_character_modifier = { modifier = murim_teaching_sword_master years = 20 }

			limit = { has_trait = murim_clan_wudang }
			add_prowess = 3
			add_opinion = { modifier = murim_grateful_student target = root opinion = 25 }
		}
	}
}

		name = murim_wudang_exp4.0010.a

			list = wudang_disciples

		add_martial = 5
