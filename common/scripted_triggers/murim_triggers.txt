		has_trait = qi_condensation_realm
		has_trait = foundation_realm
		has_trait = core_formation_realm

		has_trait = murim_foundation
		has_trait = murim_golden_core
		has_trait = murim_nascent_soul

		has_trait = murim_heavenly_tribulation

# COMBAT & CHALLENGE ELIGIBILITY
######################################

murim_can_challenge_for_rank_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	NOT = { has_trait = craven }
	murim_is_martial_artist_trigger = yes
	NOT = {
		has_character_modifier = murim_recent_duel_cooldown
	}
	prowess >= 10
}

murim_can_be_challenged_for_rank_trigger = {
	is_alive = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	murim_is_top_50_martial_artist_trigger = yes
	NOT = {
		has_character_modifier = murim_recent_duel_cooldown
	}
}

murim_ai_should_challenge_for_rank_trigger = {
	is_ai = yes
	murim_can_challenge_for_rank_trigger = yes
	prowess >= 15
	OR = {
		has_trait = ambitious
		has_trait = brave
		has_trait = wrathful
	}
}

murim_has_high_aptitude_trigger = {
	OR = {
		has_trait = genius
		has_trait = intelligent
		has_trait = quick
		has_trait = cultivation_prodigy
		learning >= 15
	}
}

murim_has_low_aptitude_trigger = {
	OR = {
		has_trait = dull
		has_trait = slow
		has_trait = stupid
		learning < 5
	}
}
