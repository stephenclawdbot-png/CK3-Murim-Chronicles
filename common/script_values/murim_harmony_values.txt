# Murim Chronicles - Sect Harmony Values
# Tracks internal stability of each sect (0-100 scale)
# INTEGRATION: Used by murim_clan_triggers.txt (harmony_is_high/low),
#              murim_sect_politics_events.txt (golden age/discord triggers),
#              murim_clan_membership_effects.txt (betrayal modifies harmony),
#              murim_game_start_effects.txt (initial harmony set to 60-80)
# Cross-refs: murim_seeding_values.txt (power score includes clan prestige),
#             murim_on_actions.txt (yearly harmony recalculation)
# RULE: Harmony affects training speed, loyalty, recruitment, vulnerability.
#       High harmony (>=70) = bonuses. Low harmony (<=30) = penalties.

# ═══════════════════════════════════════════
#  HARMONY SCORE CALCULATION (recalculated yearly)
#  Base: 50 (neutral). Range: 0-100.
# ═══════════════════════════════════════════
murim_calculate_sect_harmony_value = {
	value = 50 # Base neutral

	# ── POSITIVE FACTORS ──

	# Stable leadership (leader alive and in position 5+ years)
	if = {
		limit = {
			any_living_character = {
				murim_is_sect_leader_trigger = yes
				has_character_flag = sect_leader_tenure_5yr
			}
		}
		add = 10
	}

	# Leader has high diplomacy (charismatic leader)
	if = {
		limit = {
			any_living_character = {
				murim_is_sect_leader_trigger = yes
				diplomacy >= 15
			}
		}
		add = 5
	}

	# No betrayals in last 5 years
	if = {
		limit = {
			NOT = { has_global_variable = sect_recent_betrayal }
		}
		add = 8
	}

	# Won recent tournament (sect member placed in top 3 grand tournament)
	if = {
		limit = {
			has_global_variable = sect_tournament_glory
		}
		add = 7
	}

	# Multiple elders alive (strong advisory council: 3+ elders)
	if = {
		limit = {
			any_living_character = {
				count >= 3
				murim_is_sect_elder_trigger = yes
			}
		}
		add = 5
	}

	# Growing membership (more outer disciples than 2 years ago)
	if = {
		limit = {
			has_global_variable = sect_membership_growing
		}
		add = 5
	}

	# Successful missions / sect wars won
	if = {
		limit = {
			has_global_variable = sect_recent_victory
		}
		add = 8
	}

	# ── NEGATIVE FACTORS ──

	# No leader (succession crisis)
	if = {
		limit = {
			NOT = {
				any_living_character = {
					murim_is_sect_leader_trigger = yes
				}
			}
		}
		add = -20
	}

	# Recent betrayal (within 3 years)
	if = {
		limit = {
			has_global_variable = sect_recent_betrayal
		}
		add = -12
	}

	# Multiple betrayals (2+ in 5 years)
	if = {
		limit = {
			has_global_variable = sect_multiple_betrayals
		}
		add = -8 # Stacks with single betrayal
	}

	# Failed succession (contested or violent)
	if = {
		limit = {
			has_global_variable = sect_contested_succession
		}
		add = -15
	}

	# Lost sect war
	if = {
		limit = {
			has_global_variable = sect_recent_defeat
		}
		add = -10
	}

	# Members dying in large numbers (3+ sect members died this year)
	if = {
		limit = {
			has_global_variable = sect_heavy_losses
		}
		add = -12
	}

	# Leader is cruel or tyrannical
	if = {
		limit = {
			any_living_character = {
				murim_is_sect_leader_trigger = yes
				OR = {
					has_trait = cruel
					has_trait = tyrannical
					has_trait = callous
				}
			}
		}
		add = -8
	}

	# Internal faction split (two or more power blocs)
	if = {
		limit = {
			has_global_variable = sect_faction_split
		}
		add = -10
	}

	# Clamp to 0-100
	min = 0
	max = 100
}

# ═══════════════════════════════════════════
#  HARMONY-BASED MODIFIERS (applied to sect members)
# ═══════════════════════════════════════════

# Training speed bonus at high harmony
murim_harmony_training_bonus_value = {
	value = 0
	if = {
		limit = { scope:sect_harmony_score >= 70 }
		add = 2 # +2 prowess gain rate from training events
	}
	if = {
		limit = { scope:sect_harmony_score >= 85 }
		add = 1 # Additional +1 at very high harmony (total +3)
	}
}

# Loyalty modifier (affects betrayal chance)
murim_harmony_loyalty_opinion = {
	value = 0
	# High harmony reduces betrayal chance
	if = {
		limit = { scope:sect_harmony_score >= 70 }
		add = -2 # -2% to betrayal chance (base 3%, so now 1%)
	}
	# Low harmony increases betrayal chance
	if = {
		limit = { scope:sect_harmony_score <= 30 }
		add = 3 # +3% to betrayal chance (base 3%, so now 6%)
	}
	if = {
		limit = { scope:sect_harmony_score <= 15 }
		add = 4 # +4% additional (total +7%, base 3% = 10% betrayal!)
	}
}

# Recruitment modifier (affects how easily sect attracts new members)
murim_harmony_recruitment_opinion = {
	value = 0
	# Golden age: everyone wants to join
	if = {
		limit = { scope:sect_harmony_score >= 80 }
		add = 30 # +30% recruitment chance
	}
	if = {
		limit = { scope:sect_harmony_score >= 60 }
		add = 10 # +10% recruitment
	}
	# Discord: nobody wants to join a mess
	if = {
		limit = { scope:sect_harmony_score <= 30 }
		add = -20 # -20% recruitment
	}
	if = {
		limit = { scope:sect_harmony_score <= 15 }
		add = -30 # Additional -30% (total -50% recruitment!)
	}
}

# Defection chance (members leaving to join other sects)
murim_harmony_defection_chance_value = {
	value = 2 # Base 2% yearly for outer/associate members
	# Low harmony = more defection
	if = {
		limit = { scope:sect_harmony_score <= 30 }
		add = 5 # 7% defection at low harmony
	}
	if = {
		limit = { scope:sect_harmony_score <= 15 }
		add = 8 # 15% defection at critical harmony!
	}
	# High harmony = almost no defection
	if = {
		limit = { scope:sect_harmony_score >= 70 }
		add = -1 # 1% defection
	}
	if = {
		limit = { scope:sect_harmony_score >= 85 }
		add = -1 # 0% defection at very high harmony (clamped)
	}
	min = 0
}

# Vulnerability to external attack (rival sects or players)
murim_harmony_vulnerability_value = {
	value = 0
	# Low harmony = vulnerable to sect wars
	if = {
		limit = { scope:sect_harmony_score <= 30 }
		add = 20 # +20% chance rival sects launch attack
	}
	if = {
		limit = { scope:sect_harmony_score <= 15 }
		add = 30 # +30% additional (total +50%)
	}
	# High harmony = deterrent
	if = {
		limit = { scope:sect_harmony_score >= 70 }
		add = -15 # -15% attack chance (rivals know better)
	}
}

# ═══════════════════════════════════════════
#  HARMONY RECOVERY / DECAY RATES (yearly)
# ═══════════════════════════════════════════

murim_harmony_yearly_drift_value = {
	value = 0
	# Natural drift toward 50 (regression to mean)
	if = {
		limit = { scope:sect_harmony_score > 50 }
		add = -2 # Slowly decays without active maintenance
	}
	if = {
		limit = { scope:sect_harmony_score < 50 }
		add = 3 # Recovers slightly faster than it decays (hope)
	}
	# Shaolin special: Buddhist discipline resists decay
	if = {
		limit = {
			scope:sect_type = flag:shaolin
			scope:sect_harmony_score > 50
		}
		add = 1 # Partially offsets decay
	}
	# Heavenly Demon special: power struggles accelerate decay
	if = {
		limit = {
			scope:sect_type = flag:heavenly_demon
			scope:sect_harmony_score > 50
		}
		add = -1 # Additional decay for demon sect
	}
}
