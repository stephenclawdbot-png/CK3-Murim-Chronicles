### EVENT 1: The Supreme Path - Declaring Yourself Grandmaster ###
murim_custom_clan_exp3.0001 = {
	type = character_event
	title = murim_custom_clan_exp3.0001.t
	desc = murim_custom_clan_exp3.0001.desc
	theme = martial_arts
	
	trigger = {
		has_character_flag = custom_clan_founded
		has_character_flag = supreme_path_chosen
		NOT = { has_character_flag = grandmaster_declared }
		martial >= 20
		prowess >= 18
	}
	
	weight_multiplier = {
		base = 2
		modifier = { add = 3 has_trait = ambitious }
		modifier = { add = 2 prowess >= 22 }
	}
	
	# Option A: Declare supreme authority - all must obey
	option = {
		name = murim_custom_clan_exp3.0001.a
		add_character_flag = { flag = grandmaster_declared days = -1 }
		add_trait = supreme_grandmaster
		add_prestige = 1500
		add_dread = 50
		add_martial = 3
		hidden_effect = {
			set_variable = { name = clan_obedience value = 80 }
			set_variable = { name = clan_fear value = 60 }
		}
		ai_chance = { base = 50 }
	}
	
	# Option B: Earn respect through demonstration of power
	option = {
		name = murim_custom_clan_exp3.0001.b
		add_character_flag = { flag = grandmaster_declared days = -1 }
		add_trait = proven_grandmaster
		add_prestige = 1000
		add_prowess = 5
		hidden_effect = {
			set_variable = { name = clan_obedience value = 90 }
			set_variable = { name = clan_fear value = 30 }
		}
		ai_chance = { base = 35 }
	}
	
	# Option C: Rule through wisdom, not force
	option = {
		name = murim_custom_clan_exp3.0001.c
		trigger = { learning >= 18 }
		add_character_flag = { flag = grandmaster_declared days = -1 }
		add_trait = scholarly_grandmaster

murim_custom_clan_exp3.0003 = {
	type = character_event
	title = murim_custom_clan_exp3.0003.t
	desc = murim_custom_clan_exp3.0003.desc
	theme = martial_arts
	
	trigger = {
		has_character_flag = grandmaster_declared
		has_character_flag = custom_clan_founded
		NOT = { has_character_flag = first_disciple_tested }
	}
	
	immediate = {
		random_courtier = {
			limit = { age >= 16 age <= 30 martial >= 8 }
			save_scope_as = disciple_candidate
		}
	}
	
	# Option A: Brutal trial by combat
	option = {
		name = murim_custom_clan_exp3.0003.a
		add_character_flag = { flag = first_disciple_tested days = -1 }
		duel = {
			skill = prowess
			target = scope:disciple_candidate
			70 = {
				desc = murim_custom_clan_exp3.0003.a.success
				scope:disciple_candidate = {
					add_trait = custom_clan_disciple
					add_martial = 2
					add_prowess = 3
				}
				root = { add_prestige = 300 }
			}
			30 = {
				desc = murim_custom_clan_exp3.0003.a.failure

					increase_wounds_effect = { reason = training }
