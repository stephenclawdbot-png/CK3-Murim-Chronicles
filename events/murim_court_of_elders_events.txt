# EVENT 0001 - Elder Council Convenes
# Triggered: on_action yearly pulse, sect leaders
# AI: Full council governance with all elders voting
murim_elders.0001 = {
    type = character_event
    title = murim_elders.0001.t
    desc = murim_elders.0001.desc
    theme = diplomacy
    
    left_portrait = {
        character = root
        animation = personality_diplomacy
    }
    right_portrait = {
        character = scope:senior_elder
        animation = thinking
    }
    
    trigger = {
        is_alive = yes
        has_trait = murim_sect_leader
        NOT = { has_character_flag = elder_council_recently }
        any_courtier = {
            is_alive = yes
            age >= 40
            OR = {
                murim_is_martial_artist_trigger = yes
                has_trait = murim_righteous_elder
                has_trait = murim_unorthodox_elder
            }
        }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 2.0
            has_global_variable = murim_great_calamity_approaching
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = elder_council_recently
            days = 365
        }
        ordered_courtier = {
            limit = {
                is_alive = yes
                age >= 40
                OR = {
                    murim_is_martial_artist_trigger = yes
                    has_trait = murim_righteous_elder
                }
            }
            order_by = age
            max = 1
            save_scope_as = senior_elder
        }
        # Determine council topic
        random_list = {
            20 = { set_variable = { name = council_topic value = 1 } } # Sect policy reform
            20 = { set_variable = { name = council_topic value = 2 } } # Alliance negotiations
            20 = { set_variable = { name = council_topic value = 3 } } # War declaration
            20 = { set_variable = { name = council_topic value = 4 } } # Succession planning
            20 = { set_variable = { name = council_topic value = 5 } } # Technique teaching rights
        }
    }
    
    # Option A: Follow the elders' consensus
    option = {
        name = murim_elders.0001.a
        add_prestige = 100
        add_piety = 50
        add_character_modifier = {
            modifier = murim_elder_approved_leader
            years = 3
        }
        every_courtier = {
            limit = {
                is_alive = yes
                age >= 40
            }
            add_opinion = {
                target = root
                modifier = murim_respects_elder_council
                opinion = 10
            }
        }
        ai_chance = {
            base = 40
            modifier = {
                factor = 2.0
                has_trait = patient
            }
            modifier = {
                factor = 2.0
                has_trait = just
            }
            modifier = {
                factor = 0.5
                has_trait = ambitious
            }
        }
    }
    
    # Option B: Override the elders with your authority
    option = {
        name = murim_elders.0001.b
        add_prestige = 50
        add_dread = 10
        add_character_modifier = {
            modifier = murim_assertive_leader
            years = 3
        }
        every_courtier = {
            limit = {
                is_alive = yes
                age >= 40
            }
            add_opinion = {
                target = root
                modifier = murim_overruled_elders
                opinion = -15
            }
        }
        # Younger members approve
        every_courtier = {
            limit = {
                is_alive = yes
                age < 35
            }
            add_opinion = {
                target = root
                modifier = murim_strong_leadership
                opinion = 10
            }
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                has_trait = ambitious
            }
            modifier = {
                factor = 2.0
                has_trait = brave
            }
        }
    }
    
    # Option C: Propose a compromise
    option = {
        name = murim_elders.0001.c
        trigger = {
            diplomacy >= 14
        }
        add_prestige = 150
        add_diplomacy_skill = 1
        add_character_modifier = {
            modifier = murim_diplomatic_leader
            years = 3
        }
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                diplomacy >= 18
            }
            modifier = {
                factor = 1.5
                has_trait = gregarious
            }
        }
    }
}

# EVENT 0002 - Succession Crisis: Who Leads Next?
# Triggered: when sect leader is old or dying
# AI: Full succession vote system
murim_elders.0002 = {
    type = character_event
    title = murim_elders.0002.t
    desc = murim_elders.0002.desc
    theme = diplomacy
    
    left_portrait = {
        character = root
        animation = worry
    }
    right_portrait = {
        character = scope:succession_candidate_a
        animation = personality_bold
    }
    
    trigger = {
        is_alive = yes
        has_trait = murim_sect_leader
        OR = {
            age >= 60
            health <= 2
            has_trait = ill
        }
        NOT = { has_character_flag = succession_planned }
    }
    
    weight_multiplier = {
        base = 1
        modifier = {
            factor = 3.0
            age >= 70
        }
        modifier = {
            factor = 2.0
            health <= 1
        }
    }
    
    immediate = {
        add_character_flag = {
            flag = succession_planned
            days = 3650
        }
        # Find top two candidates
        ordered_courtier = {
            limit = {
                is_alive = yes
                age >= 25
                age <= 55
                prowess >= 12
            }
            order_by = prowess
            max = 1
            save_scope_as = succession_candidate_a
        }
        ordered_courtier = {
            limit = {
                is_alive = yes
                age >= 25
                age <= 55
                diplomacy >= 12
                NOT = { this = scope:succession_candidate_a }
            }
            order_by = diplomacy
            max = 1
            save_scope_as = succession_candidate_b
        }
    }
    
    # Option A: Name the strongest martial artist
    option = {
        name = murim_elders.0002.a
        trigger = {
            exists = scope:succession_candidate_a
        }
        scope:succession_candidate_a = {
            add_character_modifier = {
                modifier = murim_designated_successor
                years = 10
            }
            add_prestige = 300
            add_opinion = {
                target = root
                modifier = murim_chose_me_as_successor
                opinion = 40
            }
        }
        if = {
            limit = { exists = scope:succession_candidate_b }
            scope:succession_candidate_b = {
                add_opinion = {
                    target = root
                    modifier = murim_passed_over_for_succession
                    opinion = -30
                }
            }
        }
        add_prestige = 100
        ai_chance = {
            base = 35
            modifier = {
                factor = 2.0
                martial >= 14
            }
        }
    }
    
    # Option B: Name the wisest diplomat
    option = {
        name = murim_elders.0002.b
        trigger = {
            exists = scope:succession_candidate_b
        }
        scope:succession_candidate_b = {
            add_character_modifier = {
                modifier = murim_designated_successor
                years = 10
            }
            add_prestige = 300
            add_opinion = {
                target = root
                modifier = murim_chose_me_as_successor
                opinion = 40
            }
        }
        if = {
            limit = { exists = scope:succession_candidate_a }
            scope:succession_candidate_a = {
                add_opinion = {
                    target = root
                    modifier = murim_passed_over_for_succession
                    opinion = -30
                }
            }
        }
        add_prestige = 100
        ai_chance = {
            base = 30
            modifier = {
                factor = 2.0
                diplomacy >= 16
            }
        }
    }
    
    # Option C: Let the elders decide through tournament
    option = {
        name = murim_elders.0002.c
        add_character_modifier = {
            modifier = murim_succession_tournament_declared
            years = 2
        }
        add_prestige = 200
        trigger_event = {
            id = murim_elders.0003
            days = { 30 90 }
        }
        ai_chance = {
            base = 25
            modifier = {
                factor = 2.0
                has_trait = just
            }
        }
    }
    
    # Option D: Refuse to name successor - let fate decide
    option = {
        name = murim_elders.0002.d
        add_stress = 15
        every_courtier = {
            limit = { is_alive = yes age >= 25 }
            add_opinion = {
                target = root
                modifier = murim_no_succession_plan
                opinion = -10
            }
        }
        ai_chance = {
            base = 10
            modifier = {
                factor = 2.0
                has_trait = lazy
            }
        }
    }
